 257     private void justifyRows(int from, int to) {
 258         // Sometimes the DefaultTableModel is subclassed
 259         // instead of the AbstractTableModel by mistake.
 260         // Set the number of rows for the case when getRowCount
 261         // is overridden.
 262         dataVector.setSize(getRowCount());
 263 
 264         for (int i = from; i < to; i++) {
 265             if (dataVector.elementAt(i) == null) {
 266                 dataVector.setElementAt(new Vector(), i);
 267             }
 268             ((Vector)dataVector.elementAt(i)).setSize(getColumnCount());
 269         }
 270     }

 392     private static int gcd(int i, int j) {
 393         return (j == 0) ? i : gcd(j, i%j);
 394     }

 396     private static void rotate(Vector v, int a, int b, int shift) {
 397         int size = b - a;
 398         int r = size - shift;
 399         int g = gcd(size, r);
 400         for(int i = 0; i < g; i++) {
 401             int to = i;
 402             Object tmp = v.elementAt(a + to);
 403             for(int from = (to + r) % size; from != i; from = (to + r) % size) {
 404                 v.setElementAt(v.elementAt(a + from), a + to);
 405                 to = from;
 406             }
 407             v.setElementAt(tmp, a + to);
 408         }
 409     }

 411     /**
 412      *  Moves one or more rows from the inclusive range <code>start</code> to
 413      *  <code>end</code> to the <code>to</code> position in the model.
 414      *  After the move, the row that was at index <code>start</code>
 415      *  will be at index <code>to</code>.
 416      *  This method will send a <code>tableChanged</code> notification
 417        message to all the listeners.
 418      *
 419      *  <pre>
 420      *  Examples of moves:
 421      *
 422      *  1. moveRow(1,3,5);
 423      *          a|B|C|D|e|f|g|h|i|j|k   - before
 424      *          a|e|f|g|h|B|C|D|i|j|k   - after
 425      *
 426      *  2. moveRow(6,7,1);
 427      *          a|b|c|d|e|f|G|H|i|j|k   - before
 428      *          a|G|H|b|c|d|e|f|i|j|k   - after
 429      *  </pre>
 430      *
 431      * @param   start       the starting row index to be moved
 432      * @param   end         the ending row index to be moved
 433      * @param   to          the destination of the rows to be moved
 434      * @exception  ArrayIndexOutOfBoundsException  if any of the elements
 435      * would be moved out of the table's range
 436      *
 437      */
 438     public void moveRow(int start, int end, int to) {
 439         int shift = to - start;
 440         int first, last;
 441         if (shift < 0) {
 442             first = to;
 443             last = end;
 444         }
 445         else {
 446             first = start;
 447             last = to + end - start;
 448         }
 449         rotate(dataVector, first, last + 1, shift);
 450 
 451         fireTableRowsUpdated(first, last);
 452     }

 536     /**
 537      *  Adds a column to the model.  The new column will have the
 538      *  identifier <code>columnName</code>, which may be null.
 539      *  <code>columnData</code> is the
 540      *  optional vector of data for the column.  If it is <code>null</code>
 541      *  the column is filled with <code>null</code> values.  Otherwise,
 542      *  the new data will be added to model starting with the first
 543      *  element going to row 0, etc.  This method will send a
 544      *  <code>tableChanged</code> notification message to all the listeners.
 545      *
 546      * @param   columnName the identifier of the column being added
 547      * @param   columnData       optional data of the column being added
 548      */
 549     public void addColumn(Object columnName, Vector columnData) {
 550         columnIdentifiers.addElement(columnName);
 551         if (columnData != null) {
 552             int columnSize = columnData.size();
 553             if (columnSize > getRowCount()) {
 554                 dataVector.setSize(columnSize);
 555             }
 556             justifyRows(0, getRowCount());
 557             int newColumn = getColumnCount() - 1;
 558             for(int i = 0; i < columnSize; i++) {
 559                   Vector row = (Vector)dataVector.elementAt(i);
 560                   row.setElementAt(columnData.elementAt(i), newColumn);
 561             }
 562         }
 563         else {
 564             justifyRows(0, getRowCount());
 565         }
 566 
 567         fireTableStructureChanged();
 568     }

 673     /**
 674      * Returns a vector that contains the same objects as the array.
 675      * @param anArray  the array to be converted
 676      * @return  the new vector; if <code>anArray</code> is <code>null</code>,
 677      *                          returns <code>null</code>
 678      */
 679     protected static Vector convertToVector(Object[] anArray) {
 680         if (anArray == null) {
 681             return null;
 682         }
 683         Vector<Object> v = new Vector<Object>(anArray.length);
 684         for (Object o : anArray) {
 685             v.addElement(o);
 686         }
 687         return v;
 688     }

 690     /**
 691      * Returns a vector of vectors that contains the same objects as the array.
 692      * @param anArray  the double array to be converted
 693      * @return the new vector of vectors; if <code>anArray</code> is
 694      *                          <code>null</code>, returns <code>null</code>

 697         if (anArray == null) {
 698             return null;

 701         for (Object[] o : anArray) {
 702             v.addElement(convertToVector(o));
