  27 /**
  28  * This class is in support of SerializerToHTML, and acts as a sort

  38   /** Element's flags, describing the role this element plays during

  43    * responsibility to disambiguate these if necessary. */

 115   /**
 116    * Construct an ElementDescription with an initial set of flags.
 117    *
 118    * @param flags Element flags
 119    * @see m_flags
 120    */
 121   ElemDesc(int flags)
 122   {
 123     m_flags = flags;
 124   }
 125 
 126   /**
 127    * "is (this element described by these flags)".
 128    *
 129    * This might more properly be called areFlagsSet(). It accepts an
 130    * integer (being used as a bitvector) and checks whether all the
 131    * corresponding bits are set in our internal flags. Note that this
 132    * test is performed as a bitwise AND, not an equality test, so a
 133    * 0 bit in the input means "don't test", not "must be set false".
 134    *
 135    * @param flags Vector of flags to compare against this element's flags
 136    *
 137    * @return true if the flags set in the parameter are also set in the
 138    * element's stored flags.
 139    *
 140    * @see m_flags
 141    * @see isAttrFlagSet
 142    */
 143   boolean is(int flags)
 144   {
 145     // int which = (m_flags & flags);
 146     return (m_flags & flags) != 0;
 147   }
 148 
 149   /**
 150    * Set a new attribute for this element
 151    *
 152    *
 153    * @param name Attribute name
 154    * @param flags Attibute flags
 155    */
 156   void setAttr(String name, int flags)
 157   {
 158 
 159     if (null == m_attrs)
 160       m_attrs = new Hashtable();
 161 
 162     m_attrs.put(name, new Integer(flags));
 163   }
 164 
 165   /**
 166    * Find out if a flag is set in a given attribute of this element
 167    *
 168    *
 169    * @param name Attribute name
 170    * @param flags Flag to check
 171    *
 172    * @return True if the flag is set in the attribute. Returns false
 173    * if the attribute is not found
 174    * @see m_flags
 175    */
 176   boolean isAttrFlagSet(String name, int flags)
 177   {
 178 
 179     if (null != m_attrs)
 180     {
 181       Integer _flags = (Integer) m_attrs.get(name);
 182 
 183       if (null != _flags)
 184       {
 185         return (_flags.intValue() & flags) != 0;
 186       }
 187     }
 188 
 189     return false;
 190   }
