  23 package com.sun.org.apache.xml.internal.utils;

  25 /**
  26  * A very simple table that stores a list of strings, optimized
  27  * for small lists.
  28  * @xsl.usage internal
  29  */
  30 public class StringVector implements java.io.Serializable
  31 {
  32     static final long serialVersionUID = 4995234972032919748L;
  33 
  34   /** @serial Size of blocks to allocate           */
  35   protected int m_blocksize;
  36 
  37   /** @serial Array of strings this contains          */
  38   protected String m_map[];
  39 
  40   /** @serial Number of strings this contains          */
  41   protected int m_firstFree = 0;
  42 
  43   /** @serial Size of the array          */
  44   protected int m_mapSize;
  45 
  46   /**
  47    * Default constructor.  Note that the default
  48    * block size is very small, for small lists.
  49    */
  50   public StringVector()
  51   {
  52 
  53     m_blocksize = 8;
  54     m_mapSize = m_blocksize;
  55     m_map = new String[m_blocksize];
  56   }
  57 
  58   /**
  59    * Construct a StringVector, using the given block size.
  60    *
  61    * @param blocksize Size of the blocks to allocate
  62    */
  63   public StringVector(int blocksize)
  64   {
  65 
  66     m_blocksize = blocksize;
  67     m_mapSize = blocksize;
  68     m_map = new String[blocksize];
  69   }
  70 
  71   /**
  72    * Get the length of the list.
  73    *
  74    * @return Number of strings in the list
  75    */
  76   public int getLength()
  77   {
  78     return m_firstFree;
  79   }
  80 
  81   /**
  82    * Get the length of the list.
  83    *
  84    * @return Number of strings in the list
  85    */
  86   public final int size()
  87   {
  88     return m_firstFree;
  89   }
  90 
  91   /**
  92    * Append a string onto the vector.
  93    *
  94    * @param value Sting to add to the vector
  95    */
  96   public final void addElement(String value)
  97   {
  98 
  99     if ((m_firstFree + 1) >= m_mapSize)
 100     {
 101       m_mapSize += m_blocksize;
 102 
 103       String newMap[] = new String[m_mapSize];
 104 
 105       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
 106 
 107       m_map = newMap;
 108     }
 109 
 110     m_map[m_firstFree] = value;
 111 
 112     m_firstFree++;
 113   }
 114 
 115   /**
 116    * Get the nth element.
 117    *
 118    * @param i Index of string to find
 119    *
 120    * @return String at given index
 121    */
 122   public final String elementAt(int i)
 123   {
 124     return m_map[i];
 125   }
 126 
 127   /**
 128    * Tell if the table contains the given string.
 129    *
 130    * @param s String to look for
 131    *
 132    * @return True if the string is in this table
 133    */
 134   public final boolean contains(String s)
 135   {
 136 
 137     if (null == s)
 138       return false;
 139 
 140     for (int i = 0; i < m_firstFree; i++)
 141     {
 142       if (m_map[i].equals(s))
 143         return true;
 144     }
 145 
 146     return false;
 147   }
 148 
 149   /**
 150    * Tell if the table contains the given string. Ignore case.
 151    *
 152    * @param s String to find
 153    *
 154    * @return True if the String is in this vector
 155    */
 156   public final boolean containsIgnoreCase(String s)
 157   {
 158 
 159     if (null == s)
 160       return false;
 161 
 162     for (int i = 0; i < m_firstFree; i++)
 163     {
 164       if (m_map[i].equalsIgnoreCase(s))
 165         return true;
 166     }
 167 
 168     return false;
 169   }
 170 
 171   /**
 172    * Tell if the table contains the given string.
 173    *
 174    * @param s String to push into the vector
 175    */
 176   public final void push(String s)
 177   {
 178 
 179     if ((m_firstFree + 1) >= m_mapSize)
 180     {
 181       m_mapSize += m_blocksize;
 182 
 183       String newMap[] = new String[m_mapSize];
 184 
 185       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
 186 
 187       m_map = newMap;
 188     }
 189 
 190     m_map[m_firstFree] = s;
 191 
 192     m_firstFree++;
 193   }
 194 
 195   /**
 196    * Pop the tail of this vector.
 197    *
 198    * @return The String last added to this vector or null not found.
 199    * The string is removed from the vector.
 200    */
 201   public final String pop()
 202   {
 203 
 204     if (m_firstFree <= 0)
 205       return null;
 206 
 207     m_firstFree--;
 208 
 209     String s = m_map[m_firstFree];
 210 
 211     m_map[m_firstFree] = null;
 212 
 213     return s;
 214   }
 215 
 216   /**
 217    * Get the string at the tail of this vector without popping.
 218    *
 219    * @return The string at the tail of this vector.
 220    */
 221   public final String peek()
 222   {
 223     return (m_firstFree <= 0) ? null : m_map[m_firstFree - 1];
 224   }
 225 }
