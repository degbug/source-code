 127   /**
 128    * Perform a pre-order traversal non-recursive style.
 129    *
 130    * Note that TreeWalker assumes that the subtree is intended to represent
 131    * a complete (though not necessarily well-formed) document and, during a
 132    * traversal, startDocument and endDocument will always be issued to the
 133    * SAX listener.
 134    *
 135    * @param pos Node in the tree where to start traversal
 136    *
 137    * @throws TransformerException
 138    */
 139   public void traverse(Node pos) throws org.xml.sax.SAXException
 140   {
 141 
 142     this.m_contentHandler.startDocument();
 143 
 144     Node top = pos;
 145 
 146     while (null != pos)
 147     {
 148       startNode(pos);
 149 
 150       Node nextNode = pos.getFirstChild();
 151 
 152       while (null == nextNode)
 153       {
 154         endNode(pos);
 155 
 156         if (top.equals(pos))
 157           break;
 158 
 159         nextNode = pos.getNextSibling();
 160 
 161         if (null == nextNode)
 162         {
 163           pos = pos.getParentNode();
 164 
 165           if ((null == pos) || (top.equals(pos)))
 166           {
 167             if (null != pos)
 168               endNode(pos);
 169 
 170             nextNode = null;
 171 
 172             break;
 173           }
 174         }
 175       }
 176 
 177       pos = nextNode;
 178     }
 179     this.m_contentHandler.endDocument();
 180   }

 182   /**
 183    * Perform a pre-order traversal non-recursive style.
 184 
 185    * Note that TreeWalker assumes that the subtree is intended to represent
 186    * a complete (though not necessarily well-formed) document and, during a
 187    * traversal, startDocument and endDocument will always be issued to the
 188    * SAX listener.
 189    *
 190    * @param pos Node in the tree where to start traversal
 191    * @param top Node in the tree where to end traversal
 192    *
 193    * @throws TransformerException
 194    */
 195   public void traverse(Node pos, Node top) throws org.xml.sax.SAXException
 196   {
 197 
 198     this.m_contentHandler.startDocument();
 199 
 200     while (null != pos)
 201     {
 202       startNode(pos);
 203 
 204       Node nextNode = pos.getFirstChild();
 205 
 206       while (null == nextNode)
 207       {
 208         endNode(pos);
 209 
 210         if ((null != top) && top.equals(pos))
 211           break;
 212 
 213         nextNode = pos.getNextSibling();
 214 
 215         if (null == nextNode)
 216         {
 217           pos = pos.getParentNode();
 218 
 219           if ((null == pos) || ((null != top) && top.equals(pos)))
 220           {
 221             nextNode = null;
 222 
 223             break;
 224           }
 225         }
 226       }
 227 
 228       pos = nextNode;
 229     }
 230     this.m_contentHandler.endDocument();
 231   }

 249       this.m_contentHandler.characters(data.toCharArray(), 0, data.length());

 253   /**
 254    * Start processing given node
 255    *
 256    *
 257    * @param node Node to process
 258    *
 259    * @throws org.xml.sax.SAXException
 260    */
 261   protected void startNode(Node node) throws org.xml.sax.SAXException
 262   {
 263 
 264 //   TODO: 
 265 //    A Serializer implements ContentHandler, but not NodeConsumer
 266 //    so drop this reference to NodeConsumer which would otherwise
 267 //    pull in all sorts of things
 268 //    if (m_contentHandler instanceof NodeConsumer)
 269 //    {
 270 //      ((NodeConsumer) m_contentHandler).setOriginatingNode(node);
 271 //    }
 272 //    TODO: 
 273 
 274                 if (node instanceof Locator)
 275                 {
 276                         Locator loc = (Locator)node;
 277                         m_locator.setColumnNumber(loc.getColumnNumber());
 278                         m_locator.setLineNumber(loc.getLineNumber());
 279                         m_locator.setPublicId(loc.getPublicId());
 280                         m_locator.setSystemId(loc.getSystemId());
 281                 }
 282                 else
 283                 {
 284                         m_locator.setColumnNumber(0);
 285       m_locator.setLineNumber(0);
 286                 }
 287 
 288     switch (node.getNodeType())
 289     {
 290     case Node.COMMENT_NODE :
 291     {
 292       String data = ((Comment) node).getData();
 293 
 294       if (m_contentHandler instanceof LexicalHandler)
 295       {
 296         LexicalHandler lh = ((LexicalHandler) this.m_contentHandler);
 297 
 298         lh.comment(data.toCharArray(), 0, data.length());
 299       }
 300     }
 301     break;
 302     case Node.DOCUMENT_FRAGMENT_NODE :
 303 
 304       // ??;
 305       break;
 306     case Node.DOCUMENT_NODE :
 307 
 308       break;
 309     case Node.ELEMENT_NODE :
 310       Element elem_node = (Element) node;
 311       {
 312           // Make sure the namespace node
 313           // for the element itself is declared
 314           // to the ContentHandler
 315           String uri = elem_node.getNamespaceURI();
 316           if (uri != null) {
 317               String prefix = elem_node.getPrefix();
 318               if (prefix==null)
 319                 prefix="";
 320               this.m_contentHandler.startPrefixMapping(prefix,uri);
 321           }
 322       }
 323       NamedNodeMap atts = elem_node.getAttributes();
 324       int nAttrs = atts.getLength();
 325       // System.out.println("TreeWalker#startNode: "+node.getNodeName());
 326 
 327 
 328       // Make sure the namespace node of
 329       // each attribute is declared to the ContentHandler
 330       for (int i = 0; i < nAttrs; i++)
 331       {
 332         final Node attr = atts.item(i);
 333         final String attrName = attr.getNodeName();
 334         final int colon = attrName.indexOf(':');
 335         final String prefix;
 336 
 337         // System.out.println("TreeWalker#startNode: attr["+i+"] = "+attrName+", "+attr.getNodeValue());
 338         if (attrName.equals("xmlns") || attrName.startsWith("xmlns:"))
 339         {
 340           // Use "" instead of null, as Xerces likes "" for the
 341           // name of the default namespace.  Fix attributed
 342           // to "Steven Murray" .
 343           if (colon < 0)
 344             prefix = "";
 345           else
 346             prefix = attrName.substring(colon + 1);
 347 
 348           this.m_contentHandler.startPrefixMapping(prefix,
 349                                                    attr.getNodeValue());
 350         }
 351         else if (colon > 0) {
 352             prefix = attrName.substring(0,colon);
 353             String uri = attr.getNamespaceURI();
 354             if (uri != null)
 355                 this.m_contentHandler.startPrefixMapping(prefix,uri);
 356         }
 357       }
 358 
 359       String ns = m_dh.getNamespaceOfNode(node);
 360       if(null == ns)
 361         ns = "";
 362       this.m_contentHandler.startElement(ns,
 363                                          m_dh.getLocalNameOfNode(node),
 364                                          node.getNodeName(),
 365                                          new AttList(atts, m_dh));
 366       break;
 367     case Node.PROCESSING_INSTRUCTION_NODE :
 368     {
 369       ProcessingInstruction pi = (ProcessingInstruction) node;
 370       String name = pi.getNodeName();
 371 
 372       // String data = pi.getData();
 373       if (name.equals("xslt-next-is-raw"))
 374       {
 375         nextIsRaw = true;
 376       }
 377       else
 378       {
 379         this.m_contentHandler.processingInstruction(pi.getNodeName(),
 380                                                     pi.getData());
 381       }
 382     }
 383     break;
 384     case Node.CDATA_SECTION_NODE :
 385     {
 386       boolean isLexH = (m_contentHandler instanceof LexicalHandler);
 387       LexicalHandler lh = isLexH
 388                           ? ((LexicalHandler) this.m_contentHandler) : null;
 389 
 390       if (isLexH)
 391       {
 392         lh.startCDATA();
 393       }
 394 
 395       dispatachChars(node);
 396 
 397       {
 398         if (isLexH)
 399         {
 400           lh.endCDATA();
 401         }
 402       }
 403     }
 404     break;
 405     case Node.TEXT_NODE :
 406     {
 407       //String data = ((Text) node).getData();
 408 
 409       if (nextIsRaw)
 410       {
 411         nextIsRaw = false;
 412 
 413         m_contentHandler.processingInstruction(javax.xml.transform.Result.PI_DISABLE_OUTPUT_ESCAPING, "");
 414         dispatachChars(node);
 415         m_contentHandler.processingInstruction(javax.xml.transform.Result.PI_ENABLE_OUTPUT_ESCAPING, "");
 416       }
 417       else
 418       {
 419         dispatachChars(node);
 420       }
 421     }
 422     break;
 423     case Node.ENTITY_REFERENCE_NODE :
 424     {
 425       EntityReference eref = (EntityReference) node;
 426 
 427       if (m_contentHandler instanceof LexicalHandler)
 428       {
 429         ((LexicalHandler) this.m_contentHandler).startEntity(
 430           eref.getNodeName());
 431       }
 432       else
 433       {
 434 
 435         // warning("Can not output entity to a pure SAX ContentHandler");
 436       }
 437     }
 438     break;
 439     default :
 440     }

 443   /**
 444    * End processing of given node
 445    *
 446    *
 447    * @param node Node we just finished processing
 448    *
 449    * @throws org.xml.sax.SAXException
 450    */
 451   protected void endNode(Node node) throws org.xml.sax.SAXException
 452   {
 453 
 454     switch (node.getNodeType())
 455     {
 456     case Node.DOCUMENT_NODE :
 457       break;
 458 
 459     case Node.ELEMENT_NODE :
 460       String ns = m_dh.getNamespaceOfNode(node);
 461       if(null == ns)
 462         ns = "";
 463       this.m_contentHandler.endElement(ns,
 464                                          m_dh.getLocalNameOfNode(node),
 465                                          node.getNodeName());
 466 
 467       if (m_Serializer == null) {
 468       // Don't bother with endPrefixMapping calls if the ContentHandler is a
 469       // SerializationHandler because SerializationHandler's ignore the
 470       // endPrefixMapping() calls anyways. . . .  This is an optimization.
 471       Element elem_node = (Element) node;
 472       NamedNodeMap atts = elem_node.getAttributes();
 473       int nAttrs = atts.getLength();
 474 
 475       // do the endPrefixMapping calls in reverse order
 476       // of the startPrefixMapping calls
 477       for (int i = (nAttrs-1); 0 <= i; i--)
 478       {
 479         final Node attr = atts.item(i);
 480         final String attrName = attr.getNodeName();
 481         final int colon = attrName.indexOf(':');
 482         final String prefix;
 483 
 484         if (attrName.equals("xmlns") || attrName.startsWith("xmlns:"))
 485         {
 486           // Use "" instead of null, as Xerces likes "" for the
 487           // name of the default namespace.  Fix attributed
 488           // to "Steven Murray" .
 489           if (colon < 0)
 490             prefix = "";
 491           else
 492             prefix = attrName.substring(colon + 1);
 493 
 494           this.m_contentHandler.endPrefixMapping(prefix);
 495         }
 496         else if (colon > 0) {
 497             prefix = attrName.substring(0, colon);
 498             this.m_contentHandler.endPrefixMapping(prefix);
 499         }
 500       }
 501       {
 502           String uri = elem_node.getNamespaceURI();
 503           if (uri != null) {
 504               String prefix = elem_node.getPrefix();
 505               if (prefix==null)
 506                 prefix="";
 507               this.m_contentHandler.endPrefixMapping(prefix);
 508           }
 509       }
 510       }
 511       break;
 512     case Node.CDATA_SECTION_NODE :
 513       break;
 514     case Node.ENTITY_REFERENCE_NODE :
 515     {
 516       EntityReference eref = (EntityReference) node;
 517 
 518       if (m_contentHandler instanceof LexicalHandler)
 519       {
 520         LexicalHandler lh = ((LexicalHandler) this.m_contentHandler);
 521 
 522         lh.endEntity(eref.getNodeName());
 523       }
 524     }
 525     break;
 526     default :
 527     }
 528   }
