  53 /* We supports warmup for UI stack that is performed in parallel
  54  * to VM initialization.
  55  * This helps to improve startup of UI application as warmup phase
  56  * might be long due to initialization of OS or hardware resources.
  57  * It is not CPU bound and therefore it does not interfere with VM init.
  58  * Obviously such warmup only has sense for UI apps and therefore it needs
  59  * to be explicitly requested by passing -Dsun.awt.warmup=true property
  60  * (this is always the case for plugin/javaws).

  62  * Implementation launches new thread after VM starts and use it to perform

  64  * This thread is later reused as AWT toolkit thread as graphics toolkit

 104 /* Extracts value of a parameter with the specified name
 105  * from command line argument (returns pointer in the argument).
 106  * Returns NULL if the argument does not contains the parameter.
 107  * e.g.:
 108  * GetParamValue("theParam", "theParam=value") returns pointer to "value".
 109  */
 110 const char * GetParamValue(const char *paramName, const char *arg) {
 111     int nameLen = JLI_StrLen(paramName);
 112     if (JLI_StrNCmp(paramName, arg, nameLen) == 0) {
 113         /* arg[nameLen] is valid (may contain final NULL) */
 114         if (arg[nameLen] == '=') {
 115             return arg + nameLen + 1;
 116         }
 117     }
 118     return NULL;
 119 }
 120 
 121 /* Checks if commandline argument contains property specified
 122  * and analyze it as boolean property (true/false).
 123  * Returns -1 if the argument does not contain the parameter;
 124  * Returns 1 if the argument contains the parameter and its value is "true";
 125  * Returns 0 if the argument contains the parameter and its value is "false".
 126  */
 127 int GetBoolParamValue(const char *paramName, const char *arg) {
 128     const char * paramValue = GetParamValue(paramName, arg);
 129     if (paramValue != NULL) {
 130         if (JLI_StrCaseCmp(paramValue, "true") == 0) {
 131             return 1;
 132         }
 133         if (JLI_StrCaseCmp(paramValue, "false") == 0) {
 134             return 0;
 135         }
 136     }
 137     return -1;
 138 }

 171 CreateExecutionEnvironment(int *pargc, char ***pargv,
 172                            char *jrepath, jint so_jrepath,
 173                            char *jvmpath, jint so_jvmpath,
 174                            char *jvmcfg,  jint so_jvmcfg) {
 175     char * jvmtype;
 176     int i = 0;
 177     int running = CURRENT_DATA_MODEL;
 178 
 179     int wanted = running;
 180 
 181     char** argv = *pargv;
 182     for (i = 1; i < *pargc ; i++) {
 183         if (JLI_StrCmp(argv[i], "-J-d64") == 0 || JLI_StrCmp(argv[i], "-d64") == 0) {
 184             wanted = 64;
 185             continue;
 186         }
 187         if (JLI_StrCmp(argv[i], "-J-d32") == 0 || JLI_StrCmp(argv[i], "-d32") == 0) {
 188             wanted = 32;
 189             continue;
 190         }
 191 
 192         if (IsJavaArgs() && argv[i][0] != '-')
 193             continue;
 194         if (argv[i][0] != '-')
 195             break;
 196     }
 197     if (running != wanted) {
 198         JLI_ReportErrorMessage(JRE_ERROR2, wanted);
 199         exit(1);
 200     }
 201 
 202     /* Find out where the JRE is that we will be using. */
 203     if (!GetJREPath(jrepath, so_jrepath)) {
 204         JLI_ReportErrorMessage(JRE_ERROR1);
 205         exit(2);
 206     }
 207 
 208     JLI_Snprintf(jvmcfg, so_jvmcfg, "%s%slib%s%s%sjvm.cfg",
 209         jrepath, FILESEP, FILESEP, (char*)GetArch(), FILESEP);
 210 
 211     /* Find the specified JVM type */
 212     if (ReadKnownVMs(jvmcfg, JNI_FALSE) < 1) {
 213         JLI_ReportErrorMessage(CFG_ERROR7);
 214         exit(1);
 215     }
 216 
 217     jvmtype = CheckJvmType(pargc, pargv, JNI_FALSE);
 218     if (JLI_StrCmp(jvmtype, "ERROR") == 0) {
 219         JLI_ReportErrorMessage(CFG_ERROR9);
 220         exit(4);
 221     }
 222 
 223     jvmpath[0] = '\0';
 224     if (!GetJVMPath(jrepath, jvmtype, jvmpath, so_jvmpath)) {
 225         JLI_ReportErrorMessage(CFG_ERROR8, jvmtype, jvmpath);
 226         exit(4);
 227     }
 228     /* If we got here, jvmpath has been correctly initialized. */
 229 
 230     /* Check if we need preload AWT */
 231 #ifdef ENABLE_AWT_PRELOAD
 232     argv = *pargv;
 233     for (i = 0; i < *pargc ; i++) {
 234         /* Tests the "turn on" parameter only if not set yet. */
 235         if (awtPreloadD3D < 0) {
 236             if (GetBoolParamValue(PARAM_PRELOAD_D3D, argv[i]) == 1) {
 237                 awtPreloadD3D = 1;
 238             }
 239         }
 240         /* Test parameters which can disable preloading if not already disabled. */
 241         if (awtPreloadD3D != 0) {
 242             if (GetBoolParamValue(PARAM_NODDRAW, argv[i]) == 1
 243                 || GetBoolParamValue(PARAM_D3D, argv[i]) == 0
 244                 || GetBoolParamValue(PARAM_OPENGL, argv[i]) == 1)
 245             {
 246                 awtPreloadD3D = 0;
 247                 /* no need to test the rest of the parameters */
 248                 break;
 249             }
 250         }
 251     }
 252 #endif /* ENABLE_AWT_PRELOAD */
 253 }

 256 static jboolean
 257 LoadMSVCRT()
 258 {
 259     // Only do this once
 260     static int loaded = 0;
 261     char crtpath[MAXPATHLEN];
 262 
 263     if (!loaded) {
 264         /*
 265          * The Microsoft C Runtime Library needs to be loaded first.  A copy is
 266          * assumed to be present in the "JRE path" directory.  If it is not found
 267          * there (or "JRE path" fails to resolve), skip the explicit load and let
 268          * nature take its course, which is likely to be a failure to execute.
 269          * This is clearly completely specific to the exact compiler version
 270          * which isn't very nice, but its hardly the only place.
 271          * No attempt to look for compiler versions in between 2003 and 2010
 272          * as we aren't supporting building with those.
 273          */
 274 #ifdef _MSC_VER
 275 #if _MSC_VER < 1400
 276 #define CRT_DLL "msvcr71.dll"
 277 #endif
 278 #if _MSC_VER >= 1600
 279 #define CRT_DLL "msvcr100.dll"
 280 #endif
 281 #ifdef CRT_DLL
 282         if (GetJREPath(crtpath, MAXPATHLEN)) {
 283             if (JLI_StrLen(crtpath) + JLI_StrLen("\\bin\\") +
 284                     JLI_StrLen(CRT_DLL) >= MAXPATHLEN) {
 285                 JLI_ReportErrorMessage(JRE_ERROR11);
 286                 return JNI_FALSE;
 287             }
 288             (void)JLI_StrCat(crtpath, "\\bin\\" CRT_DLL);   /* Add crt dll */
 289             JLI_TraceLauncher("CRT path is %s\n", crtpath);
 290             if (_access(crtpath, 0) == 0) {
 291                 if (LoadLibrary(crtpath) == 0) {
 292                     JLI_ReportErrorMessage(DLL_ERROR4, crtpath);
 293                     return JNI_FALSE;
 294                 }
 295             }
 296         }
 297 #endif /* CRT_DLL */
 298 #endif /* _MSC_VER */
 299         loaded = 1;
 300     }
 301     return JNI_TRUE;
 302 }

 305 /*
 306  * Find path to JRE based on .exe's location or registry settings.
 307  */
 308 jboolean
 309 GetJREPath(char *path, jint pathsize)
 310 {
 311     char javadll[MAXPATHLEN];
 312     struct stat s;
 313 
 314     if (GetApplicationHome(path, pathsize)) {
 315         /* Is JRE co-located with the application? */
 316         JLI_Snprintf(javadll, sizeof(javadll), "%s\\bin\\" JAVA_DLL, path);
 317         if (stat(javadll, &s) == 0) {
 318             JLI_TraceLauncher("JRE path is %s\n", path);
 319             return JNI_TRUE;
 320         }
 321         /* ensure storage for path + \jre + NULL */
 322         if ((JLI_StrLen(path) + 4 + 1) > pathsize) {
 323             JLI_TraceLauncher("Insufficient space to store JRE path\n");
 324             return JNI_FALSE;
 325         }
 326         /* Does this app ship a private JRE in <apphome>\jre directory? */
 327         JLI_Snprintf(javadll, sizeof (javadll), "%s\\jre\\bin\\" JAVA_DLL, path);
 328         if (stat(javadll, &s) == 0) {
 329             JLI_StrCat(path, "\\jre");
 330             JLI_TraceLauncher("JRE path is %s\n", path);
 331             return JNI_TRUE;
 332         }
 333     }
 334 
 335     /* Look for a public JRE on this machine. */
 336     if (GetPublicJREHome(path, pathsize)) {
 337         JLI_TraceLauncher("JRE path is %s\n", path);
 338         return JNI_TRUE;
 339     }
 340 
 341     JLI_ReportErrorMessage(JRE_ERROR8 JAVA_DLL);
 342     return JNI_FALSE;
 343 
 344 }

 346 /*
 347  * Given a JRE location and a JVM type, construct what the name the
 348  * JVM shared library will be.  Return true, if such a library
 349  * exists, false otherwise.
 350  */
 351 static jboolean
 352 GetJVMPath(const char *jrepath, const char *jvmtype,
 353            char *jvmpath, jint jvmpathsize)
 354 {
 355     struct stat s;
 356     if (JLI_StrChr(jvmtype, '/') || JLI_StrChr(jvmtype, '\\')) {
 357         JLI_Snprintf(jvmpath, jvmpathsize, "%s\\" JVM_DLL, jvmtype);
 358     } else {
 359         JLI_Snprintf(jvmpath, jvmpathsize, "%s\\bin\\%s\\" JVM_DLL,

 362     if (stat(jvmpath, &s) == 0) {
 363         return JNI_TRUE;
 364     } else {
 365         return JNI_FALSE;
 366     }

 369 /*
 370  * Load a jvm from "jvmpath" and initialize the invocation functions.
 371  */
 372 jboolean
 373 LoadJavaVM(const char *jvmpath, InvocationFunctions *ifn)
 374 {
 375     HINSTANCE handle;
 376 
 377     JLI_TraceLauncher("JVM path is %s\n", jvmpath);
 378 
 379     /*
 380      * The Microsoft C Runtime Library needs to be loaded first.  A copy is
 381      * assumed to be present in the "JRE path" directory.  If it is not found
 382      * there (or "JRE path" fails to resolve), skip the explicit load and let
 383      * nature take its course, which is likely to be a failure to execute.
 384      *
 385      */
 386     LoadMSVCRT();
 387 
 388     /* Load the Java VM DLL */
 389     if ((handle = LoadLibrary(jvmpath)) == 0) {
 390         JLI_ReportErrorMessage(DLL_ERROR4, (char *)jvmpath);
 391         return JNI_FALSE;
 392     }
 393 
 394     /* Now get the function addresses */
 395     ifn->CreateJavaVM =
 396         (void *)GetProcAddress(handle, "JNI_CreateJavaVM");
 397     ifn->GetDefaultJavaVMInitArgs =
 398         (void *)GetProcAddress(handle, "JNI_GetDefaultJavaVMInitArgs");
 399     if (ifn->CreateJavaVM == 0 || ifn->GetDefaultJavaVMInitArgs == 0) {
 400         JLI_ReportErrorMessage(JNI_ERROR1, (char *)jvmpath);
 401         return JNI_FALSE;
 402     }
 403 
 404     return JNI_TRUE;
 405 }

 408  * If app is "c:\foo\bin\javac", then put "c:\foo" into buf.

 410 jboolean
 411 GetApplicationHome(char *buf, jint bufsize)
 412 {
 413     char *cp;
 414     GetModuleFileName(0, buf, bufsize);
 415     *JLI_StrRChr(buf, '\\') = '\0'; /* remove .exe file name */
 416     if ((cp = JLI_StrRChr(buf, '\\')) == 0) {
 417         /* This happens if the application is in a drive root, and
 418          * there is no bin directory. */
 419         buf[0] = '\0';
 420         return JNI_FALSE;
 421     }
 422     *cp = '\0';  /* remove the bin\ part */
 423     return JNI_TRUE;
 424 }

 432 static jboolean
 433 GetStringFromRegistry(HKEY key, const char *name, char *buf, jint bufsize)
 434 {
 435     DWORD type, size;
 436 
 437     if (RegQueryValueEx(key, name, 0, &type, 0, &size) == 0
 438         && type == REG_SZ
 439         && (size < (unsigned int)bufsize)) {
 440         if (RegQueryValueEx(key, name, 0, 0, buf, &size) == 0) {
 441             return JNI_TRUE;
 442         }
 443     }
 444     return JNI_FALSE;

 447 static jboolean
 448 GetPublicJREHome(char *buf, jint bufsize)
 449 {
 450     HKEY key, subkey;
 451     char version[MAXPATHLEN];
 452 
 453     /*
 454      * Note: There is a very similar implementation of the following
 455      * registry reading code in the Windows java control panel (javacp.cpl).
 456      * If there are bugs here, a similar bug probably exists there.  Hence,
 457      * changes here require inspection there.
 458      */
 459 
 460     /* Find the current version of the JRE */
 461     if (RegOpenKeyEx(HKEY_LOCAL_MACHINE, JRE_KEY, 0, KEY_READ, &key) != 0) {
 462         JLI_ReportErrorMessage(REG_ERROR1, JRE_KEY);
 463         return JNI_FALSE;
 464     }
 465 
 466     if (!GetStringFromRegistry(key, "CurrentVersion",
 467                                version, sizeof(version))) {
 468         JLI_ReportErrorMessage(REG_ERROR2, JRE_KEY);
 469         RegCloseKey(key);
 470         return JNI_FALSE;
 471     }
 472 
 473     if (JLI_StrCmp(version, GetDotVersion()) != 0) {
 474         JLI_ReportErrorMessage(REG_ERROR3, JRE_KEY, version, GetDotVersion()
 475         );
 476         RegCloseKey(key);
 477         return JNI_FALSE;
 478     }
 479 
 480     /* Find directory where the current version is installed. */
 481     if (RegOpenKeyEx(key, version, 0, KEY_READ, &subkey) != 0) {
 482         JLI_ReportErrorMessage(REG_ERROR1, JRE_KEY, version);
 483         RegCloseKey(key);
 484         return JNI_FALSE;
 485     }
 486 
 487     if (!GetStringFromRegistry(subkey, "JavaHome", buf, bufsize)) {
 488         JLI_ReportErrorMessage(REG_ERROR4, JRE_KEY, version);
 489         RegCloseKey(key);
 490         RegCloseKey(subkey);
 491         return JNI_FALSE;
 492     }
 493 
 494     if (JLI_IsTraceLauncher()) {
 495         char micro[MAXPATHLEN];
 496         if (!GetStringFromRegistry(subkey, "MicroVersion", micro,
 497                                    sizeof(micro))) {
 498             printf("Warning: Can't read MicroVersion\n");
 499             micro[0] = '\0';
 500         }
 501         printf("Version major.minor.micro = %s.%s\n", version, micro);
 502     }
 503 
 504     RegCloseKey(key);
 505     RegCloseKey(subkey);
 506     return JNI_TRUE;
 507 }

 509 /*
 510  * Support for doing cheap, accurate interval timing.
 511  */
 512 static jboolean counterAvailable = JNI_FALSE;
 513 static jboolean counterInitialized = JNI_FALSE;
 514 static LARGE_INTEGER counterFrequency;
 515 
 516 jlong CounterGet()
 517 {
 518     LARGE_INTEGER count;
 519 
 520     if (!counterInitialized) {
 521         counterAvailable = QueryPerformanceFrequency(&counterFrequency);
 522         counterInitialized = JNI_TRUE;
 523     }
 524     if (!counterAvailable) {
 525         return 0;
 526     }
 527     QueryPerformanceCounter(&count);
 528     return (jlong)(count.QuadPart);
 529 }

 531 jlong Counter2Micros(jlong counts)
 532 {
 533     if (!counterAvailable || !counterInitialized) {
 534         return 0;
 535     }
 536     return (counts * 1000 * 1000)/counterFrequency.QuadPart;
 537 }
 538 /*
 539  * windows snprintf does not guarantee a null terminator in the buffer,
 540  * if the computed size is equal to or greater than the buffer size,
 541  * as well as error conditions. This function guarantees a null terminator
 542  * under all these conditions. An unreasonable buffer or size will return
 543  * an error value. Under all other conditions this function will return the
 544  * size of the bytes actually written minus the null terminator, similar
 545  * to ansi snprintf api. Thus when calling this function the caller must
 546  * ensure storage for the null terminator.
 547  */
 548 int
 549 JLI_Snprintf(char* buffer, size_t size, const char* format, ...) {
 550     int rc;
 551     va_list vl;
 552     if (size == 0 || buffer == NULL)
 553         return -1;
 554     buffer[0] = '\0';
 555     va_start(vl, format);
 556     rc = vsnprintf(buffer, size, format, vl);
 557     va_end(vl);
 558     /* force a null terminator, if something is amiss */
 559     if (rc < 0) {
 560         /* apply ansi semantics */
 561         buffer[size - 1] = '\0';
 562         return size;
 563     } else if (rc == size) {
 564         /* force a null terminator */
 565         buffer[size - 1] = '\0';
 566     }
 567     return rc;
 568 }

 594 /*
 595  * Just like JLI_ReportErrorMessage, except that it concatenates the system
 596  * error message if any, its upto the calling routine to correctly
 597  * format the separation of the messages.
 598  */
 599 void
 600 JLI_ReportErrorMessageSys(const char *fmt, ...)
 601 {
 602     va_list vl;
 603 
 604     int save_errno = errno;
 605     DWORD       errval;
 606     jboolean freeit = JNI_FALSE;
 607     char  *errtext = NULL;
 608 
 609     va_start(vl, fmt);
 610 
 611     if ((errval = GetLastError()) != 0) {               /* Platform SDK / DOS Error */
 612         int n = FormatMessage(FORMAT_MESSAGE_FROM_SYSTEM|
 613             FORMAT_MESSAGE_IGNORE_INSERTS|FORMAT_MESSAGE_ALLOCATE_BUFFER,
 614             NULL, errval, 0, (LPTSTR)&errtext, 0, NULL);
 615         if (errtext == NULL || n == 0) {                /* Paranoia check */
 616             errtext = "";
 617             n = 0;
 618         } else {
 619             freeit = JNI_TRUE;
 620             if (n > 2) {                                /* Drop final CR, LF */
 621                 if (errtext[n - 1] == '\n') n--;
 622                 if (errtext[n - 1] == '\r') n--;
 623                 errtext[n] = '\0';
 624             }
 625         }
 626     } else {   /* C runtime error that has no corresponding DOS error code */
 627         errtext = strerror(save_errno);
 628     }
 629 
 630     if (IsJavaw()) {
 631         char *message;
 632         int mlen;
 633         /* get the length of the string we need */
 634         int len = mlen =  _vscprintf(fmt, vl) + 1;
 635         if (freeit) {
 636            mlen += (int)JLI_StrLen(errtext);
 637         }
 638 
 639         message = (char *)JLI_MemAlloc(mlen);
 640         _vsnprintf(message, len, fmt, vl);
 641         message[len]='\0';
 642 
 643         if (freeit) {
 644            JLI_StrCat(message, errtext);
 645         }
 646 
 647         MessageBox(NULL, message, "Java Virtual Machine Launcher",
 648             (MB_OK|MB_ICONSTOP|MB_APPLMODAL));
 649 
 650         JLI_MemFree(message);

 681 /*
 682  * Determine if there is an acceptable JRE in the registry directory top_key.
 683  * Upon locating the "best" one, return a fully qualified path to it.
 684  * "Best" is defined as the most advanced JRE meeting the constraints
 685  * contained in the manifest_info. If no JRE in this directory meets the
 686  * constraints, return NULL.
 687  *
 688  * It doesn't matter if we get an error reading the registry, or we just
 689  * don't find anything interesting in the directory.  We just return NULL
 690  * in either case.
 691  */
 692 static char *
 693 ProcessDir(manifest_info* info, HKEY top_key) {
 694     DWORD   index = 0;
 695     HKEY    ver_key;
 696     char    name[MAXNAMELEN];
 697     int     len;
 698     char    *best = NULL;
 699 
 700     /*
 701      * Enumerate "<top_key>/SOFTWARE/JavaSoft/Java Runtime Environment"
 702      * searching for the best available version.
 703      */
 704     while (RegEnumKey(top_key, index, name, MAXNAMELEN) == ERROR_SUCCESS) {
 705         index++;
 706         if (JLI_AcceptableRelease(name, info->jre_version))
 707             if ((best == NULL) || (JLI_ExactVersionId(name, best) > 0)) {
 708                 if (best != NULL)
 709                     JLI_MemFree(best);
 710                 best = JLI_StringDup(name);
 711             }
 712     }
 713 
 714     /*
 715      * Extract "JavaHome" from the "best" registry directory and return
 716      * that path.  If no appropriate version was located, or there is an
 717      * error in extracting the "JavaHome" string, return null.
 718      */
 719     if (best == NULL)
 720         return (NULL);
 721     else {
 722         if (RegOpenKeyEx(top_key, best, 0, KEY_READ, &ver_key)
 723           != ERROR_SUCCESS) {
 724             JLI_MemFree(best);
 725             if (ver_key != NULL)
 726                 RegCloseKey(ver_key);
 727             return (NULL);
 728         }
 729         JLI_MemFree(best);
 730         len = MAXNAMELEN;
 731         if (RegQueryValueEx(ver_key, "JavaHome", NULL, NULL, (LPBYTE)name, &len)
 732           != ERROR_SUCCESS) {
 733             if (ver_key != NULL)
 734                 RegCloseKey(ver_key);
 735             return (NULL);
 736         }
 737         if (ver_key != NULL)
 738             RegCloseKey(ver_key);
 739         return (JLI_StringDup(name));
 740     }
 741 }

 743 /*
 744  * This is the global entry point. It examines the host for the optimal
 745  * JRE to be used by scanning a set of registry entries.  This set of entries
 746  * is hardwired on Windows as "Software\JavaSoft\Java Runtime Environment"
 747  * under the set of roots "{ HKEY_CURRENT_USER, HKEY_LOCAL_MACHINE }".
 748  *
 749  * This routine simply opens each of these registry directories before passing
 750  * control onto ProcessDir().
 751  */
 752 char *
 753 LocateJRE(manifest_info* info) {
 754     HKEY    key = NULL;
 755     char    *path;
 756     int     key_index;
 757     HKEY    root_keys[2] = { HKEY_CURRENT_USER, HKEY_LOCAL_MACHINE };
 758 
 759     for (key_index = 0; key_index <= 1; key_index++) {
 760         if (RegOpenKeyEx(root_keys[key_index], JRE_KEY, 0, KEY_READ, &key)
 761           == ERROR_SUCCESS)
 762             if ((path = ProcessDir(info, key)) != NULL) {
 763                 if (key != NULL)
 764                     RegCloseKey(key);
 765                 return (path);
 766             }
 767         if (key != NULL)
 768             RegCloseKey(key);
 769     }
 770     return NULL;
 771 }

 773 /*
 774  * Local helper routine to isolate a single token (option or argument)
 775  * from the command line.
 776  *
 777  * This routine accepts a pointer to a character pointer.  The first
 778  * token (as defined by MSDN command-line argument syntax) is isolated
 779  * from that string.
 780  *
 781  * Upon return, the input character pointer pointed to by the parameter s
 782  * is updated to point to the remainding, unscanned, portion of the string,
 783  * or to a null character if the entire string has been consummed.
 784  *
 785  * This function returns a pointer to a null-terminated string which
 786  * contains the isolated first token, or to the null character if no
 787  * token could be isolated.
 788  *
 789  * Note the side effect of modifying the input string s by the insertion
 790  * of a null character, making it two strings.
 791  *
 792  * See "Parsing C Command-Line Arguments" in the MSDN Library for the
 793  * parsing rule details.  The rule summary from that specification is:
 794  *
 795  *  * Arguments are delimited by white space, which is either a space or a tab.
 796  *
 797  *  * A string surrounded by double quotation marks is interpreted as a single
 798  *    argument, regardless of white space contained within. A quoted string can
 799  *    be embedded in an argument. Note that the caret (^) is not recognized as
 800  *    an escape character or delimiter.
 801  *
 802  *  * A double quotation mark preceded by a backslash, \", is interpreted as a
 803  *    literal double quotation mark (").
 804  *
 805  *  * Backslashes are interpreted literally, unless they immediately precede a
 806  *    double quotation mark.
 807  *
 808  *  * If an even number of backslashes is followed by a double quotation mark,
 809  *    then one backslash (\) is placed in the argv array for every pair of
 810  *    backslashes (\\), and the double quotation mark (") is interpreted as a
 811  *    string delimiter.
 812  *
 813  *  * If an odd number of backslashes is followed by a double quotation mark,
 814  *    then one backslash (\) is placed in the argv array for every pair of
 815  *    backslashes (\\) and the double quotation mark is interpreted as an
 816  *    escape sequence by the remaining backslash, causing a literal double
 817  *    quotation mark (") to be placed in argv.
 818  */
 819 static char*
 820 nextarg(char** s) {
 821     char    *p = *s;
 822     char    *head;
 823     int     slashes = 0;
 824     int     inquote = 0;
 825 
 826     /*
 827      * Strip leading whitespace, which MSDN defines as only space or tab.
 828      * (Hence, no locale specific "isspace" here.)
 829      */
 830     while (*p != (char)0 && (*p == ' ' || *p == '\t'))
 831         p++;
 832     head = p;                   /* Save the start of the token to return */
 833 
 834     /*
 835      * Isolate a token from the command line.
 836      */
 837     while (*p != (char)0 && (inquote || !(*p == ' ' || *p == '\t'))) {
 838         if (*p == '\\' && *(p+1) == '"' && slashes % 2 == 0)
 839             p++;
 840         else if (*p == '"')
 841             inquote = !inquote;
 842         slashes = (*p++ == '\\') ? slashes + 1 : 0;
 843     }
 844 
 845     /*
 846      * If the token isolated isn't already terminated in a "char zero",
 847      * then replace the whitespace character with one and move to the
 848      * next character.
 849      */
 850     if (*p != (char)0)
 851         *p++ = (char)0;
 852 
 853     /*
 854      * Update the parameter to point to the head of the remaining string
 855      * reflecting the command line and return a pointer to the leading
 856      * token which was isolated from the command line.
 857      */
 858     *s = p;
 859     return (head);
 860 }

 862 /*
 863  * Local helper routine to return a string equivalent to the input string
 864  * s, but with quotes removed so the result is a string as would be found
 865  * in argv[].  The returned string should be freed by a call to JLI_MemFree().
 866  *
 867  * The rules for quoting (and escaped quotes) are:
 868  *
 869  *  1 A double quotation mark preceded by a backslash, \", is interpreted as a
 870  *    literal double quotation mark (").
 871  *
 872  *  2 Backslashes are interpreted literally, unless they immediately precede a
 873  *    double quotation mark.
 874  *
 875  *  3 If an even number of backslashes is followed by a double quotation mark,
 876  *    then one backslash (\) is placed in the argv array for every pair of
 877  *    backslashes (\\), and the double quotation mark (") is interpreted as a
 878  *    string delimiter.
 879  *
 880  *  4 If an odd number of backslashes is followed by a double quotation mark,
 881  *    then one backslash (\) is placed in the argv array for every pair of
 882  *    backslashes (\\) and the double quotation mark is interpreted as an
 883  *    escape sequence by the remaining backslash, causing a literal double
 884  *    quotation mark (") to be placed in argv.
 885  */
 886 static char*
 887 unquote(const char *s) {
 888     const char *p = s;          /* Pointer to the tail of the original string */
 889     char *un = (char*)JLI_MemAlloc(JLI_StrLen(s) + 1);  /* Ptr to unquoted string */
 890     char *pun = un;             /* Pointer to the tail of the unquoted string */
 891 
 892     while (*p != '\0') {
 893         if (*p == '"') {
 894             p++;
 895         } else if (*p == '\\') {
 896             const char *q = p + JLI_StrSpn(p,"\\");
 897             if (*q == '"')
 898                 do {
 899                     *pun++ = '\\';
 900                     p += 2;
 901                  } while (*p == '\\' && p < q);
 902             else
 903                 while (p < q)
 904                     *pun++ = *p++;
 905         } else {
 906             *pun++ = *p++;
 907         }
 908     }
 909     *pun = '\0';
 910     return un;
 911 }

 913 /*
 914  * Given a path to a jre to execute, this routine checks if this process
 915  * is indeed that jre.  If not, it exec's that jre.
 916  *
 917  * We want to actually check the paths rather than just the version string
 918  * built into the executable, so that given version specification will yield
 919  * the exact same Java environment, regardless of the version of the arbitrary
 920  * launcher we start with.
 921  */
 922 void
 923 ExecJRE(char *jre, char **argv) {
 924     jint     len;
 925     char    path[MAXPATHLEN + 1];
 926 
 927     const char *progname = GetProgramName();
 928 
 929     /*
 930      * Resolve the real path to the currently running launcher.
 931      */
 932     len = GetModuleFileName(NULL, path, MAXPATHLEN + 1);
 933     if (len == 0 || len > MAXPATHLEN) {
 934         JLI_ReportErrorMessageSys(JRE_ERROR9, progname);
 935         exit(1);
 936     }
 937 
 938     JLI_TraceLauncher("ExecJRE: old: %s\n", path);
 939     JLI_TraceLauncher("ExecJRE: new: %s\n", jre);
 940 
 941     /*
 942      * If the path to the selected JRE directory is a match to the initial
 943      * portion of the path to the currently executing JRE, we have a winner!
 944      * If so, just return.
 945      */
 946     if (JLI_StrNCaseCmp(jre, path, JLI_StrLen(jre)) == 0)
 947         return;                 /* I am the droid you were looking for */
 948 
 949     /*
 950      * If this isn't the selected version, exec the selected version.
 951      */
 952     JLI_Snprintf(path, sizeof(path), "%s\\bin\\%s.exe", jre, progname);
 953 
 954     /*
 955      * Although Windows has an execv() entrypoint, it doesn't actually
 956      * overlay a process: it can only create a new process and terminate
 957      * the old process.  Therefore, any processes waiting on the initial
 958      * process wake up and they shouldn't.  Hence, a chain of pseudo-zombie
 959      * processes must be retained to maintain the proper wait semantics.
 960      * Fortunately the image size of the launcher isn't too large at this
 961      * time.
 962      *
 963      * If it weren't for this semantic flaw, the code below would be ...
 964      *
 965      *     execv(path, argv);
 966      *     JLI_ReportErrorMessage("Error: Exec of %s failed\n", path);
 967      *     exit(1);
 968      *
 969      * The incorrect exec semantics could be addressed by:
 970      *
 971      *     exit((int)spawnv(_P_WAIT, path, argv));
 972      *
 973      * Unfortunately, a bug in Windows spawn/exec impementation prevents
 974      * this from completely working.  All the Windows POSIX process creation
 975      * interfaces are implemented as wrappers around the native Windows
 976      * function CreateProcess().  CreateProcess() takes a single string
 977      * to specify command line options and arguments, so the POSIX routine
 978      * wrappers build a single string from the argv[] array and in the
 979      * process, any quoting information is lost.
 980      *
 981      * The solution to this to get the original command line, to process it
 982      * to remove the new multiple JRE options (if any) as was done for argv
 983      * in the common SelectVersion() routine and finally to pass it directly
 984      * to the native CreateProcess() Windows process control interface.
 985      */
 986     {
 987         char    *cmdline;
 988         char    *p;
 989         char    *np;
 990         char    *ocl;
 991         char    *ccl;
 992         char    *unquoted;
 993         DWORD   exitCode;
 994         STARTUPINFO si;
 995         PROCESS_INFORMATION pi;
 996 
 997         /*
 998          * The following code block gets and processes the original command
 999          * line, replacing the argv[0] equivalent in the command line with
1000          * the path to the new executable and removing the appropriate
1001          * Multiple JRE support options. Note that similar logic exists
1002          * in the platform independent SelectVersion routine, but is
1003          * replicated here due to the syntax of CreateProcess().
1004          *
1005          * The magic "+ 4" characters added to the command line length are
1006          * 2 possible quotes around the path (argv[0]), a space after the
1007          * path and a terminating null character.
1008          */
1009         ocl = GetCommandLine();
1010         np = ccl = JLI_StringDup(ocl);
1011         p = nextarg(&np);               /* Discard argv[0] */
1012         cmdline = (char *)JLI_MemAlloc(JLI_StrLen(path) + JLI_StrLen(np) + 4);
1013         if (JLI_StrChr(path, (int)' ') == NULL && JLI_StrChr(path, (int)'\t') == NULL)
1014             cmdline = JLI_StrCpy(cmdline, path);
1015         else
1016             cmdline = JLI_StrCat(JLI_StrCat(JLI_StrCpy(cmdline, "\""), path), "\"");
1017 
1018         while (*np != (char)0) {                /* While more command-line */
1019             p = nextarg(&np);
1020             if (*p != (char)0) {                /* If a token was isolated */
1021                 unquoted = unquote(p);
1022                 if (*unquoted == '-') {         /* Looks like an option */
1023                     if (JLI_StrCmp(unquoted, "-classpath") == 0 ||
1024                       JLI_StrCmp(unquoted, "-cp") == 0) {       /* Unique cp syntax */
1025                         cmdline = JLI_StrCat(JLI_StrCat(cmdline, " "), p);
1026                         p = nextarg(&np);
1027                         if (*p != (char)0)      /* If a token was isolated */
1028                             cmdline = JLI_StrCat(JLI_StrCat(cmdline, " "), p);
1029                     } else if (JLI_StrNCmp(unquoted, "-version:", 9) != 0 &&
1030                       JLI_StrCmp(unquoted, "-jre-restrict-search") != 0 &&
1031                       JLI_StrCmp(unquoted, "-no-jre-restrict-search") != 0) {
1032                         cmdline = JLI_StrCat(JLI_StrCat(cmdline, " "), p);
1033                     }
1034                 } else {                        /* End of options */
1035                     cmdline = JLI_StrCat(JLI_StrCat(cmdline, " "), p);
1036                     cmdline = JLI_StrCat(JLI_StrCat(cmdline, " "), np);
1037                     JLI_MemFree((void *)unquoted);
1038                     break;
1039                 }
1040                 JLI_MemFree((void *)unquoted);
1041             }
1042         }
1043         JLI_MemFree((void *)ccl);
1044 
1045         if (JLI_IsTraceLauncher()) {
1046             np = ccl = JLI_StringDup(cmdline);
1047             p = nextarg(&np);
1048             printf("ReExec Command: %s (%s)\n", path, p);
1049             printf("ReExec Args: %s\n", np);
1050             JLI_MemFree((void *)ccl);
1051         }
1052         (void)fflush(stdout);
1053         (void)fflush(stderr);
1054 
1055         /*
1056          * The following code is modeled after a model presented in the
1057          * Microsoft Technical Article "Moving Unix Applications to
1058          * Windows NT" (March 6, 1994) and "Creating Processes" on MSDN
1059          * (Februrary 2005).  It approximates UNIX spawn semantics with
1060          * the parent waiting for termination of the child.
1061          */
1062         memset(&si, 0, sizeof(si));
1063         si.cb =sizeof(STARTUPINFO);
1064         memset(&pi, 0, sizeof(pi));
1065 
1066         if (!CreateProcess((LPCTSTR)path,       /* executable name */
1067           (LPTSTR)cmdline,                      /* command line */
1068           (LPSECURITY_ATTRIBUTES)NULL,          /* process security attr. */
1069           (LPSECURITY_ATTRIBUTES)NULL,          /* thread security attr. */
1070           (BOOL)TRUE,                           /* inherits system handles */
1071           (DWORD)0,                             /* creation flags */
1072           (LPVOID)NULL,                         /* environment block */
1073           (LPCTSTR)NULL,                        /* current directory */
1074           (LPSTARTUPINFO)&si,                   /* (in) startup information */
1075           (LPPROCESS_INFORMATION)&pi)) {        /* (out) process information */
1076             JLI_ReportErrorMessageSys(SYS_ERROR1, path);
1077             exit(1);
1078         }
1079 
1080         if (WaitForSingleObject(pi.hProcess, INFINITE) != WAIT_FAILED) {
1081             if (GetExitCodeProcess(pi.hProcess, &exitCode) == FALSE)
1082                 exitCode = 1;
1083         } else {
1084             JLI_ReportErrorMessage(SYS_ERROR2);
1085             exitCode = 1;
1086         }
1087 
1088         CloseHandle(pi.hThread);
1089         CloseHandle(pi.hProcess);
1090 
1091         exit(exitCode);
1092     }
1093 }

1148 /*
1149  * Block current thread and continue execution in a new thread
1150  */
1151 int
1152 ContinueInNewThread0(int (JNICALL *continuation)(void *), jlong stack_size, void * args) {
1153     int rslt = 0;
1154     unsigned thread_id;
1155 
1156 #ifndef STACK_SIZE_PARAM_IS_A_RESERVATION
1157 #define STACK_SIZE_PARAM_IS_A_RESERVATION  (0x10000)
1158 #endif
1159 
1160     /*
1161      * STACK_SIZE_PARAM_IS_A_RESERVATION is what we want, but it's not
1162      * supported on older version of Windows. Try first with the flag; and
1163      * if that fails try again without the flag. See MSDN document or HotSpot
1164      * source (os_win32.cpp) for details.
1165      */
1166     HANDLE thread_handle =
1167       (HANDLE)_beginthreadex(NULL,
1168                              (unsigned)stack_size,
1169                              continuation,
1170                              args,
1171                              STACK_SIZE_PARAM_IS_A_RESERVATION,
1172                              &thread_id);
1173     if (thread_handle == NULL) {
1174       thread_handle =
1175       (HANDLE)_beginthreadex(NULL,
1176                              (unsigned)stack_size,
1177                              continuation,
1178                              args,
1179                              0,
1180                              &thread_id);
1181     }
1182 
1183     /* AWT preloading (AFTER main thread start) */
1184 #ifdef ENABLE_AWT_PRELOAD
1185     /* D3D preloading */
1186     if (awtPreloadD3D != 0) {
1187         char *envValue;
1188         /* D3D routines checks env.var J2D_D3D if no appropriate
1189          * command line params was specified
1190          */
1191         envValue = getenv("J2D_D3D");
1192         if (envValue != NULL && JLI_StrCaseCmp(envValue, "false") == 0) {
1193             awtPreloadD3D = 0;
1194         }
1195         /* Test that AWT preloading isn't disabled by J2D_D3D_PRELOAD env.var */
1196         envValue = getenv("J2D_D3D_PRELOAD");
1197         if (envValue != NULL && JLI_StrCaseCmp(envValue, "false") == 0) {
1198             awtPreloadD3D = 0;
1199         }
1200         if (awtPreloadD3D < 0) {
1201             /* If awtPreloadD3D is still undefined (-1), test
1202              * if it is turned on by J2D_D3D_PRELOAD env.var.
1203              * By default it's turned OFF.
1204              */
1205             awtPreloadD3D = 0;
1206             if (envValue != NULL && JLI_StrCaseCmp(envValue, "true") == 0) {
1207                 awtPreloadD3D = 1;
1208             }
1209          }
1210     }
1211     if (awtPreloadD3D) {
1212         AWTPreload(D3D_PRELOAD_FUNC);
1213     }
1214 #endif /* ENABLE_AWT_PRELOAD */
1215 
1216     if (thread_handle) {
1217       WaitForSingleObject(thread_handle, INFINITE);
1218       GetExitCodeThread(thread_handle, &rslt);
1219       CloseHandle(thread_handle);
1220     } else {
1221       rslt = continuation(args);
1222     }
1223 
1224 #ifdef ENABLE_AWT_PRELOAD
1225     if (awtPreloaded) {
1226         AWTPreloadStop();
1227     }
1228 #endif /* ENABLE_AWT_PRELOAD */
1229 
1230     return rslt;
1231 }

1279 /* AWT preloading */
1280 #ifdef ENABLE_AWT_PRELOAD
1281 
1282 typedef int FnPreloadStart(void);
1283 typedef void FnPreloadStop(void);
1284 static FnPreloadStop *fnPreloadStop = NULL;
1285 static HMODULE hPreloadAwt = NULL;
1286 
1287 /*
1288  * Starts AWT preloading
1289  */
1290 int AWTPreload(const char *funcName)
1291 {
1292     int result = -1;
1293     /* load AWT library once (if several preload function should be called) */
1294     if (hPreloadAwt == NULL) {
1295         /* awt.dll is not loaded yet */
1296         char libraryPath[MAXPATHLEN];
1297         int jrePathLen = 0;
1298         HMODULE hJava = NULL;
1299         HMODULE hVerify = NULL;
1300 
1301         while (1) {
1302             /* awt.dll depends on jvm.dll & java.dll;
1303              * jvm.dll is already loaded, so we need only java.dll;
1304              * java.dll depends on MSVCRT lib & verify.dll.
1305              */
1306             if (!GetJREPath(libraryPath, MAXPATHLEN)) {
1307                 break;
1308             }
1309 
1310             /* save path length */
1311             jrePathLen = JLI_StrLen(libraryPath);
1312 
1313             if (jrePathLen + JLI_StrLen("\\bin\\verify.dll") >= MAXPATHLEN) {
1314               /* jre path is too long, the library path will not fit there;
1315                * report and abort preloading
1316                */
1317               JLI_ReportErrorMessage(JRE_ERROR11);
1318               break;
1319             }
1320 
1321             /* load msvcrt 1st */
1322             LoadMSVCRT();
1323 
1324             /* load verify.dll */
1325             JLI_StrCat(libraryPath, "\\bin\\verify.dll");
1326             hVerify = LoadLibrary(libraryPath);
1327             if (hVerify == NULL) {
1328                 break;
1329             }
1330 
1331             /* restore jrePath */
1332             libraryPath[jrePathLen] = 0;
1333             /* load java.dll */
1334             JLI_StrCat(libraryPath, "\\bin\\" JAVA_DLL);
1335             hJava = LoadLibrary(libraryPath);
1336             if (hJava == NULL) {
1337                 break;
1338             }
1339 
1340             /* restore jrePath */
1341             libraryPath[jrePathLen] = 0;
1342             /* load awt.dll */
1343             JLI_StrCat(libraryPath, "\\bin\\awt.dll");
1344             hPreloadAwt = LoadLibrary(libraryPath);
1345             if (hPreloadAwt == NULL) {
1346                 break;
1347             }
1348 
1349             /* get "preloadStop" func ptr */
1350             fnPreloadStop = (FnPreloadStop *)GetProcAddress(hPreloadAwt, "preloadStop");
1351 
1352             break;
1353         }
1354     }
1355 
1356     if (hPreloadAwt != NULL) {
1357         FnPreloadStart *fnInit = (FnPreloadStart *)GetProcAddress(hPreloadAwt, funcName);
1358         if (fnInit != NULL) {
1359             /* don't forget to stop preloading */
1360             awtPreloaded = 1;
1361 
1362             result = fnInit();
1363         }
1364     }
1365 
1366     return result;
1367 }

1369 /*
1370  * Terminates AWT preloading
1371  */
1372 void AWTPreloadStop() {
1373     if (fnPreloadStop != NULL) {
1374         fnPreloadStop();
1375     }
1376 }

1410 int
1411 filterArgs(StdArg *stdargs, const int nargc, StdArg **pargv) {
1412     StdArg* argv = NULL;
1413     int nargs = 0;
1414     int i;
1415 
1416     /* Copy the non-vm args */
1417     for (i = 0; i < nargc ; i++) {
1418         const char *arg = stdargs[i].arg;
1419         if (arg[0] == '-' && arg[1] == 'J')
1420             continue;
1421         argv = (StdArg*) JLI_MemRealloc(argv, (nargs+1) * sizeof(StdArg));
1422         argv[nargs].arg = JLI_StringDup(arg);
1423         argv[nargs].has_wildcard = stdargs[i].has_wildcard;
1424         nargs++;
1425     }
1426     *pargv = argv;
1427     return nargs;
1428 }
1429 
1430 /*
1431  * At this point we have the arguments to the application, and we need to
1432  * check with original stdargs in order to compare which of these truly
1433  * needs expansion. cmdtoargs will specify this if it finds a bare
1434  * (unquoted) argument containing a glob character(s) ie. * or ?
1435  */
1436 jobjectArray
1437 CreateApplicationArgs(JNIEnv *env, char **strv, int argc)
1438 {
1439     int i, j, idx, tlen;
1440     jobjectArray outArray, inArray;
1441     char *ostart, *astart, **nargv;
1442     jboolean needs_expansion = JNI_FALSE;
1443     jmethodID mid;
1444     int filteredargc, stdargc;
1445     StdArg *stdargs;
1446     StdArg *filteredargs;
1447     jclass cls = GetLauncherHelperClass(env);
1448     NULL_CHECK0(cls);
1449 
1450     if (argc == 0) {
1451         return NewPlatformStringArray(env, strv, argc);
1452     }
1453     // the holy grail we need to compare with.
1454     stdargs = JLI_GetStdArgs();
1455     stdargc = JLI_GetStdArgc();
1456 
1457     filteredargc = filterArgs(stdargs, stdargc, &filteredargs);
1458 
1459     // sanity check, this should never happen
1460     if (argc > stdargc) {
1461         JLI_TraceLauncher("Warning: app args is larger than the original, %d %d\n", argc, stdargc);
1462         JLI_TraceLauncher("passing arguments as-is.\n");
1463         return NewPlatformStringArray(env, strv, argc);
1464     }
1465 
1466     // sanity check, match the args we have, to the holy grail
1467     idx = filteredargc - argc;
1468     ostart = filteredargs[idx].arg;
1469     astart = strv[0];
1470     // sanity check, ensure that the first argument of the arrays are the same
1471     if (JLI_StrCmp(ostart, astart) != 0) {
1472         // some thing is amiss the args don't match
1473         JLI_TraceLauncher("Warning: app args parsing error\n");
1474         JLI_TraceLauncher("passing arguments as-is\n");
1475         return NewPlatformStringArray(env, strv, argc);
1476     }
1477 
1478     // make a copy of the args which will be expanded in java if required.
1479     nargv = (char **)JLI_MemAlloc(argc * sizeof(char*));
1480     for (i = 0, j = idx; i < argc; i++, j++) {
1481         jboolean arg_expand = (JLI_StrCmp(filteredargs[j].arg, strv[i]) == 0)
1482                                 ? filteredargs[j].has_wildcard
1483                                 : JNI_FALSE;
1484         if (needs_expansion == JNI_FALSE)
1485             needs_expansion = arg_expand;
1486 
1487         // indicator char + String + NULL terminator, the java method will strip
1488         // out the first character, the indicator character, so no matter what
1489         // we add the indicator
1490         tlen = 1 + JLI_StrLen(strv[i]) + 1;
1491         nargv[i] = (char *) JLI_MemAlloc(tlen);
1492         if (JLI_Snprintf(nargv[i], tlen, "%c%s", arg_expand ? 'T' : 'F',
1493                          strv[i]) < 0) {
1494             return NULL;
1495         }
1496         JLI_TraceLauncher("%s\n", nargv[i]);
1497     }
1498 
1499     if (!needs_expansion) {
1500         // clean up any allocated memory and return back the old arguments
1501         for (i = 0 ; i < argc ; i++) {
1502             JLI_MemFree(nargv[i]);
1503         }
1504         JLI_MemFree(nargv);
1505         return NewPlatformStringArray(env, strv, argc);
1506     }
1507     NULL_CHECK0(mid = (*env)->GetStaticMethodID(env, cls,
1508                                                 "expandArgs",
1509                                                 "([Ljava/lang/String;)[Ljava/lang/String;"));
1510 
1511     // expand the arguments that require expansion, the java method will strip
1512     // out the indicator character.
1513     inArray = NewPlatformStringArray(env, nargv, argc);
1514     outArray = (*env)->CallStaticObjectMethod(env, cls, mid, inArray);
1515     for (i = 0; i < argc; i++) {
1516         JLI_MemFree(nargv[i]);
1517     }
1518     JLI_MemFree(nargv);
1519     JLI_MemFree(filteredargs);
1520     return outArray;
1521 }
