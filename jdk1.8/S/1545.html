  A helper class to map a zone name to metazone and back to the
  appropriate zone id for the particular locale.
  

  The zid<->metazone mappings are based on CLDR metaZones.xml.
  The alias mappings are based on Link entries in tzdb data files.

  43     public static String toZid(String zid, Locale locale) {
  44         String mzone = zidToMzone.get(zid);
  45         if (mzone == null && aliases.containsKey(zid)) {
  46             zid = aliases.get(zid);
  47             mzone = zidToMzone.get(zid);
  48         }
  49         if (mzone != null) {
  50             Map<String, String> map = mzoneToZidL.get(mzone);
  51             if (map != null && map.containsKey(locale.getCountry())) {
  52                 zid = map.get(locale.getCountry());
  53             } else {
  54                 zid = mzoneToZid.get(mzone);
  55             }
  56         }
  57         return toZid(zid);
  58     }
  59 
  60     public static String toZid(String zid) {
  61         if (aliases.containsKey(zid)) {
  62             return aliases.get(zid);
  63         }
  64         return zid;
  65     }
