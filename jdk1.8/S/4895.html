 842   /**
 843    * Conditionally trim all leading and trailing whitespace in the specified String.
 844    * All strings of white space are
 845    * replaced by a single space character (#x20), except spaces after punctuation which
 846    * receive double spaces if doublePunctuationSpaces is true.
 847    * This function may be useful to a formatter, but to get first class
 848    * results, the formatter should probably do it's own white space handling
 849    * based on the semantics of the formatting object.
 850    *
 851    * @param   trimHead    Trim leading whitespace?
 852    * @param   trimTail    Trim trailing whitespace?
 853    * @param   doublePunctuationSpaces    Use double spaces for punctuation?
 854    * @return              The trimmed string.
 855    */
 856   public XMLString fixWhiteSpace(boolean trimHead, boolean trimTail,
 857                                  boolean doublePunctuationSpaces)
 858   {
 859 
 860     int end = m_length + m_start;
 861     char[] buf = new char[m_length];
 862     FastStringBuffer fsb = fsb();
 863     boolean edit = false;
 864 
 865     /* replace S to ' '. and ' '+ -> single ' '. */
 866     int d = 0;
 867     boolean pres = false;
 868 
 869     for (int s = m_start; s < end; s++)
 870     {
 871       char c = fsb.charAt(s);
 872 
 873       if (isSpace(c))
 874       {
 875         if (!pres)
 876         {
 877           if (' ' != c)
 878           {
 879             edit = true;
 880           }
 881 
 882           buf[d++] = ' ';
 883 
 884           if (doublePunctuationSpaces && (d != 0))
 885           {
 886             char prevChar = buf[d - 1];
 887 
 888             if (!((prevChar == '.') || (prevChar == '!')
 889                   || (prevChar == '?')))
 890             {
 891               pres = true;
 892             }
 893           }
 894           else
 895           {
 896             pres = true;
 897           }
 898         }
 899         else
 900         {
 901           edit = true;
 902           pres = true;
 903         }
 904       }
 905       else
 906       {
 907         buf[d++] = c;
 908         pres = false;
 909       }
 910     }
 911 
 912     if (trimTail && 1 <= d && ' ' == buf[d - 1])
 913     {
 914       edit = true;
 915 
 916       d--;
 917     }
 918 
 919     int start = 0;
 920 
 921     if (trimHead && 0 < d && ' ' == buf[0])
 922     {
 923       edit = true;
 924 
 925       start++;
 926     }
 927 
 928     XMLStringFactory xsf = XMLStringFactoryImpl.getFactory();
 929 
 930     return edit ? xsf.newstr(buf, start, d - start) : this;
 931   }

 933   /**
 934    * Convert a string to a double -- Allowed input is in fixed
 935    * notation ddd.fff.
 936    *
 937    * %OPT% CHECK PERFORMANCE against generating a Java String and
 938    * converting it to double. The advantage of running in native
 939    * machine code -- perhaps even microcode, on some systems -- may
 940    * more than make up for the cost of allocating and discarding the
 941    * additional object. We need to benchmark this.
 942    *
 943    * %OPT% More importantly, we need to decide whether we _care_ about
 944    * the performance of this operation. Does XString.toDouble constitute
 945    * any measurable percentage of our typical runtime? I suspect not!
 946    *
 947    * @return A double value representation of the string, or return Double.NaN
 948    * if the string can not be converted.  */
 949   public double toDouble()
 950   {
 951     if(m_length == 0)
 952       return Double.NaN;
 953     int i;
 954     char c;
 955     String valueString = fsb().getString(m_start,m_length);
 956 
 957     // The following are permitted in the Double.valueOf, but not by the XPath spec:
 958     // - a plus sign
 959     // - The use of e or E to indicate exponents
 960     // - trailing f, F, d, or D
 961     // See function comments; not sure if this is slower than actually doing the
 962     // conversion ourselves (as was before).
 963 
 964     for (i=0;i<m_length;i++)
 965       if (!XMLCharacterRecognizer.isWhiteSpace(valueString.charAt(i)))
 966         break;
 967     if (i == m_length) return Double.NaN;
 968     if (valueString.charAt(i) == '-')
 969       i++;
 970     for (;i<m_length;i++) {
 971       c = valueString.charAt(i);
 972       if (c != '.' && (c < '0' || c > '9'))
 973         break;
 974     }
 975     for (;i<m_length;i++)
 976       if (!XMLCharacterRecognizer.isWhiteSpace(valueString.charAt(i)))
 977         break;
 978     if (i != m_length)
 979       return Double.NaN;
 980 
 981     try {
 982       return Double.parseDouble(valueString);
 983     } catch (NumberFormatException nfe) {
 984       // This should catch double periods, empty strings.
 985       return Double.NaN;
 986     }
 987   }
