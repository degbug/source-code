 602     private String fileNameString(File file) {
 603         if (file == null) {
 604             return null;
 605         } else {
 606             JFileChooser fc = getFileChooser();
 607             if ((fc.isDirectorySelectionEnabled() && !fc.isFileSelectionEnabled()) ||
 608                 (fc.isDirectorySelectionEnabled() && fc.isFileSelectionEnabled()
 609                  && fc.getFileSystemView().isFileSystemRoot(file))) {
 610                 return file.getPath();
 611             } else {
 612                 return file.getName();
 613             }
 614         }
 615     }

 617     private String fileNameString(File[] files) {
 618         StringBuffer buf = new StringBuffer();
 619         for (int i = 0; files != null && i < files.length; i++) {
 620             if (i > 0) {
 621                 buf.append(" ");
 622             }
 623             if (files.length > 1) {
 624                 buf.append("\"");
 625             }
 626             buf.append(fileNameString(files[i]));
 627             if (files.length > 1) {
 628                 buf.append("\"");
 629             }
 630         }
 631         return buf.toString();
 632     }

          Adds the directory to the model and sets it to be selected,
          additionally clears out the previous selected directory and
          the paths leading up to it, if any.

 936         private void addItem(File directory) {
 937 
 938             if(directory == null) {
 939                 return;
 940             }
 941 
 942             boolean useShellFolder = FilePane.usesShellFolder(chooser);
 943 
 944             directories.clear();
 945 
 946             File[] baseFolders = (useShellFolder)
 947                     ? (File[]) ShellFolder.get("fileChooserComboBoxFolders")
 948                     : fsv.getRoots();
 949             directories.addAll(Arrays.asList(baseFolders));
 950 
 951             // Get the canonical (full) path. This has the side
 952             // benefit of removing extraneous chars from the path,
 953             // for example /foo/bar/ becomes /foo/bar
 954             File canonical;
 955             try {
 956                 canonical = ShellFolder.getNormalizedFile(directory);
 957             } catch (IOException e) {
 958                 // Maybe drive is not ready. Can't abort here.
 959                 canonical = directory;
 960             }
 961 
 962             // create File instances of each directory leading up to the top
 963             try {
 964                 File sf = useShellFolder ? ShellFolder.getShellFolder(canonical)
 965                                          : canonical;
 966                 File f = sf;
 967                 Vector<File> path = new Vector<File>(10);
 968                 do {
 969                     path.addElement(f);
 970                 } while ((f = f.getParentFile()) != null);
 971 
 972                 int pathCount = path.size();
 973                 // Insert chain at appropriate place in vector
 974                 for (int i = 0; i < pathCount; i++) {
 975                     f = path.get(i);
 976                     if (directories.contains(f)) {
 977                         int topIndex = directories.indexOf(f);
 978                         for (int j = i-1; j >= 0; j--) {
 979                             directories.insertElementAt(path.get(j), topIndex+i-j);
 980                         }
 981                         break;
 982                     }
 983                 }
 984                 calculateDepths();
 985                 setSelectedItem(sf);
 986             } catch (FileNotFoundException ex) {
 987                 calculateDepths();
 988             }
 989         }
 990 
 991         private void calculateDepths() {
 992             depths = new int[directories.size()];
 993             for (int i = 0; i < depths.length; i++) {
 994                 File dir = directories.get(i);
 995                 File parent = dir.getParentFile();
 996                 depths[i] = 0;
 997                 if (parent != null) {
 998                     for (int j = i-1; j >= 0; j--) {
 999                         if (parent.equals(directories.get(j))) {
1000                             depths[i] = depths[j] + 1;
1001                             break;
1002                         }
1003                     }
1004                 }
1005             }
1006         }
1007 
1008         public int getDepth(int i) {
1009             return (depths != null && i >= 0 && i < depths.length) ? depths[i] : 0;
1010         }

      ButtonAreaLayout behaves in a similar manner to
      FlowLayout. It lays out all components from left to
      right, flushed right. The widths of all components will be set
      to the largest preferred size width.

1167     private static class ButtonAreaLayout implements LayoutManager {
1168         private int hGap = 5;
1169         private int topMargin = 17;
1170 
1171         public void addLayoutComponent(String string, Component comp) {
1172         }
1173 
1174         public void layoutContainer(Container container) {
1175             Component[] children = container.getComponents();
1176 
1177             if (children != null && children.length > 0) {
1178                 int         numChildren = children.length;
1179                 Dimension[] sizes = new Dimension[numChildren];
1180                 Insets      insets = container.getInsets();
1181                 int         yLocation = insets.top + topMargin;
1182                 int         maxWidth = 0;
1183 
1184                 for (int counter = 0; counter < numChildren; counter++) {
1185                     sizes[counter] = children[counter].getPreferredSize();
1186                     maxWidth = Math.max(maxWidth, sizes[counter].width);
1187                 }
1188                 int xLocation, xOffset;
1189                 if (container.getComponentOrientation().isLeftToRight()) {
1190                     xLocation = container.getSize().width - insets.left - maxWidth;
1191                     xOffset = hGap + maxWidth;
1192                 } else {
1193                     xLocation = insets.left;
1194                     xOffset = -(hGap + maxWidth);
1195                 }
1196                 for (int counter = numChildren - 1; counter >= 0; counter--) {
1197                     children[counter].setBounds(xLocation, yLocation,
1198                                                 maxWidth, sizes[counter].height);
1199                     xLocation -= xOffset;
1200                 }
1201             }
1202         }
1203 
1204         public Dimension minimumLayoutSize(Container c) {
1205             if (c != null) {
1206                 Component[] children = c.getComponents();
1207 
1208                 if (children != null && children.length > 0) {
1209                     int       numChildren = children.length;
1210                     int       height = 0;
1211                     Insets    cInsets = c.getInsets();
1212                     int       extraHeight = topMargin + cInsets.top + cInsets.bottom;
1213                     int       extraWidth = cInsets.left + cInsets.right;
1214                     int       maxWidth = 0;
1215 
1216                     for (int counter = 0; counter < numChildren; counter++) {
1217                         Dimension aSize = children[counter].getPreferredSize();
1218                         height = Math.max(height, aSize.height);
1219                         maxWidth = Math.max(maxWidth, aSize.width);
1220                     }
1221                     return new Dimension(extraWidth + numChildren * maxWidth +
1222                                          (numChildren - 1) * hGap,
1223                                          extraHeight + height);
1224                 }
1225             }
1226             return new Dimension(0, 0);
1227         }

1230             return minimumLayoutSize(c);
1231         }
