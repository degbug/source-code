  26 /**

  30  * @author Santiago Pericas-Geertsen
  31  */
  32 public class Node {
  33     public int node;
  34     public int type;
  35 
  36     public Node(int n, int t) {
  37         node = n;
  38         type = t;
  39     }
  40 }
