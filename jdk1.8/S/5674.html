  70         if (log.isLoggable(java.util.logging.Level.FINE)) {
  71             log.log(java.util.logging.Level.FINE, "Created SignatureDSA using " + algorithmID);
  72         }
  73 
  74         String provider = JCEMapper.getProviderId();
  75         try {
  76             if (provider == null) {
  77                 this.signatureAlgorithm = Signature.getInstance(algorithmID);
  78             } else {
  79                 this.signatureAlgorithm =
  80                     Signature.getInstance(algorithmID, provider);
  81             }
  82         } catch (java.security.NoSuchAlgorithmException ex) {
  83             Object[] exArgs = { algorithmID, ex.getLocalizedMessage() };
  84             throw new XMLSignatureException("algorithms.NoSuchAlgorithm", exArgs);
  85         } catch (java.security.NoSuchProviderException ex) {
  86             Object[] exArgs = { algorithmID, ex.getLocalizedMessage() };
  87             throw new XMLSignatureException("algorithms.NoSuchAlgorithm", exArgs);
  88         }

 123     /**
 124      * @inheritDoc
 125      */
 126     protected void engineInitVerify(Key publicKey) throws XMLSignatureException {
 127         if (!(publicKey instanceof PublicKey)) {
 128             String supplied = publicKey.getClass().getName();
 129             String needed = PublicKey.class.getName();
 130             Object exArgs[] = { supplied, needed };
 131 
 132             throw new XMLSignatureException("algorithms.WrongKeyForThisOperation", exArgs);
 133         }
 134 
 135         try {
 136             this.signatureAlgorithm.initVerify((PublicKey) publicKey);
 137         } catch (InvalidKeyException ex) {
 138             // reinstantiate Signature object to work around bug in JDK
 139             // see: http://bugs.sun.com/view_bug.do?bug_id=4953555
 140             Signature sig = this.signatureAlgorithm;
 141             try {
 142                 this.signatureAlgorithm = Signature.getInstance(signatureAlgorithm.getAlgorithm());
 143             } catch (Exception e) {
 144                 // this shouldn't occur, but if it does, restore previous
 145                 // Signature
 146                 if (log.isLoggable(java.util.logging.Level.FINE)) {
 147                     log.log(java.util.logging.Level.FINE, "Exception when reinstantiating Signature:" + e);
 148                 }
 149                 this.signatureAlgorithm = sig;
 150             }
 151             throw new XMLSignatureException("empty", ex);
 152         }
 153     }

 260     /**
 261      * Converts an ASN.1 DSA value to a XML Signature DSA Value.
 262      *
 263      * The JAVA JCE DSA Signature algorithm creates ASN.1 encoded (r,s) value
 264      * pairs; the XML Signature requires the core BigInteger values.
 265      *
 266      * @param asn1Bytes
 267      * @return the decode bytes
 268      *
 269      * @throws IOException
 270      * @see 6.4.1 DSA
 271      */
 272     private static byte[] convertASN1toXMLDSIG(byte asn1Bytes[]) throws IOException {
 273 
 274         byte rLength = asn1Bytes[3];
 275         int i;
 276 
 277         for (i = rLength; (i > 0) && (asn1Bytes[(4 + rLength) - i] == 0); i--);
 278 
 279         byte sLength = asn1Bytes[5 + rLength];
 280         int j;
 281 
 282         for (j = sLength;
 283             (j > 0) && (asn1Bytes[(6 + rLength + sLength) - j] == 0); j--);
 284 
 285         if ((asn1Bytes[0] != 48) || (asn1Bytes[1] != asn1Bytes.length - 2)
 286             || (asn1Bytes[2] != 2) || (i > 20)
 287             || (asn1Bytes[4 + rLength] != 2) || (j > 20)) {
 288             throw new IOException("Invalid ASN.1 format of DSA signature");
 289         }
 290         byte xmldsigBytes[] = new byte[40];
 291 
 292         System.arraycopy(asn1Bytes, (4 + rLength) - i, xmldsigBytes, 20 - i, i);
 293         System.arraycopy(asn1Bytes, (6 + rLength + sLength) - j, xmldsigBytes,
 294                          40 - j, j);
 295 
 296         return xmldsigBytes;
 297     }

 299     /**
 300      * Converts a XML Signature DSA Value to an ASN.1 DSA value.
 301      *
 302      * The JAVA JCE DSA Signature algorithm creates ASN.1 encoded (r,s) value
 303      * pairs; the XML Signature requires the core BigInteger values.
 304      *
 305      * @param xmldsigBytes
 306      * @return the encoded ASN.1 bytes
 307      *
 308      * @throws IOException
 309      * @see 6.4.1 DSA
 310      */
 311     private static byte[] convertXMLDSIGtoASN1(byte xmldsigBytes[]) throws IOException {
 312 
 313         if (xmldsigBytes.length != 40) {
 314             throw new IOException("Invalid XMLDSIG format of DSA signature");
 315         }
 316 
 317         int i;
 318 
 319         for (i = 20; (i > 0) && (xmldsigBytes[20 - i] == 0); i--);
 320 
 321         int j = i;
 322 
 323         if (xmldsigBytes[20 - i] < 0) {
 324             j += 1;
 325         }
 326 
 327         int k;
 328 
 329         for (k = 20; (k > 0) && (xmldsigBytes[40 - k] == 0); k--);
 330 
 331         int l = k;
 332 
 333         if (xmldsigBytes[40 - k] < 0) {
 334             l += 1;
 335         }
 336 
 337         byte asn1Bytes[] = new byte[6 + j + l];
 338 
 339         asn1Bytes[0] = 48;
 340         asn1Bytes[1] = (byte) (4 + j + l);
 341         asn1Bytes[2] = 2;
 342         asn1Bytes[3] = (byte) j;
 343 
 344         System.arraycopy(xmldsigBytes, 20 - i, asn1Bytes, (4 + j) - i, i);
 345 
 346         asn1Bytes[4 + j] = 2;
 347         asn1Bytes[5 + j] = (byte) l;
 348 
 349         System.arraycopy(xmldsigBytes, 40 - k, asn1Bytes, (6 + j + l) - k, k);
 350 
 351         return asn1Bytes;
 352     }
