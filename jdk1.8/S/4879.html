 102   /**
 103    * Do a deep clone of this walker, including next and previous walkers.
 104    * If the this AxesWalker is on the clone list, don't clone but
 105    * return the already cloned version.
 106    *
 107    * @param cloneOwner non-null reference to the cloned location path
 108    *                   iterator to which this clone will be added.
 109    * @param cloneList non-null vector of sources in odd elements, and the
 110    *                  corresponding clones in even vectors.
 111    *
 112    * @return non-null clone, which may be a new clone, or may be a clone
 113    *         contained on the cloneList.
 114    */
 115   AxesWalker cloneDeep(WalkingIterator cloneOwner, Vector cloneList)
 116      throws CloneNotSupportedException
 117   {
 118     AxesWalker clone = findClone(this, cloneList);
 119     if(null != clone)
 120       return clone;
 121     clone = (AxesWalker)this.clone();
 122     clone.setLocPathIterator(cloneOwner);
 123     if(null != cloneList)
 124     {
 125       cloneList.addElement(this);
 126       cloneList.addElement(clone);
 127     }
 128 
 129     if(wi().m_lastUsedWalker == this)
 130       cloneOwner.m_lastUsedWalker = clone;
 131 
 132     if(null != m_nextWalker)
 133       clone.m_nextWalker = m_nextWalker.cloneDeep(cloneOwner, cloneList);
 134 
 135     // If you don't check for the cloneList here, you'll go into an
 136     // recursive infinate loop.
 137     if(null != cloneList)
 138     {
 139       if(null != m_prevWalker)
 140         clone.m_prevWalker = m_prevWalker.cloneDeep(cloneOwner, cloneList);
 141     }
 142     else
 143     {
 144       if(null != m_nextWalker)
 145         clone.m_nextWalker.m_prevWalker = clone;
 146     }
 147     return clone;
 148   }

 150   /**
 151    * Find a clone that corresponds to the key argument.
 152    *
 153    * @param key The original AxesWalker for which there may be a clone.
 154    * @param cloneList vector of sources in odd elements, and the
 155    *                  corresponding clones in even vectors, may be null.
 156    *
 157    * @return A clone that corresponds to the key, or null if key not found.
 158    */
 159   static AxesWalker findClone(AxesWalker key, Vector cloneList)
 160   {
 161     if(null != cloneList)
 162     {
 163       // First, look for clone on list.
 164       int n = cloneList.size();
 165       for (int i = 0; i < n; i+=2)
 166       {
 167         if(key == cloneList.elementAt(i))
 168           return (AxesWalker)cloneList.elementAt(i+1);
 169       }
 170     }
 171     return null;
 172   }

 317   /**
 318    * Get the next node in document order on the axes.
 319    *
 320    * @return the next node in document order on the axes, or null.
 321    */
 322   protected int getNextNode()
 323   {
 324     if (m_foundLast)
 325       return DTM.NULL;
 326 
 327     if (m_isFresh)
 328     {
 329       m_currentNode = m_traverser.first(m_root);
 330       m_isFresh = false;
 331     }
 332     // I shouldn't have to do this the check for current node, I think.
 333     // numbering\numbering24.xsl fails if I don't do this.  I think
 334     // it occurs as the walkers are backing up. -sb
 335     else if(DTM.NULL != m_currentNode)
 336     {
 337       m_currentNode = m_traverser.next(m_root, m_currentNode);
 338     }
 339 
 340     if (DTM.NULL == m_currentNode)
 341       this.m_foundLast = true;
 342 
 343     return m_currentNode;
 344   }

 346   /**
 347    *  Moves the <code>TreeWalker</code> to the next visible node in document
 348    * order relative to the current node, and returns the new node. If the
 349    * current node has no next node,  or if the search for nextNode attempts
 350    * to step upward from the TreeWalker's root node, returns
 351    * <code>null</code> , and retains the current node.
 352    * @return  The new node, or <code>null</code> if the current node has no
 353    *   next node  in the TreeWalker's logical view.
 354    */
 355   public int nextNode()
 356   {
 357     int nextNode = DTM.NULL;
 358     AxesWalker walker = wi().getLastUsedWalker();
 359 
 360     while (true)
 361     {
 362       if (null == walker)
 363         break;
 364 
 365       nextNode = walker.getNextNode();
 366 
 367       if (DTM.NULL == nextNode)
 368       {
 369 
 370         walker = walker.m_prevWalker;
 371       }
 372       else
 373       {
 374         if (walker.acceptNode(nextNode) != DTMIterator.FILTER_ACCEPT)
 375         {
 376           continue;
 377         }
 378 
 379         if (null == walker.m_nextWalker)
 380         {
 381           wi().setLastUsedWalker(walker);
 382 
 383           // return walker.returnNextNode(nextNode);
 384           break;
 385         }
 386         else
 387         {
 388           AxesWalker prev = walker;
 389 
 390           walker = walker.m_nextWalker;
 391 
 392           walker.setRoot(nextNode);
 393 
 394           walker.m_prevWalker = prev;
 395 
 396           continue;
 397         }
 398       }  // if(null != nextNode)
 399     }  // while(null != walker)
 400 
 401     return nextNode;
 402   }

 406   /**
 407    * Get the index of the last node that can be itterated to.
 408    *
 409    *
 410    * @param xctxt XPath runtime context.
 411    *
 412    * @return the index of the last node that can be itterated to.
 413    */
 414   public int getLastPos(XPathContext xctxt)
 415   {
 416 
 417     int pos = getProximityPosition();
 418 
 419     AxesWalker walker;
 420 
 421     try
 422     {
 423       walker = (AxesWalker) clone();
 424     }
 425     catch (CloneNotSupportedException cnse)
 426     {
 427       return -1;
 428     }
 429 
 430     walker.setPredicateCount(m_predicateIndex);
 431     walker.setNextWalker(null);
 432     walker.setPrevWalker(null);
 433 
 434     WalkingIterator lpi = wi();
 435     AxesWalker savedWalker = lpi.getLastUsedWalker();
 436 
 437     try
 438     {
 439       lpi.setLastUsedWalker(walker);
 440 
 441       int next;
 442 
 443       while (DTM.NULL != (next = walker.nextNode()))
 444       {
 445         pos++;
 446       }
 447 
 448       // TODO: Should probably save this in the iterator.
 449     }
 450     finally
 451     {
 452       lpi.setLastUsedWalker(savedWalker);
 453     }
 454 
 455     // System.out.println("pos: "+pos);
 456     return pos;
 457   }
