  84     @Override
  85     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  86         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  87         componentColors = extendedCacheKeys;
  88         //generate this entire method. Each state/bg/fg/border combo that has
  89         //been painted gets its own KEY and paint method.
  90         switch(state) {
  91             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
  92             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  93             case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
  94             case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
  95             case BACKGROUND_SELECTED_FOCUSED: paintBackgroundSelectedAndFocused(g); break;
  96 
  97         }
  98     }
