  70     @Override
  71     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  72         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  73         componentColors = extendedCacheKeys;
  74         //generate this entire method. Each state/bg/fg/border combo that has
  75         //been painted gets its own KEY and paint method.
  76         switch(state) {
  77             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  78 
  79         }
  80     }

  89     private void paintBackgroundEnabled(Graphics2D g) {
  90         roundRect = decodeRoundRect1();
  91         g.setPaint(color1);
  92         g.fill(roundRect);
  93         roundRect = decodeRoundRect2();
  94         g.setPaint(decodeGradient1(roundRect));
  95         g.fill(roundRect);
  96         rect = decodeRect1();
  97         g.setPaint(decodeGradient2(rect));
  98         g.fill(rect);
  99 
 100     }

 104     private RoundRectangle2D decodeRoundRect1() {
 105         roundRect.setRoundRect(decodeX(0.4f), //x
 106                                decodeY(0.0f), //y
 107                                decodeX(2.8f) - decodeX(0.4f), //width
 108                                decodeY(2.6f) - decodeY(0.0f), //height
 109                                4.8333335f, 4.8333335f); //rounding

 132     private Paint decodeGradient1(Shape s) {
 133         Rectangle2D bounds = s.getBounds2D();
 134         float x = (float)bounds.getX();
 135         float y = (float)bounds.getY();
 136         float w = (float)bounds.getWidth();
 137         float h = (float)bounds.getHeight();
 138         return decodeGradient((0.5f * w) + x, (0.0f * h) + y, (0.5f * w) + x, (1.0f * h) + y,
 139                 new float[] { 0.0f,0.5f,1.0f },
 140                 new Color[] { color2,
 141                             decodeColor(color2,color3,0.5f),
 142                             color3});
 143     }

 145     private Paint decodeGradient2(Shape s) {

 147         float x = (float)bounds.getX();
 148         float y = (float)bounds.getY();
 149         float w = (float)bounds.getWidth();
 150         float h = (float)bounds.getHeight();
 151         return decodeGradient((0.5f * w) + x, (0.0f * h) + y, (0.5f * w) + x, (1.0f * h) + y,
 152                 new float[] { 0.0f,0.24f,1.0f },
 153                 new Color[] { color4,
 154                             decodeColor(color4,color5,0.5f),
