 103   public void addAttributes(Attributes atts)
 104   {
 105 
 106     int nAtts = atts.getLength();
 107 
 108     for (int i = 0; i < nAtts; i++)
 109     {
 110       String uri = atts.getURI(i);
 111 
 112       if (null == uri)
 113         uri = "";
 114 
 115       String localName = atts.getLocalName(i);
 116       String qname = atts.getQName(i);
 117       int index = this.getIndex(uri, localName);
 118       // System.out.println("MutableAttrListImpl#addAttributes: "+uri+":"+localName+", "+index+", "+atts.getQName(i)+", "+this);
 119       if (index >= 0)
 120         this.setAttribute(index, uri, localName, qname, atts.getType(i),
 121                           atts.getValue(i));
 122       else
 123         addAttribute(uri, localName, qname, atts.getType(i),
 124                      atts.getValue(i));
 125     }
 126   }
