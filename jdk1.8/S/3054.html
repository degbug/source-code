  99     @Override
 100     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
 101         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 102         componentColors = extendedCacheKeys;
 103         //generate this entire method. Each state/bg/fg/border combo that has
 104         //been painted gets its own KEY and paint method.
 105         switch(state) {
 106             case FOREGROUND_ENABLED: paintForegroundEnabled(g); break;
 107             case FOREGROUND_DISABLED: paintForegroundDisabled(g); break;
 108             case FOREGROUND_MOUSEOVER: paintForegroundMouseOver(g); break;
 109             case FOREGROUND_PRESSED: paintForegroundPressed(g); break;
 110 
 111         }
 112     }
