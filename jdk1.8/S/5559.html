  23 package com.sun.org.apache.xml.internal.serializer;
  24 
  25 import java.util.Hashtable;
  26 
  27 /**
  28  * This class contains utilities used by the serializer
  29  */
  30 class Utils
  31 {
  32 
  33     /**
  34      * This nested class acts as a way to lazy load the hashtable
  35      * in a thread safe way.
  36      */
  37     static private class CacheHolder
  38     {
  39         static final Hashtable cache;
  40         static {
  41             cache = new Hashtable();
  42         }
  43     }
  44     /**
  45      * Load the class by name.
  46      *
  47      * This implementation, for performance reasons,
  48      * caches all classes loaded by name and
  49      * returns the cached Class object if it can previously
  50      * loaded classes that were load by name.  If not previously loaded
  51      * an attempt is made to load with Class.forName(classname)
  52      * @param classname the name of the class to be loaded
  53      * @return the loaded class, never null. If the class could not be
  54      * loaded a ClassNotFound exception is thrown.
  55      * @throws ClassNotFoundException if the class was not loaded
  56      */
  57     static Class ClassForName(String classname) throws ClassNotFoundException
  58     {
  59         Class c;
  60         // the first time the next line runs will reference
  61         // CacheHolder, causing the class to load and create the
  62         // Hashtable.
  63         Object o = CacheHolder.cache.get(classname);
  64         if (o == null)
  65         {
  66             // class was not in the cache, so try to load it
  67             c = Class.forName(classname);
  68             // if the class is not found we will have thrown a
  69             // ClassNotFoundException on the statement above
  70 
  71             // if we get here c is not null
  72             CacheHolder.cache.put(classname, c);
  73         }
  74         else
  75         {
  76             c = (Class)o;
  77         }
  78         return c;
  79     }
  80 }
