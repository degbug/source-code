  47 /**
  48  * A bounded {@linkplain BlockingQueue blocking queue} backed by an
  49  * array.  This queue orders elements FIFO (first-in-first-out).  The
  50  * head of the queue is that element that has been on the
  51  * queue the longest time.  The tail of the queue is that
  52  * element that has been on the queue the shortest time. New elements
  53  * are inserted at the tail of the queue, and the queue retrieval
  54  * operations obtain elements at the head of the queue.
  55  *
  56  * This is a classic "bounded buffer", in which a
  57  * fixed-sized array holds elements inserted by producers and
  58  * extracted by consumers.  Once created, the capacity cannot be
  59  * changed.  Attempts to {@code put} an element into a full queue
  60  * will result in the operation blocking; attempts to {@code take} an
  61  * element from an empty queue will similarly block.
  62  *
  63  * This class supports an optional fairness policy for ordering
  64  * waiting producer and consumer threads.  By default, this ordering
  65  * is not guaranteed. However, a queue constructed with fairness set
  66  * to {@code true} grants threads access in FIFO order. Fairness
  67  * generally decreases throughput but reduces variability and avoids
  68  * starvation.
  69  *
  70  * This class and its iterator implement all of the
  71  * optional methods of the {@link Collection} and {@link
  72  * Iterator} interfaces.
  73  *
  74  * This class is a member of the
  75  * 
  76  * Java Collections Framework.
  77  *
  78  * @since 1.5
  79  * @author Doug Lea
  80  * @param  the type of elements held in this collection
  81  */

 153     /**
 154      * Inserts element at current put position, advances, and signals.
 155      * Call only when holding lock.
 156      */
 157     private void enqueue(E x) {
 158         // assert lock.getHoldCount() == 1;
 159         // assert items[putIndex] == null;
 160         final Object[] items = this.items;
 161         items[putIndex] = x;
 162         if (++putIndex == items.length)
 163             putIndex = 0;
 164         count++;
 165         notEmpty.signal();
 166     }

 168     /**
 169      * Extracts element at current take position, advances, and signals.
 170      * Call only when holding lock.
 171      */
 172     private E dequeue() {
 173         // assert lock.getHoldCount() == 1;
 174         // assert items[takeIndex] != null;
 175         final Object[] items = this.items;
 176         @SuppressWarnings("unchecked")
 177         E x = (E) items[takeIndex];
 178         items[takeIndex] = null;
 179         if (++takeIndex == items.length)
 180             takeIndex = 0;
 181         count--;
 182         if (itrs != null)
 183             itrs.elementDequeued();
 184         notFull.signal();
 185         return x;
 186     }

 188     /**
 189      * Deletes item at array index removeIndex.
 190      * Utility for remove(Object) and iterator.remove.
 191      * Call only when holding lock.
 192      */
 193     void removeAt(final int removeIndex) {
 194         // assert lock.getHoldCount() == 1;
 195         // assert items[removeIndex] != null;
 196         // assert removeIndex >= 0 && removeIndex < items.length;
 197         final Object[] items = this.items;
 198         if (removeIndex == takeIndex) {
 199             // removing front item; just advance
 200             items[takeIndex] = null;
 201             if (++takeIndex == items.length)
 202                 takeIndex = 0;
 203             count--;
 204             if (itrs != null)
 205                 itrs.elementDequeued();
 206         } else {
 207             // an "interior" remove
 208 
 209             // slide over all others up through putIndex.
 210             final int putIndex = this.putIndex;
 211             for (int i = removeIndex;;) {
 212                 int next = i + 1;
 213                 if (next == items.length)
 214                     next = 0;
 215                 if (next != putIndex) {
 216                     items[i] = items[next];
 217                     i = next;
 218                 } else {
 219                     items[i] = null;
 220                     this.putIndex = i;
 221                     break;
 222                 }
 223             }
 224             count--;
 225             if (itrs != null)
 226                 itrs.removedAt(removeIndex);
 227         }
 228         notFull.signal();
 229     }

 262      * Creates an {@code ArrayBlockingQueue} with the given (fixed)
 263      * capacity, the specified access policy and initially containing the
 264      * elements of the given collection,
 265      * added in traversal order of the collection's iterator.
 266      *
 267      * @param capacity the capacity of this queue
 268      * @param fair if {@code true} then queue accesses for threads blocked
 269      *        on insertion or removal, are processed in FIFO order;
 270      *        if {@code false} the access order is unspecified.
 271      * @param c the collection of elements to initially contain
 272      * @throws IllegalArgumentException if {@code capacity} is less than
 273      *         {@code c.size()}, or less than 1.
 274      * @throws NullPointerException if the specified collection or any
 275      *         of its elements are null
 276      */
 277     public ArrayBlockingQueue(int capacity, boolean fair,
 278                               Collection<? extends E> c) {
 279         this(capacity, fair);
 280 
 281         final ReentrantLock lock = this.lock;
 282         lock.lock(); // Lock only for visibility, not mutual exclusion
 283         try {
 284             int i = 0;
 285             try {
 286                 for (E e : c) {
 287                     checkNotNull(e);
 288                     items[i++] = e;
 289                 }
 290             } catch (ArrayIndexOutOfBoundsException ex) {
 291                 throw new IllegalArgumentException();
 292             }
 293             count = i;
 294             putIndex = (i == capacity) ? 0 : i;
 295         } finally {
 296             lock.unlock();
 297         }
 298     }

 476     /**
 477      * Removes a single instance of the specified element from this queue,
 478      * if it is present.  More formally, removes an element {@code e} such
 479      * that {@code o.equals(e)}, if this queue contains one or more such
 480      * elements.
 481      * Returns {@code true} if this queue contained the specified element
 482      * (or equivalently, if this queue changed as a result of the call).
 483      *
 484      * Removal of interior elements in circular array based queues
 485      * is an intrinsically slow and disruptive operation, so should
 486      * be undertaken only in exceptional circumstances, ideally
 487      * only when the queue is known not to be accessible by other
 488      * threads.
 489      *
 490      * @param o element to be removed from this queue, if present
 491      * @return {@code true} if this queue changed as a result of the call
 492      */
 493     public boolean remove(Object o) {
 494         if (o == null) return false;
 495         final Object[] items = this.items;
 496         final ReentrantLock lock = this.lock;
 497         lock.lock();
 498         try {
 499             if (count > 0) {
 500                 final int putIndex = this.putIndex;
 501                 int i = takeIndex;
 502                 do {
 503                     if (o.equals(items[i])) {
 504                         removeAt(i);
 505                         return true;
 506                     }
 507                     if (++i == items.length)
 508                         i = 0;
 509                 } while (i != putIndex);
 510             }
 511             return false;
 512         } finally {
 513             lock.unlock();
 514         }
 515     }

 580     /**
 581      * Returns an array containing all of the elements in this queue, in
 582      * proper sequence; the runtime type of the returned array is that of
 583      * the specified array.  If the queue fits in the specified array, it
 584      * is returned therein.  Otherwise, a new array is allocated with the
 585      * runtime type of the specified array and the size of this queue.
 586      *
 587      * If this queue fits in the specified array with room to spare
 588      * (i.e., the array has more elements than this queue), the element in
 589      * the array immediately following the end of the queue is set to
 590      * {@code null}.
 591      *
 592      * Like the {@link #toArray()} method, this method acts as bridge between
 593      * array-based and collection-based APIs.  Further, this method allows
 594      * precise control over the runtime type of the output array, and may,
 595      * under certain circumstances, be used to save allocation costs.
 596      *
 597      * Suppose {@code x} is a queue known to contain only strings.
 598      * The following code can be used to dump the queue into a newly
 599      * allocated array of {@code String}:
 600      *
 601      *  
 {@code String[] y = x.toArray(new String[0]);}

 602      *
 603      * Note that {@code toArray(new Object[0])} is identical in function to
 604      * {@code toArray()}.
 605      *
 606      * @param a the array into which the elements of the queue are to
 607      *          be stored, if it is big enough; otherwise, a new array of the
 608      *          same runtime type is allocated for this purpose
 609      * @return an array containing all of the elements in this queue
 610      * @throws ArrayStoreException if the runtime type of the specified array
 611      *         is not a supertype of the runtime type of every element in
 612      *         this queue
 613      * @throws NullPointerException if the specified array is null
 614      */
 615     @SuppressWarnings("unchecked")
 616     public <T> T[] toArray(T[] a) {
 617         final Object[] items = this.items;
 618         final ReentrantLock lock = this.lock;
 619         lock.lock();
 620         try {
 621             final int count = this.count;
 622             final int len = a.length;
 623             if (len < count)
 624                 a = (T[])java.lang.reflect.Array.newInstance(
 625                     a.getClass().getComponentType(), count);
 626             int n = items.length - takeIndex;
 627             if (count <= n)
 628                 System.arraycopy(items, takeIndex, a, 0, count);
 629             else {
 630                 System.arraycopy(items, takeIndex, a, 0, n);
 631                 System.arraycopy(items, 0, a, n, count - n);
 632             }
 633             if (len > count)
 634                 a[count] = null;
 635         } finally {
 636             lock.unlock();
 637         }
 638         return a;
 639     }

 643         lock.lock();

 666     /**
 667      * Atomically removes all of the elements from this queue.
 668      * The queue will be empty after this call returns.
 669      */
 670     public void clear() {
 671         final Object[] items = this.items;
 672         final ReentrantLock lock = this.lock;
 673         lock.lock();
 674         try {
 675             int k = count;
 676             if (k > 0) {
 677                 final int putIndex = this.putIndex;
 678                 int i = takeIndex;
 679                 do {
 680                     items[i] = null;
 681                     if (++i == items.length)
 682                         i = 0;
 683                 } while (i != putIndex);
 684                 takeIndex = putIndex;
 685                 count = 0;
 686                 if (itrs != null)
 687                     itrs.queueIsEmpty();
 688                 for (; k > 0 && lock.hasWaiters(notFull); k--)
 689                     notFull.signal();
 690             }
 691         } finally {
 692             lock.unlock();
 693         }

 706     /**
 707      * @throws UnsupportedOperationException {@inheritDoc}
 708      * @throws ClassCastException            {@inheritDoc}
 709      * @throws NullPointerException          {@inheritDoc}
 710      * @throws IllegalArgumentException      {@inheritDoc}
 711      */
 712     public int drainTo(Collection<? super E> c, int maxElements) {
 713         checkNotNull(c);
 714         if (c == this)
 715             throw new IllegalArgumentException();
 716         if (maxElements <= 0)
 717             return 0;
 718         final Object[] items = this.items;
 719         final ReentrantLock lock = this.lock;
 720         lock.lock();
 721         try {
 722             int n = Math.min(maxElements, count);
 723             int take = takeIndex;
 724             int i = 0;
 725             try {
 726                 while (i < n) {
 727                     @SuppressWarnings("unchecked")
 728                     E x = (E) items[take];
 729                     c.add(x);
 730                     items[take] = null;
 731                     if (++take == items.length)
 732                         take = 0;
 733                     i++;
 734                 }
 735                 return n;
 736             } finally {
 737                 // Restore invariants even if c.add() threw
 738                 if (i > 0) {
 739                     count -= i;
 740                     takeIndex = take;
 741                     if (itrs != null) {
 742                         if (count == 0)
 743                             itrs.queueIsEmpty();
 744                         else if (i > take)
 745                             itrs.takeIndexWrapped();
 746                     }
 747                     for (; i > 0 && lock.hasWaiters(notFull); i--)
 748                         notFull.signal();
 749                 }
 750             }
 751         } finally {
 752             lock.unlock();
 753         }
 754     }

 769     /**
 770      * Shared data between iterators and their queue, allowing queue
 771      * modifications to update iterators when elements are removed.
 772      *
 773      * This adds a lot of complexity for the sake of correctly
 774      * handling some uncommon operations, but the combination of
 775      * circular-arrays and supporting interior removes (i.e., those
 776      * not at head) would cause iterators to sometimes lose their
 777      * places and/or (re)report elements they shouldn't.  To avoid
 778      * this, when a queue has one or more iterators, it keeps iterator
 779      * state consistent by:
 780      *
 781      * (1) keeping track of the number of "cycles", that is, the
 782      *     number of times takeIndex has wrapped around to 0.
 783      * (2) notifying all iterators via the callback removedAt whenever
 784      *     an interior element is removed (and thus other elements may
 785      *     be shifted).
 786      *
 787      * These suffice to eliminate iterator inconsistencies, but
 788      * unfortunately add the secondary responsibility of maintaining
 789      * the list of iterators.  We track all active iterators in a
 790      * simple linked list (accessed only when the queue's lock is
 791      * held) of weak references to Itr.  The list is cleaned up using
 792      * 3 different mechanisms:
 793      *
 794      * (1) Whenever a new iterator is created, do some O(1) checking for
 795      *     stale list elements.
 796      *
 797      * (2) Whenever takeIndex wraps around to 0, check for iterators
 798      *     that have been unused for more than one wrap-around cycle.
 799      *
 800      * (3) Whenever the queue becomes empty, all iterators are notified
 801      *     and this entire data structure is discarded.
 802      *
 803      * So in addition to the removedAt callback that is necessary for
 804      * correctness, iterators have the shutdown and takeIndexWrapped
 805      * callbacks that help remove stale iterators from the list.
 806      *
 807      * Whenever a list element is examined, it is expunged if either
 808      * the GC has determined that the iterator is discarded, or if the
 809      * iterator reports that it is "detached" (does not need any
 810      * further state updates).  Overhead is maximal when takeIndex
 811      * never advances, iterators are discarded before they are
 812      * exhausted, and all removals are interior removes, in which case
 813      * all stale iterators are discovered by the GC.  But even in this
 814      * case we don't increase the amortized complexity.
 815      *
 816      * Care must be taken to keep list sweeping methods from
 817      * reentrantly invoking another such method, causing subtle
 818      * corruption bugs.
 819      */
 820     class Itrs {
 821 
 822         /**
 823          * Node in a linked list of weak iterator references.
 824          */
 825         private class Node extends WeakReference<Itr> {
 826             Node next;
 827 
 828             Node(Itr iterator, Node next) {
 829                 super(iterator);
 830                 this.next = next;
 831             }
 832         }
 833 
 834         /** Incremented whenever takeIndex wraps around to 0 */
 835         int cycles = 0;
 836 
 837         /** Linked list of weak iterator references */
 838         private Node head;
 839 
 840         /** Used to expunge stale iterators */
 841         private Node sweeper = null;
 842 
 843         private static final int SHORT_SWEEP_PROBES = 4;
 844         private static final int LONG_SWEEP_PROBES = 16;
 845 
 846         Itrs(Itr initial) {
 847             register(initial);
 848         }

 850         /**
 851          * Sweeps itrs, looking for and expunging stale iterators.
 852          * If at least one was found, tries harder to find more.
 853          * Called only from iterating thread.
 854          *
 855          * @param tryHarder whether to start in try-harder mode, because
 856          * there is known to be at least one iterator to collect
 857          */
 858         void doSomeSweeping(boolean tryHarder) {
 859             // assert lock.getHoldCount() == 1;
 860             // assert head != null;
 861             int probes = tryHarder ? LONG_SWEEP_PROBES : SHORT_SWEEP_PROBES;
 862             Node o, p;
 863             final Node sweeper = this.sweeper;
 864             boolean passedGo;   // to limit search to one full sweep
 865 
 866             if (sweeper == null) {
 867                 o = null;
 868                 p = head;
 869                 passedGo = true;
 870             } else {
 871                 o = sweeper;
 872                 p = o.next;
 873                 passedGo = false;
 874             }
 875 
 876             for (; probes > 0; probes--) {
 877                 if (p == null) {
 878                     if (passedGo)
 879                         break;
 880                     o = null;
 881                     p = head;
 882                     passedGo = true;
 883                 }
 884                 final Itr it = p.get();
 885                 final Node next = p.next;
 886                 if (it == null || it.isDetached()) {
 887                     // found a discarded/exhausted iterator
 888                     probes = LONG_SWEEP_PROBES; // "try harder"
 889                     // unlink p
 890                     p.clear();
 891                     p.next = null;
 892                     if (o == null) {
 893                         head = next;
 894                         if (next == null) {
 895                             // We've run out of iterators to track; retire
 896                             itrs = null;
 897                             return;
 898                         }
 899                     }
 900                     else
 901                         o.next = next;
 902                 } else {
 903                     o = p;
 904                 }
 905                 p = next;
 906             }
 907 
 908             this.sweeper = (p == null) ? null : o;
 909         }

 911         /**
 912          * Adds a new iterator to the linked list of tracked iterators.
 913          */
 914         void register(Itr itr) {
 915             // assert lock.getHoldCount() == 1;
 916             head = new Node(itr, head);
 917         }

 919         /**
 920          * Called whenever takeIndex wraps around to 0.
 921          *
 922          * Notifies all iterators, and expunges any that are now stale.
 923          */
 924         void takeIndexWrapped() {
 925             // assert lock.getHoldCount() == 1;
 926             cycles++;
 927             for (Node o = null, p = head; p != null;) {
 928                 final Itr it = p.get();
 929                 final Node next = p.next;
 930                 if (it == null || it.takeIndexWrapped()) {
 931                     // unlink p
 932                     // assert it == null || it.isDetached();
 933                     p.clear();
 934                     p.next = null;
 935                     if (o == null)
 936                         head = next;
 937                     else
 938                         o.next = next;
 939                 } else {
 940                     o = p;
 941                 }
 942                 p = next;
 943             }
 944             if (head == null)   // no more iterators to track
 945                 itrs = null;
 946         }
 947 
 948         /**
 949          * Called whenever an interior remove (not at takeIndex) occurred.
 950          *
 951          * Notifies all iterators, and expunges any that are now stale.
 952          */
 953         void removedAt(int removedIndex) {
 954             for (Node o = null, p = head; p != null;) {
 955                 final Itr it = p.get();
 956                 final Node next = p.next;
 957                 if (it == null || it.removedAt(removedIndex)) {
 958                     // unlink p
 959                     // assert it == null || it.isDetached();
 960                     p.clear();
 961                     p.next = null;
 962                     if (o == null)
 963                         head = next;
 964                     else
 965                         o.next = next;
 966                 } else {
 967                     o = p;
 968                 }
 969                 p = next;
 970             }
 971             if (head == null)   // no more iterators to track
 972                 itrs = null;
 973         }
 974 
 975         /**
 976          * Called whenever the queue becomes empty.
 977          *
 978          * Notifies all active iterators that the queue is empty,
 979          * clears all weak refs, and unlinks the itrs datastructure.
 980          */
 981         void queueIsEmpty() {
 982             // assert lock.getHoldCount() == 1;
 983             for (Node p = head; p != null; p = p.next) {
 984                 Itr it = p.get();
 985                 if (it != null) {
 986                     p.clear();
 987                     it.shutdown();
 988                 }
 989             }
 990             head = null;
 991             itrs = null;
 992         }
 993 
 994         /**
 995          * Called whenever an element has been dequeued (at takeIndex).
 996          */
 997         void elementDequeued() {
 998             // assert lock.getHoldCount() == 1;
 999             if (count == 0)
1000                 queueIsEmpty();
1001             else if (takeIndex == 0)
1002                 takeIndexWrapped();
1003         }

1006     /**
1007      * Iterator for ArrayBlockingQueue.
1008      *
1009      * To maintain weak consistency with respect to puts and takes, we
1010      * read ahead one slot, so as to not report hasNext true but then
1011      * not have an element to return.
1012      *
1013      * We switch into "detached" mode (allowing prompt unlinking from
1014      * itrs without help from the GC) when all indices are negative, or
1015      * when hasNext returns false for the first time.  This allows the
1016      * iterator to track concurrent updates completely accurately,
1017      * except for the corner case of the user calling Iterator.remove()
1018      * after hasNext() returned false.  Even in this case, we ensure
1019      * that we don't remove the wrong element by keeping track of the
1020      * expected element to remove, in lastItem.  Yes, we may fail to
1021      * remove lastItem from the queue if it moved due to an interleaved
1022      * interior remove while in detached mode.
1023      */
1024     private class Itr implements Iterator<E> {
1025         /** Index to look for new nextItem; NONE at end */
1026         private int cursor;
1027 
1028         /** Element to be returned by next call to next(); null if none */
1029         private E nextItem;
1030 
1031         /** Index of nextItem; NONE if none, REMOVED if removed elsewhere */
1032         private int nextIndex;
1033 
1034         /** Last element returned; null if none or not detached. */
1035         private E lastItem;
1036 
1037         /** Index of lastItem, NONE if none, REMOVED if removed elsewhere */
1038         private int lastRet;
1039 
1040         /** Previous value of takeIndex, or DETACHED when detached */
1041         private int prevTakeIndex;
1042 
1043         /** Previous value of iters.cycles */
1044         private int prevCycles;
1045 
1046         /** Special index value indicating "not available" or "undefined" */
1047         private static final int NONE = -1;
1048 
1049         /**
1050          * Special index value indicating "removed elsewhere", that is,
1051          * removed by some operation other than a call to this.remove().
1052          */
1053         private static final int REMOVED = -2;
1054 
1055         /** Special value for prevTakeIndex indicating "detached mode" */
1056         private static final int DETACHED = -3;
1057 
1058         Itr() {
1059             // assert lock.getHoldCount() == 0;
1060             lastRet = NONE;
1061             final ReentrantLock lock = ArrayBlockingQueue.this.lock;
1062             lock.lock();
1063             try {
1064                 if (count == 0) {
1065                     // assert itrs == null;
1066                     cursor = NONE;
1067                     nextIndex = NONE;
1068                     prevTakeIndex = DETACHED;
1069                 } else {
1070                     final int takeIndex = ArrayBlockingQueue.this.takeIndex;
1071                     prevTakeIndex = takeIndex;
1072                     nextItem = itemAt(nextIndex = takeIndex);
1073                     cursor = incCursor(takeIndex);
1074                     if (itrs == null) {
1075                         itrs = new Itrs(this);
1076                     } else {
1077                         itrs.register(this); // in this order
1078                         itrs.doSomeSweeping(false);
1079                     }
1080                     prevCycles = itrs.cycles;
1081                     // assert takeIndex >= 0;
1082                     // assert prevTakeIndex == takeIndex;
1083                     // assert nextIndex >= 0;
1084                     // assert nextItem != null;
1085                 }
1086             } finally {
1087                 lock.unlock();
1088             }
1089         }
1090 
1091         boolean isDetached() {
1092             // assert lock.getHoldCount() == 1;
1093             return prevTakeIndex < 0;
1094         }
1095 
1096         private int incCursor(int index) {
1097             // assert lock.getHoldCount() == 1;
1098             if (++index == items.length)
1099                 index = 0;
1100             if (index == putIndex)
1101                 index = NONE;
1102             return index;
1103         }
1104 
1105         /**
1106          * Returns true if index is invalidated by the given number of
1107          * dequeues, starting from prevTakeIndex.
1108          */
1109         private boolean invalidated(int index, int prevTakeIndex,
1110                                     long dequeues, int length) {
1111             if (index < 0)
1112                 return false;
1113             int distance = index - prevTakeIndex;
1114             if (distance < 0)
1115                 distance += length;
1116             return dequeues > distance;
1117         }
1118 
1119         /**
1120          * Adjusts indices to incorporate all dequeues since the last
1121          * operation on this iterator.  Call only from iterating thread.
1122          */
1123         private void incorporateDequeues() {
1124             // assert lock.getHoldCount() == 1;
1125             // assert itrs != null;
1126             // assert !isDetached();
1127             // assert count > 0;
1128 
1129             final int cycles = itrs.cycles;
1130             final int takeIndex = ArrayBlockingQueue.this.takeIndex;
1131             final int prevCycles = this.prevCycles;
1132             final int prevTakeIndex = this.prevTakeIndex;
1133 
1134             if (cycles != prevCycles || takeIndex != prevTakeIndex) {
1135                 final int len = items.length;
1136                 // how far takeIndex has advanced since the previous
1137                 // operation of this iterator
1138                 long dequeues = (cycles - prevCycles) * len
1139                     + (takeIndex - prevTakeIndex);
1140 
1141                 // Check indices for invalidation
1142                 if (invalidated(lastRet, prevTakeIndex, dequeues, len))
1143                     lastRet = REMOVED;
1144                 if (invalidated(nextIndex, prevTakeIndex, dequeues, len))
1145                     nextIndex = REMOVED;
1146                 if (invalidated(cursor, prevTakeIndex, dequeues, len))
1147                     cursor = takeIndex;
1148 
1149                 if (cursor < 0 && nextIndex < 0 && lastRet < 0)
1150                     detach();
1151                 else {
1152                     this.prevCycles = cycles;
1153                     this.prevTakeIndex = takeIndex;
1154                 }
1155             }
1156         }
1157 
1158         /**
1159          * Called when itrs should stop tracking this iterator, either
1160          * because there are no more indices to update (cursor < 0 &&
1161          * nextIndex < 0 && lastRet < 0) or as a special exception, when
1162          * lastRet >= 0, because hasNext() is about to return false for the
1163          * first time.  Call only from iterating thread.
1164          */
1165         private void detach() {
1166             // Switch to detached mode
1167             // assert lock.getHoldCount() == 1;
1168             // assert cursor == NONE;
1169             // assert nextIndex < 0;
1170             // assert lastRet < 0 || nextItem == null;
1171             // assert lastRet < 0 ^ lastItem != null;
1172             if (prevTakeIndex >= 0) {
1173                 // assert itrs != null;
1174                 prevTakeIndex = DETACHED;
1175                 // try to unlink from itrs (but not too hard)
1176                 itrs.doSomeSweeping(true);
1177             }
1178         }
1179 
1180         /**
1181          * For performance reasons, we would like not to acquire a lock in
1182          * hasNext in the common case.  To allow for this, we only access
1183          * fields (i.e. nextItem) that are not modified by update operations
1184          * triggered by queue modifications.
1185          */
1186         public boolean hasNext() {
1187             // assert lock.getHoldCount() == 0;
1188             if (nextItem != null)
1189                 return true;
1190             noNext();
1191             return false;
1192         }
1193 
1194         private void noNext() {
1195             final ReentrantLock lock = ArrayBlockingQueue.this.lock;
1196             lock.lock();
1197             try {
1198                 // assert cursor == NONE;
1199                 // assert nextIndex == NONE;
1200                 if (!isDetached()) {
1201                     // assert lastRet >= 0;
1202                     incorporateDequeues(); // might update lastRet
1203                     if (lastRet >= 0) {
1204                         lastItem = itemAt(lastRet);
1205                         // assert lastItem != null;
1206                         detach();
1207                     }
1208                 }
1209                 // assert isDetached();
1210                 // assert lastRet < 0 ^ lastItem != null;
1211             } finally {
1212                 lock.unlock();
1213             }
1214         }

1216         public E next() {
1217             // assert lock.getHoldCount() == 0;
1218             final E x = nextItem;
1219             if (x == null)
1220                 throw new NoSuchElementException();
1221             final ReentrantLock lock = ArrayBlockingQueue.this.lock;
1222             lock.lock();
1223             try {
1224                 if (!isDetached())
1225                     incorporateDequeues();
1226                 // assert nextIndex != NONE;
1227                 // assert lastItem == null;
1228                 lastRet = nextIndex;
1229                 final int cursor = this.cursor;
1230                 if (cursor >= 0) {
1231                     nextItem = itemAt(nextIndex = cursor);
1232                     // assert nextItem != null;
1233                     this.cursor = incCursor(cursor);
1234                 } else {
1235                     nextIndex = NONE;
1236                     nextItem = null;
1237                 }
1238             } finally {
1239                 lock.unlock();
1240             }
1241             return x;
1242         }

1244         public void remove() {
1245             // assert lock.getHoldCount() == 0;
1246             final ReentrantLock lock = ArrayBlockingQueue.this.lock;
1247             lock.lock();
1248             try {
1249                 if (!isDetached())
1250                     incorporateDequeues(); // might update lastRet or detach
1251                 final int lastRet = this.lastRet;
1252                 this.lastRet = NONE;
1253                 if (lastRet >= 0) {
1254                     if (!isDetached())
1255                         removeAt(lastRet);
1256                     else {
1257                         final E lastItem = this.lastItem;
1258                         // assert lastItem != null;
1259                         this.lastItem = null;
1260                         if (itemAt(lastRet) == lastItem)
1261                             removeAt(lastRet);
1262                     }
1263                 } else if (lastRet == NONE)
1264                     throw new IllegalStateException();
1265                 // else lastRet == REMOVED and the last returned element was
1266                 // previously asynchronously removed via an operation other
1267                 // than this.remove(), so nothing to do.
1268 
1269                 if (cursor < 0 && nextIndex < 0)
1270                     detach();
1271             } finally {
1272                 lock.unlock();
1273                 // assert lastRet == NONE;
1274                 // assert lastItem == null;
1275             }
1276         }

1278         /**
1279          * Called to notify the iterator that the queue is empty, or that it
1280          * has fallen hopelessly behind, so that it should abandon any
1281          * further iteration, except possibly to return one more element
1282          * from next(), as promised by returning true from hasNext().
1283          */
1284         void shutdown() {
1285             // assert lock.getHoldCount() == 1;
1286             cursor = NONE;
1287             if (nextIndex >= 0)
1288                 nextIndex = REMOVED;
1289             if (lastRet >= 0) {
1290                 lastRet = REMOVED;
1291                 lastItem = null;
1292             }
1293             prevTakeIndex = DETACHED;
1294             // Don't set nextItem to null because we must continue to be
1295             // able to return it on next().
1296             //
1297             // Caller will unlink from itrs when convenient.
1298         }

1300         private int distance(int index, int prevTakeIndex, int length) {
1301             int distance = index - prevTakeIndex;
1302             if (distance < 0)
1303                 distance += length;
1304             return distance;
1305         }

1307         /**
1308          * Called whenever an interior remove (not at takeIndex) occurred.
1309          *
1310          * @return true if this iterator should be unlinked from itrs
1311          */
1312         boolean removedAt(int removedIndex) {
1313             // assert lock.getHoldCount() == 1;
1314             if (isDetached())
1315                 return true;
1316 
1317             final int cycles = itrs.cycles;
1318             final int takeIndex = ArrayBlockingQueue.this.takeIndex;
1319             final int prevCycles = this.prevCycles;
1320             final int prevTakeIndex = this.prevTakeIndex;
1321             final int len = items.length;
1322             int cycleDiff = cycles - prevCycles;
1323             if (removedIndex < takeIndex)
1324                 cycleDiff++;
1325             final int removedDistance =
1326                 (cycleDiff * len) + (removedIndex - prevTakeIndex);
1327             // assert removedDistance >= 0;
1328             int cursor = this.cursor;
1329             if (cursor >= 0) {
1330                 int x = distance(cursor, prevTakeIndex, len);
1331                 if (x == removedDistance) {
1332                     if (cursor == putIndex)
1333                         this.cursor = cursor = NONE;
1334                 }
1335                 else if (x > removedDistance) {
1336                     // assert cursor != prevTakeIndex;
1337                     this.cursor = cursor = dec(cursor);
1338                 }
1339             }
1340             int lastRet = this.lastRet;
1341             if (lastRet >= 0) {
1342                 int x = distance(lastRet, prevTakeIndex, len);
1343                 if (x == removedDistance)
1344                     this.lastRet = lastRet = REMOVED;
1345                 else if (x > removedDistance)
1346                     this.lastRet = lastRet = dec(lastRet);
1347             }
1348             int nextIndex = this.nextIndex;
1349             if (nextIndex >= 0) {
1350                 int x = distance(nextIndex, prevTakeIndex, len);
1351                 if (x == removedDistance)
1352                     this.nextIndex = nextIndex = REMOVED;
1353                 else if (x > removedDistance)
1354                     this.nextIndex = nextIndex = dec(nextIndex);
1355             }
1356             else if (cursor < 0 && nextIndex < 0 && lastRet < 0) {
1357                 this.prevTakeIndex = DETACHED;
1358                 return true;
1359             }
1360             return false;
1361         }
1362 
1363         /**
1364          * Called whenever takeIndex wraps around to zero.
1365          *
1366          * @return true if this iterator should be unlinked from itrs
1367          */
1368         boolean takeIndexWrapped() {
1369             // assert lock.getHoldCount() == 1;
1370             if (isDetached())
1371                 return true;
1372             if (itrs.cycles - prevCycles > 1) {
1373                 // All the elements that existed at the time of the last
1374                 // operation are gone, so abandon further iteration.
1375                 shutdown();
1376                 return true;
1377             }
1378             return false;
1379         }
