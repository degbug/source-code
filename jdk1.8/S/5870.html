  23 package com.sun.org.apache.xml.internal.utils;
  24 
  25 import java.util.EmptyStackException;
  26 
  27 /**
  28  * Implement a stack of simple integers.
  29  *
  30  * %OPT%
  31  * This is currently based on IntVector, which permits fast acess but pays a
  32  * heavy recopying penalty if/when its size is increased. If we expect deep
  33  * stacks, we should consider a version based on ChunkedIntVector.
  34  * @xsl.usage internal
  35  */
  36 public class IntStack extends IntVector
  37 {
  38 
  39   /**
  40    * Default constructor.  Note that the default
  41    * block size is very small, for small lists.
  42    */
  43   public IntStack()
  44   {
  45     super();
  46   }
  47 
  48   /**
  49    * Construct a IntVector, using the given block size.
  50    *
  51    * @param blocksize Size of block to allocate
  52    */
  53   public IntStack(int blocksize)
  54   {
  55     super(blocksize);
  56   }
  57 
  58   /**
  59    * Copy constructor for IntStack
  60    *
  61    * @param v IntStack to copy
  62    */
  63   public IntStack (IntStack v)
  64   {
  65         super(v);
  66   }
  67 
  68   /**
  69    * Pushes an item onto the top of this stack.
  70    *
  71    * @param   i   the int to be pushed onto this stack.
  72    * @return  the <code>item</code> argument.
  73    */
  74   public int push(int i)
  75   {
  76 
  77     if ((m_firstFree + 1) >= m_mapSize)
  78     {
  79       m_mapSize += m_blocksize;
  80 
  81       int newMap[] = new int[m_mapSize];
  82 
  83       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
  84 
  85       m_map = newMap;
  86     }
  87 
  88     m_map[m_firstFree] = i;
  89 
  90     m_firstFree++;
  91 
  92     return i;
  93   }
  94 
  95   /**
  96    * Removes the object at the top of this stack and returns that
  97    * object as the value of this function.
  98    *
  99    * @return     The object at the top of this stack.
 100    */
 101   public final int pop()
 102   {
 103     return m_map[--m_firstFree];
 104   }
 105 
 106   /**
 107    * Quickly pops a number of items from the stack.
 108    */
 109 
 110   public final void quickPop(int n)
 111   {
 112     m_firstFree -= n;
 113   }
 114 
 115   /**
 116    * Looks at the object at the top of this stack without removing it
 117    * from the stack.
 118    *
 119    * @return     the object at the top of this stack.
 120    * @throws  EmptyStackException  if this stack is empty.
 121    */
 122   public final int peek()
 123   {
 124     try {
 125       return m_map[m_firstFree - 1];
 126     }
 127     catch (ArrayIndexOutOfBoundsException e)
 128     {
 129       throw new EmptyStackException();
 130     }
 131   }
 132 
 133   /**
 134    * Looks at the object at the position the stack counting down n items.
 135    *
 136    * @param n The number of items down, indexed from zero.
 137    * @return     the object at n items down.
 138    * @throws  EmptyStackException  if this stack is empty.
 139    */
 140   public int peek(int n)
 141   {
 142     try {
 143       return m_map[m_firstFree-(1+n)];
 144     }
 145     catch (ArrayIndexOutOfBoundsException e)
 146     {
 147       throw new EmptyStackException();
 148     }
 149   }
 150 
 151   /**
 152    * Sets an object at a the top of the statck
 153    *
 154    *
 155    * @param val object to set at the top
 156    * @throws  EmptyStackException  if this stack is empty.
 157    */
 158   public void setTop(int val)
 159   {
 160     try {
 161       m_map[m_firstFree - 1] = val;
 162     }
 163     catch (ArrayIndexOutOfBoundsException e)
 164     {
 165       throw new EmptyStackException();
 166     }
 167   }
 168 
 169   /**
 170    * Tests if this stack is empty.
 171    *
 172    * @return  <code>true</code> if this stack is empty;
 173    *          <code>false</code> otherwise.
 174    * @since   JDK1.0
 175    */
 176   public boolean empty()
 177   {
 178     return m_firstFree == 0;
 179   }
 180 
 181   /**
 182    * Returns where an object is on this stack.
 183    *
 184    * @param   o   the desired object.
 185    * @return  the distance from the top of the stack where the object is]
 186    *          located; the return value <code>-1</code> indicates that the
 187    *          object is not on the stack.
 188    * @since   JDK1.0
 189    */
 190   public int search(int o)
 191   {
 192 
 193     int i = lastIndexOf(o);
 194 
 195     if (i >= 0)
 196     {
 197       return size() - i;
 198     }
 199 
 200     return -1;
 201   }
 202 
 203   /**
 204    * Returns clone of current IntStack
 205    *
 206    * @return clone of current IntStack
 207    */
 208   public Object clone()
 209     throws CloneNotSupportedException
 210   {
 211         return (IntStack) super.clone();
 212   }
 213 }
