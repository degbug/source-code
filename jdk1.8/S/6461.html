 113 class SendQ extends Vector<SnmpInformRequest> {
 114 
 115     SendQ(int initialCapacity, int capacityIncr) {
 116         super(initialCapacity , capacityIncr) ;
 117     }
 118 
 119     private synchronized void notifyClients() {
 120         this.notifyAll() ;
 121     }
 122 
 123     public synchronized void addRequest(SnmpInformRequest req) {
 124 
 125         long nextPoll = req.getAbsNextPollTime() ;
 126 
 127         int i ;
 128         for (i = size() ; i > 0 ; i--) {
 129             if (nextPoll < getRequestAt(i-1).getAbsNextPollTime())
 130                 break ;
 131         }
 132         if (i == size()) {
 133             addElement(req) ;
 134             notifyClients() ;
 135         } else
 136             insertElementAt(req, i) ;
 137     }
 138 
 139     public synchronized boolean waitUntilReady() {
 140         while (true) {
 141             if (isBeingDestroyed == true)
 142                 return false;
 143             long tmp = 0 ;
 144             if (isEmpty() == false) {
 145                 long currTime = System.currentTimeMillis() ;
 146                 SnmpInformRequest req = lastElement() ;
 147                 tmp = req.getAbsNextPollTime() - currTime ;
 148                 if (tmp <= 0) {
 149                     return true ;
 150                 }
 151             }
 152             waitOnThisQueue(tmp) ;
 153         }
 154     }
 155 
 156     public synchronized Vector<SnmpInformRequest> getAllOutstandingRequest(long margin) {
 157         int i ;
 158         Vector<SnmpInformRequest> outreq = new Vector<>();
 159         while (true) {
 160             if (waitUntilReady() == true) {
 161                 long refTime = System.currentTimeMillis() + margin ;
 162 
 163                 for (i = size() ; i > 0 ; i--) {
 164                     SnmpInformRequest req = getRequestAt(i-1) ;
 165                     if (req.getAbsNextPollTime() > refTime)
 166                         break ;
 167                     outreq.addElement(req) ;
 168                 }
 169 
 170                 if (! outreq.isEmpty()) {
 171                     elementCount -= outreq.size() ;
 172                     return outreq ;
 173                 }
 174             }
 175             else
 176                 return null;
 177         }
 178     }
 179 
 180     public synchronized void waitOnThisQueue(long time) {
 181         if (time == 0 && !isEmpty()) {
 182             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 183                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpQManager.class.getName(),
 184                     "waitOnThisQueue", "[" + Thread.currentThread().toString() + "]:" +
 185                     "Fatal BUG :: Blocking on newq permenantly. But size = " + size());
 186             }
 187         }
 188 
 189         try {
 190             this.wait(time) ;
 191         } catch (InterruptedException e) {
 192         }
 193     }
 194 
 195     public SnmpInformRequest getRequestAt(int idx) {
 196         return elementAt(idx) ;
 197     }
 198 
 199     public synchronized SnmpInformRequest removeRequest(long reqid) {
 200         int max= size() ;
 201         for (int i = 0 ; i < max ; i++) {
 202             SnmpInformRequest reqc = getRequestAt(i) ;
 203             if (reqid == reqc.getRequestId()) {
 204                 removeElementAt(i) ;
 205                 return reqc ;
 206             }
 207         }
 208         return null ;
 209     }
