  89      * Creates and returns the GraphicsEnvironment, according to the
  90      * system property 'java.awt.graphicsenv'.
  91      *
  92      * @return the graphics environment
  93      */
  94     private static GraphicsEnvironment createGE() {
  95         GraphicsEnvironment ge;
  96         String nm = AccessController.doPrivileged(new GetPropertyAction("java.awt.graphicsenv", null));
  97         try {
  98 //          long t0 = System.currentTimeMillis();
  99             Class<GraphicsEnvironment> geCls;
 100             try {
 101                 // First we try if the bootclassloader finds the requested
 102                 // class. This way we can avoid to run in a privileged block.
 103                 geCls = (Class<GraphicsEnvironment>)Class.forName(nm);
 104             } catch (ClassNotFoundException ex) {
 105                 // If the bootclassloader fails, we try again with the
 106                 // application classloader.
 107                 ClassLoader cl = ClassLoader.getSystemClassLoader();
 108                 geCls = (Class<GraphicsEnvironment>)Class.forName(nm, true, cl);
 109             }
 110             ge = geCls.newInstance();
 111 //          long t1 = System.currentTimeMillis();
 112 //          System.out.println("GE creation took " + (t1-t0)+ "ms.");
 113             if (isHeadless()) {
 114                 ge = new HeadlessGraphicsEnvironment(ge);
 115             }
 116         } catch (ClassNotFoundException e) {
 117             throw new Error("Could not find class: "+nm);
 118         } catch (InstantiationException e) {
 119             throw new Error("Could not instantiate Graphics Environment: "
 120                             + nm);
 121         } catch (IllegalAccessException e) {
 122             throw new Error ("Could not access Graphics Environment: "
 123                              + nm);
 124         }
 125         return ge;
 126     }

 158     /**
 159      * @return the value of the property "java.awt.headless"
 160      * @since 1.4
 161      */
 162     private static boolean getHeadlessProperty() {
 163         if (headless == null) {
 164             AccessController.doPrivileged((PrivilegedAction<Void>) () -> {
 165                 String nm = System.getProperty("java.awt.headless");
 166 
 167                 if (nm == null) {
 168                     /* No need to ask for DISPLAY when run in a browser */
 169                     if (System.getProperty("javaplugin.version") != null) {
 170                         headless = defaultHeadless = Boolean.FALSE;
 171                     } else {
 172                         String osName = System.getProperty("os.name");
 173                         if (osName.contains("OS X") && "sun.awt.HToolkit".equals(
 174                                 System.getProperty("awt.toolkit")))
 175                         {
 176                             headless = defaultHeadless = Boolean.TRUE;
 177                         } else {
 178                             final String display = System.getenv("DISPLAY");
 179                             headless = defaultHeadless =
 180                                 ("Linux".equals(osName) ||
 181                                  "SunOS".equals(osName) ||
 182                                  "FreeBSD".equals(osName) ||
 183                                  "NetBSD".equals(osName) ||
 184                                  "OpenBSD".equals(osName) ||
 185                                  "AIX".equals(osName)) &&
 186                                  (display == null || display.trim().isEmpty());
 187                         }
 188                     }
 189                 } else {
 190                     headless = Boolean.valueOf(nm);
 191                 }
 192                 return null;
 193             });
 194         }
 195         return headless;
 196     }

 331     /**
 332      * Registers a created Fontin this
 333      * GraphicsEnvironment.
 334      * A created font is one that was returned from calling
 335      * {@link Font#createFont}, or derived from a created font by
 336      * calling {@link Font#deriveFont}.
 337      * After calling this method for such a font, it is available to
 338      * be used in constructing new Fonts by name or family name,
 339      * and is enumerated by {@link #getAvailableFontFamilyNames} and
 340      * {@link #getAllFonts} within the execution context of this
 341      * application or applet. This means applets cannot register fonts in
 342      * a way that they are visible to other applets.
 343      * 
 344      * Reasons that this method might not register the font and therefore
 345      * return false are:
 346      * 
    
 347      * 
  • The font is not a created Font.
    • 
 348      * 
  • The font conflicts with a non-created Font already
    • 
 349      * in this GraphicsEnvironment. For example if the name
 350      * is that of a system font, or a logical font as described in the
 351      * documentation of the {@link Font} class. It is implementation dependent
 352      * whether a font may also conflict if it has the same family name
 353      * as a system font.
 354      * Notice that an application can supersede the registration
 355      * of an earlier created font with a new one.
 356      * 
 357      * @return true if the font is successfully
 358      * registered in this GraphicsEnvironment.
 359      * @throws NullPointerException if font is null
 360      * @since 1.6
 361      */
 362     public boolean registerFont(Font font) {
 363         if (font == null) {
 364             throw new NullPointerException("font cannot be null.");
 365         }
 366         FontManager fm = FontManagerFactory.getInstance();
 367         return fm.registerFont(font);
 368     }

 371      * Indicates a preference for locale-specific fonts in the mapping of
 372      * logical fonts to physical fonts. Calling this method indicates that font
 373      * rendering should primarily use fonts specific to the primary writing
 374      * system (the one indicated by the default encoding and the initial
 375      * default locale). For example, if the primary writing system is
 376      * Japanese, then characters should be rendered using a Japanese font
 377      * if possible, and other fonts should only be used for characters for
 378      * which the Japanese font doesn't have glyphs.
 379      * 
 380      * The actual change in font rendering behavior resulting from a call
 381      * to this method is implementation dependent; it may have no effect at
 382      * all, or the requested behavior may already match the default behavior.
 383      * The behavior may differ between font rendering in lightweight
 384      * and peered components.  Since calling this method requests a
 385      * different font, clients should expect different metrics, and may need
 386      * to recalculate window sizes and layout. Therefore this method should
 387      * be called before user interface initialisation.

 426     public Point getCenterPoint() throws HeadlessException {

 429         Rectangle usableBounds =
 430          SunGraphicsEnvironment.getUsableBounds(getDefaultScreenDevice());
 431         return new Point((usableBounds.width / 2) + usableBounds.x,
