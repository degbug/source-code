 409     @Override
 410     public final OptionalDouble average() {
 411         long[] avg = collect(() -> new long[2],
 412                              (ll, i) -> {
 413                                  ll[0]++;
 414                                  ll[1] += i;
 415                              },
 416                              (ll, rr) -> {
 417                                  ll[0] += rr[0];
 418                                  ll[1] += rr[1];
 419                              });
 420         return avg[0] > 0
 421                ? OptionalDouble.of((double) avg[1] / avg[0])
 422                : OptionalDouble.empty();
