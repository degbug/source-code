  27 /**
  28  * Holds information about a given encoding, which is the Java name for the
  29  * encoding, the equivalent ISO name.
  30  * 
  31  * An object of this type has two useful methods
  32  * 

  33  * isInEncoding(char ch);
  34  * 
  35  * which can be called if the character is not the high one in
  36  * a surrogate pair and:
  37  * 

  38  * isInEncoding(char high, char low);
  39  * 
  40  * which can be called if the two characters from a high/low surrogate pair.
  41  * 
  42  * An EncodingInfo object is a node in a binary search tree. Such a node
  43  * will answer if a character is in the encoding, and do so for a given
  44  * range of unicode values (m_first to
  45  * m_last). It will handle a certain range of values
  46  * explicitly (m_explFirst to m_explLast).
  47  * If the unicode point is before that explicit range, that is it
  48  * is in the range m_first <= value < m_explFirst, then it will delegate to another EncodingInfo object for The root
  49  * of such a tree, m_before.  Likewise for values in the range
  50  * m_explLast < value <= m_last, but delgating to m_after
  51  * 
  52  * Actually figuring out if a code point is in the encoding is expensive. So the
  53  * purpose of this tree is to cache such determinations, and not to build the
  54  * entire tree of information at the start, but only build up as much of the
  55  * tree as is used during the transformation.
  56  * 
  57  * This Class is not a public API, and should only be used internally within
  58  * the serializer.
  59  *
  60  * @xsl.usage internal
  61  */
  62 public final class EncodingInfo extends Object

 166         public boolean isInEncoding(char ch1) {
 167             final boolean ret;
 168             int codePoint = Encodings.toCodePoint(ch1);
 169             if (codePoint < m_explFirst) {
 170                 // The unicode value is before the range
 171                 // that we explictly manage, so we delegate the answer.
 172 
 173                 // If we don't have an m_before object to delegate to, make one.
 174                 if (m_before == null)
 175                     m_before =
 176                         new EncodingImpl(
 177                             m_encoding,
 178                             m_first,
 179                             m_explFirst - 1,
 180                             codePoint);
 181                 ret = m_before.isInEncoding(ch1);
 182             } else if (m_explLast < codePoint) {
 183                 // The unicode value is after the range
 184                 // that we explictly manage, so we delegate the answer.
 185 
 186                 // If we don't have an m_after object to delegate to, make one.
 187                 if (m_after == null)
 188                     m_after =
 189                         new EncodingImpl(
 190                             m_encoding,
 191                             m_explLast + 1,
 192                             m_last,
 193                             codePoint);
 194                 ret = m_after.isInEncoding(ch1);
 195             } else {
 196                 // The unicode value is in the range we explitly handle
 197                 final int idx = codePoint - m_explFirst;
 198 
 199                 // If we already know the answer, just return it.
 200                 if (m_alreadyKnown[idx])
 201                     ret = m_isInEncoding[idx];
 202                 else {
 203                     // We don't know the answer, so find out,
 204                     // which may be expensive, then cache the answer
 205                     ret = inEncoding(ch1, m_encoding);
 206                     m_alreadyKnown[idx] = true;
 207                     m_isInEncoding[idx] = ret;
 208                 }
 209             }
 210             return ret;
 211         }

 316         private EncodingImpl() {
 317             // This object will answer whether any unicode value
 318             // is in the encoding, it handles values 0 through Integer.MAX_VALUE
 319             this(javaName, 0, Integer.MAX_VALUE, (char) 0);
 320         }

 322         private EncodingImpl(String encoding, int first, int last, int codePoint) {
 323             // Set the range of unicode values that this object manages
 324             // either explicitly or implicitly.
 325             m_first = first;
 326             m_last = last;
 327 
 328             // Set the range of unicode values that this object
 329             // explicitly manages. Align the explicitly managed values
 330             // to RANGE so multiple EncodingImpl objects dont manage the same
 331             // values.
 332             m_explFirst = codePoint / RANGE * RANGE;
 333             m_explLast = m_explFirst + (RANGE-1);
 334 
 335             m_encoding = encoding;
 336 
 337             if (javaName != null)
 338             {
 339                 // Some optimization.
 340                 if (0 <= m_explFirst && m_explFirst <= 127) {
 341                     // This particular EncodingImpl explicitly handles
 342                     // characters in the low range.
 343                     if ("UTF8".equals(javaName)
 344                         || "UTF-16".equals(javaName)
 345                         || "ASCII".equals(javaName)
 346                         || "US-ASCII".equals(javaName)
 347                         || "Unicode".equals(javaName)
 348                         || "UNICODE".equals(javaName)
 349                         || javaName.startsWith("ISO8859")) {
 350 
 351                         // Not only does this EncodingImpl object explicitly
 352                         // handle chracters in the low range, it is
 353                         // also one that we know something about, without
 354                         // needing to call inEncoding(char ch, String encoding)
 355                         // for this low range
 356                         //
 357                         // By initializing the table ahead of time
 358                         // for these low values, we prevent the expensive
 359                         // inEncoding(char ch, String encoding)
 360                         // from being called, at least for these common
 361                         // encodings.
 362                         for (int unicode = 1; unicode < 127; unicode++) {
 363                             final int idx = unicode - m_explFirst;
 364                             if (0 <= idx && idx < RANGE) {
 365                                 m_alreadyKnown[idx] = true;
 366                                 m_isInEncoding[idx] = true;
 367                             }
 368                         }
 369                     }
 370                 }
 371 
 372                 /* A little bit more than optimization.
 373                  *
 374                  * We will say that any character is in the encoding if
 375                  * we don't have an encoding.
 376                  * This is meaningful when the serializer is being used
 377                  * in temporary output state, where we are not writing to
 378                  * the final output tree.  It is when writing to the
 379                  * final output tree that we need to worry about the output
 380                  * encoding
 381                  */
 382                 if (javaName == null) {
 383                     for (int idx = 0; idx < m_alreadyKnown.length; idx++) {
 384                         m_alreadyKnown[idx] = true;
 385                         m_isInEncoding[idx] = true;
 386                     }
 387                 }
 388             }
 389         }

 392     /**
 393      * This is heart of the code that determines if a given character
 394      * is in the given encoding. This method is probably expensive,
 395      * and the answer should be cached.
 396      * 
 397      * This method is not a public API,
 398      * and should only be used internally within the serializer.
 399      * @param ch the char in question, that is not a high char of
 400      * a high/low surrogate pair.
 401      * @param encoding the Java name of the enocding.
 402      *
 403      * @xsl.usage internal
 404      *
 405      */
 406     private static boolean inEncoding(char ch, String encoding) {
 407         boolean isInEncoding;
 408         try {
 409             char cArray[] = new char[1];
 410             cArray[0] = ch;
 411             // Construct a String from the char
 412             String s = new String(cArray);
 413             // Encode the String into a sequence of bytes
 414             // using the given, named charset.
 415             byte[] bArray = s.getBytes(encoding);
 416             isInEncoding = inEncoding(ch, bArray);
 417 
 418         } catch (Exception e) {
 419             isInEncoding = false;
 420 
 421             // If for some reason the encoding is null, e.g.
 422             // for a temporary result tree, we should just
 423             // say that every character is in the encoding.
 424             if (encoding == null)
 425                 isInEncoding = true;
 426         }
 427         return isInEncoding;
 428     }

 464     /**
 465      * This method is the core of determining if character
 466      * is in the encoding. The method is not foolproof, because
 467      * s.getBytes(encoding) has specified behavior only if the
 468      * characters are in the specified encoding. However this
 469      * method tries it's best.
 470      * @param ch the char that was converted using getBytes, or
 471      * the first char of a high/low pair that was converted.
 472      * @param data the bytes written out by the call to s.getBytes(encoding);
 473      * @return true if the character is in the encoding.
 474      */
 475     private static boolean inEncoding(char ch, byte[] data) {
 476         final boolean isInEncoding;
 477         // If the string written out as data is not in the encoding,
 478         // the output is not specified according to the documentation
 479         // on the String.getBytes(encoding) method,
 480         // but we do our best here.
 481         if (data==null || data.length == 0) {
 482             isInEncoding = false;
 483         }
 484         else {
 485             if (data[0] == 0)
 486                 isInEncoding = false;
 487             else if (data[0] == '?' && ch != '?')
 488                 isInEncoding = false;
 489             /*
 490              * else if (isJapanese) {
 491              *   // isJapanese is really
 492              *   //   (    "EUC-JP".equals(javaName)
 493              *   //    ||  "EUC_JP".equals(javaName)
 494              *  //     ||  "SJIS".equals(javaName)   )
 495              *
 496              *   // Work around some bugs in JRE for Japanese
 497              *   if(data[0] == 0x21)
 498              *     isInEncoding = false;
 499              *   else if (ch == 0xA5)
 500              *     isInEncoding = false;
 501              *   else
 502              *     isInEncoding = true;
 503              * }
 504              */
 505 
 506             else {
 507                 // We don't know for sure, but it looks like it is in the encoding
 508                 isInEncoding = true;
 509             }
 510         }
 511         return isInEncoding;
 512     }
