  58    * DOM Level 1 did not have a standard mechanism for creating a new
  59    * Document object. This function provides a DOM-implementation-independent
  60    * abstraction for that for that concept. It's typically used when
  61    * outputting a new DOM as the result of an operation.
  62    * 
  63    * TODO: This isn't directly compatable with DOM Level 2.
  64    * The Level 2 createDocument call also creates the root
  65    * element, and thus requires that you know what that element will be

  67    * to change this code, and the callers, so we can use the DOM's own

  69    * sequence, but you may give up some intelligence in the DOM by

  73    *
  74    * @param isSecureProcessing state of the secure processing feature.
  75    * @return The newly created DOM Document object, with no children, or
  76    * null if we can't find a DOM implementation that permits creating
  77    * new empty Documents.
  78    */
  79   public static Document createDocument(boolean isSecureProcessing)
  80   {
  81 
  82     try
  83     {
  84 
  85       // Use an implementation of the JAVA API for XML Parsing 1.0 to
  86       // create a DOM Document node to contain the result.
  87       DocumentBuilderFactory dfactory = DocumentBuilderFactory.newInstance();
  88 
  89       dfactory.setNamespaceAware(true);
  90       dfactory.setValidating(true);
  91 
  92       if (isSecureProcessing)
  93       {
  94         try
  95         {
  96           dfactory.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, true);
  97         }
  98         catch (ParserConfigurationException pce) {}
  99       }
 100 
 101       DocumentBuilder docBuilder = dfactory.newDocumentBuilder();
 102       Document outNode = docBuilder.newDocument();
 103 
 104       return outNode;
 105     }
 106     catch (ParserConfigurationException pce)
 107     {
 108       throw new RuntimeException(
 109         XMLMessages.createXMLMessage(
 110           XMLErrorResources.ER_CREATEDOCUMENT_NOT_SUPPORTED, null));  //"createDocument() not supported in XPathContext!");

 152    * Supports the XPath function GenerateID by returning a unique
 153    * identifier string for any given DOM Node.
 154    * 
 155    * Warning: The base implementation uses the Node object's hashCode(),
 156    * which is NOT guaranteed to be unique. If that method hasn't been
 157    * overridden in this DOM ipmlementation, most Java implementions will

 181   /**
 182    * Figure out whether node2 should be considered as being later
 183    * in the document than node1, in Document Order as defined
 184    * by the XPath model. This may not agree with the ordering defined
 185    * by other XML applications.
 186    * 
 187    * There are some cases where ordering isn't defined, and neither are
 188    * the results of this function -- though we'll generally return true.
 189    *
 190    * TODO: Make sure this does the right thing with attribute nodes!!!
 191    *
 192    * @param node1 DOM Node to perform position comparison on.
 193    * @param node2 DOM Node to perform position comparison on .
 194    *
 195    * @return false if node2 comes before node1, otherwise return true.
 196    * You can think of this as
 197    * (node1.documentOrderPosition <= node2.documentOrderPosition).
 198    */
 199   public static boolean isNodeAfter(Node node1, Node node2)
 200   {
 201     if (node1 == node2 || isNodeTheSame(node1, node2))
 202       return true;
 203 
 204         // Default return value, if there is no defined ordering
 205     boolean isNodeAfter = true;
 206 
 207     Node parent1 = getParentOfNode(node1);
 208     Node parent2 = getParentOfNode(node2);
 209 
 210     // Optimize for most common case
 211     if (parent1 == parent2 || isNodeTheSame(parent1, parent2))  // then we know they are siblings
 212     {
 213       if (null != parent1)
 214         isNodeAfter = isNodeAfterSibling(parent1, node1, node2);
 215       else
 216       {
 217                   // If both parents are null, ordering is not defined.
 218                   // We're returning a value in lieu of throwing an exception.
 219                   // Not a case we expect to arise in XPath, but beware if you
 220                   // try to reuse this method.
 221 
 222                   // We can just fall through in this case, which allows us
 223                   // to hit the debugging code at the end of the function.
 224           //return isNodeAfter;
 225       }
 226     }
 227     else
 228     {
 229 
 230       // General strategy: Figure out the lengths of the two
 231       // ancestor chains, reconcile the lengths, and look for
 232           // the lowest common ancestor. If that ancestor is one of
 233           // the nodes being compared, it comes before the other.
 234       // Otherwise perform a sibling compare.
 235                 //
 236                 // NOTE: If no common ancestor is found, ordering is undefined
 237                 // and we return the default value of isNodeAfter.
 238 
 239       // Count parents in each ancestor chain
 240       int nParents1 = 2, nParents2 = 2;  // include node & parent obtained above
 241 
 242       while (parent1 != null)
 243       {
 244         nParents1++;
 245 
 246         parent1 = getParentOfNode(parent1);
 247       }
 248 
 249       while (parent2 != null)
 250       {
 251         nParents2++;
 252 
 253         parent2 = getParentOfNode(parent2);
 254       }
 255 
 256           // Initially assume scan for common ancestor starts with
 257           // the input nodes.
 258       Node startNode1 = node1, startNode2 = node2;
 259 
 260       // If one ancestor chain is longer, adjust its start point
 261           // so we're comparing at the same depths
 262       if (nParents1 < nParents2)
 263       {
 264         // Adjust startNode2 to depth of startNode1
 265         int adjust = nParents2 - nParents1;
 266 
 267         for (int i = 0; i < adjust; i++)
 268         {
 269           startNode2 = getParentOfNode(startNode2);
 270         }
 271       }
 272       else if (nParents1 > nParents2)
 273       {
 274         // adjust startNode1 to depth of startNode2
 275         int adjust = nParents1 - nParents2;
 276 
 277         for (int i = 0; i < adjust; i++)
 278         {
 279           startNode1 = getParentOfNode(startNode1);
 280         }
 281       }
 282 
 283       Node prevChild1 = null, prevChild2 = null;  // so we can "back up"
 284 
 285       // Loop up the ancestor chain looking for common parent
 286       while (null != startNode1)
 287       {
 288         if (startNode1 == startNode2 || isNodeTheSame(startNode1, startNode2))  // common parent?
 289         {
 290           if (null == prevChild1)  // first time in loop?
 291           {
 292 
 293             // Edge condition: one is the ancestor of the other.
 294             isNodeAfter = (nParents1 < nParents2) ? true : false;
 295 
 296             break;  // from while loop
 297           }
 298           else
 299           {
 300                         // Compare ancestors below lowest-common as siblings
 301             isNodeAfter = isNodeAfterSibling(startNode1, prevChild1,
 302                                              prevChild2);
 303 
 304             break;  // from while loop
 305           }
 306         }  // end if(startNode1 == startNode2)
 307 
 308                 // Move up one level and try again
 309         prevChild1 = startNode1;
 310         startNode1 = getParentOfNode(startNode1);
 311         prevChild2 = startNode2;
 312         startNode2 = getParentOfNode(startNode2);
 313       }  // end while(parents exist to examine)
 314     }  // end big else (not immediate siblings)
 315 
 316         // WARNING: The following diagnostic won't report the early
 317         // "same node" case. Fix if/when needed.
 318 
 319     /* -- please do not remove... very useful for diagnostics --
 320     System.out.println("node1 = "+node1.getNodeName()+"("+node1.getNodeType()+")"+
 321     ", node2 = "+node2.getNodeName()
 322     +"("+node2.getNodeType()+")"+
 323     ", isNodeAfter = "+isNodeAfter); */
 324     return isNodeAfter;
 325   }  // end isNodeAfter(Node node1, Node node2)

 342   /**
 343    * Figure out if child2 is after child1 in document order.
 344    * 
 345    * Warning: Some aspects of "document order" are not well defined.
 346    * For example, the order of attributes is considered
 347    * meaningless in XML, and the order reported by our model will
 348    * be consistant for a given invocation but may not
 349    * match that of either the source file or the serialized output.
 350    *
 351    * @param parent Must be the parent of both child1 and child2.
 352    * @param child1 Must be the child of parent and not equal to child2.
 353    * @param child2 Must be the child of parent and not equal to child1.
 354    * @return true if child 2 is after child1 in document order.
 355    */
 356   private static boolean isNodeAfterSibling(Node parent, Node child1,
 357                                             Node child2)
 358   {
 359 
 360     boolean isNodeAfterSibling = false;
 361     short child1type = child1.getNodeType();
 362     short child2type = child2.getNodeType();
 363 
 364     if ((Node.ATTRIBUTE_NODE != child1type)
 365             && (Node.ATTRIBUTE_NODE == child2type))
 366     {
 367 
 368       // always sort attributes before non-attributes.
 369       isNodeAfterSibling = false;
 370     }
 371     else if ((Node.ATTRIBUTE_NODE == child1type)
 372              && (Node.ATTRIBUTE_NODE != child2type))
 373     {
 374 
 375       // always sort attributes before non-attributes.
 376       isNodeAfterSibling = true;
 377     }
 378     else if (Node.ATTRIBUTE_NODE == child1type)
 379     {
 380       NamedNodeMap children = parent.getAttributes();
 381       int nNodes = children.getLength();
 382       boolean found1 = false, found2 = false;
 383 
 384           // Count from the start until we find one or the other.
 385       for (int i = 0; i < nNodes; i++)
 386       {
 387         Node child = children.item(i);
 388 
 389         if (child1 == child || isNodeTheSame(child1, child))
 390         {
 391           if (found2)
 392           {
 393             isNodeAfterSibling = false;
 394 
 395             break;
 396           }
 397 
 398           found1 = true;
 399         }
 400         else if (child2 == child || isNodeTheSame(child2, child))
 401         {
 402           if (found1)
 403           {
 404             isNodeAfterSibling = true;
 405 
 406             break;
 407           }
 408 
 409           found2 = true;
 410         }
 411       }
 412     }
 413     else
 414     {
 415                 // TODO: Check performance of alternate solution:
 416                 // There are two choices here: Count from the start of
 417                 // the document until we find one or the other, or count
 418                 // from one until we find or fail to find the other.
 419                 // Either can wind up scanning all the siblings in the worst
 420                 // case, which on a wide document can be a lot of work but
 421                 // is more typically is a short list.
 422                 // Scanning from the start involves two tests per iteration,
 423                 // but it isn't clear that scanning from the middle doesn't
 424                 // yield more iterations on average.
 425                 // We should run some testcases.
 426       Node child = parent.getFirstChild();
 427       boolean found1 = false, found2 = false;
 428 
 429       while (null != child)
 430       {
 431 
 432         // Node child = children.item(i);
 433         if (child1 == child || isNodeTheSame(child1, child))
 434         {
 435           if (found2)
 436           {
 437             isNodeAfterSibling = false;
 438 
 439             break;
 440           }
 441 
 442           found1 = true;
 443         }
 444         else if (child2 == child || isNodeTheSame(child2, child))
 445         {
 446           if (found1)
 447           {
 448             isNodeAfterSibling = true;
 449 
 450             break;
 451           }
 452 
 453           found2 = true;
 454         }
 455 
 456         child = child.getNextSibling();
 457       }
 458     }
 459 
 460     return isNodeAfterSibling;
 461   }  // end isNodeAfterSibling(Node parent, Node child1, Node child2)

 488   /**
 489    * Given an XML Namespace prefix and a context in which the prefix
 490    * is to be evaluated, return the Namespace Name this prefix was
 491    * bound to. Note that DOM Level 3 is expected to provide a version of
 492    * this which deals with the DOM's "early binding" behavior.
 493    *
 494    * Default handling:
 495    *
 496    * @param prefix String containing namespace prefix to be resolved,
 497    * without the ':' which separates it from the localname when used
 498    * in a Node Name. The empty sting signifies the default namespace
 499    * at this point in the document.
 500    * @param namespaceContext Element which provides context for resolution.
 501    * (We could extend this to work for other nodes by first seeking their
 502    * nearest Element ancestor.)
 503    *
 504    * @return a String containing the Namespace URI which this prefix
 505    * represents in the specified context.
 506    */
 507   public String getNamespaceForPrefix(String prefix, Element namespaceContext)
 508   {
 509 
 510     int type;
 511     Node parent = namespaceContext;
 512     String namespace = null;
 513 
 514     if (prefix.equals("xml"))
 515     {
 516       namespace = QName.S_XMLNAMESPACEURI; // Hardcoded, per Namespace spec
 517     }
 518         else if(prefix.equals("xmlns"))
 519     {
 520           // Hardcoded in the DOM spec, expected to be adopted by
 521           // Namespace spec. NOTE: Namespace declarations _must_ use
 522           // the xmlns: prefix; other prefixes declared as belonging
 523           // to this namespace will not be recognized and should
 524           // probably be rejected by parsers as erroneous declarations.
 525       namespace = "http://www.w3.org/2000/xmlns/";
 526     }
 527     else
 528     {
 529           // Attribute name for this prefix's declaration
 530           String declname=(prefix=="")
 531                         ? "xmlns"
 532                         : "xmlns:"+prefix;
 533 
 534           // Scan until we run out of Elements or have resolved the namespace
 535       while ((null != parent) && (null == namespace)
 536              && (((type = parent.getNodeType()) == Node.ELEMENT_NODE)
 537                  || (type == Node.ENTITY_REFERENCE_NODE)))
 538       {
 539         if (type == Node.ELEMENT_NODE)
 540         {
 541 
 542                         // Look for the appropriate Namespace Declaration attribute,
 543                         // either "xmlns:prefix" or (if prefix is "") "xmlns".
 544                         // TODO: This does not handle "implicit declarations"
 545                         // which may be created when the DOM is edited. DOM Level
 546                         // 3 will define how those should be interpreted. But
 547                         // this issue won't arise in freshly-parsed DOMs.
 548 
 549                 // NOTE: declname is set earlier, outside the loop.
 550                         Attr attr=((Element)parent).getAttributeNode(declname);
 551                         if(attr!=null)
 552                         {
 553                 namespace = attr.getNodeValue();
 554                 break;
 555                         }
 556                 }
 557 
 558         parent = getParentOfNode(parent);
 559       }
 560     }
 561 
 562     return namespace;
 563   }

 604   /**
 605    * Returns the namespace of the given node. Differs from simply getting
 606    * the node's prefix and using getNamespaceForPrefix in that it attempts
 607    * to cache some of the data in NSINFO objects, to avoid repeated lookup.
 608    * TODO: Should we consider moving that logic into getNamespaceForPrefix?
 609    *
 610    * @param n Node to be examined.
 611    *
 612    * @return String containing the Namespace Name (uri) for this node.
 613    * Note that this is undefined for any nodes other than Elements and
 614    * Attributes.
 615    */
 616   public String getNamespaceOfNode(Node n)
 617   {
 618 
 619     String namespaceOfPrefix;
 620     boolean hasProcessedNS;
 621     NSInfo nsInfo;
 622     short ntype = n.getNodeType();
 623 
 624     if (Node.ATTRIBUTE_NODE != ntype)
 625     {
 626       Object nsObj = m_NSInfos.get(n);  // return value
 627 
 628       nsInfo = (nsObj == null) ? null : (NSInfo) nsObj;
 629       hasProcessedNS = (nsInfo == null) ? false : nsInfo.m_hasProcessedNS;
 630     }
 631     else
 632     {
 633       hasProcessedNS = false;
 634       nsInfo = null;
 635     }
 636 
 637     if (hasProcessedNS)
 638     {
 639       namespaceOfPrefix = nsInfo.m_namespace;
 640     }
 641     else
 642     {
 643       namespaceOfPrefix = null;
 644 
 645       String nodeName = n.getNodeName();
 646       int indexOfNSSep = nodeName.indexOf(':');
 647       String prefix;
 648 
 649       if (Node.ATTRIBUTE_NODE == ntype)
 650       {
 651         if (indexOfNSSep > 0)
 652         {
 653           prefix = nodeName.substring(0, indexOfNSSep);
 654         }
 655         else
 656         {
 657 
 658           // Attributes don't use the default namespace, so if
 659           // there isn't a prefix, we're done.
 660           return namespaceOfPrefix;
 661         }
 662       }
 663       else
 664       {
 665         prefix = (indexOfNSSep >= 0)
 666                  ? nodeName.substring(0, indexOfNSSep) : "";
 667       }
 668 
 669       boolean ancestorsHaveXMLNS = false;
 670       boolean nHasXMLNS = false;
 671 
 672       if (prefix.equals("xml"))
 673       {
 674         namespaceOfPrefix = QName.S_XMLNAMESPACEURI;
 675       }
 676       else
 677       {
 678         int parentType;
 679         Node parent = n;
 680 
 681         while ((null != parent) && (null == namespaceOfPrefix))
 682         {
 683           if ((null != nsInfo)
 684                   && (nsInfo.m_ancestorHasXMLNSAttrs
 685                       == NSInfo.ANCESTORNOXMLNS))
 686           {
 687             break;
 688           }
 689 
 690           parentType = parent.getNodeType();
 691 
 692           if ((null == nsInfo) || nsInfo.m_hasXMLNSAttrs)
 693           {
 694             boolean elementHasXMLNS = false;
 695 
 696             if (parentType == Node.ELEMENT_NODE)
 697             {
 698               NamedNodeMap nnm = parent.getAttributes();
 699 
 700               for (int i = 0; i < nnm.getLength(); i++)
 701               {
 702                 Node attr = nnm.item(i);
 703                 String aname = attr.getNodeName();
 704 
 705                 if (aname.charAt(0) == 'x')
 706                 {
 707                   boolean isPrefix = aname.startsWith("xmlns:");
 708 
 709                   if (aname.equals("xmlns") || isPrefix)
 710                   {
 711                     if (n == parent)
 712                       nHasXMLNS = true;
 713 
 714                     elementHasXMLNS = true;
 715                     ancestorsHaveXMLNS = true;
 716 
 717                     String p = isPrefix ? aname.substring(6) : "";
 718 
 719                     if (p.equals(prefix))
 720                     {
 721                       namespaceOfPrefix = attr.getNodeValue();
 722 
 723                       break;
 724                     }
 725                   }
 726                 }
 727               }
 728             }
 729 
 730             if ((Node.ATTRIBUTE_NODE != parentType) && (null == nsInfo)
 731                     && (n != parent))
 732             {
 733               nsInfo = elementHasXMLNS
 734                        ? m_NSInfoUnProcWithXMLNS : m_NSInfoUnProcWithoutXMLNS;
 735 
 736               m_NSInfos.put(parent, nsInfo);
 737             }
 738           }
 739 
 740           if (Node.ATTRIBUTE_NODE == parentType)
 741           {
 742             parent = getParentOfNode(parent);
 743           }
 744           else
 745           {
 746             m_candidateNoAncestorXMLNS.addElement(parent);
 747             m_candidateNoAncestorXMLNS.addElement(nsInfo);
 748 
 749             parent = parent.getParentNode();
 750           }
 751 
 752           if (null != parent)
 753           {
 754             Object nsObj = m_NSInfos.get(parent);  // return value
 755 
 756             nsInfo = (nsObj == null) ? null : (NSInfo) nsObj;
 757           }
 758         }
 759 
 760         int nCandidates = m_candidateNoAncestorXMLNS.size();
 761 
 762         if (nCandidates > 0)
 763         {
 764           if ((false == ancestorsHaveXMLNS) && (null == parent))
 765           {
 766             for (int i = 0; i < nCandidates; i += 2)
 767             {
 768               Object candidateInfo = m_candidateNoAncestorXMLNS.elementAt(i
 769                                        + 1);
 770 
 771               if (candidateInfo == m_NSInfoUnProcWithoutXMLNS)
 772               {
 773                 m_NSInfos.put(m_candidateNoAncestorXMLNS.elementAt(i),
 774                               m_NSInfoUnProcNoAncestorXMLNS);
 775               }
 776               else if (candidateInfo == m_NSInfoNullWithoutXMLNS)
 777               {
 778                 m_NSInfos.put(m_candidateNoAncestorXMLNS.elementAt(i),
 779                               m_NSInfoNullNoAncestorXMLNS);
 780               }
 781             }
 782           }
 783 
 784           m_candidateNoAncestorXMLNS.removeAllElements();
 785         }
 786       }
 787 
 788       if (Node.ATTRIBUTE_NODE != ntype)
 789       {
 790         if (null == namespaceOfPrefix)
 791         {
 792           if (ancestorsHaveXMLNS)
 793           {
 794             if (nHasXMLNS)
 795               m_NSInfos.put(n, m_NSInfoNullWithXMLNS);
 796             else
 797               m_NSInfos.put(n, m_NSInfoNullWithoutXMLNS);
 798           }
 799           else
 800           {
 801             m_NSInfos.put(n, m_NSInfoNullNoAncestorXMLNS);
 802           }
 803         }
 804         else
 805         {
 806           m_NSInfos.put(n, new NSInfo(namespaceOfPrefix, nHasXMLNS));
 807         }
 808       }
 809     }
 810 
 811     return namespaceOfPrefix;
 812   }

 979   /**
 980    * Obtain the XPath-model parent of a DOM node -- ownerElement for Attrs,
 981    * parent for other nodes.
 982    * 
 983    * Background: The DOM believes that you must be your Parent's
 984    * Child, and thus Attrs don't have parents. XPath said that Attrs
 985    * do have their owning Element as their parent. This function
 986    * bridges the difference, either by using the DOM Level 2 ownerElement
 987    * function or by using a "silly and expensive function" in Level 1
 988    * DOMs.
 989    * 
 990    * (There's some discussion of future DOMs generalizing ownerElement
 991    * into ownerNode and making it work on all types of nodes. This
 992    * still wouldn't help the users of Level 1 or Level 2 DOMs)
 993    * 
 994    *
 995    * @param node Node whose XPath parent we want to obtain
 996    *
 997    * @return the parent of the node, or the ownerElement if it's an
 998    * Attr node, or null if the node is an orphan.
 999    *
1000    * @throws RuntimeException if the Document has no root element.
1001    * This can't arise if the Document was created
1002    * via the DOM Level 2 factory methods, but is possible if other
1003    * mechanisms were used to obtain it
1004    */

1006   {
1007     Node parent;
1008     short nodeType = node.getNodeType();
1009 
1010     if (Node.ATTRIBUTE_NODE == nodeType)
1011     {
1012       Document doc = node.getOwnerDocument();
1013           /*
1014       TBD:
1015       if(null == doc)
1016       {
1017         throw new RuntimeException(XSLMessages.createXPATHMessage(XPATHErrorResources.ER_CHILD_HAS_NO_OWNER_DOCUMENT, null));//"Attribute child does not have an owner document!");
1018       }
1019       */
1020 
1021           // Given how expensive the tree walk may be, we should first ask
1022           // whether this DOM can answer the question for us. The additional
1023           // test does slow down Level 1 DOMs slightly. DOMHelper2, which
1024           // is currently specialized for Xerces, assumes it can use the
1025           // Level 2 solution. We might want to have an intermediate stage,
1026           // which would assume DOM Level 2 but not assume Xerces.
1027           //
1028           // (Shouldn't have to check whether impl is null in a compliant DOM,

1033                   parent=((Attr)node).getOwnerElement();

1044           XMLMessages.createXMLMessage(

1050 
1051         }
1052     else
1053     {
1054       parent = node.getParentNode();
1055 
1056       // if((Node.DOCUMENT_NODE != nodeType) && (null == parent))
1057       // {
1058       //   throw new RuntimeException("Child does not have parent!");
1059       // }
1060     }
1061 
1062     return parent;
1063   }

1088   /**
1089    * The getUnparsedEntityURI function returns the URI of the unparsed
1090    * entity with the specified name in the same document as the context
1091    * node (see [3.3 Unparsed Entities]). It returns the empty string if
1092    * there is no such entity.
1093    * 
1094    * XML processors may choose to use the System Identifier (if one
1095    * is provided) to resolve the entity, rather than the URI in the
1096    * Public Identifier. The details are dependent on the processor, and
1097    * we would have to support some form of plug-in resolver to handle
1098    * this properly. Currently, we simply return the System Identifier if
1099    * present, and hope that it a usable URI or that our caller can
1100    * map it to one.
1101    * TODO: Resolve Public Identifiers... or consider changing function name.
1102    * 
1103    * If we find a relative URI
1104    * reference, XML expects it to be resolved in terms of the base URI
1105    * of the document. The DOM doesn't do that for us, and it isn't
1106    * entirely clear whether that should be done here; currently that's
1107    * pushed up to a higher levelof our application. (Note that DOM Level
1108    * 1 didn't store the document's base URI.)
1109    * TODO: Consider resolving Relative URIs.
1110    * 
1111    * (The DOM's statement that "An XML processor may choose to
1112    * completely expand entities before the structure model is passed
1113    * to the DOM" refers only to parsed entities, not unparsed, and hence
1114    * doesn't affect this function.)
1115    *
1116    * @param name A string containing the Entity Name of the unparsed
1117    * entity.
1118    * @param doc Document node for the document to be searched.
1119    *
1120    * @return String containing the URI of the Unparsed Entity, or an
1121    * empty string if no such entity exists.
1122    */
1123   public String getUnparsedEntityURI(String name, Document doc)
1124   {
1125 
1126     String url = "";
1127     DocumentType doctype = doc.getDoctype();
1128 
1129     if (null != doctype)
1130     {
1131       NamedNodeMap entities = doctype.getEntities();
1132       if(null == entities)
1133         return url;
1134       Entity entity = (Entity) entities.getNamedItem(name);
1135       if(null == entity)
1136         return url;
1137 
1138       String notationName = entity.getNotationName();
1139 
1140       if (null != notationName)  // then it's unparsed
1141       {
1142         // The draft says: "The XSLT processor may use the public
1143         // identifier to generate a URI for the entity instead of the URI
1144         // specified in the system identifier. If the XSLT processor does
1145         // not use the public identifier to generate the URI, it must use
1146         // the system identifier; if the system identifier is a relative
1147         // URI, it must be resolved into an absolute URI using the URI of
1148         // the resource containing the entity declaration as the base
1149         // URI [RFC2396]."
1150         // So I'm falling a bit short here.
1151         url = entity.getSystemId();
1152 
1153         if (null == url)
1154         {
1155           url = entity.getPublicId();
1156         }
1157         else
1158         {
1159           // This should be resolved to an absolute URL, but that's hard
1160           // to do from here.
1161         }
1162       }
1163     }
1164 
1165     return url;
1166   }

1168   /**
1169    * Support for getParentOfNode; walks a DOM tree until it finds
1170    * the Element which owns the Attr. This is hugely expensive, and
1171    * if at all possible you should use the DOM Level 2 Attr.ownerElement()
1172    * method instead.
1173    *  
1174    * The DOM Level 1 developers expected that folks would keep track
1175    * of the last Element they'd seen and could recover the info from
1176    * that source. Obviously that doesn't work very well if the only
1177    * information you've been presented with is the Attr. The DOM Level 2
1178    * getOwnerElement() method fixes that, but only for Level 2 and
1179    * later DOMs.
1180    *
1181    * @param elem Element whose subtree is to be searched for this Attr
1182    * @param attr Attr whose owner is to be located.
1183    *
1184    * @return the first Element whose attribute list includes the provided
1185    * attr. In modern DOMs, this will also be the only such Element. (Early
1186    * DOMs had some hope that Attrs might be sharable, but this idea has
1187    * been abandoned.)
1188    */
1189   private static Node locateAttrParent(Element elem, Node attr)
1190   {
1191 
1192     Node parent = null;
1193 
1194         // This should only be called for Level 1 DOMs, so we don't have to
1195         // worry about namespace issues. In later levels, it's possible
1196         // for a DOM to have two Attrs with the same NodeName but
1197         // different namespaces, and we'd need to get getAttributeNodeNS...
1198         // but later levels also have Attr.getOwnerElement.
1199         Attr check=elem.getAttributeNode(attr.getNodeName());
1200         if(check==attr)
1201                 parent = elem;
1202 
1203     if (null == parent)
1204     {
1205       for (Node node = elem.getFirstChild(); null != node;
1206               node = node.getNextSibling())
1207       {
1208         if (Node.ELEMENT_NODE == node.getNodeType())
1209         {
1210           parent = locateAttrParent((Element) node, attr);
1211 
1212           if (null != parent)
1213             break;
1214         }
1215       }
1216     }
1217 
1218     return parent;
1219   }

1288   /**
1289    * Retrieve the text content of a DOM subtree, appending it into a
1290    * user-supplied FastStringBuffer object. Note that attributes are
1291    * not considered part of the content of an element.
1292    * 
1293    * There are open questions regarding whitespace stripping.
1294    * Currently we make no special effort in that regard, since the standard
1295    * DOM doesn't yet provide DTD-based information to distinguish
1296    * whitespace-in-element-context from genuine #PCDATA. Note that we
1297    * should probably also consider xml:space if/when we address this.
1298    * DOM Level 3 may solve the problem for us.
1299    *
1300    * @param node Node whose subtree is to be walked, gathering the
1301    * contents of all Text or CDATASection nodes.
1302    * @param buf FastStringBuffer into which the contents of the text
1303    * nodes are to be concatenated.
1304    */
1305   public static void getNodeData(Node node, FastStringBuffer buf)
1306   {
1307 
1308     switch (node.getNodeType())
1309     {
1310     case Node.DOCUMENT_FRAGMENT_NODE :
1311     case Node.DOCUMENT_NODE :
1312     case Node.ELEMENT_NODE :
1313     {
1314       for (Node child = node.getFirstChild(); null != child;
1315               child = child.getNextSibling())
1316       {
1317         getNodeData(child, buf);
1318       }
1319     }
1320     break;
1321     case Node.TEXT_NODE :
1322     case Node.CDATA_SECTION_NODE :
1323       buf.append(node.getNodeValue());
1324       break;
1325     case Node.ATTRIBUTE_NODE :
1326       buf.append(node.getNodeValue());
1327       break;
1328     case Node.PROCESSING_INSTRUCTION_NODE :
1329       // warning(XPATHErrorResources.WG_PARSING_AND_PREPARING);
1330       break;
