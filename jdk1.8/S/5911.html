  31 /**
  32  * DTMNamedNodeMap is a quickie (as opposed to quick) implementation of the DOM's
  33  * NamedNodeMap interface, intended to support DTMProxy's getAttributes()
  34  * call.
  35  * <p>
  36  * ***** Note: this does _not_ current attempt to cache any of the data;
  37  * if you ask for attribute 27 and then 28, you'll have to rescan the first
  38  * 27. It should probably at least keep track of the last one retrieved,
  39  * and possibly buffer the whole array.
  40  * <p>
  41  * ***** Also note that there's no fastpath for the by-name query; we search
  42  * linearly until we find it or fail to find it. Again, that could be
  43  * optimized at some cost in object creation/storage.
  44  * @xsl.usage internal
  45  */

  70   /**
  71    * Return the number of Attributes on this Element
  72    *
  73    * @return The number of nodes in this map.
  74    */
  75   public int getLength()
  76   {
  77 
  78     if (m_count == -1)
  79     {
  80       short count = 0;
  81 
  82       for (int n = dtm.getFirstAttribute(element); n != -1;
  83               n = dtm.getNextAttribute(n))
  84       {
  85         ++count;
  86       }
  87 
  88       m_count = count;
  89     }
  90 
  91     return (int) m_count;
  92   }

  94   /**
  95    * Retrieves a node specified by name.
  96    * @param name The <code>nodeName</code> of a node to retrieve.
  97    * @return A <code>Node</code> (of any type) with the specified
  98    *   <code>nodeName</code>, or <code>null</code> if it does not identify
  99    *   any node in this map.
 100    */
 101   public Node getNamedItem(String name)
 102   {
 103 
 104     for (int n = dtm.getFirstAttribute(element); n != DTM.NULL;
 105             n = dtm.getNextAttribute(n))
 106     {
 107       if (dtm.getNodeName(n).equals(name))
 108         return dtm.getNode(n);
 109     }
 110 
 111     return null;
 112   }
 113 
 114   /**
 115    * Returns the <code>index</code>th item in the map. If <code>index</code>
 116    * is greater than or equal to the number of nodes in this map, this
 117    * returns <code>null</code>.
 118    * @param i The index of the requested item.
 119    * @return The node at the <code>index</code>th position in the map, or
 120    *   <code>null</code> if that is not a valid index.
 121    */
 122   public Node item(int i)
 123   {
 124 
 125     int count = 0;
 126 
 127     for (int n = dtm.getFirstAttribute(element); n != -1;
 128             n = dtm.getNextAttribute(n))
 129     {
 130       if (count == i)
 131         return dtm.getNode(n);
 132       else
 133         ++count;
 134     }
 135 
 136     return null;
 137   }

 188   /**
 189    * Retrieves a node specified by local name and namespace URI. HTML-only
 190    * DOM implementations do not need to implement this method.
 191    * @param namespaceURI The namespace URI of the node to retrieve.
 192    * @param localName The local name of the node to retrieve.
 193    *
 194    * @return A <code>Node</code> (of any type) with the specified local
 195    *   name and namespace URI, or <code>null</code> if they do not
 196    *   identify any node in this map.
 197    * @since DOM Level 2
 198    */
 199   public Node getNamedItemNS(String namespaceURI, String localName)
 200   {
 201        Node retNode = null;
 202        for (int n = dtm.getFirstAttribute(element); n != DTM.NULL;
 203                        n = dtm.getNextAttribute(n))
 204        {
 205          if (localName.equals(dtm.getLocalName(n)))
 206          {
 207            String nsURI = dtm.getNamespaceURI(n);
 208            if ((namespaceURI == null && nsURI == null)
 209                   || (namespaceURI != null && namespaceURI.equals(nsURI)))
 210            {
 211              retNode = dtm.getNode(n);
 212              break;
 213            }
 214          }
 215        }
 216        return retNode;
 217   }
