  66 /**
  67  * Abstract super class for instructions that use an index into the
  68  * constant pool such as LDC, INVOKEVIRTUAL, etc.
  69  *
  70  * @see ConstantPoolGen
  71  * @see LDC
  72  * @see INVOKEVIRTUAL
  73  *
  74  * @author  <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
  75  */
  76 public abstract class CPInstruction extends Instruction
  77   implements TypedInstruction, IndexedInstruction
  78 {
  79   protected int index; // index to constant pool
  80 
  81   /**
  82    * Empty constructor needed for the Class.newInstance() statement in
  83    * Instruction.readInstruction(). Not to be used otherwise.
  84    */
  85   CPInstruction() {}
  86 
  87   /**
  88    * @param index to constant pool
  89    */
  90   protected CPInstruction(short opcode, int index) {
  91     super(opcode, (short)3);
  92     setIndex(index);
  93   }
  94 
  95   /**
  96    * Dump instruction as byte code to stream out.
  97    * @param out Output stream
  98    */
  99   public void dump(DataOutputStream out) throws IOException {
 100     out.writeByte(opcode);
 101     out.writeShort(index);
 102   }
 103 
 104   /**
 105    * Long output format:
 106    *
 107    * &lt;name of opcode&gt; "["&lt;opcode number&gt;"]"
 108    * "("&lt;length of instruction&gt;")" "&lt;"&lt; constant pool index&gt;"&gt;"
 109    *
 110    * @param verbose long/short format switch
 111    * @return mnemonic for instruction
 112    */
 113   public String toString(boolean verbose) {
 114     return super.toString(verbose) + " " + index;
 115   }
 116 
 117   /**
 118    * @return mnemonic for instruction with symbolic references resolved
 119    */
 120   public String toString(ConstantPool cp) {
 121     Constant c   = cp.getConstant(index);
 122     String   str = cp.constantToString(c);
 123 
 124     if(c instanceof ConstantClass)
 125       str = str.replace('.', '/');
 126 
 127     return com.sun.org.apache.bcel.internal.Constants.OPCODE_NAMES[opcode] + " " + str;
 128   }
 129 
 130   /**
 131    * Read needed data (i.e., index) from file.
 132    * @param bytes input stream
 133    * @param wide wide prefix?
 134    */
 135   protected void initFromFile(ByteSequence bytes, boolean wide)
 136        throws IOException
 137   {
 138     setIndex(bytes.readUnsignedShort());
 139     length = 3;
 140   }
 141 
 142   /**
 143    * @return index in constant pool referred by this instruction.
 144    */
 145   public final int getIndex() { return index; }
 146 
 147   /**
 148    * Set the index to constant pool.
 149    * @param index in  constant pool.
 150    */
 151   public void setIndex(int index) {
 152     if(index < 0)
 153       throw new ClassGenException("Negative index value: " + index);
 154 
 155     this.index = index;
 156   }
 157 
 158   /** @return type related with this instruction.
 159    */
 160   public Type getType(ConstantPoolGen cpg) {
 161     ConstantPool cp   = cpg.getConstantPool();
 162     String       name = cp.getConstantString(index, com.sun.org.apache.bcel.internal.Constants.CONSTANT_Class);
 163 
 164     if(!name.startsWith("["))
 165       name = "L" + name + ";";
 166 
 167     return Type.getType(name);
 168   }
 169 }
