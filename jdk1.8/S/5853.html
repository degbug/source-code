  41  public static Locale langToLocale(String lang) {
  42        if((lang == null) || lang.equals(EMPTY_STRING)){ // not specified => getDefault
  43             return Locale.getDefault();
  44        }
  45         String language = EMPTY_STRING;
  46         String country =  EMPTY_STRING;
  47         String variant =  EMPTY_STRING;
  48 
  49         int i1 = lang.indexOf(IETF_SEPARATOR);
  50         if (i1 < 0) {
  51             language = lang;
  52         } else {
  53             language = lang.substring(0, i1);
  54             ++i1;
  55             int i2 = lang.indexOf(IETF_SEPARATOR, i1);
  56             if (i2 < 0) {
  57                 country = lang.substring(i1);
  58             } else {
  59                 country = lang.substring(i1, i2);
  60                 variant = lang.substring(i2+1);
  61             }
  62         }
  63 
  64         if(language.length() == 2){
  65            language = language.toLowerCase();
  66         }else {
  67           language = EMPTY_STRING;
  68         }
  69 
  70         if(country.length() == 2){
  71            country = country.toUpperCase();
  72         }else {
  73           country = EMPTY_STRING;
  74         }
  75 
  76         if((variant.length() > 0) &&
  77         ((language.length() == 2) ||(country.length() == 2))){
  78            variant = variant.toUpperCase();
  79         }else{
  80             variant = EMPTY_STRING;
  81         }
  82 
  83         return new Locale(language, country, variant );
  84     }
