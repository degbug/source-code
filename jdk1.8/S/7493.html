  40 /** Base template for creating TaggedProfiles.  A TaggedProfile will often contain
  41 * tagged components.  A template that does not contain components acts like
  42 * an empty immutable list.
  43 *
  44 * @author Ken Cavanaugh
  45 */
  46 public interface TaggedProfileTemplate extends List, Identifiable,
  47     WriteContents, MakeImmutable
  48 {
  49     /** Return an iterator that iterates over tagged components with
  50     * identifier id.  It is not possible to modify the list through this
  51     * iterator.
  52     */
  53     public Iterator iteratorById( int id ) ;
  54 
  55     /** Create a TaggedProfile from this template.
  56     */
  57     TaggedProfile create( ObjectKeyTemplate oktemp, ObjectId id ) ;
  58 
  59     /** Write the profile create( oktemp, id ) to the OutputStream os.
  60     */
  61     void write( ObjectKeyTemplate oktemp, ObjectId id, OutputStream os) ;
  62 
  63     /** Return true if temp is equivalent to this template.  Equivalence
  64      * means that in some sense an invocation on a profile created by this
  65      * template has the same results as an invocation on a profile
  66      * created from temp.  Equivalence may be weaker than equality.
  67      */
  68     boolean isEquivalent( TaggedProfileTemplate temp );
  69 
  70     /** Return the tagged components in this profile (if any)
  71      * in the GIOP marshalled form, which is required for Portable
  72      * Interceptors.  Returns null if either the profile has no
  73      * components, or if this type of profile can never contain
  74      * components.
  75      */
  76     org.omg.IOP.TaggedComponent[] getIOPComponents(
  77         ORB orb, int id );
  78 }
