  32 /**
  33  * A SpinnerModel for sequences of Dates.
  34  * The upper and lower bounds of the sequence are defined by properties called
  35  * start and end and the size
  36  * of the increase or decrease computed by the nextValue
  37  * and previousValue methods is defined by a property
  38  * called calendarField.  The start
  39  * and end properties can be null to
  40  * indicate that the sequence has no lower or upper limit.
  41  * 
  42  * The value of the calendarField property must be one of the
  43  * java.util.Calendar constants that specify a field
  44  * within a Calendar.  The getNextValue
  45  * and getPreviousValue
  46  * methods change the date forward or backwards by this amount.
  47  * For example, if calendarField is Calendar.DAY_OF_WEEK,
  48  * then nextValue produces a Date that's 24
  49  * hours after the current value, and previousValue
  50  * produces a Date that's 24 hours earlier.
  51  * 
  52  * The legal values for calendarField are:

  54  *   
  • Calendar.ERA

  • Calendar.YEAR

  • Calendar.MONTH

    • 
  58  *   
  • Calendar.WEEK_OF_MONTH
    • 
  59  *   
  • Calendar.DAY_OF_MONTH
    • 
  60  *   
  • Calendar.DAY_OF_YEAR
    • 
  61  *   
  • Calendar.DAY_OF_WEEK
    • 
  62  *   
  • Calendar.DAY_OF_WEEK_IN_MONTH
    • 
  63  *   
  • Calendar.AM_PM
    • 
  64  *   
  • Calendar.HOUR
    • 
  65  *   
  • Calendar.HOUR_OF_DAY

  66  *   
  • Calendar.MINUTE

  70  * However some UIs may set the calendarField before committing the edit

 120     /**
 121      * Creates a SpinnerDateModel that represents a sequence of dates
 122      * between start and end.  The
 123      * nextValue and previousValue methods
 124      * compute elements of the sequence by advancing or reversing
 125      * the current date value by the
 126      * calendarField time unit.  For a precise description
 127      * of what it means to increment or decrement a Calendar
 128      * field, see the add method in
 129      * java.util.Calendar.
 130      * 
 131      * The start and end parameters can be
 132      * null to indicate that the range doesn't have an
 133      * upper or lower bound.  If value or
 134      * calendarField is null, or if both
 135      * start and end are specified and
 136      * minimum > maximum then an
 137      * IllegalArgumentException is thrown.
 138      * Similarly if (minimum <= value <= maximum) is false,
 139      * an IllegalArgumentException is thrown.
 140      *
 141      * @param value the current (non null) value of the model
 142      * @param start the first date in the sequence or null
 143      * @param end the last date in the sequence or null
 144      * @param calendarField one of
 145      *   
      
 146      *    
    • Calendar.ERA
      • 
 147      *    
    • Calendar.YEAR
      • 
 148      *    
    • Calendar.MONTH
      • 
 149      *    
    • Calendar.WEEK_OF_YEAR
      • 
 150      *    
    • Calendar.WEEK_OF_MONTH
      • 
 151      *    
    • Calendar.DAY_OF_MONTH
      • 
 152      *    
    • Calendar.DAY_OF_YEAR
      • 
 153      *    
    • Calendar.DAY_OF_WEEK
      • 
 154      *    
    • Calendar.DAY_OF_WEEK_IN_MONTH
      • 
 155      *    
    • Calendar.AM_PM
      • 
 156      *    
    • Calendar.HOUR
      • 
 157      *    
    • Calendar.HOUR_OF_DAY
      • 
 158      *    
    • Calendar.MINUTE
      • 
 159      *    
    • Calendar.SECOND
      • 
 160      *    
    • Calendar.MILLISECOND
      • 
 161      *   
 162      *
 163      * @throws IllegalArgumentException if value or
 164      *    calendarField are null,
 165      *    if calendarField isn't valid,
 166      *    or if the following expression is
 167      *    false: (start <= value <= end).
 168      *
 169      * @see Calendar#add
 170      * @see #setValue
 171      * @see #setStart
 172      * @see #setEnd
 173      * @see #setCalendarField
 174      */
 175     public SpinnerDateModel(Date value, Comparable start, Comparable end, int calendarField) {
 176         if (value == null) {
 177             throw new IllegalArgumentException("value is null");
 178         }
 179         if (!calendarFieldOK(calendarField)) {
 180             throw new IllegalArgumentException("invalid calendarField");
 181         }
 182         if (!(((start == null) || (start.compareTo(value) <= 0)) &&
 183               ((end == null) || (end.compareTo(value) >= 0)))) {
 184             throw new IllegalArgumentException("(start <= value <= end) is false");
 185         }
 186         this.value = Calendar.getInstance();
 187         this.start = start;
 188         this.end = end;
 189         this.calendarField = calendarField;
 190 
 191         this.value.setTime(value);
 192     }

 206     /**
 207      * Changes the lower limit for Dates in this sequence.
 208      * If start is null,
 209      * then there is no lower limit.  No bounds checking is done here:
 210      * the new start value may invalidate the
 211      * (start <= value <= end)
 212      * invariant enforced by the constructors.  This is to simplify updating
 213      * the model.  Naturally one should ensure that the invariant is true
 214      * before calling the nextValue, previousValue,
 215      * or setValue methods.
 216      * 
 217      * Typically this property is a Date however it's possible to use
 218      * a Comparable with a compareTo method for Dates.
 219      * For example start might be an instance of a class like this:
 220      * 
      
 221      * MyStartDate implements Comparable {
 222      *     long t = 12345;
 223      *     public int compareTo(Date d) {
 224      *            return (t < d.getTime() ? -1 : (t == d.getTime() ? 0 : 1));
 225      *     }
 226      *     public int compareTo(Object o) {
 227      *            return compareTo((Date)o);
 228      *     }
 229      * }
 230      * 
 231      * Note that the above example will throw a ClassCastException
 232      * if the Object passed to compareTo(Object)
 233      * is not a Date.
 234      * 
 235      * This method fires a ChangeEvent if the
 236      * start has changed.
 237      *
 238      * @param start defines the first date in the sequence
 239      * @see #getStart
 240      * @see #setEnd
 241      * @see #addChangeListener
 242      */
 243     public void setStart(Comparable start) {
 244         if ((start == null) ? (this.start != null) : !start.equals(this.start)) {
 245             this.start = start;

 247         }

 368     /**
 369      * Returns the next Date in the sequence, or null if
 370      * the next date is after end.
 371      *
 372      * @return the next Date in the sequence, or null if
 373      *     the next date is after end.
 374      *
 375      * @see SpinnerModel#getNextValue
 376      * @see #getPreviousValue
 377      * @see #setCalendarField
 378      */
 379     public Object getNextValue() {
 380         Calendar cal = Calendar.getInstance();
 381         cal.setTime(value.getTime());
 382         cal.add(calendarField, 1);
 383         Date next = cal.getTime();
 384         return ((end == null) || (end.compareTo(next) >= 0)) ? next : null;
 385     }
