 108         void getXmlnsAttr(Collection<Attr> col) {
 109             int size = levels.size() - 1;
 110             if (cur == null) {
 111                 cur = new XmlsStackElement();
 112                 cur.level = currentLevel;
 113                 lastlevel = currentLevel;
 114                 levels.add(cur);
 115             }
 116             boolean parentRendered = false;
 117             XmlsStackElement e = null;
 118             if (size == -1) {
 119                 parentRendered = true;
 120             } else {
 121                 e = levels.get(size);
 122                 if (e.rendered && e.level + 1 == currentLevel) {
 123                     parentRendered = true;
 124                 }
 125             }
 126             if (parentRendered) {
 127                 col.addAll(cur.nodes);
 128                 cur.rendered = true;
 129                 return;
 130             }
 131 
 132             Map<String, Attr> loa = new HashMap<String, Attr>();
 133             List<Attr> baseAttrs = new ArrayList<Attr>();
 134             boolean successiveOmitted = true;
 135             for (; size >= 0; size--) {
 136                 e = levels.get(size);
 137                 if (e.rendered) {
 138                     successiveOmitted = false;
 139                 }
 140                 Iterator<Attr> it = e.nodes.iterator();
 141                 while (it.hasNext() && successiveOmitted) {
 142                     Attr n = it.next();
 143                     if (n.getLocalName().equals("base") && !e.rendered) {
 144                         baseAttrs.add(n);
 145                     } else if (!loa.containsKey(n.getName())) {
 146                         loa.put(n.getName(), n);
 147                     }
 148                 }
 149             }
 150             if (!baseAttrs.isEmpty()) {
 151                 Iterator<Attr> it = col.iterator();
 152                 String base = null;
 153                 Attr baseAttr = null;
 154                 while (it.hasNext()) {
 155                     Attr n = it.next();
 156                     if (n.getLocalName().equals("base")) {
 157                         base = n.getValue();
 158                         baseAttr = n;
 159                         break;
 160                     }
 161                 }
 162                 it = baseAttrs.iterator();
 163                 while (it.hasNext()) {
 164                     Attr n = it.next();
 165                     if (base == null) {
 166                         base = n.getValue();
 167                         baseAttr = n;
 168                     } else {
 169                         try {
 170                             base = joinURI(n.getValue(), base);
 171                         } catch (URISyntaxException ue) {
 172                             if (log.isLoggable(java.util.logging.Level.FINE)) {
 173                                 log.log(java.util.logging.Level.FINE, ue.getMessage(), ue);
 174                             }
 175                         }
 176                     }
 177                 }
 178                 if (base != null && base.length() != 0) {
 179                     baseAttr.setValue(base);
 180                     col.add(baseAttr);
 181                 }
 182             }
 183 
 184             cur.rendered = true;
 185             col.addAll(loa.values());
 186         }

 243     @Override
 244     protected Iterator<Attr> handleAttributesSubtree(Element element, NameSpaceSymbTable ns)
 245         throws CanonicalizationException {
 246         if (!element.hasAttributes() && !firstCall) {
 247             return null;
 248         }
 249         // result will contain the attrs which have to be output
 250         final SortedSet<Attr> result = this.result;
 251         result.clear();
 252 
 253         if (element.hasAttributes()) {
 254             NamedNodeMap attrs = element.getAttributes();
 255             int attrsLength = attrs.getLength();
 256 
 257             for (int i = 0; i < attrsLength; i++) {
 258                 Attr attribute = (Attr) attrs.item(i);
 259                 String NUri = attribute.getNamespaceURI();
 260                 String NName = attribute.getLocalName();
 261                 String NValue = attribute.getValue();
 262 
 263                 if (!XMLNS_URI.equals(NUri)) {
 264                     // It's not a namespace attr node. Add to the result and continue.
 265                     result.add(attribute);
 266                 } else if (!(XML.equals(NName) && XML_LANG_URI.equals(NValue))) {
 267                     // The default mapping for xml must not be output.
 268                     Node n = ns.addMappingAndRender(NName, NValue, attribute);
 269 
 270                     if (n != null) {
 271                         // Render the ns definition
 272                         result.add((Attr)n);
 273                         if (C14nHelper.namespaceIsRelative(attribute)) {
 274                             Object exArgs[] = {element.getTagName(), NName, attribute.getNodeValue()};
 275                             throw new CanonicalizationException(
 276                                 "c14n.Canonicalizer.RelativeNamespace", exArgs
 277                             );
 278                         }
 279                     }
 280                 }
 281             }
 282         }
 283 
 284         if (firstCall) {
 285             // It is the first node of the subtree
 286             // Obtain all the namespaces defined in the parents, and added to the output.
 287             ns.getUnrenderedNodes(result);
 288             // output the attributes in the xml namespace.
 289             xmlattrStack.getXmlnsAttr(result);
 290             firstCall = false;
 291         }
 292 
 293         return result.iterator();
 294     }

 415     protected void handleParent(Element e, NameSpaceSymbTable ns) {
 416         if (!e.hasAttributes() && e.getNamespaceURI() == null) {
 417             return;
 418         }
 419         xmlattrStack.push(-1);
 420         NamedNodeMap attrs = e.getAttributes();
 421         int attrsLength = attrs.getLength();
 422         for (int i = 0; i < attrsLength; i++) {
 423             Attr attribute = (Attr) attrs.item(i);
 424             String NName = attribute.getLocalName();
 425             String NValue = attribute.getNodeValue();
 426 
 427             if (Constants.NamespaceSpecNS.equals(attribute.getNamespaceURI())) {
 428                 if (!XML.equals(NName) || !Constants.XML_LANG_SPACE_SpecNS.equals(NValue)) {
 429                     ns.addMapping(NName, NValue, attribute);
 430                 }
 431             } else if (!"id".equals(NName) && XML_LANG_URI.equals(attribute.getNamespaceURI())) {
 432                 xmlattrStack.addXmlnsAttr(attribute);
 433             }
 434         }
 435         if (e.getNamespaceURI() != null) {
 436             String NName = e.getPrefix();
 437             String NValue = e.getNamespaceURI();
 438             String Name;
 439             if (NName == null || NName.equals("")) {
 440                 NName = "xmlns";

 442             } else {
 443                 Name = "xmlns:" + NName;
 444             }
 445             Attr n = e.getOwnerDocument().createAttributeNS("http://www.w3.org/2000/xmlns/", Name);
 446             n.setValue(NValue);
 447             ns.addMapping(NName, NValue, n);

 451     private static String joinURI(String baseURI, String relativeURI) throws URISyntaxException {
 452         String bscheme = null;
 453         String bauthority = null;
 454         String bpath = "";
 455         String bquery = null;
 456 
 457         // pre-parse the baseURI
 458         if (baseURI != null) {
 459             if (baseURI.endsWith("..")) {
 460                 baseURI = baseURI + "/";
 461             }
 462             URI base = new URI(baseURI);
 463             bscheme = base.getScheme();
 464             bauthority = base.getAuthority();
 465             bpath = base.getPath();
 466             bquery = base.getQuery();
 467         }
 468 
 469         URI r = new URI(relativeURI);
 470         String rscheme = r.getScheme();
 471         String rauthority = r.getAuthority();
 472         String rpath = r.getPath();
 473         String rquery = r.getQuery();
 474 
 475         String tscheme, tauthority, tpath, tquery;
 476         if (rscheme != null && rscheme.equals(bscheme)) {
 477             rscheme = null;
 478         }
 479         if (rscheme != null) {
 480             tscheme = rscheme;
 481             tauthority = rauthority;
 482             tpath = removeDotSegments(rpath);
 483             tquery = rquery;
 484         } else {
 485             if (rauthority != null) {
 486                 tauthority = rauthority;
 487                 tpath = removeDotSegments(rpath);
 488                 tquery = rquery;
 489             } else {
 490                 if (rpath.length() == 0) {
 491                     tpath = bpath;
 492                     if (rquery != null) {
 493                         tquery = rquery;
 494                     } else {
 495                         tquery = bquery;
 496                     }
 497                 } else {
 498                     if (rpath.startsWith("/")) {
 499                         tpath = removeDotSegments(rpath);
 500                     } else {
 501                         if (bauthority != null && bpath.length() == 0) {
 502                             tpath = "/" + rpath;
 503                         } else {
 504                             int last = bpath.lastIndexOf('/');
 505                             if (last == -1) {
 506                                 tpath = rpath;
 507                             } else {
 508                                 tpath = bpath.substring(0, last+1) + rpath;
 509                             }
 510                         }
 511                         tpath = removeDotSegments(tpath);
 512                     }
 513                     tquery = rquery;
 514                 }
 515                 tauthority = bauthority;
 516             }
 517             tscheme = bscheme;
 518         }
 519         return new URI(tscheme, tauthority, tpath, tquery, null).toString();
 520     }

 522     private static String removeDotSegments(String path) {
 523         if (log.isLoggable(java.util.logging.Level.FINE)) {
 524             log.log(java.util.logging.Level.FINE, "STEP   OUTPUT BUFFER\t\tINPUT BUFFER");
 525         }
 526 
 527         // 1. The input buffer is initialized with the now-appended path
 528         // components then replace occurrences of "//" in the input buffer
 529         // with "/" until no more occurrences of "//" are in the input buffer.
 530         String input = path;
 531         while (input.indexOf("//") > -1) {
 532             input = input.replaceAll("//", "/");
 533         }
 534 
 535         // Initialize the output buffer with the empty string.
 536         StringBuilder output = new StringBuilder();
 537 
 538         // If the input buffer starts with a root slash "/" then move this
 539         // character to the output buffer.
 540         if (input.charAt(0) == '/') {
 541             output.append("/");
 542             input = input.substring(1);
 543         }
 544 
 545         printStep("1 ", output.toString(), input);
 546 
 547         // While the input buffer is not empty, loop as follows
 548         while (input.length() != 0) {
 549             // 2A. If the input buffer begins with a prefix of "./",
 550             // then remove that prefix from the input buffer
 551             // else if the input buffer begins with a prefix of "../", then
 552             // if also the output does not contain the root slash "/" only,
 553             // then move this prefix to the end of the output buffer else
 554             // remove that prefix
 555             if (input.startsWith("./")) {
 556                 input = input.substring(2);
 557                 printStep("2A", output.toString(), input);
 558             } else if (input.startsWith("../")) {
 559                 input = input.substring(3);
 560                 if (!output.toString().equals("/")) {
 561                     output.append("../");
 562                 }
 563                 printStep("2A", output.toString(), input);
 564                 // 2B. if the input buffer begins with a prefix of "/./" or "/.",
 565                 // where "." is a complete path segment, then replace that prefix
 566                 // with "/" in the input buffer; otherwise,
 567             } else if (input.startsWith("/./")) {
 568                 input = input.substring(2);
 569                 printStep("2B", output.toString(), input);
 570             } else if (input.equals("/.")) {
 571                 // FIXME: what is complete path segment?
 572                 input = input.replaceFirst("/.", "/");
 573                 printStep("2B", output.toString(), input);
 574                 // 2C. if the input buffer begins with a prefix of "/../" or "/..",
 575                 // where ".." is a complete path segment, then replace that prefix
 576                 // with "/" in the input buffer and if also the output buffer is
 577                 // empty, last segment in the output buffer equals "../" or "..",
 578                 // where ".." is a complete path segment, then append ".." or "/.."
 579                 // for the latter case respectively to the output buffer else
 580                 // remove the last segment and its preceding "/" (if any) from the
 581                 // output buffer and if hereby the first character in the output
 582                 // buffer was removed and it was not the root slash then delete a
 583                 // leading slash from the input buffer; otherwise,
 584             } else if (input.startsWith("/../")) {
 585                 input = input.substring(3);
 586                 if (output.length() == 0) {
 587                     output.append("/");
 588                 } else if (output.toString().endsWith("../")) {
 589                     output.append("..");
 590                 } else if (output.toString().endsWith("..")) {
 591                     output.append("/..");
 592                 } else {
 593                     int index = output.lastIndexOf("/");
 594                     if (index == -1) {
 595                         output = new StringBuilder();
 596                         if (input.charAt(0) == '/') {
 597                             input = input.substring(1);
 598                         }
 599                     } else {
 600                         output = output.delete(index, output.length());
 601                     }
 602                 }
 603                 printStep("2C", output.toString(), input);
 604             } else if (input.equals("/..")) {
 605                 // FIXME: what is complete path segment?
 606                 input = input.replaceFirst("/..", "/");
 607                 if (output.length() == 0) {
 608                     output.append("/");
 609                 } else if (output.toString().endsWith("../")) {
 610                     output.append("..");
 611                 } else if (output.toString().endsWith("..")) {
 612                     output.append("/..");
 613                 } else {
 614                     int index = output.lastIndexOf("/");
 615                     if (index == -1) {
 616                         output = new StringBuilder();
 617                         if (input.charAt(0) == '/') {
 618                             input = input.substring(1);
 619                         }
 620                     } else {
 621                         output = output.delete(index, output.length());
 622                     }
 623                 }
 624                 printStep("2C", output.toString(), input);
 625                 // 2D. if the input buffer consists only of ".", then remove
 626                 // that from the input buffer else if the input buffer consists
 627                 // only of ".." and if the output buffer does not contain only
 628                 // the root slash "/", then move the ".." to the output buffer
 629                 // else delte it.; otherwise,
 630             } else if (input.equals(".")) {
 631                 input = "";
 632                 printStep("2D", output.toString(), input);
 633             } else if (input.equals("..")) {
 634                 if (!output.toString().equals("/")) {
 635                     output.append("..");
 636                 }
 637                 input = "";
 638                 printStep("2D", output.toString(), input);
 639                 // 2E. move the first path segment (if any) in the input buffer
 640                 // to the end of the output buffer, including the initial "/"
 641                 // character (if any) and any subsequent characters up to, but not
 642                 // including, the next "/" character or the end of the input buffer.
 643             } else {
 644                 int end = -1;
 645                 int begin = input.indexOf('/');
 646                 if (begin == 0) {
 647                     end = input.indexOf('/', 1);
 648                 } else {
 649                     end = begin;
 650                     begin = 0;
 651                 }
 652                 String segment;
 653                 if (end == -1) {
 654                     segment = input.substring(begin);
 655                     input = "";
 656                 } else {
 657                     segment = input.substring(begin, end);
 658                     input = input.substring(end);
 659                 }
 660                 output.append(segment);
 661                 printStep("2E", output.toString(), input);
 662             }
 663         }
 664 
 665         // 3. Finally, if the only or last segment of the output buffer is
 666         // "..", where ".." is a complete path segment not followed by a slash
 667         // then append a slash "/". The output buffer is returned as the result
 668         // of remove_dot_segments
 669         if (output.toString().endsWith("..")) {
 670             output.append("/");
 671             printStep("3 ", output.toString(), input);
 672         }
 673 
 674         return output.toString();
 675     }

 677     private static void printStep(String step, String output, String input) {
 678         if (log.isLoggable(java.util.logging.Level.FINE)) {
 679             log.log(java.util.logging.Level.FINE, " " + step + ":   " + output);
