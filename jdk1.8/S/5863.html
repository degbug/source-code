  25 /**
  26  * A very simple table that stores a list of StringToStringTables, optimized
  27  * for small lists.
  28  * @xsl.usage internal
  29  */
  30 public class StringToStringTableVector
  31 {
  32 
  33   /** Size of blocks to allocate         */
  34   private int m_blocksize;
  35 
  36   /** Array of StringToStringTable objects          */
  37   private StringToStringTable m_map[];
  38 
  39   /** Number of StringToStringTable objects in this array          */
  40   private int m_firstFree = 0;
  41 
  42   /** Size of this array          */
  43   private int m_mapSize;
  44 
  45   /**
  46    * Default constructor.  Note that the default
  47    * block size is very small, for small lists.
  48    */
  49   public StringToStringTableVector()
  50   {
  51 
  52     m_blocksize = 8;
  53     m_mapSize = m_blocksize;
  54     m_map = new StringToStringTable[m_blocksize];
  55   }
  56 
  57   /**
  58    * Construct a StringToStringTableVector, using the given block size.
  59    *
  60    * @param blocksize Size of blocks to allocate
  61    */
  62   public StringToStringTableVector(int blocksize)
  63   {
  64 
  65     m_blocksize = blocksize;
  66     m_mapSize = blocksize;
  67     m_map = new StringToStringTable[blocksize];
  68   }
  69 
  70   /**
  71    * Get the length of the list.
  72    *
  73    * @return Number of StringToStringTable objects in the list
  74    */
  75   public final int getLength()
  76   {
  77     return m_firstFree;
  78   }
  79 
  80   /**
  81    * Get the length of the list.
  82    *
  83    * @return Number of StringToStringTable objects in the list
  84    */
  85   public final int size()
  86   {
  87     return m_firstFree;
  88   }
  89 
  90   /**
  91    * Append a StringToStringTable object onto the vector.
  92    *
  93    * @param value StringToStringTable object to add
  94    */
  95   public final void addElement(StringToStringTable value)
  96   {
  97 
  98     if ((m_firstFree + 1) >= m_mapSize)
  99     {
 100       m_mapSize += m_blocksize;
 101 
 102       StringToStringTable newMap[] = new StringToStringTable[m_mapSize];
 103 
 104       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
 105 
 106       m_map = newMap;
 107     }
 108 
 109     m_map[m_firstFree] = value;
 110 
 111     m_firstFree++;
 112   }
 113 
 114   /**
 115    * Given a string, find the last added occurance value
 116    * that matches the key.
 117    *
 118    * @param key String to look up
 119    *
 120    * @return the last added occurance value that matches the key
 121    * or null if not found.
 122    */
 123   public final String get(String key)
 124   {
 125 
 126     for (int i = m_firstFree - 1; i >= 0; --i)
 127     {
 128       String nsuri = m_map[i].get(key);
 129 
 130       if (nsuri != null)
 131         return nsuri;
 132     }
 133 
 134     return null;
 135   }
 136 
 137   /**
 138    * Given a string, find out if there is a value in this table
 139    * that matches the key.
 140    *
 141    * @param key String to look for
 142    *
 143    * @return True if the string was found in table, null if not
 144    */
 145   public final boolean containsKey(String key)
 146   {
 147 
 148     for (int i = m_firstFree - 1; i >= 0; --i)
 149     {
 150       if (m_map[i].get(key) != null)
 151         return true;
 152     }
 153 
 154     return false;
 155   }
 156 
 157   /**
 158    * Remove the last element.
 159    */
 160   public final void removeLastElem()
 161   {
 162 
 163     if (m_firstFree > 0)
 164     {
 165       m_map[m_firstFree] = null;
 166 
 167       m_firstFree--;
 168     }
 169   }
 170 
 171   /**
 172    * Get the nth element.
 173    *
 174    * @param i Index of element to find
 175    *
 176    * @return The StringToStringTable object at the given index
 177    */
 178   public final StringToStringTable elementAt(int i)
 179   {
 180     return m_map[i];
 181   }
 182 
 183   /**
 184    * Tell if the table contains the given StringToStringTable.
 185    *
 186    * @param s The StringToStringTable to find
 187    *
 188    * @return True if the StringToStringTable is found
 189    */
 190   public final boolean contains(StringToStringTable s)
 191   {
 192 
 193     for (int i = 0; i < m_firstFree; i++)
 194     {
 195       if (m_map[i].equals(s))
 196         return true;
 197     }
 198 
 199     return false;
 200   }
