 114    * Append a Node onto the vector.

 116    * @param value Node to add to the vector

 118   public void addElement(int value)
 119   {
 120 
 121     if ((m_firstFree + 1) >= m_mapSize)
 122     {
 123       if (null == m_map)
 124       {
 125         m_map = new int[m_blocksize];
 126         m_mapSize = m_blocksize;
 127       }
 128       else
 129       {
 130         m_mapSize += m_blocksize;
 131 
 132         int newMap[] = new int[m_mapSize];
 133 
 134         System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
 135 
 136         m_map = newMap;
 137       }
 138     }
 139 
 140     m_map[m_firstFree] = value;
 141 
 142     m_firstFree++;
 143   }

 364   public void insertElementAt(int value, int at)
 365   {
 366 
 367     if (null == m_map)
 368     {
 369       m_map = new int[m_blocksize];
 370       m_mapSize = m_blocksize;
 371     }
 372     else if ((m_firstFree + 1) >= m_mapSize)
 373     {
 374       m_mapSize += m_blocksize;
 375 
 376       int newMap[] = new int[m_mapSize];
 377 
 378       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
 379 
 380       m_map = newMap;
 381     }
 382 
 383     if (at <= (m_firstFree - 1))
 384     {
 385       System.arraycopy(m_map, at, m_map, at + 1, m_firstFree - at);
 386     }
 387 
 388     m_map[at] = value;
 389 
 390     m_firstFree++;

 404     {
 405       m_mapSize = nNodes + m_blocksize;
 406       m_map = new int[m_mapSize];
 407     }
 408     else if ((m_firstFree + nNodes) >= m_mapSize)
 409     {
 410       m_mapSize += (nNodes + m_blocksize);
 411 
 412       int newMap[] = new int[m_mapSize];
 413 
 414       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + nNodes);
 415 
 416       m_map = newMap;
 417     }
 418 
 419     System.arraycopy(nodes.m_map, 0, m_map, m_firstFree, nNodes);
 420 
 421     m_firstFree += nNodes;

 456   /**
 457    * Removes the first occurrence of the argument from this vector.
 458    * If the object is found in this vector, each component in the vector
 459    * with an index greater or equal to the object's index is shifted
 460    * downward to have an index one smaller than the value it had
 461    * previously.
 462    *
 463    * @param s Node to remove from the list
 464    *
 465    * @return True if the node was successfully removed
 466    */
 467   public boolean removeElement(int s)
 468   {
 469 
 470     if (null == m_map)
 471       return false;
 472 
 473     for (int i = 0; i < m_firstFree; i++)
 474     {
 475       int node = m_map[i];
 476 
 477       if (node == s)
 478       {
 479         if (i > m_firstFree)
 480           System.arraycopy(m_map, i + 1, m_map, i - 1, m_firstFree - i);
 481         else
 482           m_map[i] = DTM.NULL;
 483 
 484         m_firstFree--;
 485 
 486         return true;
 487       }
 488     }
 489 
 490     return false;
 491   }

 633   /**
 634    * Sort an array using a quicksort algorithm.
 635    *
 636    * @param a The array to be sorted.
 637    * @param lo0  The low index.
 638    * @param hi0  The high index.
 639    *
 640    * @throws Exception
 641    */
 642   public void sort(int a[], int lo0, int hi0) throws Exception
 643   {
 644 
 645     int lo = lo0;
 646     int hi = hi0;
 647 
 648     // pause(lo, hi);
 649     if (lo >= hi)
 650     {
 651       return;
 652     }
 653     else if (lo == hi - 1)
 654     {
 655 
 656       /*
 657        *  sort a two element list by swapping if necessary
 658        */
 659       if (a[lo] > a[hi])
 660       {
 661         int T = a[lo];
 662 
 663         a[lo] = a[hi];
 664         a[hi] = T;
 665       }
 666 
 667       return;
 668     }
 669 
 670     /*
 671      *  Pick a pivot and move it out of the way
 672      */
 673     int pivot = a[(lo + hi) / 2];
 674 
 675     a[(lo + hi) / 2] = a[hi];
 676     a[hi] = pivot;
 677 
 678     while (lo < hi)
 679     {
 680 
 681       /*
 682        *  Search forward from a[lo] until an element is found that
 683        *  is greater than the pivot or lo >= hi
 684        */
 685       while (a[lo] <= pivot && lo < hi)
 686       {
 687         lo++;
 688       }
 689 
 690       /*
 691        *  Search backward from a[hi] until element is found that
 692        *  is less than the pivot, or lo >= hi
 693        */
 694       while (pivot <= a[hi] && lo < hi)
 695       {
 696         hi--;
 697       }
 698 
 699       /*
 700        *  Swap elements a[lo] and a[hi]
 701        */
 702       if (lo < hi)
 703       {
 704         int T = a[lo];
 705 
 706         a[lo] = a[hi];
 707         a[hi] = T;
 708 
 709         // pause();
 710       }
 711 
 712       // if (stopRequested) {
 713       //    return;
 714       // }
 715     }
 716 
 717     /*
 718      *  Put the median in the "center" of the list
 719      */
 720     a[hi0] = a[hi];
 721     a[hi] = pivot;
 722 
 723     /*
 724      *  Recursive calls, elements a[lo0] to a[lo-1] are less than or
 725      *  equal to pivot, elements a[hi+1] to a[hi0] are greater than
 726      *  pivot.
 727      */
 728     sort(a, lo0, lo - 1);
 729     sort(a, hi + 1, hi0);
 730   }
