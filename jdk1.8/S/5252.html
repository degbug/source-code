 471     public void getTokens(){
 472         fSchemaPointer = fSchemaPointer.substring(fSchemaPointer.indexOf("(")+1, fSchemaPointer.length());
 473         StringTokenizer st = new StringTokenizer(fSchemaPointer, "/");
 474         String tempToken;
 475         Integer integerToken =null;
 476         Stack tempPointerToken = new Stack();
 477         if(fPointerToken == null){
 478             fPointerToken = new Stack();
 479         }
 480         while(st.hasMoreTokens()){
 481             tempToken=st.nextToken();
 482             try {
 483                 integerToken = Integer.valueOf(tempToken);
 484                 tempPointerToken.push(integerToken);
 485             }catch(NumberFormatException e){
 486                 tempPointerToken.push(tempToken);
 487             }
 488         }
 489         while(!tempPointerToken.empty()){
 490             fPointerToken.push(tempPointerToken.pop());
 491         }
 492     }//getTokens

 502     public boolean getNextToken(){
 503         Object currentToken;
 504         if (!fPointerToken.isEmpty()){
 505             currentToken = fPointerToken.pop();
 506             if(currentToken instanceof Integer){
 507                 fCurrentTokenint = ((Integer)currentToken).intValue();
 508                 fCurrentTokenType = 1;
 509             }
 510             else{
 511                 fCurrentTokenString = ((String)currentToken).toString();
 512                 fCurrentTokenType = 2;
 513             }
 514             return true;
 515         }
 516         else {
 517             return false;
 518         }
 519     }

 528     public boolean checkStringToken(QName element, XMLAttributes attributes){
 529         QName cacheQName = null;
 530         String id =null;
 531         String rawname =null;
 532         QName attrName = new QName();
 533         String attrType = null;
 534         String attrValue = null;
 535         int attrCount = attributes.getLength();
 536         for (int i = 0; i < attrCount; i++) {
 537             Augmentations aaugs = attributes.getAugmentations(i);
 538             attributes.getName(i,attrName);
 539             attrType = attributes.getType(i);
 540             attrValue = attributes.getValue(i);
 541             if(attrType != null && attrValue!= null && isIdAttribute(attributes,aaugs,i) && attrValue.equals(fCurrentTokenString)){
 542                 if(hasMoreToken()){
 543                     fCurrentTokenType = 0;
 544                     fCurrentTokenString = null;
 545                     return true;
 546                 }
 547                 else{
 548                     foundElement = element;
 549                     includeElement = true;
 550                     fCurrentTokenType = 0;
 551                     fCurrentTokenString = null;
 552                     fSubResourceIdentified = true;
 553                     return true;
 554                 }
 555             }
 556         }
 557         return false;
 558     }

 560     public boolean checkIntegerToken(QName element){
 561         if(!skip){
 562             fElementCount++;
 563             if(fCurrentTokenint == fElementCount){
 564                 if(hasMoreToken()){
 565                     fElementCount=0;
 566                     fCurrentTokenType = 0;
 567                     return true;
 568                 }
 569                 else{
 570                     foundElement = element;
 571                     includeElement = true;
 572                     fCurrentTokenType = 0;
 573                     fElementCount=0;
 574                     fSubResourceIdentified =true;
 575                     return true;
 576                 }
 577             }else{
 578                 addQName(element);
 579                 skip = true;
 580                 return false;
 581             }
 582         }
 583         return false;
 584     }

 634     public void startElement(QName element, XMLAttributes attributes,
 635     Augmentations augs)throws XNIException {
 636 
 637         boolean requiredToken=false;
 638         if(fCurrentTokenType == 0)
 639             getNextToken();
 640         if(fCurrentTokenType ==1)
 641             requiredToken = checkIntegerToken(element);
 642         else if (fCurrentTokenType ==2)
 643             requiredToken = checkStringToken(element, attributes);
 644         if(requiredToken && hasMoreToken())
 645             getNextToken();
 646         if(fDocumentHandler != null && includeElement){
 647             elemCount++;
 648             fDocumentHandler.startElement(element, attributes, augs);
 649         }
 650 
 651     }

 664                 ftempCurrentElement.pop();
