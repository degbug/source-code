  28 /**
  29  * This class implements a generic PrefixResolver that
  30  * can be used to perform prefix-to-namespace lookup
  31  * for the XPath object.
  32  * @xsl.usage general
  33  */
  34 public class PrefixResolverDefault implements PrefixResolver
  35 {
  36 
  37   /**
  38    * The context to resolve the prefix from, if the context
  39    * is not given.
  40    */
  41   Node m_context;
  42 
  43   /**
  44    * Construct a PrefixResolverDefault object.
  45    * @param xpathExpressionContext The context from
  46    * which XPath expression prefixes will be resolved.
  47    * Warning: This will not work correctly if xpathExpressionContext
  48    * is an attribute node.
  49    */
  50   public PrefixResolverDefault(Node xpathExpressionContext)
  51   {
  52     m_context = xpathExpressionContext;
  53   }
  54 
  55   /**
  56    * Given a namespace, get the corrisponding prefix.  This assumes that
  57    * the PrevixResolver hold's it's own namespace context, or is a namespace
  58    * context itself.
  59    * @param prefix Prefix to resolve.
  60    * @return Namespace that prefix resolves to, or null if prefix
  61    * is not bound.
  62    */
  63   public String getNamespaceForPrefix(String prefix)
  64   {
  65     return getNamespaceForPrefix(prefix, m_context);
  66   }
  67 
  68   /**
  69    * Given a namespace, get the corrisponding prefix.
  70    * Warning: This will not work correctly if namespaceContext
  71    * is an attribute node.
  72    * @param prefix Prefix to resolve.
  73    * @param namespaceContext Node from which to start searching for a
  74    * xmlns attribute that binds a prefix to a namespace.
  75    * @return Namespace that prefix resolves to, or null if prefix
  76    * is not bound.
  77    */
  78   public String getNamespaceForPrefix(String prefix,
  79                                       org.w3c.dom.Node namespaceContext)
  80   {
  81 
  82     Node parent = namespaceContext;
  83     String namespace = null;
  84 
  85     if (prefix.equals("xml"))
  86     {
  87       namespace = Constants.S_XMLNAMESPACEURI;
  88     }
  89     else
  90     {
  91       int type;
  92 
  93       while ((null != parent) && (null == namespace)
  94              && (((type = parent.getNodeType()) == Node.ELEMENT_NODE)
  95                  || (type == Node.ENTITY_REFERENCE_NODE)))
  96       {
  97         if (type == Node.ELEMENT_NODE)
  98         {
  99                 if (parent.getNodeName().indexOf(prefix+":") == 0)
 100                         return parent.getNamespaceURI();
 101           NamedNodeMap nnm = parent.getAttributes();
 102 
 103           for (int i = 0; i < nnm.getLength(); i++)
 104           {
 105             Node attr = nnm.item(i);
 106             String aname = attr.getNodeName();
 107             boolean isPrefix = aname.startsWith("xmlns:");
 108 
 109             if (isPrefix || aname.equals("xmlns"))
 110             {
 111               int index = aname.indexOf(':');
 112               String p = isPrefix ? aname.substring(index + 1) : "";
 113 
 114               if (p.equals(prefix))
 115               {
 116                 namespace = attr.getNodeValue();
 117 
 118                 break;
 119               }
 120             }
 121           }
 122         }
 123 
 124         parent = parent.getParentNode();
 125       }
 126     }
 127 
 128     return namespace;
 129   }
