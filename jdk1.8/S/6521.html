  28 /**
  29  * Provides methods to convert between character positions and line numbers
  30  * for a compilation unit.
  31  *
  32  * @since 1.6
  33  */
  34 @jdk.Exported
  35 public interface LineMap {
  36     /**
  37      * Find the start position of a line.
  38      *
  39      * @param line line number (beginning at 1)
  40      * @return     position of first character in line
  41      * @throws  IndexOutOfBoundsException
  42      *           if {@code lineNumber < 1}
  43      *           if {@code lineNumber > no. of lines}
  44      */
  45     long getStartPosition(long line);
  46 
  47     /**
  48      * Find the position corresponding to a (line,column).
  49      *
  50      * @param   line    line number (beginning at 1)
  51      * @param   column  tab-expanded column number (beginning 1)
  52      *
  53      * @return  position of character
  54      * @throws  IndexOutOfBoundsException
  55      *           if {@code line < 1}
  56      *           if {@code line > no. of lines}
  57      */
  58     long getPosition(long line, long column);
  59 
  60     /**
  61      * Find the line containing a position; a line termination
  62      * character is on the line it terminates.
  63      *
  64      * @param   pos  character offset of the position
  65      * @return the line number of pos (first line is 1)
  66      */
  67     long getLineNumber(long pos);
  68 
  69     /**
  70      * Find the column for a character position.
  71      * Tab characters preceding the position on the same line
  72      * will be expanded when calculating the column number.
  73      *
  74      * @param  pos   character offset of the position
  75      * @return       the tab-expanded column number of pos (first column is 1)
  76      */
  77     long getColumnNumber(long pos);
  78 
  79 }
