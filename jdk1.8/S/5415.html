     /** Constructs a component manager suitable for Xerces' schema validator. */

 203     public XMLSchemaValidatorComponentManager(XSGrammarPoolContainer grammarContainer) {
 204         // setup components
 205         fEntityManager = new XMLEntityManager();
 206         fComponents.put(ENTITY_MANAGER, fEntityManager);
 207 
 208         fErrorReporter = new XMLErrorReporter();
 209         fComponents.put(ERROR_REPORTER, fErrorReporter);
 210 
 211         fNamespaceContext = new NamespaceSupport();
 212         fComponents.put(NAMESPACE_CONTEXT, fNamespaceContext);
 213 
 214         fSchemaValidator = new XMLSchemaValidator();
 215         fComponents.put(SCHEMA_VALIDATOR, fSchemaValidator);
 216 
 217         fValidationManager = new ValidationManager();
 218         fComponents.put(VALIDATION_MANAGER, fValidationManager);
 219 
 220         // setup other properties
 221         fComponents.put(ENTITY_RESOLVER, null);
 222         fComponents.put(ERROR_HANDLER, null);
 223 
 224         fComponents.put(SYMBOL_TABLE, new SymbolTable());
 225 
 226         // setup grammar pool
 227         fComponents.put(XMLGRAMMAR_POOL, grammarContainer.getGrammarPool());
 228         fUseGrammarPoolOnly = grammarContainer.isFullyComposed();
 229 
 230         // add schema message formatter to error reporter
 231         fErrorReporter.putMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN, new XSMessageFormatter());
 232 
 233         // add all recognized features and properties and apply their defaults
 234         addRecognizedParamsAndSetDefaults(fEntityManager, grammarContainer);
 235         addRecognizedParamsAndSetDefaults(fErrorReporter, grammarContainer);
 236         addRecognizedParamsAndSetDefaults(fSchemaValidator, grammarContainer);
 237 
 238         boolean secureProcessing = grammarContainer.getFeature(XMLConstants.FEATURE_SECURE_PROCESSING);
 239         if (System.getSecurityManager() != null) {
 240             _isSecureMode = true;
 241             secureProcessing = true;
 242         }
 243 
 244         fInitSecurityManager = (XMLSecurityManager)
 245                 grammarContainer.getProperty(SECURITY_MANAGER);
 246         if (fInitSecurityManager != null ) {
 247             fInitSecurityManager.setSecureProcessing(secureProcessing);
 248         } else {
 249             fInitSecurityManager = new XMLSecurityManager(secureProcessing);
 250         }
 251 
 252         setProperty(SECURITY_MANAGER, fInitSecurityManager);
 253 
 254         //pass on properties set on SchemaFactory
 255         fSecurityPropertyMgr = (XMLSecurityPropertyManager)
 256                 grammarContainer.getProperty(Constants.XML_SECURITY_PROPERTY_MANAGER);
 257         setProperty(XML_SECURITY_PROPERTY_MANAGER, fSecurityPropertyMgr);
 258     }

      Returns the state of a feature.
      @param featureId The feature identifier.
      @return true if the feature is supported
      @throws XMLConfigurationException Thrown for configuration error.
                                        In general, components should
                                        only throw this exception if
                                        it is really
                                        a critical error.

 272     public FeatureState getFeatureState(String featureId)
 273             throws XMLConfigurationException {
 274         if (PARSER_SETTINGS.equals(featureId)) {
 275             return FeatureState.is(fConfigUpdated);
 276         }
 277         else if (VALIDATION.equals(featureId) || SCHEMA_VALIDATION.equals(featureId)) {
 278             return FeatureState.is(true);
 279         }
 280         else if (USE_GRAMMAR_POOL_ONLY.equals(featureId)) {
 281             return FeatureState.is(fUseGrammarPoolOnly);
 282         }
 283         else if (XMLConstants.FEATURE_SECURE_PROCESSING.equals(featureId)) {
 284             return FeatureState.is(fInitSecurityManager.isSecureProcessing());
 285         }
 286         else if (SCHEMA_ELEMENT_DEFAULT.equals(featureId)) {
 287             return FeatureState.is(true); //pre-condition: VALIDATION and SCHEMA_VALIDATION are always true
 288         }
 289         return super.getFeatureState(featureId);
 290     }

      Set the state of a feature.
      @param featureId The unique identifier (URI) of the feature.
      @param state The requested state of the feature (true or false).
      @exception XMLConfigurationException If the requested feature is not known.

 300     public void setFeature(String featureId, boolean value) throws XMLConfigurationException {
 301         if (PARSER_SETTINGS.equals(featureId)) {
 302             throw new XMLConfigurationException(Status.NOT_SUPPORTED, featureId);
 303         }
 304         else if (value == false && (VALIDATION.equals(featureId) || SCHEMA_VALIDATION.equals(featureId))) {
 305             throw new XMLConfigurationException(Status.NOT_SUPPORTED, featureId);
 306         }
 307         else if (USE_GRAMMAR_POOL_ONLY.equals(featureId) && value != fUseGrammarPoolOnly) {
 308             throw new XMLConfigurationException(Status.NOT_SUPPORTED, featureId);
 309         }
 310         if (XMLConstants.FEATURE_SECURE_PROCESSING.equals(featureId)) {
 311             if (_isSecureMode && !value) {
 312                 throw new XMLConfigurationException(Status.NOT_ALLOWED, XMLConstants.FEATURE_SECURE_PROCESSING);
 313             }
 314 
 315             fInitSecurityManager.setSecureProcessing(value);
 316             setProperty(SECURITY_MANAGER, fInitSecurityManager);
 317 
 318             if (value && Constants.IS_JDK8_OR_ABOVE) {
 319                 fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_DTD,
 320                         XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
 321                 fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_SCHEMA,
 322                         XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
 323                 setProperty(XML_SECURITY_PROPERTY_MANAGER, fSecurityPropertyMgr);
 324             }
 325 
 326             return;
 327         }
 328         fConfigUpdated = true;
 329         fEntityManager.setFeature(featureId, value);
 330         fErrorReporter.setFeature(featureId, value);
 331         fSchemaValidator.setFeature(featureId, value);
 332         if (!fInitFeatures.containsKey(featureId)) {
 333             boolean current = super.getFeature(featureId);
 334             fInitFeatures.put(featureId, current ? Boolean.TRUE : Boolean.FALSE);
 335         }
 336         super.setFeature(featureId, value);
 337     }

      Returns the value of a property.
      @param propertyId The property identifier.

 351     public PropertyState getPropertyState(String propertyId)
 352             throws XMLConfigurationException {
 353         if (LOCALE.equals(propertyId)) {
 354             return PropertyState.is(getLocale());
 355         }
 356         final Object component = fComponents.get(propertyId);
 357         if (component != null) {
 358             return PropertyState.is(component);
 359         }
 360         else if (fComponents.containsKey(propertyId)) {
 361             return PropertyState.is(null);
 362         }
 363         return super.getPropertyState(propertyId);
 364     }

      Sets the state of a property.
      @param propertyId The unique identifier (URI) of the property.
      @param value The requested state of the property.
      @exception XMLConfigurationException If the requested property is not known.

 374     public void setProperty(String propertyId, Object value) throws XMLConfigurationException {
 375         if ( ENTITY_MANAGER.equals(propertyId) || ERROR_REPORTER.equals(propertyId) ||
 376              NAMESPACE_CONTEXT.equals(propertyId) || SCHEMA_VALIDATOR.equals(propertyId) ||
 377              SYMBOL_TABLE.equals(propertyId) || VALIDATION_MANAGER.equals(propertyId) ||
 378              XMLGRAMMAR_POOL.equals(propertyId)) {
 379             throw new XMLConfigurationException(Status.NOT_SUPPORTED, propertyId);
 380         }
 381         fConfigUpdated = true;
 382         fEntityManager.setProperty(propertyId, value);
 383         fErrorReporter.setProperty(propertyId, value);
 384         fSchemaValidator.setProperty(propertyId, value);
 385         if (ENTITY_RESOLVER.equals(propertyId) || ERROR_HANDLER.equals(propertyId) ||
 386                 SECURITY_MANAGER.equals(propertyId)) {
 387             fComponents.put(propertyId, value);
 388             return;
 389         }
 390         else if (LOCALE.equals(propertyId)) {
 391             setLocale((Locale) value);
 392             fComponents.put(propertyId, value);
 393             return;
 394         }
 395         //check if the property is managed by security manager
 396         if (fInitSecurityManager == null ||
 397                 !fInitSecurityManager.setLimit(propertyId, XMLSecurityManager.State.APIPROPERTY, value)) {
 398             //check if the property is managed by security property manager
 399             if (fSecurityPropertyMgr == null ||
 400                     !fSecurityPropertyMgr.setValue(propertyId, XMLSecurityPropertyManager.State.APIPROPERTY, value)) {
 401                 //fall back to the existing property manager
 402                 if (!fInitProperties.containsKey(propertyId)) {
 403                     fInitProperties.put(propertyId, super.getProperty(propertyId));
 404                 }
 405                 super.setProperty(propertyId, value);
 406             }
 407         }
 408     }

     /** Cleans out configuration, restoring it to its initial state. */

 474     void restoreInitialState() {
 475         fConfigUpdated = true;
 476 
 477         // Remove error resolver and error handler
 478         fComponents.put(ENTITY_RESOLVER, null);
 479         fComponents.put(ERROR_HANDLER, null);
 480 
 481         // Set the Locale back to null.
 482         setLocale(null);
 483         fComponents.put(LOCALE, null);
 484 
 485         // Restore initial security manager
 486         fComponents.put(SECURITY_MANAGER, fInitSecurityManager);
 487 
 488         // Set the Locale back to null.
 489         setLocale(null);
 490         fComponents.put(LOCALE, null);
 491 
 492         // Reset feature and property values to their initial values
 493         if (!fInitFeatures.isEmpty()) {
 494             Iterator iter = fInitFeatures.entrySet().iterator();
 495             while (iter.hasNext()) {
 496                 Map.Entry entry = (Map.Entry) iter.next();
 497                 String name = (String) entry.getKey();
 498                 boolean value = ((Boolean) entry.getValue()).booleanValue();
 499                 super.setFeature(name, value);
 500             }
 501             fInitFeatures.clear();
 502         }
 503         if (!fInitProperties.isEmpty()) {
 504             Iterator iter = fInitProperties.entrySet().iterator();
 505             while (iter.hasNext()) {
 506                 Map.Entry entry = (Map.Entry) iter.next();
 507                 String name = (String) entry.getKey();
 508                 Object value = entry.getValue();
 509                 super.setProperty(name, value);
 510             }
 511             fInitProperties.clear();
 512         }
 513     }

     /** Sets feature defaults for the given component on this configuration. */

 516     private void setFeatureDefaults(final XMLComponent component,
 517             final String [] recognizedFeatures, XSGrammarPoolContainer grammarContainer) {
 518         if (recognizedFeatures != null) {
 519             for (int i = 0; i < recognizedFeatures.length; ++i) {
 520                 String featureId = recognizedFeatures[i];
 521                 Boolean state = grammarContainer.getFeature(featureId);
 522                 if (state == null) {
 523                     state = component.getFeatureDefault(featureId);
 524                 }
 525                 if (state != null) {
 526                     // Do not overwrite values already set on the configuration.
 527                     if (!fFeatures.containsKey(featureId)) {
 528                         fFeatures.put(featureId, state);
 529                         // For newly added components who recognize this feature
 530                         // but did not offer a default value, we need to make
 531                         // sure these components will get an opportunity to read
 532                         // the value before parsing begins.
 533                         fConfigUpdated = true;
 534                     }
 535                 }
 536             }
 537         }
 538     }

     /** Sets property defaults for the given component on this configuration. */

 541     private void setPropertyDefaults(final XMLComponent component, final String [] recognizedProperties) {
 542         if (recognizedProperties != null) {
 543             for (int i = 0; i < recognizedProperties.length; ++i) {
 544                 String propertyId = recognizedProperties[i];
 545                 Object value = component.getPropertyDefault(propertyId);
 546                 if (value != null) {
 547                     // Do not overwrite values already set on the configuration.
 548                     if (!fProperties.containsKey(propertyId)) {
 549                         fProperties.put(propertyId, value);
 550                         // For newly added components who recognize this property
 551                         // but did not offer a default value, we need to make
 552                         // sure these components will get an opportunity to read
 553                         // the value before parsing begins.
 554                         fConfigUpdated = true;
 555                     }
 556                 }
 557             }
 558         }
