  94     @Override
  95     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  96         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  97         componentColors = extendedCacheKeys;
  98         //generate this entire method. Each state/bg/fg/border combo that has
  99         //been painted gets its own KEY and paint method.
 100         switch(state) {
 101             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
 102             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
 103             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
 104 
 105         }
 106     }
