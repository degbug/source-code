 230     /*
 231      * Check the version of the current JDK against that specified in the
 232      * parameter
 233      *
 234      * There is a proposal to change the java version string to:
 235      * MAJOR.MINOR.FU.CPU.PSU-BUILDNUMBER_BUGIDNUMBER_OPTIONAL
 236      * This method would work with both the current format and that proposed
 237      *
 238      * @param compareTo a JDK version to be compared to
 239      * @return true if the current version is the same or above that represented
 240      * by the parameter
 241      */
 242     public static boolean isJavaVersionAtLeast(int compareTo) {
 243         String javaVersion = SecuritySupport.getSystemProperty("java.version");
 244         String versions[] = javaVersion.split("\\.", 3);
 245         if (Integer.parseInt(versions[0]) >= compareTo ||
 246             Integer.parseInt(versions[1]) >= compareTo) {
 247             return true;
 248         }
 249         return false;
 250     }
