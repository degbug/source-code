  71     /**
  72      * Constructs a new <CODE>SnmpOid</CODE> from a dot-formatted <CODE>String</CODE> or a MIB variable
  73      * name. It generates an exception if the variable name cannot be resolved, or
  74      * if the dot-formatted <CODE>String</CODE> has an invalid subidentifier.
  75      * This constructor helps build an OID object with a <CODE>String</CODE> like .1.2.3.4 or 1.2.3.4
  76      * or <CODE>ifInOctets</CODE> or <CODE>ifInOctets</CODE>.0.
  77      * @param s <CODE>String</CODE> or MIB variable of the form .1.2.3 or 1.2.3 or <CODE>ifInOctets</CODE>.
  78      * @exception IllegalArgumentException The subidentifier is neither a numeric <CODE>String</CODE>
  79      * nor a <CODE>String</CODE> of the MIB database.
  80      */
  81     public SnmpOid(String s) throws IllegalArgumentException {
  82         String dotString = s ;
  83 
  84         if (s.startsWith(".") == false) {
  85             try {
  86                 dotString = resolveVarName(s);
  87             } catch(SnmpStatusException e) {
  88                 throw new IllegalArgumentException(e.getMessage());
  89             }
  90         }
  91 
  92         StringTokenizer st = new StringTokenizer(dotString, ".", false) ;
  93         componentCount= st.countTokens();
  94 
  95         // Now extract the ids
  96         //
  97         if (componentCount == 0) {
  98             components = new long[15] ;
  99         }  else {
 100             components = new long[componentCount] ;
 101             try {
 102                 for (int i = 0 ; i < componentCount ; i++) {
 103                     try {
 104                         components[i] = Long.parseLong(st.nextToken()) ;
 105                     }
 106                     catch(NoSuchElementException e) {}
 107                 }
 108             }
 109             catch(NumberFormatException e) {
 110                 throw new IllegalArgumentException(s) ;
 111             }
 112         }
 113     }

 449    /**
 450      * Compares two OIDs lexicographically.
 451      * @param other The OID to be compared.
 452      * @return
 453      * The value 0 if the parameter <CODE>other</CODE> is equal to this <CODE>SnmpOid</CODE>.
 454      * A value smaller than 0 if this <CODE>SnmpOid</CODE> is lexicographically smaller than <CODE>other</CODE>.
 455      * A value larger than 0 if this <CODE>SnmpOid</CODE> is lexicographically larger than <CODE>other</CODE>.
 456      */
 457     public int compareTo(SnmpOid other) {
 458         int result = 0 ;
 459         int i = 0 ;
 460         int cmplen = Math.min(componentCount, other.componentCount) ;
 461         long[] otheroid = other.components;
 462 
 463         for (i = 0; i < cmplen; i++) {
 464             if (components[i] != otheroid[i]) {
 465                 break ;
 466             }
 467         }
 468         if ((i == componentCount) && (i == other.componentCount)) {
 469             result = 0 ;
 470         }
 471         else if (i == componentCount) {
 472             result = -1 ;
 473         }
 474         else if (i == other.componentCount) {
 475             result = 1 ;
 476         }
 477         else {
 478             result = (components[i] < otheroid[i]) ? -1 : 1 ;
 479         }
 480         return result ;
 481     }

 483     /**
 484      * Resolves a MIB variable <CODE>String</CODE> with the MIB database.
 485      * @param s The variable name to resolve.
 486      * @exception SnmpStatusException If the variable is not found in the MIB database.
 487      */
 488     public String resolveVarName(String s) throws SnmpStatusException {
 489         int index = s.indexOf('.') ;
 490 
 491         // First handle the case where oid is expressed as 1.2.3.4
 492         //
 493         try {
 494             return handleLong(s, index);
 495         } catch(NumberFormatException e) {}
 496 
 497         SnmpOidTable table = getSnmpOidTable();
 498         // if we are here, it means we have something to resolve..
 499         //
 500         if (table == null)
 501           throw new SnmpStatusException(SnmpStatusException.noSuchName);
 502 
 503         // Ok assume there is a variable name to resolve ...
 504         //
 505         if (index <= 0) {
 506             SnmpOidRecord rec = table.resolveVarName(s);
 507             return rec.getOid();
 508 
 509         } else {
 510             SnmpOidRecord rec = table.resolveVarName(s.substring(0, index));
 511             return (rec.getOid()+ s.substring(index));
 512 
 513         }
 514     }

 575     /**
 576      * Checks if there is enough space in the components
 577      * array to insert n new subids. If not, it increases the size of
 578      * the array.
 579      * In fact it reallocates a new array and copy the old one into the new one.
 580      * @param n The number of subids to insert.
 581      */
 582     private void enlargeIfNeeded(int n) {
 583         int neededSize = components.length ;
 584         while (componentCount + n > neededSize) {
 585             neededSize = neededSize * 2 ;
 586         }
 587         if (neededSize > components.length) {
 588             long[] newComponents = new long[neededSize] ;
 589             for (int i = 0 ; i < components.length ; i++) {
 590                 newComponents[i] = components[i] ;
 591             }
 592             components = newComponents ;
 593         }
 594     }
