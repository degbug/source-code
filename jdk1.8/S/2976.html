  92     private void updateStyle(JMenuItem mi) {
  93         SynthContext context = getContext(mi, ENABLED);
  94         SynthStyle oldStyle = style;
  95 
  96         style = SynthLookAndFeel.updateStyle(context, this);
  97         if (oldStyle != style) {
  98             String prefix = getPropertyPrefix();
  99 
 100             Object value = style.get(context, prefix + ".textIconGap");
 101             if (value != null) {
 102                 LookAndFeel.installProperty(mi, "iconTextGap", value);
 103             }
 104             defaultTextIconGap = mi.getIconTextGap();
 105 
 106             if (menuItem.getMargin() == null ||
 107                          (menuItem.getMargin() instanceof UIResource)) {
 108                 Insets insets = (Insets)style.get(context, prefix + ".margin");
 109 
 110                 if (insets == null) {
 111                     // Some places assume margins are non-null.
 112                     insets = SynthLookAndFeel.EMPTY_UIRESOURCE_INSETS;
 113                 }
 114                 menuItem.setMargin(insets);
 115             }
 116             acceleratorDelimiter = style.getString(context, prefix +
 117                                             ".acceleratorDelimiter", "+");
 118 
 119             arrowIcon = style.getIcon(context, prefix + ".arrowIcon");
 120 
 121             checkIcon = style.getIcon(context, prefix + ".checkIcon");
 122             if (oldStyle != null) {
 123                 uninstallKeyboardActions();
 124                 installKeyboardActions();
 125             }
 126         }
 127         context.dispose();
 128 
 129         SynthContext accContext = getContext(mi, Region.MENU_ITEM_ACCELERATOR,
 130                                              ENABLED);
 131 
 132         accStyle = SynthLookAndFeel.updateStyle(accContext, this);
 133         accContext.dispose();
 134     }

 184     private int getComponentState(JComponent c) {
 185         int state;
 186 
 187         if (!c.isEnabled()) {
 188             state = DISABLED;
 189         }
 190         else if (menuItem.isArmed()) {
 191             state = MOUSE_OVER;
 192         }
 193         else {
 194             state = SynthLookAndFeel.getComponentState(c);
 195         }
 196         if (menuItem.isSelected()) {
 197             state |= SELECTED;
 198         }
 199         return state;
 200     }
 201 
 202     private int getComponentState(JComponent c, Region region) {
 203         return getComponentState(c);
 204     }

 212                                                      Icon arrowIcon,
 213                                                      int defaultTextIconGap) {
 214         SynthContext context = getContext(c);
 215         SynthContext accContext = getContext(c, Region.MENU_ITEM_ACCELERATOR);
 216         Dimension value = SynthGraphicsUtils.getPreferredMenuItemSize(
 217                 context, accContext, c, checkIcon, arrowIcon,
 218                 defaultTextIconGap, acceleratorDelimiter,
 219                 MenuItemLayoutHelper.useCheckAndArrow(menuItem),
 220                 getPropertyPrefix());
 221         context.dispose();
 222         accContext.dispose();
 223         return value;

 227     /**
 228      * Notifies this UI delegate to repaint the specified component.
 229      * This method paints the component background, then calls
 230      * the {@link #paint(SynthContext,Graphics)} method.
 231      *
 232      * <p>In general, this method does not need to be overridden by subclasses.
 233      * All Look and Feel rendering code should reside in the {@code paint} method.

 235      * @param g the {@code Graphics} object used for painting
 236      * @param c the component being painted
 237      * @see #paint(SynthContext,Graphics)
 238      */
 239     @Override
 240     public void update(Graphics g, JComponent c) {
 241         SynthContext context = getContext(c);
 242 
 243         SynthLookAndFeel.update(context, g);
 244         paintBackground(context, g, c);
 245         paint(context, g);
 246         context.dispose();
 247     }

 266     /**
 267      * Paints the specified component.
 268      *
 269      * @param context context for the component being painted
 270      * @param g the {@code Graphics} object used for painting
 271      * @see #update(Graphics,JComponent)
 272      */
 273     protected void paint(SynthContext context, Graphics g) {
 274         SynthContext accContext = getContext(menuItem,
 275                                              Region.MENU_ITEM_ACCELERATOR);
 276 
 277         // Refetch the appropriate check indicator for the current state
 278         String prefix = getPropertyPrefix();
 279         Icon checkIcon = style.getIcon(context, prefix + ".checkIcon");
 280         Icon arrowIcon = style.getIcon(context, prefix + ".arrowIcon");
 281         SynthGraphicsUtils.paint(context, accContext, g, checkIcon, arrowIcon,
 282               acceleratorDelimiter, defaultTextIconGap, getPropertyPrefix());
 283         accContext.dispose();
 284     }
