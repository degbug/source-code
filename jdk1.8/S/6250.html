      Instantiate Security Manager in accordance with the status of
      secure processing
      @param secureProcessing

 179     public XMLSecurityManager(boolean secureProcessing) {
 180         values = new int[Limit.values().length];
 181         states = new State[Limit.values().length];
 182         isSet = new boolean[Limit.values().length];
 183         for (Limit limit : Limit.values()) {
 184             if (secureProcessing) {
 185                 values[limit.ordinal()] = limit.secureValue();
 186                 states[limit.ordinal()] = State.FSP;
 187             } else {
 188                 values[limit.ordinal()] = limit.defaultValue();
 189                 states[limit.ordinal()] = State.DEFAULT;
 190             }
 191         }
 192         //read system properties or jaxp.properties
 193         readSystemProperties();
 194     }

 273                 values[index] = value;
 274                 states[index] = state;

      Read from system properties, or those in jaxp.properties

 394     private void readSystemProperties() {
 395 
 396         for (Limit limit : Limit.values()) {
 397             if (!getSystemProperty(limit, limit.systemProperty())) {
 398                 //if system property is not found, try the older form if any
 399                 for (NameMap nameMap : NameMap.values()) {
 400                     String oldName = nameMap.getOldName(limit.systemProperty());
 401                     if (oldName != null) {
 402                         getSystemProperty(limit, oldName);
 403                     }
 404                 }
 405             }
 406         }
 407 
 408     }

      Read from system properties, or those in jaxp.properties
      @param property the type of the property
      @param sysPropertyName the name of system property

 416     private boolean getSystemProperty(Limit limit, String sysPropertyName) {
 417         try {
 418             String value = SecuritySupport.getSystemProperty(sysPropertyName);
 419             if (value != null && !value.equals("")) {
 420                 values[limit.ordinal()] = Integer.parseInt(value);
 421                 states[limit.ordinal()] = State.SYSTEMPROPERTY;
 422                 return true;
 423             }
 424 
 425             value = SecuritySupport.readJAXPProperty(sysPropertyName);
 426             if (value != null && !value.equals("")) {
 427                 values[limit.ordinal()] = Integer.parseInt(value);
 428                 states[limit.ordinal()] = State.JAXPDOTPROPERTIES;
 429                 return true;
 430             }
 431         } catch (NumberFormatException e) {
 432             //invalid setting
 433             throw new NumberFormatException("Invalid setting for system property: " + limit.systemProperty());
 434         }
 435         return false;
 436     }
