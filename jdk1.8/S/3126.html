  37 /**
  38  * JFormattedTextField extends JTextField adding
  39  * support for formatting arbitrary values, as well as retrieving a particular
  40  * object once the user has edited the text. The following illustrates
  41  * configuring a JFormattedTextField to edit dates:
  42  * 

  43  *   JFormattedTextField ftf = new JFormattedTextField();
  44  *   ftf.setValue(new Date());
  45  * 
  46  * 
  47  * Once a JFormattedTextField has been created, you can
  48  * listen for editing changes by way of adding
  49  * a PropertyChangeListener and listening for
  50  * PropertyChangeEvents with the property name value.
  51  * 
  52  * JFormattedTextField allows
  53  * configuring what action should be taken when focus is lost. The possible
  54  * configurations are:
  55  * 

  56  * 
Value
Description

  57  * JFormattedTextField.REVERT
  58  *            Revert the display to match that of getValue,
  59  *                possibly losing the current edit.
  60  *        JFormattedTextField.COMMIT
  61  *            Commits the current value. If the value being edited
  62  *                isn't considered a legal value by the
  63  *                AbstractFormatter that is, a
  64  *                ParseException is thrown, then the value
  65  *                will not change, and then edited value will persist.
  66  *        JFormattedTextField.COMMIT_OR_REVERT
  67  *            Similar to COMMIT, but if the value isn't
  68  *                legal, behave like REVERT.
  69  *        JFormattedTextField.PERSIST
  70  *            Do nothing, don't obtain a new
  71  *                AbstractFormatter, and don't update the value.
  72  * 
  73  * The default is JFormattedTextField.COMMIT_OR_REVERT,
  74  * refer to {@link #setFocusLostBehavior} for more information on this.
  75  * 
  76  * JFormattedTextField allows the focus to leave, even if
  77  * the currently edited value is invalid. To lock the focus down while the
  78  * JFormattedTextField is an invalid edit state
  79  * you can attach an InputVerifier. The following code snippet
  80  * shows a potential implementation of such an InputVerifier:
  81  * 

  82  * public class FormattedTextFieldVerifier extends InputVerifier {
  83  *     public boolean verify(JComponent input) {
  84  *         if (input instanceof JFormattedTextField) {
  85  *             JFormattedTextField ftf = (JFormattedTextField)input;
  86  *             AbstractFormatter formatter = ftf.getFormatter();
  87  *             if (formatter != null) {
  88  *                 String text = ftf.getText();
  89  *                 try {
  90  *                      formatter.stringToValue(text);
  91  *                      return true;
  92  *                  } catch (ParseException pe) {
  93  *                      return false;
  94  *                  }
  95  *              }
  96  *          }
  97  *          return true;
  98  *      }
  99  *      public boolean shouldYieldFocus(JComponent input) {
 100  *          return verify(input);
 101  *      }
 102  *  }
 103  * 
 104  * 
 105  * Alternatively, you could invoke commitEdit, which would also
 106  * commit the value.
 107  * 
 108  * JFormattedTextField does not do the formatting it self,
 109  * rather formatting is done through an instance of
 110  * JFormattedTextField.AbstractFormatter which is obtained from
 111  * an instance of JFormattedTextField.AbstractFormatterFactory.
 112  * Instances of JFormattedTextField.AbstractFormatter are
 113  * notified when they become active by way of the
 114  * install method, at which point the
 115  * JFormattedTextField.AbstractFormatter can install whatever
 116  * it needs to, typically a DocumentFilter. Similarly when
 117  * JFormattedTextField no longer
 118  * needs the AbstractFormatter, it will invoke
 119  * uninstall.
 120  * 
 121  * JFormattedTextField typically
 122  * queries the AbstractFormatterFactory for an
 123  * AbstractFormat when it gains or loses focus. Although this
 124  * can change based on the focus lost policy. If the focus lost
 125  * policy is JFormattedTextField.PERSIST
 126  * and the JFormattedTextField has been edited, the
 127  * AbstractFormatterFactory will not be queried until the
 128  * value has been committed. Similarly if the focus lost policy is
 129  * JFormattedTextField.COMMIT and an exception
 130  * is thrown from stringToValue, the
 131  * AbstractFormatterFactory will not be queried when focus is
 132  * lost or gained.
 133  * 
 134  * JFormattedTextField.AbstractFormatter
 135  * is also responsible for determining when values are committed to
 136  * the JFormattedTextField. Some
 137  * JFormattedTextField.AbstractFormatters will make new values
 138  * available on every edit, and others will never commit the value. You can
 139  * force the current value to be obtained
 140  * from the current JFormattedTextField.AbstractFormatter
 141  * by way of invoking commitEdit. commitEdit will
 142  * be invoked whenever return is pressed in the
 143  * JFormattedTextField.
 144  * 
 145  * If an AbstractFormatterFactory has not been explicitly
 146  * set, one will be set based on the Class of the value type after
 147  * setValue has been invoked (assuming value is non-null).
 148  * For example, in the following code an appropriate
 149  * AbstractFormatterFactory and AbstractFormatter
 150  * will be created to handle formatting of numbers:
 151  * 

 152  *   JFormattedTextField tf = new JFormattedTextField();
 153  *   tf.setValue(100);
 154  * 
 155  * 
 156  * Warning: As the AbstractFormatter will
 157  * typically install a DocumentFilter on the
 158  * Document, and a NavigationFilter on the
 159  * JFormattedTextField you should not install your own. If you do,
 160  * you are likely to see odd behavior in that the editing policy of the
 161  * AbstractFormatter will not be enforced.
 162  * 
 163  * Warning: Swing is not thread safe. For more
 164  * information see

 579     /**
 580      * Processes any input method events, such as
 581      * InputMethodEvent.INPUT_METHOD_TEXT_CHANGED or
 582      * InputMethodEvent.CARET_POSITION_CHANGED.
 583      *
 584      * @param e the InputMethodEvent
 585      * @see InputMethodEvent
 586      */
 587     protected void processInputMethodEvent(InputMethodEvent e) {
 588         AttributedCharacterIterator text = e.getText();
 589         int commitCount = e.getCommittedCharacterCount();
 590 
 591         // Keep track of the composed text
 592         if (text != null) {
 593             int begin = text.getBeginIndex();
 594             int end = text.getEndIndex();
 595             composedTextExists = ((end - begin) > commitCount);
 596         } else {
 597             composedTextExists = false;
 598         }
 599 
 600         super.processInputMethodEvent(e);
 601     }

 603     /**
 604      * Processes any focus events, such as
 605      * FocusEvent.FOCUS_GAINED or
 606      * FocusEvent.FOCUS_LOST.
 607      *
 608      * @param e the FocusEvent
 609      * @see FocusEvent
 610      */
 611     protected void processFocusEvent(FocusEvent e) {
 612         super.processFocusEvent(e);
 613 
 614         // ignore temporary focus event
 615         if (e.isTemporary()) {
 616             return;
 617         }
 618 
 619         if (isEdited() && e.getID() == FocusEvent.FOCUS_LOST) {
 620             InputContext ic = getInputContext();
 621             if (focusLostHandler == null) {
 622                 focusLostHandler = new FocusLostHandler();
 623             }
 624 
 625             // if there is a composed text, process it first
 626             if ((ic != null) && composedTextExists) {
 627                 ic.endComposition();
 628                 EventQueue.invokeLater(focusLostHandler);
 629             } else {
 630                 focusLostHandler.run();
 631             }
 632         }
 633         else if (!isEdited()) {
 634             // reformat
 635             setValue(getValue(), true, true);
 636         }
 637     }

 639     /**
 640      * FOCUS_LOST behavior implementation
 641      */
 642     private class FocusLostHandler implements Runnable, Serializable {
 643         public void run() {
 644             int fb = JFormattedTextField.this.getFocusLostBehavior();
 645             if (fb == JFormattedTextField.COMMIT ||
 646                 fb == JFormattedTextField.COMMIT_OR_REVERT) {
 647                 try {
 648                     JFormattedTextField.this.commitEdit();
 649                     // Give it a chance to reformat.
 650                     JFormattedTextField.this.setValue(
 651                         JFormattedTextField.this.getValue(), true, true);
 652                 } catch (ParseException pe) {
 653                     if (fb == JFormattedTextField.this.COMMIT_OR_REVERT) {
 654                         JFormattedTextField.this.setValue(
 655                             JFormattedTextField.this.getValue(), true, true);
 656                     }
 657                 }
 658             }
 659             else if (fb == JFormattedTextField.REVERT) {
 660                 JFormattedTextField.this.setValue(
 661                     JFormattedTextField.this.getValue(), true, true);
 662             }
 663         }
 664     }

 730     /**
 731      * Resets the Actions that come from the TextFormatter to
 732      * actions.
 733      */
 734     private void setFormatterActions(Action[] actions) {
 735         if (actions == null) {
 736             if (textFormatterActionMap != null) {
 737                 textFormatterActionMap.clear();
 738             }
 739         }
 740         else {
 741             if (textFormatterActionMap == null) {
 742                 ActionMap map = getActionMap();
 743 
 744                 textFormatterActionMap = new ActionMap();
 745                 while (map != null) {
 746                     ActionMap parent = map.getParent();
 747 
 748                     if (parent instanceof UIResource || parent == null) {
 749                         map.setParent(textFormatterActionMap);
 750                         textFormatterActionMap.setParent(parent);
 751                         break;
 752                     }
 753                     map = parent;
 754                 }
 755             }
 756             for (int counter = actions.length - 1; counter >= 0;
 757                  counter--) {
 758                 Object key = actions[counter].getValue(Action.NAME);
 759 
 760                 if (key != null) {
 761                     textFormatterActionMap.put(key, actions[counter]);
 762                 }
 763             }
 764         }
 765     }

 767     /**
 768      * Does the setting of the value. If createFormat is true,
 769      * this will also obtain a new AbstractFormatter from the
 770      * current factory. The property change event will be fired if
 771      * firePC is true.
 772      */
 773     private void setValue(Object value, boolean createFormat, boolean firePC) {
 774         Object oldValue = this.value;
 775 
 776         this.value = value;
 777 
 778         if (createFormat) {
 779             AbstractFormatterFactory factory = getFormatterFactory();
 780             AbstractFormatter atf;
 781 
 782             if (factory != null) {
 783                 atf = factory.getFormatter(this);
 784             }
 785             else {
 786                 atf = null;
 787             }
 788             setFormatter(atf);
 789         }
 790         else {
 791             // Assumed to be valid
 792             setEditValid(true);
 793         }
 794 
 795         setEdited(false);
 796 
 797         if (firePC) {
 798             firePropertyChange("value", oldValue, value);
 799         }
 800     }

 816     /**
 817      * Returns an AbstractFormatterFactory suitable for the passed in
 818      * Object type.
 819      */
 820     private AbstractFormatterFactory getDefaultFormatterFactory(Object type) {
 821         if (type instanceof DateFormat) {
 822             return new DefaultFormatterFactory(new DateFormatter
 823                                                ((DateFormat)type));
 824         }
 825         if (type instanceof NumberFormat) {
 826             return new DefaultFormatterFactory(new NumberFormatter(
 827                                                (NumberFormat)type));
 828         }
 829         if (type instanceof Format) {
 830             return new DefaultFormatterFactory(new InternationalFormatter(
 831                                                (Format)type));
 832         }
 833         if (type instanceof Date) {
 834             return new DefaultFormatterFactory(new DateFormatter());
 835         }
 836         if (type instanceof Number) {
 837             AbstractFormatter displayFormatter = new NumberFormatter();
 838             ((NumberFormatter)displayFormatter).setValueClass(type.getClass());
 839             AbstractFormatter editFormatter = new NumberFormatter(
 840                                   new DecimalFormat("#.#"));
 841             ((NumberFormatter)editFormatter).setValueClass(type.getClass());
 842 
 843             return new DefaultFormatterFactory(displayFormatter,
 844                                                displayFormatter,editFormatter);
 845         }
 846         return new DefaultFormatterFactory(new DefaultFormatter());
 847     }

 876     /**
 877      * Instances of AbstractFormatter are used by
 878      * JFormattedTextField to handle the conversion both
 879      * from an Object to a String, and back from a String to an Object.
 880      * AbstractFormatters can also enforce editing policies,
 881      * or navigation policies, or manipulate the
 882      * JFormattedTextField in any way it sees fit to
 883      * enforce the desired policy.
 884      * 
 885      * An AbstractFormatter can only be active in
 886      * one JFormattedTextField at a time.
 887      * JFormattedTextField invokes
 888      * install when it is ready to use it followed
 889      * by uninstall when done. Subclasses
 890      * that wish to install additional state should override
 891      * install and message super appropriately.
 892      * 
 893      * Subclasses must override the conversion methods
 894      * stringToValue and valueToString. Optionally
 895      * they can override getActions,
 896      * getNavigationFilter and getDocumentFilter
 897      * to restrict the JFormattedTextField in a particular
 898      * way.
 899      * 
 900      * Subclasses that allow the JFormattedTextField to be in
 901      * a temporarily invalid state should invoke setEditValid
 902      * at the appropriate times.
 903      * @since 1.4
 904      */
 905     public static abstract class AbstractFormatter implements Serializable {
 906         private JFormattedTextField ftf;
 907 
 908         /**
 909          * Installs the AbstractFormatter onto a particular
 910          * JFormattedTextField.
 911          * This will invoke valueToString to convert the
 912          * current value from the JFormattedTextField to
 913          * a String. This will then install the Actions from
 914          * getActions, the DocumentFilter
 915          * returned from getDocumentFilter and the
 916          * NavigationFilter returned from
 917          * getNavigationFilter onto the
 918          * JFormattedTextField.
 919          * 
 920          * Subclasses will typically only need to override this if they
 921          * wish to install additional listeners on the
 922          * JFormattedTextField.
 923          * 
 924          * If there is a ParseException in converting the
 925          * current value to a String, this will set the text to an empty
 926          * String, and mark the JFormattedTextField as being
 927          * in an invalid state.
 928          * 
 929          * While this is a public method, this is typically only useful
 930          * for subclassers of JFormattedTextField.
 931          * JFormattedTextField will invoke this method at
 932          * the appropriate times when the value changes, or its internal
 933          * state changes.  You will only need to invoke this yourself if
 934          * you are subclassing JFormattedTextField and
 935          * installing/uninstalling AbstractFormatter at a
 936          * different time than JFormattedTextField does.
 937          *
 938          * @param ftf JFormattedTextField to format for, may be null indicating
 939          *            uninstall from current JFormattedTextField.
 940          */
 941         public void install(JFormattedTextField ftf) {
 942             if (this.ftf != null) {
 943                 uninstall();
 944             }
 945             this.ftf = ftf;
 946             if (ftf != null) {
 947                 try {
 948                     ftf.setText(valueToString(ftf.getValue()));
 949                 } catch (ParseException pe) {
 950                     ftf.setText("");
 951                     setEditValid(false);
 952                 }
 953                 installDocumentFilter(getDocumentFilter());
 954                 ftf.setNavigationFilter(getNavigationFilter());
 955                 ftf.setFormatterActions(getActions());
 956             }
 957         }
 958 
 959         /**
 960          * Uninstalls any state the AbstractFormatter may have
 961          * installed on the JFormattedTextField. This resets the
 962          * DocumentFilter, NavigationFilter
 963          * and additional Actions installed on the
 964          * JFormattedTextField.
 965          */
 966         public void uninstall() {
 967             if (this.ftf != null) {
 968                 installDocumentFilter(null);
 969                 this.ftf.setNavigationFilter(null);
 970                 this.ftf.setFormatterActions(null);
 971             }
 972         }

 974         /**
 975          * Parses text returning an arbitrary Object. Some
 976          * formatters may return null.
 977          *
 978          * @throws ParseException if there is an error in the conversion
 979          * @param text String to convert
 980          * @return Object representation of text
 981          */
 982         public abstract Object stringToValue(String text) throws
 983                                      ParseException;
 984 
 985         /**
 986          * Returns the string value to display for value.
 987          *
 988          * @throws ParseException if there is an error in the conversion
 989          * @param value Value to convert
 990          * @return String representation of value
 991          */
 992         public abstract String valueToString(Object value) throws
 993                         ParseException;

1102     /**
1103      * Used to commit the edit. This extends JTextField.NotifyAction
1104      * so that isEnabled is true while a JFormattedTextField
1105      * has focus, and extends actionPerformed to invoke
1106      * commitEdit.
1107      */
1108     static class CommitAction extends JTextField.NotifyAction {
1109         public void actionPerformed(ActionEvent e) {
1110             JTextComponent target = getFocusedComponent();
1111 
1112             if (target instanceof JFormattedTextField) {
1113                 // Attempt to commit the value
1114                 try {
1115                     ((JFormattedTextField)target).commitEdit();
1116                 } catch (ParseException pe) {
1117                     ((JFormattedTextField)target).invalidEdit();
1118                     // value not committed, don't notify ActionListeners
1119                     return;
1120                 }
1121             }
1122             // Super behavior.
1123             super.actionPerformed(e);
1124         }
1125 
1126         public boolean isEnabled() {
1127             JTextComponent target = getFocusedComponent();
1128             if (target instanceof JFormattedTextField) {
1129                 JFormattedTextField ftf = (JFormattedTextField)target;
1130                 if (!ftf.isEdited()) {
1131                     return false;
1132                 }
1133                 return true;
1134             }
1135             return super.isEnabled();
1136         }
1137     }
