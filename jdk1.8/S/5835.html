  23 package com.sun.org.apache.xml.internal.utils;

  25 /**
  26  * A very simple lookup table that stores a list of strings, the even
  27  * number strings being keys, and the odd number strings being values.
  28  * @xsl.usage internal
  29  */
  30 public class StringToStringTable
  31 {
  32 
  33   /** Size of blocks to allocate          */
  34   private int m_blocksize;
  35 
  36   /** Array of strings this contains          */
  37   private String m_map[];
  38 
  39   /** Number of strings this contains           */
  40   private int m_firstFree = 0;
  41 
  42   /** Size of this table           */
  43   private int m_mapSize;
  44 
  45   /**
  46    * Default constructor.  Note that the default
  47    * block size is very small, for small lists.
  48    */
  49   public StringToStringTable()
  50   {
  51 
  52     m_blocksize = 16;
  53     m_mapSize = m_blocksize;
  54     m_map = new String[m_blocksize];
  55   }
  56 
  57   /**
  58    * Construct a StringToStringTable, using the given block size.
  59    *
  60    * @param blocksize Size of blocks to allocate
  61    */
  62   public StringToStringTable(int blocksize)
  63   {
  64 
  65     m_blocksize = blocksize;
  66     m_mapSize = blocksize;
  67     m_map = new String[blocksize];
  68   }
  69 
  70   /**
  71    * Get the length of the list.
  72    *
  73    * @return Number of strings in the list
  74    */
  75   public final int getLength()
  76   {
  77     return m_firstFree;
  78   }
  79 
  80   /**
  81    * Append a string onto the vector.
  82    * The strings go to the even locations in the array
  83    * and the values in the odd.
  84    *
  85    * @param key String to add to the list
  86    * @param value Value of the string
  87    */
  88   public final void put(String key, String value)
  89   {
  90 
  91     if ((m_firstFree + 2) >= m_mapSize)
  92     {
  93       m_mapSize += m_blocksize;
  94 
  95       String newMap[] = new String[m_mapSize];
  96 
  97       System.arraycopy(m_map, 0, newMap, 0, m_firstFree + 1);
  98 
  99       m_map = newMap;
 100     }
 101 
 102     m_map[m_firstFree] = key;
 103 
 104     m_firstFree++;
 105 
 106     m_map[m_firstFree] = value;
 107 
 108     m_firstFree++;
 109   }
 110 
 111   /**
 112    * Tell if the table contains the given string.
 113    *
 114    * @param key String to look up
 115    *
 116    * @return return the value of the string or null if not found.
 117    */
 118   public final String get(String key)
 119   {
 120 
 121     for (int i = 0; i < m_firstFree; i += 2)
 122     {
 123       if (m_map[i].equals(key))
 124         return m_map[i + 1];
 125     }
 126 
 127     return null;
 128   }
 129 
 130   /**
 131    * Remove the given string and its value from this table.
 132    *
 133    * @param key String to remove from the table
 134    */
 135   public final void remove(String key)
 136   {
 137 
 138     for (int i = 0; i < m_firstFree; i += 2)
 139     {
 140       if (m_map[i].equals(key))
 141       {
 142         if ((i + 2) < m_firstFree)
 143           System.arraycopy(m_map, i + 2, m_map, i, m_firstFree - (i + 2));
 144 
 145         m_firstFree -= 2;
 146         m_map[m_firstFree] = null;
 147         m_map[m_firstFree + 1] = null;
 148 
 149         break;
 150       }
 151     }
 152   }
 153 
 154   /**
 155    * Tell if the table contains the given string. Ignore case
 156    *
 157    * @param key String to look up
 158    *
 159    * @return The value of the string or null if not found
 160    */
 161   public final String getIgnoreCase(String key)
 162   {
 163 
 164     if (null == key)
 165       return null;
 166 
 167     for (int i = 0; i < m_firstFree; i += 2)
 168     {
 169       if (m_map[i].equalsIgnoreCase(key))
 170         return m_map[i + 1];
 171     }
 172 
 173     return null;
 174   }
 175 
 176   /**
 177    * Tell if the table contains the given string in the value.
 178    *
 179    * @param val Value of the string to look up
 180    *
 181    * @return the string associated with the given value or null if not found
 182    */
 183   public final String getByValue(String val)
 184   {
 185 
 186     for (int i = 1; i < m_firstFree; i += 2)
 187     {
 188       if (m_map[i].equals(val))
 189         return m_map[i - 1];
 190     }
 191 
 192     return null;
 193   }
 194 
 195   /**
 196    * Get the nth element.
 197    *
 198    * @param i index of the string to look up.
 199    *
 200    * @return The string at the given index.
 201    */
 202   public final String elementAt(int i)
 203   {
 204     return m_map[i];
 205   }
 206 
 207   /**
 208    * Tell if the table contains the given string.
 209    *
 210    * @param key String to look up
 211    *
 212    * @return True if the given string is in this table
 213    */
 214   public final boolean contains(String key)
 215   {
 216 
 217     for (int i = 0; i < m_firstFree; i += 2)
 218     {
 219       if (m_map[i].equals(key))
 220         return true;
 221     }
 222 
 223     return false;
 224   }
 225 
 226   /**
 227    * Tell if the table contains the given string.
 228    *
 229    * @param val value to look up
 230    *
 231    * @return True if the given value is in the table.
 232    */
 233   public final boolean containsValue(String val)
 234   {
 235 
 236     for (int i = 1; i < m_firstFree; i += 2)
 237     {
 238       if (m_map[i].equals(val))
 239         return true;
 240     }
 241 
 242     return false;
 243   }
 244 }
