  66     public void paint( Graphics g, JComponent c )
  67     {
  68         Dimension s = c.getSize();
  69 
  70         if ( ((JSeparator)c).getOrientation() == JSeparator.VERTICAL )
  71         {
  72           g.setColor( c.getForeground() );
  73           g.drawLine( 0, 0, 0, s.height );
  74 
  75           g.setColor( c.getBackground() );
  76           g.drawLine( 1, 0, 1, s.height );
  77         }
  78         else  // HORIZONTAL
  79         {
  80           g.setColor( c.getForeground() );
  81           g.drawLine( 0, 0, s.width, 0 );
  82 
  83           g.setColor( c.getBackground() );
  84           g.drawLine( 0, 1, s.width, 1 );
  85         }
  86     }

  88     public Dimension getPreferredSize( JComponent c )
  89     {
