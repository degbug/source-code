 151     /**
 152      * Create the component.  This is basically a
 153      * big switch statement based upon the tag type
 154      * and html attributes of the associated element.
 155      */
 156     protected Component createComponent() {
 157         AttributeSet attr = getElement().getAttributes();
 158         HTML.Tag t = (HTML.Tag)
 159             attr.getAttribute(StyleConstants.NameAttribute);
 160         JComponent c = null;
 161         Object model = attr.getAttribute(StyleConstants.ModelAttribute);
 162 
 163         // Remove listeners previously registered in shared model
 164         // when a new UI component is replaced.  See bug 7189299.
 165         removeStaleListenerForModel(model);
 166         if (t == HTML.Tag.INPUT) {
 167             c = createInputComponent(attr, model);
 168         } else if (t == HTML.Tag.SELECT) {
 169 
 170             if (model instanceof OptionListModel) {
 171 
 172                 JList list = new JList((ListModel) model);
 173                 int size = HTML.getIntegerAttributeValue(attr,
 174                                                          HTML.Attribute.SIZE,
 175                                                          1);
 176                 list.setVisibleRowCount(size);
 177                 list.setSelectionModel((ListSelectionModel)model);
 178                 c = new JScrollPane(list);
 179             } else {
 180                 c = new JComboBox((ComboBoxModel) model);
 181                 maxIsPreferred = 3;
 182             }
 183         } else if (t == HTML.Tag.TEXTAREA) {
 184             JTextArea area = new JTextArea((Document) model);
 185             int rows = HTML.getIntegerAttributeValue(attr,
 186                                                      HTML.Attribute.ROWS,
 187                                                      1);
 188             area.setRows(rows);
 189             int cols = HTML.getIntegerAttributeValue(attr,
 190                                                      HTML.Attribute.COLS,
 191                                                      20);
 192             maxIsPreferred = 3;
 193             area.setColumns(cols);
 194             c = new JScrollPane(area,
 195                                 JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
 196                                 JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS);
 197         }
 198 
 199         if (c != null) {
 200             c.setAlignmentY(1.0f);
 201         }
 202         return c;
 203     }

 207      * Creates a component for an &lt;INPUT&gt; element based on the
 208      * value of the "type" attribute.
 209      *
 210      * @param set of attributes associated with the &lt;INPUT&gt; element.
 211      * @param model the value of the StyleConstants.ModelAttribute
 212      * @return the component.
 213      */
 214     private JComponent createInputComponent(AttributeSet attr, Object model) {
 215         JComponent c = null;
 216         String type = (String) attr.getAttribute(HTML.Attribute.TYPE);
 217 
 218         if (type.equals("submit") || type.equals("reset")) {
 219             String value = (String)
 220                 attr.getAttribute(HTML.Attribute.VALUE);
 221             if (value == null) {
 222                 if (type.equals("submit")) {
 223                     value = UIManager.getString("FormView.submitButtonText");
 224                 } else {
 225                     value = UIManager.getString("FormView.resetButtonText");
 226                 }
 227             }
 228             JButton button = new JButton(value);
 229             if (model != null) {
 230                 button.setModel((ButtonModel)model);
 231                 button.addActionListener(this);
 232             }
 233             c = button;
 234             maxIsPreferred = 3;
 235         } else if (type.equals("image")) {
 236             String srcAtt = (String) attr.getAttribute(HTML.Attribute.SRC);
 237             JButton button;
 238             try {
 239                 URL base = ((HTMLDocument)getElement().getDocument()).getBase();
 240                 URL srcURL = new URL(base, srcAtt);
 241                 Icon icon = new ImageIcon(srcURL);
 242                 button  = new JButton(icon);
 243             } catch (MalformedURLException e) {
 244                 button = new JButton(srcAtt);
 245             }
 246             if (model != null) {
 247                 button.setModel((ButtonModel)model);
 248                 button.addMouseListener(new MouseEventListener());
 249             }
 250             c = button;
 251             maxIsPreferred = 3;
 252         } else if (type.equals("checkbox")) {
 253             c = new JCheckBox();
 254             if (model != null) {
 255                 ((JCheckBox)c).setModel((JToggleButton.ToggleButtonModel) model);
 256             }
 257             maxIsPreferred = 3;
 258         } else if (type.equals("radio")) {
 259             c = new JRadioButton();
 260             if (model != null) {
 261                 ((JRadioButton)c).setModel((JToggleButton.ToggleButtonModel)model);
 262             }
 263             maxIsPreferred = 3;
 264         } else if (type.equals("text")) {
 265             int size = HTML.getIntegerAttributeValue(attr,
 266                                                      HTML.Attribute.SIZE,
 267                                                      -1);
 268             JTextField field;
 269             if (size > 0) {
 270                 field = new JTextField();
 271                 field.setColumns(size);
 272             }
 273             else {
 274                 field = new JTextField();
 275                 field.setColumns(20);
 276             }
 277             c = field;
 278             if (model != null) {
 279                 field.setDocument((Document) model);
 280             }
 281             field.addActionListener(this);
 282             maxIsPreferred = 3;
 283         } else if (type.equals("password")) {
 284             JPasswordField field = new JPasswordField();
 285             c = field;
 286             if (model != null) {
 287                 field.setDocument((Document) model);
 288             }
 289             int size = HTML.getIntegerAttributeValue(attr,
 290                                                      HTML.Attribute.SIZE,
 291                                                      -1);
 292             field.setColumns((size > 0) ? size : 20);
 293             field.addActionListener(this);
 294             maxIsPreferred = 3;
 295         } else if (type.equals("file")) {
 296             JTextField field = new JTextField();
 297             if (model != null) {
 298                 field.setDocument((Document)model);
 299             }
 300             int size = HTML.getIntegerAttributeValue(attr, HTML.Attribute.SIZE,
 301                                                      -1);
 302             field.setColumns((size > 0) ? size : 20);
 303             JButton browseButton = new JButton(UIManager.getString
 304                                            ("FormView.browseFileButtonText"));
 305             Box box = Box.createHorizontalBox();
 306             box.add(field);
 307             box.add(Box.createHorizontalStrut(5));
 308             box.add(browseButton);
 309             browseButton.addActionListener(new BrowseFileAction(
 310                                            attr, (Document)model));
 311             c = box;
 312             maxIsPreferred = 3;
 313         }
 314         return c;
 315     }

 317     private void removeStaleListenerForModel(Object model) {
 318         if (model instanceof DefaultButtonModel) {
 319             // case of JButton whose model is DefaultButtonModel
 320             // Need to remove stale ActionListener, ChangeListener and
 321             // ItemListener that are instance of AbstractButton$Handler.
 322             DefaultButtonModel buttonModel = (DefaultButtonModel) model;
 323             String listenerClass = "javax.swing.AbstractButton$Handler";
 324             for (ActionListener listener : buttonModel.getActionListeners()) {
 325                 if (listenerClass.equals(listener.getClass().getName())) {
 326                     buttonModel.removeActionListener(listener);
 327                 }
 328             }
 329             for (ChangeListener listener : buttonModel.getChangeListeners()) {
 330                 if (listenerClass.equals(listener.getClass().getName())) {
 331                     buttonModel.removeChangeListener(listener);
 332                 }
 333             }
 334             for (ItemListener listener : buttonModel.getItemListeners()) {
 335                 if (listenerClass.equals(listener.getClass().getName())) {
 336                     buttonModel.removeItemListener(listener);
 337                 }
 338             }
 339         } else if (model instanceof AbstractListModel) {
 340             // case of JComboBox and JList
 341             // For JList, the stale ListDataListener is instance
 342             // BasicListUI$Handler.
 343             // For JComboBox, there are 2 stale ListDataListeners, which are
 344             // BasicListUI$Handler and BasicComboBoxUI$Handler.
 345             AbstractListModel listModel = (AbstractListModel) model;
 346             String listenerClass1 =
 347                     "javax.swing.plaf.basic.BasicListUI$Handler";
 348             String listenerClass2 =
 349                     "javax.swing.plaf.basic.BasicComboBoxUI$Handler";
 350             for (ListDataListener listener : listModel.getListDataListeners()) {
 351                 if (listenerClass1.equals(listener.getClass().getName())
 352                         || listenerClass2.equals(listener.getClass().getName()))
 353                 {
 354                     listModel.removeListDataListener(listener);
 355                 }
 356             }
 357         } else if (model instanceof AbstractDocument) {
 358             // case of JPasswordField, JTextField and JTextArea
 359             // All have 2 stale DocumentListeners.
 360             String listenerClass1 =
 361                     "javax.swing.plaf.basic.BasicTextUI$UpdateHandler";
 362             String listenerClass2 =
 363                     "javax.swing.text.DefaultCaret$Handler";
 364             AbstractDocument docModel = (AbstractDocument) model;
 365             for (DocumentListener listener : docModel.getDocumentListeners()) {
 366                 if (listenerClass1.equals(listener.getClass().getName())
 367                         || listenerClass2.equals(listener.getClass().getName()))
 368                 {
 369                     docModel.removeDocumentListener(listener);
 370                 }
 371             }
 372         }
 373     }

 375     /**
 376      * Determines the maximum span for this view along an
 377      * axis. For certain components, the maximum and preferred span are the
 378      * same. For others this will return the value
 379      * returned by Component.getMaximumSize along the
 380      * axis of interest.
 381      *
 382      * @param axis may be either View.X_AXIS or View.Y_AXIS
 383      * @return   the span the view would like to be rendered into &gt;= 0.
 384      *           Typically the view is told to render into the span
 385      *           that is returned, although there is no guarantee.
 386      *           The parent may choose to resize or break the view.
 387      * @exception IllegalArgumentException for an invalid axis
 388      */
 389     public float getMaximumSpan(int axis) {
 390         switch (axis) {
 391         case View.X_AXIS:
 392             if ((maxIsPreferred & 1) == 1) {
 393                 super.getMaximumSpan(axis);
 394                 return getPreferredSpan(axis);
 395             }
 396             return super.getMaximumSpan(axis);
 397         case View.Y_AXIS:
 398             if ((maxIsPreferred & 2) == 2) {
 399                 super.getMaximumSpan(axis);
 400                 return getPreferredSpan(axis);
 401             }
 402             return super.getMaximumSpan(axis);
 403         default:
 404             break;
 405         }
 406         return super.getMaximumSpan(axis);
 407     }

 410     /**
 411      * Responsible for processing the ActionEvent.
 412      * If the element associated with the FormView,
 413      * has a type of "submit", "reset", "text" or "password"
 414      * then the action is processed.  In the case of a "submit"
 415      * the form is submitted.  In the case of a "reset"
 416      * the form is reset to its original state.
 417      * In the case of "text" or "password", if the
 418      * element is the last one of type "text" or "password",
 419      * the form is submitted.  Otherwise, focus is transferred
 420      * to the next component in the form.
 421      *
 422      * @param evt the ActionEvent.
 423      */
 424     public void actionPerformed(ActionEvent evt) {
 425         Element element = getElement();
 426         StringBuilder dataBuffer = new StringBuilder();
 427         HTMLDocument doc = (HTMLDocument)getDocument();
 428         AttributeSet attr = element.getAttributes();
 429 
 430         String type = (String) attr.getAttribute(HTML.Attribute.TYPE);
 431 
 432         if (type.equals("submit")) {
 433             getFormData(dataBuffer);
 434             submitData(dataBuffer.toString());
 435         } else if (type.equals("reset")) {
 436             resetForm();
 437         } else if (type.equals("text") || type.equals("password")) {
 438             if (isLastTextOrPasswordField()) {
 439                 getFormData(dataBuffer);
 440                 submitData(dataBuffer.toString());
 441             } else {
 442                 getComponent().transferFocus();
 443             }
 444         }
 445     }

 448     /**
 449      * This method is responsible for submitting the form data.
 450      * A thread is forked to undertake the submission.
 451      */
 452     protected void submitData(String data) {
 453         Element form = getFormElement();
 454         AttributeSet attrs = form.getAttributes();
 455         HTMLDocument doc = (HTMLDocument) form.getDocument();
 456         URL base = doc.getBase();
 457 
 458         String target = (String) attrs.getAttribute(HTML.Attribute.TARGET);
 459         if (target == null) {
 460             target = "_self";
 461         }
 462 
 463         String method = (String) attrs.getAttribute(HTML.Attribute.METHOD);
 464         if (method == null) {
 465             method = "GET";
 466         }
 467         method = method.toLowerCase();
 468         boolean isPostMethod = method.equals("post");
 469         if (isPostMethod) {
 470             storePostData(doc, target, data);
 471         }
 472 
 473         String action = (String) attrs.getAttribute(HTML.Attribute.ACTION);
 474         URL actionURL;
 475         try {
 476             actionURL = (action == null)
 477                 ? new URL(base.getProtocol(), base.getHost(),
 478                                         base.getPort(), base.getFile())
 479                 : new URL(base, action);
 480             if (!isPostMethod) {
 481                 String query = data.toString();
 482                 actionURL = new URL(actionURL + "?" + query);
 483             }
 484         } catch (MalformedURLException e) {
 485             actionURL = null;
 486         }
 487         final JEditorPane c = (JEditorPane) getContainer();
 488         HTMLEditorKit kit = (HTMLEditorKit) c.getEditorKit();
 489 
 490         FormSubmitEvent formEvent = null;
 491         if (!kit.isAutoFormSubmission() || doc.isFrameDocument()) {
 492             FormSubmitEvent.MethodType methodType = isPostMethod
 493                     ? FormSubmitEvent.MethodType.POST
 494                     : FormSubmitEvent.MethodType.GET;
 495             formEvent = new FormSubmitEvent(
 496                     FormView.this, HyperlinkEvent.EventType.ACTIVATED,
 497                     actionURL, form, target, methodType, data);
 498 
 499         }
 500         // setPage() may take significant time so schedule it to run later.
 501         final FormSubmitEvent fse = formEvent;
 502         final URL url = actionURL;
 503         SwingUtilities.invokeLater(new Runnable() {
 504             public void run() {
 505                 if (fse != null) {
 506                     c.fireHyperlinkUpdate(fse);
 507                 } else {
 508                     try {
 509                         c.setPage(url);
 510                     } catch (IOException e) {
 511                         UIManager.getLookAndFeel().provideErrorFeedback(c);
 512                     }
 513                 }
 514             }
 515         });
 516     }

 518     private void storePostData(HTMLDocument doc, String target, String data) {
 519 
 520         /* POST data is stored into the document property named by constant
 521          * PostDataProperty from where it is later retrieved by method
 522          * JEditorPane.getPostData().  If the current document is in a frame,
 523          * the data is initially put into the toplevel (frameset) document
 524          * property (named <PostDataProperty>.<Target frame name>).  It is the
 525          * responsibility of FrameView which updates the target frame
 526          * to move data from the frameset document property into the frame
 527          * document property.
 528          */
 529 
 530         Document propDoc = doc;
 531         String propName = PostDataProperty;
 532 
 533         if (doc.isFrameDocument()) {
 534             // find the top-most JEditorPane holding the frameset view.
 535             FrameView.FrameEditorPane p =
 536                     (FrameView.FrameEditorPane) getContainer();
 537             FrameView v = p.getFrameView();
 538             JEditorPane c = v.getOutermostJEditorPane();
 539             if (c != null) {
 540                 propDoc = c.getDocument();
 541                 propName += ("." + target);
 542             }
 543         }
 544 
 545         propDoc.putProperty(propName, data);
 546     }

 584     /**
 585      * Extracts the value of the name attribute
 586      * associated with the input element of type
 587      * image.  If name is defined it is encoded using
 588      * the URLEncoder.encode() method and the
 589      * image data is returned in the following format:
 590      *      name + ".x" +"="+ x +"&"+ name +".y"+"="+ y
 591      * otherwise,
 592      *      "x="+ x +"&y="+ y
 593      *
 594      * @param point associated with the mouse click.
 595      * @return the image data.
 596      */
 597     private String getImageData(Point point) {
 598 
 599         String mouseCoords = point.x + ":" + point.y;
 600         int sep = mouseCoords.indexOf(':');
 601         String x = mouseCoords.substring(0, sep);
 602         String y = mouseCoords.substring(++sep);
 603         String name = (String) getElement().getAttributes().getAttribute(HTML.Attribute.NAME);
 604 
 605         String data;
 606         if (name == null || name.equals("")) {
 607             data = "x="+ x +"&y="+ y;
 608         } else {
 609             name = URLEncoder.encode(name);
 610             data = name + ".x" +"="+ x +"&"+ name +".y"+"="+ y;
 611         }
 612         return data;
 613     }

 641     /**
 642      * Iterates over the
 643      * element hierarchy, extracting data from the
 644      * models associated with the relevant form elements.
 645      * "Relevant" means the form elements that are part
 646      * of the same form whose element triggered the submit
 647      * action.
 648      *
 649      * @param buffer        the buffer that contains that data to submit
 650      * @param targetElement the element that triggered the
 651      *                      form submission
 652      */
 653     private void getFormData(StringBuilder buffer) {
 654         Element formE = getFormElement();
 655         if (formE != null) {
 656             ElementIterator it = new ElementIterator(formE);
 657             Element next;
 658 
 659             while ((next = it.next()) != null) {
 660                 if (isControl(next)) {
 661                     String type = (String)next.getAttributes().getAttribute
 662                                        (HTML.Attribute.TYPE);
 663 
 664                     if (type != null && type.equals("submit") &&
 665                         next != getElement()) {
 666                         // do nothing - this submit is not the trigger
 667                     } else if (type == null || !type.equals("image")) {
 668                         // images only result in data if they triggered
 669                         // the submit and they require that the mouse click
 670                         // coords be appended to the data.  Hence its
 671                         // processing is handled by the view.
 672                         loadElementDataIntoBuffer(next, buffer);
 673                     }
 674                 }
 675             }
 676         }
 677     }

 685      *
 686      */
 687     private void loadElementDataIntoBuffer(Element elem, StringBuilder buffer) {
 688 
 689         AttributeSet attr = elem.getAttributes();
 690         String name = (String)attr.getAttribute(HTML.Attribute.NAME);
 691         if (name == null) {
 692             return;
 693         }
 694         String value = null;
 695         HTML.Tag tag = (HTML.Tag)elem.getAttributes().getAttribute
 696                                   (StyleConstants.NameAttribute);
 697 
 698         if (tag == HTML.Tag.INPUT) {
 699             value = getInputElementData(attr);
 700         } else if (tag ==  HTML.Tag.TEXTAREA) {
 701             value = getTextAreaData(attr);
 702         } else if (tag == HTML.Tag.SELECT) {
 703             loadSelectData(attr, buffer);
 704         }
 705 
 706         if (name != null && value != null) {
 707             appendBuffer(buffer, name, value);
 708         }
 709     }

 712     /**
 713      * Returns the data associated with an &lt;INPUT&gt; form
 714      * element.  The value of "type" attributes is
 715      * used to determine the type of the model associated
 716      * with the element and then the relevant data is
 717      * extracted.
 718      */
 719     private String getInputElementData(AttributeSet attr) {
 720 
 721         Object model = attr.getAttribute(StyleConstants.ModelAttribute);
 722         String type = (String) attr.getAttribute(HTML.Attribute.TYPE);
 723         String value = null;
 724 
 725         if (type.equals("text") || type.equals("password")) {
 726             Document doc = (Document)model;
 727             try {
 728                 value = doc.getText(0, doc.getLength());
 729             } catch (BadLocationException e) {
 730                 value = null;
 731             }
 732         } else if (type.equals("submit") || type.equals("hidden")) {
 733             value = (String) attr.getAttribute(HTML.Attribute.VALUE);
 734             if (value == null) {
 735                 value = "";
 736             }
 737         } else if (type.equals("radio") || type.equals("checkbox")) {
 738             ButtonModel m = (ButtonModel)model;
 739             if (m.isSelected()) {
 740                 value = (String) attr.getAttribute(HTML.Attribute.VALUE);
 741                 if (value == null) {
 742                     value = "on";
 743                 }
 744             }
 745         } else if (type.equals("file")) {
 746             Document doc = (Document)model;
 747             String path;
 748 
 749             try {
 750                 path = doc.getText(0, doc.getLength());
 751             } catch (BadLocationException e) {
 752                 path = null;
 753             }
 754             if (path != null && path.length() > 0) {
 755                 value = path;
 756             }
 757         }
 758         return value;
 759     }

 761     /**
 762      * Returns the data associated with the &lt;TEXTAREA&gt; form
 763      * element.  This is done by getting the text stored in the
 764      * Document model.
 765      */
 766     private String getTextAreaData(AttributeSet attr) {
 767         Document doc = (Document)attr.getAttribute(StyleConstants.ModelAttribute);
 768         try {
 769             return doc.getText(0, doc.getLength());
 770         } catch (BadLocationException e) {
 771             return null;
 772         }
 773     }

 776     /**
 777      * Loads the buffer with the data associated with the Select
 778      * form element.  Basically, only items that are selected
 779      * and have their name attribute set are added to the buffer.
 780      */
 781     private void loadSelectData(AttributeSet attr, StringBuilder buffer) {
 782 
 783         String name = (String)attr.getAttribute(HTML.Attribute.NAME);
 784         if (name == null) {
 785             return;
 786         }
 787         Object m = attr.getAttribute(StyleConstants.ModelAttribute);
 788         if (m instanceof OptionListModel) {
 789             OptionListModel<Option> model = (OptionListModel<Option>) m;
 790 
 791             for (int i = 0; i < model.getSize(); i++) {
 792                 if (model.isSelectedIndex(i)) {
 793                     Option option = model.getElementAt(i);
 794                     appendBuffer(buffer, name, option.getValue());
 795                 }
 796             }
 797         } else if (m instanceof ComboBoxModel) {
 798             ComboBoxModel model = (ComboBoxModel)m;
 799             Option option = (Option)model.getSelectedItem();
 800             if (option != null) {
 801                 appendBuffer(buffer, name, option.getValue());
 802             }
 803         }
 804     }

 806     /**
 807      * Appends name / value pairs into the
 808      * buffer.  Both names and values are encoded using the
 809      * URLEncoder.encode() method before being added to the
 810      * buffer.
 811      */
 812     private void appendBuffer(StringBuilder buffer, String name, String value) {
 813         if (buffer.length() > 0) {
 814             buffer.append('&');
 815         }
 816         String encodedName = URLEncoder.encode(name);
 817         buffer.append(encodedName);
 818         buffer.append('=');
 819         String encodedValue = URLEncoder.encode(value);
 820         buffer.append(encodedValue);
 821     }

 830     /**
 831      * Iterates over the element hierarchy to determine if
 832      * the element parameter, which is assumed to be an
 833      * &lt;INPUT&gt; element of type password or text, is the last
 834      * one of either kind, in the form to which it belongs.
 835      */
 836     boolean isLastTextOrPasswordField() {
 837         Element parent = getFormElement();
 838         Element elem = getElement();
 839 
 840         if (parent != null) {
 841             ElementIterator it = new ElementIterator(parent);
 842             Element next;
 843             boolean found = false;
 844 
 845             while ((next = it.next()) != null) {
 846                 if (next == elem) {
 847                     found = true;
 848                 }
 849                 else if (found && isControl(next)) {
 850                     AttributeSet elemAttr = next.getAttributes();
 851 
 852                     if (HTMLDocument.matchNameAttribute
 853                                      (elemAttr, HTML.Tag.INPUT)) {
 854                         String type = (String)elemAttr.getAttribute
 855                                                   (HTML.Attribute.TYPE);
 856 
 857                         if ("text".equals(type) || "password".equals(type)) {
 858                             return false;
 859                         }
 860                     }
 861                 }
 862             }
 863         }
 864         return true;
 865     }

 867     /**
 868      * Resets the form
 869      * to its initial state by reinitializing the models
 870      * associated with each form element to their initial
 871      * values.
 872      *
 873      * param elem the element that triggered the reset
 874      */
 875     void resetForm() {
 876         Element parent = getFormElement();
 877 
 878         if (parent != null) {
 879             ElementIterator it = new ElementIterator(parent);
 880             Element next;
 881 
 882             while((next = it.next()) != null) {
 883                 if (isControl(next)) {
 884                     AttributeSet elemAttr = next.getAttributes();
 885                     Object m = elemAttr.getAttribute(StyleConstants.
 886                                                      ModelAttribute);
 887                     if (m instanceof TextAreaDocument) {
 888                         TextAreaDocument doc = (TextAreaDocument)m;
 889                         doc.reset();
 890                     } else if (m instanceof PlainDocument) {
 891                         try {
 892                             PlainDocument doc =  (PlainDocument)m;
 893                             doc.remove(0, doc.getLength());
 894                             if (HTMLDocument.matchNameAttribute
 895                                              (elemAttr, HTML.Tag.INPUT)) {
 896                                 String value = (String)elemAttr.
 897                                            getAttribute(HTML.Attribute.VALUE);
 898                                 if (value != null) {
 899                                     doc.insertString(0, value, null);
 900                                 }
 901                             }
 902                         } catch (BadLocationException e) {
 903                         }
 904                     } else if (m instanceof OptionListModel) {
 905                         OptionListModel model = (OptionListModel) m;
 906                         int size = model.getSize();
 907                         for (int i = 0; i < size; i++) {
 908                             model.removeIndexInterval(i, i);
 909                         }
 910                         BitSet selectionRange = model.getInitialSelection();
 911                         for (int i = 0; i < selectionRange.size(); i++) {
 912                             if (selectionRange.get(i)) {
 913                                 model.addSelectionInterval(i, i);
 914                             }
 915                         }
 916                     } else if (m instanceof OptionComboBoxModel) {
 917                         OptionComboBoxModel model = (OptionComboBoxModel) m;
 918                         Option option = model.getInitialSelection();
 919                         if (option != null) {
 920                             model.setSelectedItem(option);
 921                         }
 922                     } else if (m instanceof JToggleButton.ToggleButtonModel) {
 923                         boolean checked = ((String)elemAttr.getAttribute
 924                                            (HTML.Attribute.CHECKED) != null);
 925                         JToggleButton.ToggleButtonModel model =
 926                                         (JToggleButton.ToggleButtonModel)m;
 927                         model.setSelected(checked);
 928                     }
 929                 }
 930             }
 931         }
 932     }
