  51         PARSE,

  59         ANALYZE,

  63         GENERATE,
  64         /**
  65          * For events relating to overall annotation processing.
  66          **/
  67         ANNOTATION_PROCESSING,
  68         /**
  69          * For events relating to an individual annotation processing round.
  70          **/
  71         ANNOTATION_PROCESSING_ROUND
  72     };
