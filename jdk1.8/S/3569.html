 234      * <p>Descriptor constructor taking an XML String.</p>
 235      *
 236      * <p>The format of the XML string is not defined, but an
 237      * implementation must ensure that the string returned by
 238      * {@link #toXMLString() toXMLString()} on an existing
 239      * descriptor can be used to instantiate an equivalent
 240      * descriptor using this constructor.</p>
 241      *
 242      * <p>In this implementation, all field values will be created
 243      * as Strings.  If the field values are not Strings, the
 244      * programmer will have to reset or convert these fields
 245      * correctly.</p>
 246      *
 247      * @param inStr An XML-formatted string used to populate this
 248      * Descriptor.  The format is not defined, but any
 249      * implementation must ensure that the string returned by
 250      * method {@link #toXMLString toXMLString} on an existing
 251      * descriptor can be used to instantiate an equivalent
 252      * descriptor when instantiated using this constructor.
 253      *
 254      * @exception RuntimeOperationsException If the String inStr
 255      * passed in parameter is null
 256      * @exception XMLParseException XML parsing problem while parsing
 257      * the input String
 258      * @exception MBeanException Wraps a distributed communication Exception.
 259      */
 260     /* At some stage we should rewrite this code to be cleverer.  Using
 261        a StringTokenizer as we do means, first, that we accept a lot of
 262        bogus strings without noticing they are bogus, and second, that we
 263        split the string being parsed at characters like > even if they
 264        occur in the middle of a field value. */
 265     public DescriptorSupport(String inStr)
 266             throws MBeanException, RuntimeOperationsException,
 267                    XMLParseException {
 268         /* parse an XML-formatted string and populate internal
 269          * structure with it */
 270         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 271             MODELMBEAN_LOGGER.logp(Level.FINEST,
 272                     DescriptorSupport.class.getName(),
 273                     "Descriptor(String = '" + inStr + "')", "Constructor");
 274         }
 275         if (inStr == null) {
 276             if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 277                 MODELMBEAN_LOGGER.logp(Level.FINEST,
 278                         DescriptorSupport.class.getName(),
 279                         "Descriptor(String = null)", "Illegal arguments");
 280             }
 281             final String msg = "String in parameter is null";
 282             final RuntimeException iae = new IllegalArgumentException(msg);
 283             throw new RuntimeOperationsException(iae, msg);
 284         }
 285 
 286         final String lowerInStr = inStr.toLowerCase();
 287         if (!lowerInStr.startsWith("<descriptor>")
 288             || !lowerInStr.endsWith("</descriptor>")) {
 289             throw new XMLParseException("No <descriptor>, </descriptor> pair");
 290         }
 291 
 292         // parse xmlstring into structures
 293         init(null);
 294         // create dummy descriptor: should have same size
 295         // as number of fields in xmlstring
 296         // loop through structures and put them in descriptor
 297 
 298         StringTokenizer st = new StringTokenizer(inStr, "<> \t\n\r\f");
 299 
 300         boolean inFld = false;
 301         boolean inDesc = false;
 302         String fieldName = null;
 303         String fieldValue = null;
 304 
 305 
 306         while (st.hasMoreTokens()) {  // loop through tokens
 307             String tok = st.nextToken();
 308 
 309             if (tok.equalsIgnoreCase("FIELD")) {
 310                 inFld = true;
 311             } else if (tok.equalsIgnoreCase("/FIELD")) {
 312                 if ((fieldName != null) && (fieldValue != null)) {
 313                     fieldName =
 314                         fieldName.substring(fieldName.indexOf('"') + 1,
 315                                             fieldName.lastIndexOf('"'));
 316                     final Object fieldValueObject =
 317                         parseQuotedFieldValue(fieldValue);
 318                     setField(fieldName, fieldValueObject);
 319                 }
 320                 fieldName = null;
 321                 fieldValue = null;
 322                 inFld = false;
 323             } else if (tok.equalsIgnoreCase("DESCRIPTOR")) {
 324                 inDesc = true;
 325             } else if (tok.equalsIgnoreCase("/DESCRIPTOR")) {
 326                 inDesc = false;
 327                 fieldName = null;
 328                 fieldValue = null;
 329                 inFld = false;
 330             } else if (inFld && inDesc) {
 331                 // want kw=value, eg, name="myname" value="myvalue"
 332                 int eq_separator = tok.indexOf("=");
 333                 if (eq_separator > 0) {
 334                     String kwPart = tok.substring(0,eq_separator);
 335                     String valPart = tok.substring(eq_separator+1);
 336                     if (kwPart.equalsIgnoreCase("NAME"))
 337                         fieldName = valPart;
 338                     else if (kwPart.equalsIgnoreCase("VALUE"))
 339                         fieldValue = valPart;
 340                     else {  // xml parse exception
 341                         final String msg =
 342                             "Expected `name' or `value', got `" + tok + "'";
 343                         throw new XMLParseException(msg);
 344                     }
 345                 } else { // xml parse exception
 346                     final String msg =
 347                         "Expected `keyword=value', got `" + tok + "'";
 348                     throw new XMLParseException(msg);
 349                 }
 350             }
 351         }  // while tokens
 352         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 353             MODELMBEAN_LOGGER.logp(Level.FINEST,
 354                     DescriptorSupport.class.getName(),
 355                     "Descriptor(XMLString)", "Exit");
 356         }
 357     }

 417     /**
 418      * Constructor taking fields in the <i>fieldName=fieldValue</i>
 419      * format.
 420      *
 421      * @param fields String array with each element containing a
 422      * field name and value.  If this array is null or empty, then the
 423      * default constructor will be executed. Null strings or empty
 424      * strings will be ignored.
 425      *
 426      * <p>All field values should be Strings.  If the field values are
 427      * not Strings, the programmer will have to reset or convert these
 428      * fields correctly.
 429      *
 430      * <p>Note: Each string should be of the form
 431      * <i>fieldName=fieldValue</i>.  The field name
 432      * ends at the first {@code =} character; for example if the String
 433      * is {@code a=b=c} then the field name is {@code a} and its value
 434      * is {@code b=c}.
 435      *
 436      * @exception RuntimeOperationsException for illegal value for
 437      * field Names or field Values.  The field must contain an
 438      * "=". "=fieldValue", "fieldName", and "fieldValue" are illegal.
 439      * FieldName cannot be null.  "fieldName=" will cause the value to
 440      * be null.  If the descriptor construction fails for any reason,
 441      * this exception will be thrown.
 442      *
 443      */
 444     public DescriptorSupport(String... fields)
 445     {
 446         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 447             MODELMBEAN_LOGGER.logp(Level.FINEST,
 448                     DescriptorSupport.class.getName(),
 449                     "Descriptor(String... fields)", "Constructor");
 450         }
 451         init(null);
 452         if (( fields == null ) || ( fields.length == 0))
 453             return;
 454 
 455         init(null);
 456 
 457         for (int i=0; i < fields.length; i++) {
 458             if ((fields[i] == null) || (fields[i].equals(""))) {
 459                 continue;
 460             }
 461             int eq_separator = fields[i].indexOf("=");
 462             if (eq_separator < 0) {
 463                 // illegal if no = or is first character
 464                 if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 465                     MODELMBEAN_LOGGER.logp(Level.FINEST,
 466                             DescriptorSupport.class.getName(),
 467                             "Descriptor(String... fields)",
 468                             "Illegal arguments: field does not have " +
 469                             "'=' as a name and value separator");
 470                 }
 471                 final String msg = "Field in invalid format: no equals sign";
 472                 final RuntimeException iae = new IllegalArgumentException(msg);
 473                 throw new RuntimeOperationsException(iae, msg);
 474             }
 475 
 476             String fieldName = fields[i].substring(0,eq_separator);
 477             String fieldValue = null;
 478             if (eq_separator < fields[i].length()) {
 479                 // = is not in last character
 480                 fieldValue = fields[i].substring(eq_separator+1);
 481             }
 482 
 483             if (fieldName.equals("")) {
 484                 if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 485                     MODELMBEAN_LOGGER.logp(Level.FINEST,
 486                             DescriptorSupport.class.getName(),
 487                             "Descriptor(String... fields)",
 488                             "Illegal arguments: fieldName is empty");
 489                 }
 490 
 491                 final String msg = "Field in invalid format: no fieldName";
 492                 final RuntimeException iae = new IllegalArgumentException(msg);
 493                 throw new RuntimeOperationsException(iae, msg);
 494             }
 495 
 496             setField(fieldName,fieldValue);
 497         }
 498         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 499             MODELMBEAN_LOGGER.logp(Level.FINEST,
 500                     DescriptorSupport.class.getName(),
 501                     "Descriptor(String... fields)", "Exit");
 502         }
 503     }

 539     public synchronized void setField(String fieldName, Object fieldValue)
 540             throws RuntimeOperationsException {
 541 
 542         // field name cannot be null or empty
 543         if ((fieldName == null) || (fieldName.equals(""))) {
 544             if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 545                 MODELMBEAN_LOGGER.logp(Level.FINEST,
 546                         DescriptorSupport.class.getName(),
 547                         "setField(fieldName,fieldValue)",
 548                         "Illegal arguments: null or empty field name");
 549             }
 550 
 551             final String msg = "Field name to be set is null or empty";
 552             final RuntimeException iae = new IllegalArgumentException(msg);
 553             throw new RuntimeOperationsException(iae, msg);
 554         }
 555 
 556         if (!validateField(fieldName, fieldValue)) {
 557             if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 558                 MODELMBEAN_LOGGER.logp(Level.FINEST,
 559                         DescriptorSupport.class.getName(),
 560                         "setField(fieldName,fieldValue)",
 561                         "Illegal arguments");
 562             }
 563 
 564             final String msg =
 565                 "Field value invalid: " + fieldName + "=" + fieldValue;
 566             final RuntimeException iae = new IllegalArgumentException(msg);
 567             throw new RuntimeOperationsException(iae, msg);
 568         }
 569 
 570         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 571             MODELMBEAN_LOGGER.logp(Level.FINEST,
 572                     DescriptorSupport.class.getName(),
 573                     "setField(fieldName,fieldValue)", "Entry: setting '"
 574                     + fieldName + "' to '" + fieldValue + "'");
 575         }
 576 
 577         // Since we do not remove any existing entry with this name,
 578         // the field will preserve whatever case it had, ignoring
 579         // any difference there might be in fieldName.
 580         descriptorMap.put(fieldName, fieldValue);
 581     }

 583     public synchronized String[] getFields() {
 584         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 585             MODELMBEAN_LOGGER.logp(Level.FINEST,
 586                     DescriptorSupport.class.getName(),
 587                     "getFields()", "Entry");
 588         }
 589         int numberOfEntries = descriptorMap.size();
 590 
 591         String[] responseFields = new String[numberOfEntries];
 592         Set<Map.Entry<String, Object>> returnedSet = descriptorMap.entrySet();
 593 
 594         int i = 0;
 595 
 596         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 597             MODELMBEAN_LOGGER.logp(Level.FINEST,
 598                     DescriptorSupport.class.getName(),
 599                     "getFields()", "Returning " + numberOfEntries + " fields");
 600         }
 601         for (Iterator<Map.Entry<String, Object>> iter = returnedSet.iterator();
 602              iter.hasNext(); i++) {
 603             Map.Entry<String, Object> currElement = iter.next();
 604 
 605             if (currElement == null) {
 606                 if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 607                     MODELMBEAN_LOGGER.logp(Level.FINEST,
 608                             DescriptorSupport.class.getName(),
 609                             "getFields()", "Element is null");
 610                 }
 611             } else {
 612                 Object currValue = currElement.getValue();
 613                 if (currValue == null) {
 614                     responseFields[i] = currElement.getKey() + "=";
 615                 } else {
 616                     if (currValue instanceof java.lang.String) {
 617                         responseFields[i] =
 618                             currElement.getKey() + "=" + currValue.toString();
 619                     } else {
 620                         responseFields[i] =
 621                             currElement.getKey() + "=(" +
 622                             currValue.toString() + ")";
 623                     }
 624                 }
 625             }
 626         }
 627 
 628         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
 629             MODELMBEAN_LOGGER.logp(Level.FINEST,
 630                     DescriptorSupport.class.getName(),
 631                     "getFields()", "Exit");
 632         }
 633 
 634         return responseFields;
 635     }

 974     private boolean validateField(String fldName, Object fldValue) {
 975         if ((fldName == null) || (fldName.equals("")))
 976             return false;
 977         String SfldValue = "";
 978         boolean isAString = false;
 979         if ((fldValue != null) && (fldValue instanceof java.lang.String)) {
 980             SfldValue = (String) fldValue;
 981             isAString = true;
 982         }
 983 
 984         boolean nameOrDescriptorType =
 985             (fldName.equalsIgnoreCase("Name") ||
 986              fldName.equalsIgnoreCase("DescriptorType"));
 987         if (nameOrDescriptorType ||
 988             fldName.equalsIgnoreCase("SetMethod") ||
 989             fldName.equalsIgnoreCase("GetMethod") ||
 990             fldName.equalsIgnoreCase("Role") ||
 991             fldName.equalsIgnoreCase("Class")) {
 992             if (fldValue == null || !isAString)
 993                 return false;
 994             if (nameOrDescriptorType && SfldValue.equals(""))
 995                 return false;
 996             return true;
 997         } else if (fldName.equalsIgnoreCase("visibility")) {
 998             long v;
 999             if ((fldValue != null) && (isAString)) {
1000                 v = toNumeric(SfldValue);
1001             } else if (fldValue instanceof java.lang.Integer) {
1002                 v = ((Integer)fldValue).intValue();
1003             } else return false;
1004 
1005             if (v >= 1 &&  v <= 4)
1006                 return true;
1007             else
1008                 return false;
1009         } else if (fldName.equalsIgnoreCase("severity")) {
1010 
1011             long v;
1012             if ((fldValue != null) && (isAString)) {
1013                 v = toNumeric(SfldValue);
1014             } else if (fldValue instanceof java.lang.Integer) {
1015                 v = ((Integer)fldValue).intValue();
1016             } else return false;
1017 
1018             return (v >= 0 && v <= 6);
1019         } else if (fldName.equalsIgnoreCase("PersistPolicy")) {
1020             return (((fldValue != null) && (isAString)) &&
1021                     ( SfldValue.equalsIgnoreCase("OnUpdate") ||
1022                       SfldValue.equalsIgnoreCase("OnTimer") ||
1023                       SfldValue.equalsIgnoreCase("NoMoreOftenThan") ||
1024                       SfldValue.equalsIgnoreCase("Always") ||
1025                       SfldValue.equalsIgnoreCase("Never") ||
1026                       SfldValue.equalsIgnoreCase("OnUnregister")));
1027         } else if (fldName.equalsIgnoreCase("PersistPeriod") ||
1028                    fldName.equalsIgnoreCase("CurrencyTimeLimit") ||
1029                    fldName.equalsIgnoreCase("LastUpdatedTimeStamp") ||
1030                    fldName.equalsIgnoreCase("LastReturnedTimeStamp")) {
1031 
1032             long v;
1033             if ((fldValue != null) && (isAString)) {
1034                 v = toNumeric(SfldValue);
1035             } else if (fldValue instanceof java.lang.Number) {
1036                 v = ((Number)fldValue).longValue();
1037             } else return false;
1038 
1039             return (v >= -1);
1040         } else if (fldName.equalsIgnoreCase("log")) {
1041             return ((fldValue instanceof java.lang.Boolean) ||
1042                     (isAString &&
1043                      (SfldValue.equalsIgnoreCase("T") ||
1044                       SfldValue.equalsIgnoreCase("true") ||
1045                       SfldValue.equalsIgnoreCase("F") ||
1046                       SfldValue.equalsIgnoreCase("false") )));
1047         }
1048 
1049         // default to true, it is a field we aren't validating (user etc.)
1050         return true;
1051     }

1055     /**
1056      * <p>Returns an XML String representing the descriptor.</p>
1057      *
1058      * <p>The format is not defined, but an implementation must
1059      * ensure that the string returned by this method can be
1060      * used to build an equivalent descriptor when instantiated
1061      * using the constructor {@link #DescriptorSupport(String)
1062      * DescriptorSupport(String inStr)}.</p>
1063      *
1064      * <p>Fields which are not String objects will have toString()
1065      * called on them to create the value. The value will be
1066      * enclosed in parentheses.  It is not guaranteed that you can
1067      * reconstruct these objects unless they have been
1068      * specifically set up to support toString() in a meaningful
1069      * format and have a matching constructor that accepts a
1070      * String in the same format.</p>
1071      *
1072      * <p>If the descriptor is empty the following String is
1073      * returned: &lt;Descriptor&gt;&lt;/Descriptor&gt;</p>
1074      *
1075      * @return the XML string.
1076      *
1077      * @exception RuntimeOperationsException for illegal value for
1078      * field Names or field Values.  If the XML formatted string
1079      * construction fails for any reason, this exception will be
1080      * thrown.
1081      */
1082     public synchronized String toXMLString() {
1083         final StringBuilder buf = new StringBuilder("<Descriptor>");
1084         Set<Map.Entry<String, Object>> returnedSet = descriptorMap.entrySet();
1085         for (Map.Entry<String, Object> currElement : returnedSet) {
1086             final String name = currElement.getKey();
1087             Object value = currElement.getValue();
1088             String valueString = null;
1089             /* Set valueString to non-null if and only if this is a string that
1090                cannot be confused with the encoding of an object.  If it
1091                could be so confused (surrounded by parentheses) then we
1092                call makeFieldValue as for any non-String object and end
1093                up with an encoding like "(java.lang.String/(thing))".  */
1094             if (value instanceof String) {
1095                 final String svalue = (String) value;
1096                 if (!svalue.startsWith("(") || !svalue.endsWith(")"))
1097                     valueString = quote(svalue);
1098             }
1099             if (valueString == null)
1100                 valueString = makeFieldValue(value);
1101             buf.append("<field name=\"").append(name).append("\" value=\"")
1102                 .append(valueString).append("\"></field>");
1103         }
1104         buf.append("</Descriptor>");
1105         return buf.toString();
1106     }

1108     private static final String[] entities = {
1109         " &#32;",
1110         "\"&quot;",
1111         "<&lt;",
1112         ">&gt;",
1113         "&&amp;",
1114         "\r&#13;",
1115         "\t&#9;",
1116         "\n&#10;",
1117         "\f&#12;",
1118     };
1119     private static final Map<String,Character> entityToCharMap =
1120         new HashMap<String,Character>();
1121     private static final String[] charToEntityMap;
1122 
1123     static {
1124         char maxChar = 0;
1125         for (int i = 0; i < entities.length; i++) {
1126             final char c = entities[i].charAt(0);
1127             if (c > maxChar)
1128                 maxChar = c;
1129         }
1130         charToEntityMap = new String[maxChar + 1];
1131         for (int i = 0; i < entities.length; i++) {
1132             final char c = entities[i].charAt(0);
1133             final String entity = entities[i].substring(1);
1134             charToEntityMap[c] = entity;
1135             entityToCharMap.put(entity, c);
1136         }
1137     }

1139     private static boolean isMagic(char c) {
1140         return (c < charToEntityMap.length && charToEntityMap[c] != null);
1141     }

1143     /*
1144      * Quote the string so that it will be acceptable to the (String)
1145      * constructor.  Since the parsing code in that constructor is fairly
1146      * stupid, we're obliged to quote apparently innocuous characters like
1147      * space, <, and >.  In a future version, we should rewrite the parser
1148      * and only quote " plus either \ or & (depending on the quote syntax).
1149      */
1150     private static String quote(String s) {
1151         boolean found = false;
1152         for (int i = 0; i < s.length(); i++) {
1153             if (isMagic(s.charAt(i))) {
1154                 found = true;
1155                 break;
1156             }
1157         }
1158         if (!found)
1159             return s;
1160         final StringBuilder buf = new StringBuilder();
1161         for (int i = 0; i < s.length(); i++) {
1162             char c = s.charAt(i);
1163             if (isMagic(c))
1164                 buf.append(charToEntityMap[c]);
1165             else
1166                 buf.append(c);
1167         }
1168         return buf.toString();
1169     }

1171     private static String unquote(String s) throws XMLParseException {
1172         if (!s.startsWith("\"") || !s.endsWith("\""))
1173             throw new XMLParseException("Value must be quoted: <" + s + ">");
1174         final StringBuilder buf = new StringBuilder();
1175         final int len = s.length() - 1;
1176         for (int i = 1; i < len; i++) {
1177             final char c = s.charAt(i);
1178             final int semi;
1179             final Character quoted;
1180             if (c == '&'
1181                 && (semi = s.indexOf(';', i + 1)) >= 0
1182                 && ((quoted = entityToCharMap.get(s.substring(i, semi+1)))
1183                     != null)) {
1184                 buf.append(quoted);
1185                 i = semi;
1186             } else
1187                 buf.append(c);
1188         }
1189         return buf.toString();
1190     }

1197     private static String makeFieldValue(Object value) {
1198         if (value == null)
1199             return "(null)";
1200

1202         try {
1203             valueClass.getConstructor(String.class);

1209             throw new RuntimeOperationsException(iae,
1210                                                  "Cannot make XML descriptor");

1213             // too bad if it's not: we'll find out when we try to
1214             // reconstruct the DescriptorSupport
1215         }
1216 
1217         final String quotedValueString = quote(value.toString());
1218

1222     /*
1223      * Parse a field value from the XML produced by toXMLString().
1224      * Given a descriptor XML containing <field name="nnn" value="vvv">,
1225      * the argument to this method will be "vvv" (a string including the
1226      * containing quote characters).  If vvv begins and ends with parentheses,
1227      * then it may contain:
1228      * - the characters "null", in which case the result is null;
1229      * - a value of the form "some.class.name/xxx", in which case the
1230      * result is equivalent to `new some.class.name("xxx")';
1231      * - some other string, in which case the result is that string,
1232      * without the parentheses.
1233      */
1234     private static Object parseQuotedFieldValue(String s)
1235             throws XMLParseException {
1236         s = unquote(s);
1237         if (s.equalsIgnoreCase("(null)"))
1238             return null;
1239         if (!s.startsWith("(") || !s.endsWith(")"))
1240             return s;
1241         final int slash = s.indexOf('/');
1242         if (slash < 0) {
1243             // compatibility: old code didn't include class name
1244             return s.substring(1, s.length() - 1);
1245         }
1246         final String className = s.substring(1, slash);
1247 
1248         final Constructor<?> constr;
1249         try {
1250             ReflectUtil.checkPackageAccess(className);
1251             final ClassLoader contextClassLoader =
1252                 Thread.currentThread().getContextClassLoader();
1253             final Class<?> c =
1254                 Class.forName(className, false, contextClassLoader);
1255             constr = c.getConstructor(new Class<?>[] {String.class});
1256         } catch (Exception e) {
1257             throw new XMLParseException(e,
1258                                         "Cannot parse value: <" + s + ">");
1259         }
1260         final String arg = s.substring(slash + 1, s.length() - 1);
1261         try {
1262             return constr.newInstance(new Object[] {arg});
1263         } catch (Exception e) {
1264             final String msg =
1265                 "Cannot construct instance of " + className +
1266                 " with arg: <" + s + ">";
1267             throw new XMLParseException(e, msg);
1268         }
1269     }

1271     /**
1272      * Returns a human readable string representing the
1273      * descriptor.  The string will be in the format of
1274      * "fieldName=fieldValue,fieldName2=fieldValue2,..."<br>
1275      *
1276      * If there are no fields in the descriptor, then an empty String
1277      * is returned.<br>
1278      *
1279      * If a fieldValue is an object then the toString() method is
1280      * called on it and its returned value is used as the value for
1281      * the field enclosed in parenthesis.
1282      *
1283      * @exception RuntimeOperationsException for illegal value for
1284      * field Names or field Values.  If the descriptor string fails
1285      * for any reason, this exception will be thrown.
1286      */
1287     @Override
1288     public synchronized String toString() {
1289         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
1290             MODELMBEAN_LOGGER.logp(Level.FINEST,
1291                     DescriptorSupport.class.getName(),
1292                     "toString()", "Entry");
1293         }
1294 
1295         String respStr = "";
1296         String[] fields = getFields();
1297 
1298         if ((fields == null) || (fields.length == 0)) {
1299             if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
1300                 MODELMBEAN_LOGGER.logp(Level.FINEST,
1301                         DescriptorSupport.class.getName(),
1302                         "toString()", "Empty Descriptor");
1303             }
1304             return respStr;
1305         }
1306 
1307         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
1308             MODELMBEAN_LOGGER.logp(Level.FINEST,
1309                     DescriptorSupport.class.getName(),
1310                     "toString()", "Printing " + fields.length + " fields");
1311         }
1312 
1313         for (int i=0; i < fields.length; i++) {
1314             if (i == (fields.length - 1)) {
1315                 respStr = respStr.concat(fields[i]);
1316             } else {
1317                 respStr = respStr.concat(fields[i] + ", ");
1318             }
1319         }
1320 
1321         if (MODELMBEAN_LOGGER.isLoggable(Level.FINEST)) {
1322             MODELMBEAN_LOGGER.logp(Level.FINEST,
1323                     DescriptorSupport.class.getName(),
1324                     "toString()", "Exit returning " + respStr);
1325         }
1326 
1327         return respStr;
1328     }

1356     /**
1357      * Serializes a {@link DescriptorSupport} to an {@link ObjectOutputStream}.
1358      */
1359     /* If you set jmx.serial.form to "1.2.0" or "1.2.1", then we are
1360        bug-compatible with those versions.  Specifically, field names
1361        are forced to lower-case before being written.  This
1362        contradicts the spec, which, though it does not mention
1363        serialization explicitly, does say that the case of field names
1364        is preserved.  But in 1.2.0 and 1.2.1, this requirement was not
1365        met.  Instead, field names in the descriptor map were forced to
1366        lower case.  Those versions expect this to have happened to a
1367        descriptor they deserialize and e.g. getFieldValue will not
1368        find a field whose name is spelt with a different case.
1369     */
1370     private void writeObject(ObjectOutputStream out) throws IOException {
1371         ObjectOutputStream.PutField fields = out.putFields();
1372         boolean compat = "1.0".equals(serialForm);
1373         if (compat)
1374             fields.put("currClass", currClass);
1375 
1376         /* Purge the field "targetObject" from the DescriptorSupport before
1377          * serializing since the referenced object is typically not
1378          * serializable.  We do this here rather than purging the "descriptor"
1379          * variable below because that HashMap doesn't do case-insensitivity.
1380          * See CR 6332962.
1381          */
1382         SortedMap<String, Object> startMap = descriptorMap;
1383         if (startMap.containsKey("targetObject")) {
1384             startMap = new TreeMap<String, Object>(descriptorMap);
1385             startMap.remove("targetObject");
1386         }
1387 
1388         final HashMap<String, Object> descriptor;
1389         if (compat || "1.2.0".equals(serialForm) ||
1390                 "1.2.1".equals(serialForm)) {
1391             descriptor = new HashMap<String, Object>();
1392             for (Map.Entry<String, Object> entry : startMap.entrySet())
1393                 descriptor.put(entry.getKey().toLowerCase(), entry.getValue());
1394         } else
1395             descriptor = new HashMap<String, Object>(startMap);
1396 
1397         fields.put("descriptor", descriptor);
1398         out.writeFields();
1399     }
