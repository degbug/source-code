 102         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 103         componentColors = extendedCacheKeys;
 104         //generate this entire method. Each state/bg/fg/border combo that has
 105         //been painted gets its own KEY and paint method.
 106         switch(state) {
 107             case LEAFICON_ENABLED: paintleafIconEnabled(g); break;
 108             case CLOSEDICON_ENABLED: paintclosedIconEnabled(g); break;
 109             case OPENICON_ENABLED: paintopenIconEnabled(g); break;

 111             case COLLAPSEDICON_ENABLED_SELECTED: paintcollapsedIconEnabledAndSelected(g); break;
 112             case EXPANDEDICON_ENABLED: paintexpandedIconEnabled(g); break;
 113             case EXPANDEDICON_ENABLED_SELECTED: paintexpandedIconEnabledAndSelected(g); break;
 114 
 115         }
 116     }
