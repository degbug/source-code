 138      * Package-private routine (exposed to java.lang.Class via
 139      * ReflectAccess) which returns a copy of this Method. The copy's
 140      * "root" field points to this Method.
 141      */
 142     Method copy() {
 143         // This routine enables sharing of MethodAccessor objects
 144         // among Method objects which refer to the same underlying
 145         // method in the VM. (All of this contortion is only necessary
 146         // because of the "accessibility" bit in AccessibleObject,
 147         // which implicitly requires that new java.lang.reflect
 148         // objects be fabricated for each reflective call on Class
 149         // objects.)
 150         if (this.root != null)
 151             throw new IllegalArgumentException("Can not copy a non-root Method");
 152 
 153         Method res = new Method(clazz, name, parameterTypes, returnType,
 154                                 exceptionTypes, modifiers, slot, signature,
 155                                 annotations, parameterAnnotations, annotationDefault);
 156         res.root = this;
 157         // Might as well eagerly propagate this if already present
 158         res.methodAccessor = methodAccessor;
 159         return res;
 160     }

 425     /**
 426      * Invokes the underlying method represented by this {@code Method}
 427      * object, on the specified object with the specified parameters.
 428      * Individual parameters are automatically unwrapped to match
 429      * primitive formal parameters, and both primitive and reference
 430      * parameters are subject to method invocation conversions as
 431      * necessary.
 432      *
 433      * <p>If the underlying method is static, then the specified {@code obj}
 434      * argument is ignored. It may be null.
 435      *
 436      * <p>If the number of formal parameters required by the underlying method is
 437      * 0, the supplied {@code args} array may be of length 0 or null.
 438      *
 439      * <p>If the underlying method is an instance method, it is invoked
 440      * using dynamic method lookup as documented in The Java Language
 441      * Specification, Second Edition, section 15.12.4.4; in particular,
 442      * overriding based on the runtime type of the target object will occur.
 443      *
 444      * <p>If the underlying method is static, the class that declared
 445      * the method is initialized if it has not already been initialized.
 446      *
 447      * <p>If the method completes normally, the value it returns is
 448      * returned to the caller of invoke; if the value has a primitive
 449      * type, it is first appropriately wrapped in an object. However,
 450      * if the value has the type of an array of a primitive type, the
 451      * elements of the array are <i>not</i> wrapped in objects; in
 452      * other words, an array of primitive type is returned.  If the
 453      * underlying method return type is void, the invocation returns
 454      * null.
 455      *
 456      * @param obj  the object the underlying method is invoked from
 457      * @param args the arguments used for the method call
 458      * @return the result of dispatching the method represented by
 459      * this object on {@code obj} with parameters
 460      * {@code args}
 461      *
 462      * @exception IllegalAccessException    if this {@code Method} object
 463      *              is enforcing Java language access control and the underlying
 464      *              method is inaccessible.
 465      * @exception IllegalArgumentException  if the method is an
 466      *              instance method and the specified object argument
 467      *              is not an instance of the class or interface
 468      *              declaring the underlying method (or of a subclass
 469      *              or implementor thereof); if the number of actual
 470      *              and formal parameters differ; if an unwrapping
 471      *              conversion for primitive arguments fails; or if,
 472      *              after possible unwrapping, a parameter value
 473      *              cannot be converted to the corresponding formal
 474      *              parameter type by a method invocation conversion.
 475      * @exception InvocationTargetException if the underlying method
 476      *              throws an exception.
 477      * @exception NullPointerException      if the specified object is null
 478      *              and the method is an instance method.
 479      * @exception ExceptionInInitializerError if the initialization
 480      * provoked by this method fails.
 481      */
 482     @CallerSensitive
 483     public Object invoke(Object obj, Object... args)
 484         throws IllegalAccessException, IllegalArgumentException,
 485            InvocationTargetException
 486     {
 487         if (!override) {
 488             if (!Reflection.quickCheckMemberAccess(clazz, modifiers)) {
 489                 Class<?> caller = Reflection.getCallerClass();
 490                 checkAccess(caller, clazz, obj, modifiers);
 491             }
 492         }
 493         MethodAccessor ma = methodAccessor;             // read volatile
 494         if (ma == null) {
 495             ma = acquireMethodAccessor();
 496         }
 497         return ma.invoke(obj, args);
 498     }

 536      * is, a non-static method with a body, declared in an interface

 544         // Default methods are public non-abstract instance methods

 550     // NOTE that there is no synchronization used here. It is correct
 551     // (though not efficient) to generate more than one MethodAccessor
 552     // for a given Method. However, avoiding synchronization will
 553     // probably make the implementation more scalable.
 554     private MethodAccessor acquireMethodAccessor() {
 555         // First check to see if one has been created yet, and take it
 556         // if so
 557         MethodAccessor tmp = null;
 558         if (root != null) tmp = root.getMethodAccessor();
 559         if (tmp != null) {
 560             methodAccessor = tmp;
 561         } else {
 562             // Otherwise fabricate one and propagate it up to the root
 563             tmp = reflectionFactory.newMethodAccessor(this);
 564             setMethodAccessor(tmp);
 565         }
 566 
 567         return tmp;
 568     }

 586     /**
 587      * Returns the default value for the annotation member represented by
 588      * this {@code Method} instance.  If the member is of a primitive type,
 589      * an instance of the corresponding wrapper type is returned. Returns
 590      * null if no default is associated with the member, or if the method
 591      * instance does not represent a declared member of an annotation type.
 592      *
 593      * @return the default value for the annotation member represented
 594      *     by this {@code Method} instance.
 595      * @throws TypeNotPresentException if the annotation is of type
 596      *     {@link Class} and no definition can be found for the
 597      *     default class value.
 598      * @since  1.5
 599      */
 600     public Object getDefaultValue() {
 601         if  (annotationDefault == null)
 602             return null;
 603         Class<?> memberType = AnnotationType.invocationHandlerReturnType(
 604             getReturnType());
 605         Object result = AnnotationParser.parseMemberValue(
 606             memberType, ByteBuffer.wrap(annotationDefault),
 607             sun.misc.SharedSecrets.getJavaLangAccess().
 608                 getConstantPool(getDeclaringClass()),
 609             getDeclaringClass());
 610         if (result instanceof sun.reflect.annotation.ExceptionProxy)
 611             throw new AnnotationFormatError("Invalid default: " + this);
 612         return result;
 613     }
