 326     private void emitConst(Object con) {
 327         if (con == null) {
 328             mv.visitInsn(Opcodes.ACONST_NULL);
 329             return;
 330         }
 331         if (con instanceof Integer) {
 332             emitIconstInsn((int) con);
 333             return;
 334         }
 335         if (con instanceof Long) {
 336             long x = (long) con;
 337             if (x == (short) x) {
 338                 emitIconstInsn((int) x);
 339                 mv.visitInsn(Opcodes.I2L);
 340                 return;
 341             }
 342         }
 343         if (con instanceof Float) {
 344             float x = (float) con;
 345             if (x == (short) x) {
 346                 emitIconstInsn((int) x);
 347                 mv.visitInsn(Opcodes.I2F);
 348                 return;
 349             }
 350         }
 351         if (con instanceof Double) {
 352             double x = (double) con;
 353             if (x == (short) x) {
 354                 emitIconstInsn((int) x);
 355                 mv.visitInsn(Opcodes.I2D);
 356                 return;
 357             }
 358         }
 359         if (con instanceof Boolean) {
 360             emitIconstInsn((boolean) con ? 1 : 0);
 361             return;
 362         }
 363         // fall through:
 364         mv.visitLdcInsn(con);
 365     }

 367     private void emitIconstInsn(int i) {
 368         int opcode;
 369         switch (i) {
 370         case 0:  opcode = Opcodes.ICONST_0;  break;
 371         case 1:  opcode = Opcodes.ICONST_1;  break;
 372         case 2:  opcode = Opcodes.ICONST_2;  break;
 373         case 3:  opcode = Opcodes.ICONST_3;  break;
 374         case 4:  opcode = Opcodes.ICONST_4;  break;
 375         case 5:  opcode = Opcodes.ICONST_5;  break;
 376         default:
 377             if (i == (byte) i) {
 378                 mv.visitIntInsn(Opcodes.BIPUSH, i & 0xFF);
 379             } else if (i == (short) i) {
 380                 mv.visitIntInsn(Opcodes.SIPUSH, (char) i);
 381             } else {
 382                 mv.visitLdcInsn(i);
 383             }
 384             return;
 385         }
 386         mv.visitInsn(opcode);
 387     }

 447     private int arrayInsnOpcode(byte tcode, int aaop) throws InternalError {
 448         assert(aaop == Opcodes.AASTORE || aaop == Opcodes.AALOAD);
 449         int xas;
 450         switch (tcode) {
 451             case Opcodes.T_BOOLEAN: xas = Opcodes.BASTORE; break;

 453             case Opcodes.T_CHAR:    xas = Opcodes.CASTORE; break;

 455             case Opcodes.T_INT:     xas = Opcodes.IASTORE; break;

 467         int i = indexForFrameLocal(oldFrameLocal);
 468         if (i < 0)  return;
 469         BasicType type = localTypes[i];
 470         int newFrameLocal = makeLocalTemp(type);
 471         mv.visitVarInsn(loadInsnOpcode(type), oldFrameLocal);

 517      * This is usually a no-op, except when pclass is a subword type or a reference other than Object or an interface.

 523     private void emitImplicitConversion(BasicType ptype, Class<?> pclass, Object arg) {
 524         assert(basicType(pclass) == ptype);  // boxing/unboxing handled by caller
 525         if (pclass == ptype.basicTypeClass() && ptype != L_TYPE)
 526             return;   // nothing to do

 529                 if (VerifyType.isNullConversion(Object.class, pclass, false)) {
 530                     if (PROFILE_LEVEL > 0)

 540         }
 541         throw newInternalError("bad implicit conversion: tc="+ptype+": "+pclass);

 556     private void emitReferenceCast(Class<?> cls, Object arg) {
 557         Name writeBack = null;  // local to write back result
 558         if (arg instanceof Name) {
 559             Name n = (Name) arg;
 560             if (assertStaticType(cls, n))
 561                 return;  // this cast was already performed
 562             if (lambdaForm.useCount(n) > 1) {
 563                 // This guy gets used more than once.
 564                 writeBack = n;
 565             }
 566         }
 567         if (isStaticallyNameable(cls)) {
 568             String sig = getInternalName(cls);
 569             mv.visitTypeInsn(Opcodes.CHECKCAST, sig);
 570         } else {
 571             mv.visitLdcInsn(constantPlaceholder(cls));
 572             mv.visitTypeInsn(Opcodes.CHECKCAST, CLS);
 573             mv.visitInsn(Opcodes.SWAP);
 574             mv.visitMethodInsn(Opcodes.INVOKESTATIC, MHI, "castReference", CLL_SIG, false);
 575             if (Object[].class.isAssignableFrom(cls))
 576                 mv.visitTypeInsn(Opcodes.CHECKCAST, OBJARY);
 577             else if (PROFILE_LEVEL > 0)
 578                 mv.visitTypeInsn(Opcodes.CHECKCAST, OBJ);
 579         }
 580         if (writeBack != null) {
 581             mv.visitInsn(Opcodes.DUP);
 582             emitAstoreInsn(writeBack.index());
 583         }
 584     }

 630     /**
 631      * Generate an invoker method for the passed {@link LambdaForm}.
 632      */
 633     private byte[] generateCustomizedCodeBytes() {
 634         classFilePrologue();
 635 
 636         // Suppress this method in backtraces displayed to the user.
 637         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
 638 
 639         // Mark this method as a compiled LambdaForm
 640         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Compiled;", true);
 641 
 642         if (lambdaForm.forceInline) {
 643             // Force inlining of this invoker method.
 644             mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
 645         } else {
 646             mv.visitAnnotation("Ljava/lang/invoke/DontInline;", true);
 647         }
 648 
 649         if (lambdaForm.customized != null) {
 650             // Since LambdaForm is customized for a particular MethodHandle, it's safe to substitute
 651             // receiver MethodHandle (at slot #0) with an embedded constant and use it instead.
 652             // It enables more efficient code generation in some situations, since embedded constants
 653             // are compile-time constants for JIT compiler.
 654             mv.visitLdcInsn(constantPlaceholder(lambdaForm.customized));
 655             mv.visitTypeInsn(Opcodes.CHECKCAST, MH);
 656             assert(checkActualReceiver()); // expects MethodHandle on top of the stack
 657             mv.visitVarInsn(Opcodes.ASTORE, localsMap[0]);
 658         }
 659 
 660         // iterate over the form's names, generating bytecode instructions for each
 661         // start iterating at the first name following the arguments
 662         Name onStack = null;
 663         for (int i = lambdaForm.arity; i < lambdaForm.names.length; i++) {
 664             Name name = lambdaForm.names[i];
 665 
 666             emitStoreResult(onStack);
 667             onStack = name;  // unless otherwise modified below
 668             MethodHandleImpl.Intrinsic intr = name.function.intrinsicName();
 669             switch (intr) {
 670                 case SELECT_ALTERNATIVE:
 671                     assert isSelectAlternative(i);
 672                     if (PROFILE_GWT) {
 673                         assert(name.arguments[0] instanceof Name &&
 674                                nameRefersTo((Name)name.arguments[0], MethodHandleImpl.class, "profileBoolean"));
 675                         mv.visitAnnotation("Ljava/lang/invoke/InjectedProfile;", true);
 676                     }
 677                     onStack = emitSelectAlternative(name, lambdaForm.names[i+1]);
 678                     i++;  // skip MH.invokeBasic of the selectAlternative result
 679                     continue;
 680                 case GUARD_WITH_CATCH:
 681                     assert isGuardWithCatch(i);
 682                     onStack = emitGuardWithCatch(i);
 683                     i = i+2; // Jump to the end of GWC idiom
 684                     continue;
 685                 case NEW_ARRAY:
 686                     Class<?> rtype = name.function.methodType().returnType();
 687                     if (isStaticallyNameable(rtype)) {
 688                         emitNewArray(name);
 689                         continue;
 690                     }
 691                     break;
 692                 case ARRAY_LOAD:
 693                     emitArrayLoad(name);
 694                     continue;
 695                 case ARRAY_STORE:
 696                     emitArrayStore(name);
 697                     continue;
 698                 case IDENTITY:
 699                     assert(name.arguments.length == 1);
 700                     emitPushArguments(name);
 701                     continue;
 702                 case ZERO:
 703                     assert(name.arguments.length == 0);
 704                     emitConst(name.type.basicTypeWrapper().zero());
 705                     continue;
 706                 case NONE:
 707                     // no intrinsic associated
 708                     break;
 709                 default:
 710                     throw newInternalError("Unknown intrinsic: "+intr);
 711             }
 712 
 713             MemberName member = name.function.member();
 714             if (isStaticallyInvocable(member)) {
 715                 emitStaticInvoke(member, name);
 716             } else {
 717                 emitInvoke(name);
 718             }
 719         }
 720 
 721         // return statement
 722         emitReturn(onStack);
 723 
 724         classFileEpilogue();
 725         bogusMethod(lambdaForm);
 726 
 727         final byte[] classFile = cw.toByteArray();
 728         maybeDump(className, classFile);
 729         return classFile;
 730     }

 787     static boolean isStaticallyInvocable(MemberName member) {
 788         if (member == null)  return false;
 789         if (member.isConstructor())  return false;
 790         Class<?> cls = member.getDeclaringClass();
 791         if (cls.isArray() || cls.isPrimitive())
 792             return false;  // FIXME
 793         if (cls.isAnonymousClass() || cls.isLocalClass())
 794             return false;  // inner class of some sort
 795         if (cls.getClassLoader() != MethodHandle.class.getClassLoader())
 796             return false;  // not on BCP
 797         if (ReflectUtil.isVMAnonymousClass(cls)) // FIXME: switch to supported API once it is added
 798             return false;
 799         MethodType mtype = member.getMethodOrFieldType();
 800         if (!isStaticallyNameable(mtype.returnType()))
 801             return false;
 802         for (Class<?> ptype : mtype.parameterArray())
 803             if (!isStaticallyNameable(ptype))
 804                 return false;
 805         if (!member.isPrivate() && VerifyAccess.isSamePackage(MethodHandle.class, cls))
 806             return true;   // in java.lang.invoke package
 807         if (member.isPublic() && isStaticallyNameable(cls))
 808             return true;
 809         return false;
 810     }

 812     static boolean isStaticallyNameable(Class<?> cls) {
 813         if (cls == Object.class)
 814             return true;
 815         while (cls.isArray())
 816             cls = cls.getComponentType();
 817         if (cls.isPrimitive())
 818             return true;  // int[].class, for example
 819         if (ReflectUtil.isVMAnonymousClass(cls)) // FIXME: switch to supported API once it is added
 820             return false;
 821         // could use VerifyAccess.isClassAccessible but the following is a safe approximation
 822         if (cls.getClassLoader() != Object.class.getClassLoader())
 823             return false;
 824         if (VerifyAccess.isSamePackage(MethodHandle.class, cls))
 825             return true;
 826         if (!Modifier.isPublic(cls.getModifiers()))
 827             return false;
 828         for (Class<?> pkgcls : STATICALLY_INVOCABLE_PACKAGES) {
 829             if (VerifyAccess.isSamePackage(pkgcls, cls))
 830                 return true;
 831         }
 832         return false;
 833     }

 839     /**
 840      * Emit an invoke for the given name, using the MemberName directly.
 841      */
 842     void emitStaticInvoke(MemberName member, Name name) {
 843         assert(member.equals(name.function.member()));
 844         Class<?> defc = member.getDeclaringClass();
 845         String cname = getInternalName(defc);
 846         String mname = member.getName();
 847         String mtype;
 848         byte refKind = member.getReferenceKind();
 849         if (refKind == REF_invokeSpecial) {
 850             // in order to pass the verifier, we need to convert this to invokevirtual in all cases
 851             assert(member.canBeStaticallyBound()) : member;
 852             refKind = REF_invokeVirtual;
 853         }
 854 
 855         if (member.getDeclaringClass().isInterface() && refKind == REF_invokeVirtual) {
 856             // Methods from Object declared in an interface can be resolved by JVM to invokevirtual kind.
 857             // Need to convert it back to invokeinterface to pass verification and make the invocation works as expected.
 858             refKind = REF_invokeInterface;
 859         }
 860 
 861         // push arguments
 862         emitPushArguments(name);
 863 
 864         // invocation
 865         if (member.isMethod()) {
 866             mtype = member.getMethodType().toMethodDescriptorString();
 867             mv.visitMethodInsn(refKindOpcode(refKind), cname, mname, mtype,
 868                                member.getDeclaringClass().isInterface());
 869         } else {
 870             mtype = MethodType.toFieldDescriptorString(member.getFieldType());
 871             mv.visitFieldInsn(refKindOpcode(refKind), cname, mname, mtype);
 872         }
 873         // Issue a type assertion for the result, so we can avoid casts later.
 874         if (name.type == L_TYPE) {
 875             Class<?> rtype = member.getInvocationType().returnType();
 876             assert(!rtype.isPrimitive());
 877             if (rtype != Object.class && !rtype.isInterface()) {
 878                 assertStaticType(rtype, name);
 879             }
 880         }
 881     }

 883     void emitNewArray(Name name) throws InternalError {
 884         Class<?> rtype = name.function.methodType().returnType();
 885         if (name.arguments.length == 0) {
 886             // The array will be a constant.
 887             Object emptyArray;
 888             try {
 889                 emptyArray = name.function.resolvedHandle.invoke();
 890             } catch (Throwable ex) {
 891                 throw newInternalError(ex);
 892             }
 893             assert(java.lang.reflect.Array.getLength(emptyArray) == 0);
 894             assert(emptyArray.getClass() == rtype);  // exact typing
 895             mv.visitLdcInsn(constantPlaceholder(emptyArray));
 896             emitReferenceCast(rtype, emptyArray);
 897             return;
 898         }
 899         Class<?> arrayElementType = rtype.getComponentType();
 900         assert(arrayElementType != null);
 901         emitIconstInsn(name.arguments.length);
 902         int xas = Opcodes.AASTORE;
 903         if (!arrayElementType.isPrimitive()) {
 904             mv.visitTypeInsn(Opcodes.ANEWARRAY, getInternalName(arrayElementType));
 905         } else {
 906             byte tc = arrayTypeCode(Wrapper.forPrimitiveType(arrayElementType));
 907             xas = arrayInsnOpcode(tc, xas);
 908             mv.visitIntInsn(Opcodes.NEWARRAY, tc);
 909         }
 910         // store arguments
 911         for (int i = 0; i < name.arguments.length; i++) {
 912             mv.visitInsn(Opcodes.DUP);
 913             emitIconstInsn(i);
 914             emitPushArgument(name, i);
 915             mv.visitInsn(xas);
 916         }
 917         // the array is left on the stack
 918         assertStaticType(rtype, name);
 919     }
 920     int refKindOpcode(byte refKind) {
 921         switch (refKind) {
 922         case REF_invokeVirtual:      return Opcodes.INVOKEVIRTUAL;
 923         case REF_invokeStatic:       return Opcodes.INVOKESTATIC;
 924         case REF_invokeSpecial:      return Opcodes.INVOKESPECIAL;
 925         case REF_invokeInterface:    return Opcodes.INVOKEINTERFACE;
 926         case REF_getField:           return Opcodes.GETFIELD;
 927         case REF_putField:           return Opcodes.PUTFIELD;
 928         case REF_getStatic:          return Opcodes.GETSTATIC;
 929         case REF_putStatic:          return Opcodes.PUTSTATIC;
 930         }
 931         throw new InternalError("refKind="+refKind);
 932     }

 975     /**
 976      * Check if i-th name is a call to MethodHandleImpl.selectAlternative.
 977      */
 978     private boolean isSelectAlternative(int pos) {
 979         // selectAlternative idiom:
 980         //   t_{n}:L=MethodHandleImpl.selectAlternative(...)
 981         //   t_{n+1}:?=MethodHandle.invokeBasic(t_{n}, ...)
 982         if (pos+1 >= lambdaForm.names.length)  return false;
 983         Name name0 = lambdaForm.names[pos];
 984         Name name1 = lambdaForm.names[pos+1];
 985         return nameRefersTo(name0, MethodHandleImpl.class, "selectAlternative") &&
 986                isInvokeBasic(name1) &&
 987                name1.lastUseIndex(name0) == 0 &&        // t_{n+1}:?=MethodHandle.invokeBasic(t_{n}, ...)
 988                lambdaForm.lastUseIndex(name0) == pos+1; // t_{n} is local: used only in t_{n+1}
 989     }

 991     /**
 992      * Check if i-th name is a start of GuardWithCatch idiom.
 993      */
 994     private boolean isGuardWithCatch(int pos) {
 995         // GuardWithCatch idiom:
 996         //   t_{n}:L=MethodHandle.invokeBasic(...)
 997         //   t_{n+1}:L=MethodHandleImpl.guardWithCatch(*, *, *, t_{n});
 998         //   t_{n+2}:?=MethodHandle.invokeBasic(t_{n+1})

1006                name1.lastUseIndex(name0) == 3 &&          // t_{n+1}:L=MethodHandleImpl.guardWithCatch(*, *, *, t_{n});
1007                lambdaForm.lastUseIndex(name0) == pos+1 && // t_{n} is local: used only in t_{n+1}
1008                name2.lastUseIndex(name1) == 1 &&          // t_{n+2}:?=MethodHandle.invokeBasic(t_{n+1})
1009                lambdaForm.lastUseIndex(name1) == pos+2;   // t_{n+1} is local: used only in t_{n+2}

1012     /**
1013      * Emit bytecode for the selectAlternative idiom.
1014      *
1015      * The pattern looks like (Cf. MethodHandleImpl.makeGuardWithTest):
1016      * 
{@code

1017      *   Lambda(a0:L,a1:I)=>{
1018      *     t2:I=foo.test(a1:I);
1019      *     t3:L=MethodHandleImpl.selectAlternative(t2:I,(MethodHandle(int)int),(MethodHandle(int)int));
1020      *     t4:I=MethodHandle.invokeBasic(t3:L,a1:I);t4:I}
1021      * }
1022      */
1023     private Name emitSelectAlternative(Name selectAlternativeName, Name invokeBasicName) {
1024         assert isStaticallyInvocable(invokeBasicName);
1025 
1026         Name receiver = (Name) invokeBasicName.arguments[0];
1027 
1028         Label L_fallback = new Label();
1029         Label L_done     = new Label();
1030 
1031         // load test result
1032         emitPushArgument(selectAlternativeName, 0);
1033 
1034         // if_icmpne L_fallback
1035         mv.visitJumpInsn(Opcodes.IFEQ, L_fallback);
1036 
1037         // invoke selectAlternativeName.arguments[1]
1038         Class<?>[] preForkClasses = localClasses.clone();
1039         emitPushArgument(selectAlternativeName, 1);  // get 2nd argument of selectAlternative
1040         emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
1041         emitStaticInvoke(invokeBasicName);
1042 
1043         // goto L_done
1044         mv.visitJumpInsn(Opcodes.GOTO, L_done);
1045 
1046         // L_fallback:
1047         mv.visitLabel(L_fallback);
1048 
1049         // invoke selectAlternativeName.arguments[2]
1050         System.arraycopy(preForkClasses, 0, localClasses, 0, preForkClasses.length);
1051         emitPushArgument(selectAlternativeName, 2);  // get 3rd argument of selectAlternative
1052         emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
1053         emitStaticInvoke(invokeBasicName);
1054 
1055         // L_done:
1056         mv.visitLabel(L_done);
1057         // for now do not bother to merge typestate; just reset to the dominator state
1058         System.arraycopy(preForkClasses, 0, localClasses, 0, preForkClasses.length);
1059 
1060         return invokeBasicName;  // return what's on stack
1061     }

1063     /**
1064       * Emit bytecode for the guardWithCatch idiom.
1065       *
1066       * The pattern looks like (Cf. MethodHandleImpl.makeGuardWithCatch):
1067       * 
{@code

1068       *  guardWithCatch=Lambda(a0:L,a1:L,a2:L,a3:L,a4:L,a5:L,a6:L,a7:L)=>{
1069       *    t8:L=MethodHandle.invokeBasic(a4:L,a6:L,a7:L);
1070       *    t9:L=MethodHandleImpl.guardWithCatch(a1:L,a2:L,a3:L,t8:L);
1071       *   t10:I=MethodHandle.invokeBasic(a5:L,t9:L);t10:I}
1072       * }
1073       *
1074       * It is compiled into bytecode equivalent of the following code:
1075       * 
{@code

1076       *  try {
1077       *      return a1.invokeBasic(a6, a7);
1078       *  } catch (Throwable e) {
1079       *      if (!a2.isInstance(e)) throw e;
1080       *      return a3.invokeBasic(ex, a6, a7);
1081       *  }}
1082       */
1083     private Name emitGuardWithCatch(int pos) {
1084         Name args    = lambdaForm.names[pos];
1085         Name invoker = lambdaForm.names[pos+1];
1086         Name result  = lambdaForm.names[pos+2];
1087 
1088         Label L_startBlock = new Label();
1089         Label L_endBlock = new Label();
1090         Label L_handler = new Label();
1091         Label L_done = new Label();
1092 
1093         Class<?> returnType = result.function.resolvedHandle.type().returnType();
1094         MethodType type = args.function.resolvedHandle.type()
1095                               .dropParameterTypes(0,1)
1096                               .changeReturnType(returnType);
1097 
1098         mv.visitTryCatchBlock(L_startBlock, L_endBlock, L_handler, "java/lang/Throwable");
1099 
1100         // Normal case
1101         mv.visitLabel(L_startBlock);
1102         // load target
1103         emitPushArgument(invoker, 0);
1104         emitPushArguments(args, 1); // skip 1st argument: method handle
1105         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", type.basicType().toMethodDescriptorString(), false);
1106         mv.visitLabel(L_endBlock);
1107         mv.visitJumpInsn(Opcodes.GOTO, L_done);
1108 
1109         // Exceptional case
1110         mv.visitLabel(L_handler);
1111 
1112         // Check exception's type
1113         mv.visitInsn(Opcodes.DUP);
1114         // load exception class
1115         emitPushArgument(invoker, 1);
1116         mv.visitInsn(Opcodes.SWAP);
1117         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, "java/lang/Class", "isInstance", "(Ljava/lang/Object;)Z", false);
1118         Label L_rethrow = new Label();
1119         mv.visitJumpInsn(Opcodes.IFEQ, L_rethrow);
1120 
1121         // Invoke catcher
1122         // load catcher
1123         emitPushArgument(invoker, 2);
1124         mv.visitInsn(Opcodes.SWAP);
1125         emitPushArguments(args, 1); // skip 1st argument: method handle
1126         MethodType catcherType = type.insertParameterTypes(0, Throwable.class);
1127         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", catcherType.basicType().toMethodDescriptorString(), false);
1128         mv.visitJumpInsn(Opcodes.GOTO, L_done);
1129

1131         mv.visitInsn(Opcodes.ATHROW);

1209     /**
1210      * Emit a type conversion bytecode casting from "from" to "to".
1211      */
1212     private void emitPrimCast(Wrapper from, Wrapper to) {
1213         // Here's how.
1214         // -   indicates forbidden
1215         // <-> indicates implicit
1216         //      to ----> boolean  byte     short    char     int      long     float    double
1217         // from boolean    <->        -        -        -        -        -        -        -
1218         //      byte        -       <->       i2s      i2c      <->      i2l      i2f      i2d
1219         //      short       -       i2b       <->      i2c      <->      i2l      i2f      i2d
1220         //      char        -       i2b       i2s      <->      <->      i2l      i2f      i2d
1221         //      int         -       i2b       i2s      i2c      <->      i2l      i2f      i2d
1222         //      long        -     l2i,i2b   l2i,i2s  l2i,i2c    l2i      <->      l2f      l2d
1223         //      float       -     f2i,i2b   f2i,i2s  f2i,i2c    f2i      f2l      <->      f2d
1224         //      double      -     d2i,i2b   d2i,i2s  d2i,i2c    d2i      d2l      d2f      <->
1225         if (from == to) {
1226             // no cast required, should be dead code anyway
1227             return;
1228         }
1229         if (from.isSubwordOrInt()) {
1230             // cast from {byte,short,char,int} to anything
1231             emitI2X(to);
1232         } else {
1233             // cast from {long,float,double} to anything
1234             if (to.isSubwordOrInt()) {
1235                 // cast to {byte,short,char,int}
1236                 emitX2I(from);
1237                 if (to.bitWidth() < 32) {
1238                     // targets other than int require another conversion
1239                     emitI2X(to);
1240                 }
1241             } else {
1242                 // cast to {long,float,double} - this is verbose
1243                 boolean error = false;
1244                 switch (from) {
1245                 case LONG:
1246                     switch (to) {
1247                     case FLOAT:   mv.visitInsn(Opcodes.L2F);  break;
1248                     case DOUBLE:  mv.visitInsn(Opcodes.L2D);  break;
1249                     default:      error = true;               break;
1250                     }
1251                     break;
1252                 case FLOAT:
1253                     switch (to) {
1254                     case LONG :   mv.visitInsn(Opcodes.F2L);  break;
1255                     case DOUBLE:  mv.visitInsn(Opcodes.F2D);  break;
1256                     default:      error = true;               break;
1257                     }
1258                     break;
1259                 case DOUBLE:
1260                     switch (to) {
1261                     case LONG :   mv.visitInsn(Opcodes.D2L);  break;
1262                     case FLOAT:   mv.visitInsn(Opcodes.D2F);  break;
1263                     default:      error = true;               break;
1264                     }
1265                     break;
1266                 default:
1267                     error = true;
1268                     break;
1269                 }
1270                 if (error) {
1271                     throw new IllegalStateException("unhandled prim cast: " + from + "2" + to);
1272                 }
1273             }
1274         }
1275     }

1277     private void emitI2X(Wrapper type) {
1278         switch (type) {
1279         case BYTE:    mv.visitInsn(Opcodes.I2B);  break;
1280         case SHORT:   mv.visitInsn(Opcodes.I2S);  break;
1281         case CHAR:    mv.visitInsn(Opcodes.I2C);  break;
1282         case INT:     /* naught */                break;
1283         case LONG:    mv.visitInsn(Opcodes.I2L);  break;
1284         case FLOAT:   mv.visitInsn(Opcodes.I2F);  break;
1285         case DOUBLE:  mv.visitInsn(Opcodes.I2D);  break;
1286         case BOOLEAN:
1287             // For compatibility with ValueConversions and explicitCastArguments:
1288             mv.visitInsn(Opcodes.ICONST_1);
1289             mv.visitInsn(Opcodes.IAND);
1290             break;
1291         default:   throw new InternalError("unknown type: " + type);
1292         }
1293     }

1316     private byte[] generateLambdaFormInterpreterEntryPointBytes() {
1317         classFilePrologue();
1318 
1319         // Suppress this method in backtraces displayed to the user.
1320         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
1321 
1322         // Don't inline the interpreter entry.
1323         mv.visitAnnotation("Ljava/lang/invoke/DontInline;", true);
1324 
1325         // create parameter array
1326         emitIconstInsn(invokerType.parameterCount());
1327         mv.visitTypeInsn(Opcodes.ANEWARRAY, "java/lang/Object");
1328 
1329         // fill parameter array
1330         for (int i = 0; i < invokerType.parameterCount(); i++) {
1331             Class<?> ptype = invokerType.parameterType(i);
1332             mv.visitInsn(Opcodes.DUP);
1333             emitIconstInsn(i);
1334             emitLoadInsn(basicType(ptype), i);
1335             // box if primitive type
1336             if (ptype.isPrimitive()) {
1337                 emitBoxing(Wrapper.forPrimitiveType(ptype));
1338             }
1339             mv.visitInsn(Opcodes.AASTORE);
1340         }
1341         // invoke
1342         emitAloadInsn(0);
1343         mv.visitFieldInsn(Opcodes.GETFIELD, MH, "form", "Ljava/lang/invoke/LambdaForm;");
1344         mv.visitInsn(Opcodes.SWAP);  // swap form and array; avoid local variable
1345         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, LF, "interpretWithArguments", "([Ljava/lang/Object;)Ljava/lang/Object;", false);

1358

1374     private byte[] generateNamedFunctionInvokerImpl(MethodTypeForm typeForm) {
1375         MethodType dstType = typeForm.erasedType();
1376         classFilePrologue();
1377 
1378         // Suppress this method in backtraces displayed to the user.
1379         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
1380 
1381         // Force inlining of this invoker method.
1382         mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
1383 
1384         // Load receiver
1385         emitAloadInsn(0);
1386 
1387         // Load arguments from array
1388         for (int i = 0; i < dstType.parameterCount(); i++) {
1389             emitAloadInsn(1);
1390             emitIconstInsn(i);
1391             mv.visitInsn(Opcodes.AALOAD);
1392 
1393             // Maybe unbox
1394             Class<?> dptype = dstType.parameterType(i);
1395             if (dptype.isPrimitive()) {
1396                 Class<?> sptype = dstType.basicType().wrap().parameterType(i);
1397                 Wrapper dstWrapper = Wrapper.forBasicType(dptype);
1398                 Wrapper srcWrapper = dstWrapper.isSubwordOrInt() ? Wrapper.INT : dstWrapper;  // narrow subword from int
1399                 emitUnboxing(srcWrapper);
1400                 emitPrimCast(srcWrapper, dstWrapper);
1401             }
1402         }
1403 
1404         // Invoke
1405         String targetDesc = dstType.basicType().toMethodDescriptorString();
1406         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", targetDesc, false);
1407 
1408         // Box primitive types
1409         Class<?> rtype = dstType.returnType();
1410         if (rtype != void.class && rtype.isPrimitive()) {
1411             Wrapper srcWrapper = Wrapper.forBasicType(rtype);
1412             Wrapper dstWrapper = srcWrapper.isSubwordOrInt() ? Wrapper.INT : srcWrapper;  // widen subword to int
1413             // boolean casts not allowed
1414             emitPrimCast(srcWrapper, dstWrapper);
1415             emitBoxing(dstWrapper);
1416         }
1417 
1418         // If the return type is void we return a null reference.
1419         if (rtype == void.class) {
1420             mv.visitInsn(Opcodes.ACONST_NULL);
1421         }
1422         emitReturnInsn(L_TYPE);  // NOTE: NamedFunction invokers always return a reference value.
1423 
1424         classFileEpilogue();
1425         bogusMethod(dstType);
1426 
1427         final byte[] classFile = cw.toByteArray();
1428         maybeDump(className, classFile);
1429         return classFile;
1430     }
