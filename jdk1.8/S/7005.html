 201         List sortedKeys = new ArrayList(typeMap.keySet());
 202         Collections.sort(sortedKeys);
 203         Iterator i = sortedKeys.iterator();
 204         while (i.hasNext()) {
 205             Integer pos = (Integer)i.next();
