 255     private String getFileName(SyntaxTreeNode node) {
 256         Stylesheet stylesheet = node.getStylesheet();
 257         if (stylesheet != null)
 258             return stylesheet.getSystemId();
 259         else
 260             return null;
 261     }

 263     private String formatLine() {
 264         StringBuffer result = new StringBuffer();
 265         if (_url != null) {
 266             result.append(_url);
 267             result.append(": ");
 268         }
 269         if (_line > 0) {
 270             result.append("line ");
 271             result.append(Integer.toString(_line));
 272             result.append(": ");
 273         }
 274         return result.toString();
 275     }

 277     /**
 278      * This version of toString() uses the _params instance variable
 279      * to format the message. If the _code is negative
 280      * the use _message as the error string.
 281      */
 282     public String toString() {
 283         String suffix = (_params == null) ?
 284             (null != _code ? getErrorMessage() : _message)
 285             : MessageFormat.format(getErrorMessage(), _params);
 286         return formatLine() + suffix;
 287     }
 288 
 289     public String toString(Object obj) {
 290         Object params[] = new Object[1];
 291         params[0] = obj.toString();
 292         String suffix = MessageFormat.format(getErrorMessage(), params);
 293         return formatLine() + suffix;
 294     }
 295 
 296     public String toString(Object obj0, Object obj1) {
 297         Object params[] = new Object[2];
 298         params[0] = obj0.toString();
 299         params[1] = obj1.toString();
 300         String suffix = MessageFormat.format(getErrorMessage(), params);
 301         return formatLine() + suffix;
 302     }
