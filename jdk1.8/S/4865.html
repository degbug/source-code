  30 /**
  31  * This class defines a simplified type of union iterator that only
  32  * tests along the child axes.  If the conditions are right, it is
  33  * much faster than using a UnionPathIterator.
  34  */
  35 public class UnionChildIterator extends ChildTestIterator
  36 {
  37     static final long serialVersionUID = 3500298482193003495L;
  38   /**
  39    * Even though these may hold full LocPathIterators, this array does
  40    * not have to be cloned, since only the node test and predicate
  41    * portion are used, and these only need static information.  However,
  42    * also note that index predicates can not be used!
  43    */
  44   private PredicatedNodeTest[] m_nodeTests = null;

  46   /**
  47    * Constructor for UnionChildIterator
  48    */
  49   public UnionChildIterator()
  50   {
  51     super(null);
  52   }

  54   /**
  55    * Add a node test to the union list.
  56    *
  57    * @param test reference to a NodeTest, which will be added
  58    * directly to the list of node tests (in other words, it will
  59    * not be cloned).  The parent of this test will be set to
  60    * this object.
  61    */
  62   public void addNodeTest(PredicatedNodeTest test)
  63   {
  64 
  65     // Increase array size by only 1 at a time.  Fix this
  66     // if it looks to be a problem.
  67     if (null == m_nodeTests)
  68     {
  69       m_nodeTests = new PredicatedNodeTest[1];
  70       m_nodeTests[0] = test;
  71     }
  72     else
  73     {
  74       PredicatedNodeTest[] tests = m_nodeTests;
  75       int len = m_nodeTests.length;
  76 
  77       m_nodeTests = new PredicatedNodeTest[len + 1];
  78 
  79       System.arraycopy(tests, 0, m_nodeTests, 0, len);
  80 
  81       m_nodeTests[len] = test;
  82     }
  83     test.exprSetParent(this);
  84   }

  86   /**
  87    * This function is used to fixup variables from QNames to stack frame
  88    * indexes at stylesheet build time.
  89    * @param vars List of QNames that correspond to variables.  This list
  90    * should be searched backwards for the first qualified name that
  91    * corresponds to the variable reference qname.  The position of the
  92    * QName in the vector from the start of the vector will be its position
  93    * in the stack frame (but variables above the globalsTop value will need
  94    * to be offset to the current stack frame).
  95    */
  96   public void fixupVariables(java.util.Vector vars, int globalsSize)
  97   {
  98     super.fixupVariables(vars, globalsSize);
  99     if (m_nodeTests != null) {
 100       for (int i = 0; i < m_nodeTests.length; i++) {
 101         m_nodeTests[i].fixupVariables(vars, globalsSize);
 102       }
 103     }
 104   }

 106   /**
 107    * Test whether a specified node is visible in the logical view of a
 108    * TreeWalker or NodeIterator. This function will be called by the
 109    * implementation of TreeWalker and NodeIterator; it is not intended to
 110    * be called directly from user code.
 111    * @param n  The node to check to see if it passes the filter or not.
 112    * @return  a constant to determine whether the node is accepted,
 113    *   rejected, or skipped, as defined  above .
 114    */
 115   public short acceptNode(int n)
 116   {
 117     XPathContext xctxt = getXPathContext();
 118     try
 119     {
 120       xctxt.pushCurrentNode(n);
 121       for (int i = 0; i < m_nodeTests.length; i++)
 122       {
 123         PredicatedNodeTest pnt = m_nodeTests[i];
 124         XObject score = pnt.execute(xctxt, n);
 125         if (score != NodeTest.SCORE_NONE)
 126         {
 127           // Note that we are assuming there are no positional predicates!
 128           if (pnt.getPredicateCount() > 0)
 129           {
 130             if (pnt.executePredicates(n, xctxt))
 131               return DTMIterator.FILTER_ACCEPT;
 132           }
 133           else
 134             return DTMIterator.FILTER_ACCEPT;
 135 
 136         }
 137       }
 138     }
 139     catch (javax.xml.transform.TransformerException se)
 140     {
 141 
 142       // TODO: Fix this.
 143       throw new RuntimeException(se.getMessage());
 144     }
 145     finally
 146     {
 147       xctxt.popCurrentNode();
 148     }
 149     return DTMIterator.FILTER_SKIP;
 150   }
