  83     @Override
  84     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  85         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  86         componentColors = extendedCacheKeys;
  87         //generate this entire method. Each state/bg/fg/border combo that has
  88         //been painted gets its own KEY and paint method.
  89         switch(state) {
  90             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  91             case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
  92             case FOREGROUND_ENABLED: paintForegroundEnabled(g); break;
  93             case FOREGROUND_ENABLED_VERTICAL: paintForegroundEnabledAndVertical(g); break;
  94 
  95         }
  96     }
