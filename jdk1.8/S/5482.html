  29 /**
  30  * Attribute represents an XML-style attribute of an
  31  * Element. Typically, the allowable values are controlled by its
  32  * declaration in the Document Type Definition (DTD) governing this
  33  * kind of document.
  34  * 

  35  * If the attribute has not been explicitly assigned a value, but has
  36  * been declared in the DTD, it will exist and have that default. Only
  37  * if neither the document nor the DTD specifies a value will the
  38  * Attribute really be considered absent and have no value; in that
  39  * case, querying the attribute will return null.
  40  * 

  41  * Attributes may have multiple children that contain their data. (XML
  42  * allows attributes to contain entity references, and tokenized
  43  * attribute types such as NMTOKENS may have a child for each token.)
  44  * For convenience, the Attribute object's getValue() method returns
  45  * the string version of the attribute's value.
  46  * 

  47  * Attributes are not children of the Elements they belong to, in the
  48  * usual sense, and have no valid Parent reference. However, the spec
  49  * says they _do_ belong to a specific Element, and an INUSE exception
  50  * is to be thrown if the user attempts to explicitly share them
  51  * between elements.
  52  * 

  53  * Note that Elements do not permit attributes to appear to be shared
  54  * (see the INUSE exception), so this object's mutability is
  55  * officially not an issue.
  56  * 

  57  * DeferredAttrImpl inherits from AttrImpl which does not support
  58  * Namespaces. DeferredAttrNSImpl, which inherits from AttrNSImpl, does.
  59  * @see DeferredAttrNSImpl
  60  *
  61  * @xerces.internal
  62  *
  63  * @author Andy Clark, IBM
  64  * @author Arnaud  Le Hors, IBM
  65  * @since  PR-DOM-Level-1-19980818.
  66  */

 115     /** Synchronizes the data (name and value) for fast nodes. */
 116     protected void synchronizeData() {
 117 
 118         // no need to sync in the future
 119         needsSyncData(false);
 120 
 121         // fluff data
 122         DeferredDocumentImpl ownerDocument =
 123             (DeferredDocumentImpl) ownerDocument();
 124         name = ownerDocument.getNodeName(fNodeIndex);
 125         int extra = ownerDocument.getNodeExtra(fNodeIndex);
 126         isSpecified((extra & SPECIFIED) != 0);
 127         isIdAttribute((extra & ID) != 0);
 128 
 129         int extraNode = ownerDocument.getLastChild(fNodeIndex);
 130         type = ownerDocument.getTypeInfo(extraNode);
 131     } // synchronizeData()
 132 
 133     /**
 134      * Synchronizes the node's children with the internal structure.
 135      * Fluffing the children at once solves a lot of work to keep
 136      * the two structures in sync. The problem gets worse when
 137      * editing the tree -- this makes it a lot easier.
 138      */
