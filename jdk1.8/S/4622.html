  88       return Type.FLOAT;

  90       return Type.LONG;
  91     case Constants.F2D:  case Constants.I2D: case Constants.L2D:
  92         return Type.DOUBLE;
  93     case Constants.I2B:
  94       return Type.BYTE;
  95     case Constants.I2C:
  96       return Type.CHAR;
  97     case Constants.I2S:
  98       return Type.SHORT;

 102     }
