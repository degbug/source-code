  67     /**
  68      * Test if the DOM implementation supports a specific "feature" --
  69      * currently meaning language and level thereof.
  70      *
  71      * @param feature      The package name of the feature to test.
  72      * In Level 1, supported values are "HTML" and "XML" (case-insensitive).
  73      * At this writing, com.sun.org.apache.xerces.internal.dom supports only XML.
  74      *
  75      * @param version      The version number of the feature being tested.
  76      * This is interpreted as "Version of the DOM API supported for the
  77      * specified Feature", and in Level 1 should be "1.0"
  78      *
  79      * @return    true iff this implementation is compatable with the specified
  80      * feature and version.
  81      */
  82     public boolean hasFeature(String feature, String version) {
  83 
  84         return (feature.equalsIgnoreCase("XS-Loader") && (version == null || version.equals("1.0")) ||
  85                 super.hasFeature(feature, version));
  86     } // hasFeature(String,String):boolean

  90     /* (non-Javadoc)
  91      * @see com.sun.org.apache.xerces.internal.xs.XSImplementation#createXSLoader(com.sun.org.apache.xerces.internal.xs.StringList)
  92      */
  93     public XSLoader createXSLoader(StringList versions) throws XSException {
  94         XSLoader loader = new XSLoaderImpl();
  95         if (versions == null){
  96                         return loader;
  97         }
  98         for (int i=0; i<versions.getLength();i++){
  99                 if (!versions.item(i).equals("1.0")){
 100                                 String msg =
 101                                         DOMMessageFormatter.formatMessage(
 102                                                 DOMMessageFormatter.DOM_DOMAIN,
 103                                                 "FEATURE_NOT_SUPPORTED",
 104                                                 new Object[] { versions.item(i) });
 105                                 throw new XSException(XSException.NOT_SUPPORTED_ERR, msg);
 106                 }
 107         }
 108         return loader;
 109     }
