 283     /**
 284      * Copied code from java.util.Properties. Parsing the data ourselves is the
 285      * only way to handle duplicate keys and values.

 287     private void parseAndStoreReader(BufferedReader in) throws IOException {
 288         while (true) {
 289             // Get next line
 290             String line = in.readLine();
 291             if (line == null) {
 292                 return;
 293             }
 294 
 295             if (line.length() > 0) {
 296                 // Continue lines that end in slashes if they are not comments
 297                 char firstChar = line.charAt(0);
 298                 if (firstChar != '#' && firstChar != '!') {
 299                     while (continueLine(line)) {
 300                         String nextLine = in.readLine();
 301                         if (nextLine == null) {
 302                             nextLine = "";
 303                         }
 304                         String loppedLine =
 305                             line.substring(0, line.length() - 1);
 306                         // Advance beyond whitespace on new line
 307                         int startIndex = 0;
 308                         for(; startIndex < nextLine.length(); startIndex++) {
 309                             if (whiteSpaceChars.
 310                                     indexOf(nextLine.charAt(startIndex)) == -1)
 311                             {
 312                                 break;
 313                             }
 314                         }
 315                         nextLine = nextLine.substring(startIndex,
 316                                                       nextLine.length());
 317                         line = loppedLine+nextLine;
 318                     }
 319 
 320                     // Find start of key
 321                     int len = line.length();
 322                     int keyStart = 0;
 323                     for(; keyStart < len; keyStart++) {
 324                         if(whiteSpaceChars.
 325                                indexOf(line.charAt(keyStart)) == -1) {
 326                             break;
 327                         }
 328                     }
 329 
 330                     // Blank lines are ignored
 331                     if (keyStart == len) {
 332                         continue;
 333                     }
 334 
 335                     // Find separation between key and value
 336                     int separatorIndex = keyStart;
 337                     for(; separatorIndex < len; separatorIndex++) {
 338                         char currentChar = line.charAt(separatorIndex);
 339                         if (currentChar == '\\') {
 340                             separatorIndex++;
 341                         } else if (keyValueSeparators.
 342                                        indexOf(currentChar) != -1) {
 343                             break;
 344                         }
 345                     }
 346 
 347                     // Skip over whitespace after key if any
 348                     int valueIndex = separatorIndex;
 349                     for (; valueIndex < len; valueIndex++) {
 350                         if (whiteSpaceChars.
 351                                 indexOf(line.charAt(valueIndex)) == -1) {
 352                             break;
 353                         }
 354                     }
 355 
 356                     // Skip over one non whitespace key value separators if any
 357                     if (valueIndex < len) {
 358                         if (strictKeyValueSeparators.
 359                                 indexOf(line.charAt(valueIndex)) != -1) {
 360                             valueIndex++;
 361                         }
 362                     }
 363 
 364                     // Skip over white space after other separators if any
 365                     while (valueIndex < len) {
 366                         if (whiteSpaceChars.
 367                                 indexOf(line.charAt(valueIndex)) == -1) {
 368                             break;
 369                         }
 370                         valueIndex++;
 371                     }
 372 
 373                     String key = line.substring(keyStart, separatorIndex);
 374                     String value = (separatorIndex < len)
 375                         ? line.substring(valueIndex, len)
 376                         : "";
 377 
 378                     // Convert then store key and value
 379                     key = loadConvert(key);
 380                     value = loadConvert(value);
 381 
 382                     try {
 383                         MimeType mime = new MimeType(value);
 384                         if ("text".equals(mime.getPrimaryType())) {
 385                             String charset = mime.getParameter("charset");
 386                             if (DataTransferer.doesSubtypeSupportCharset
 387                                     (mime.getSubType(), charset))
 388                             {
 389                                 // We need to store the charset and eoln
 390                                 // parameters, if any, so that the
 391                                 // DataTransferer will have this information
 392                                 // for conversion into the native format.
 393                                 DataTransferer transferer =
 394                                     DataTransferer.getInstance();
 395                                 if (transferer != null) {
 396                                     transferer.registerTextFlavorProperties
 397                                         (key, charset,
 398                                          mime.getParameter("eoln"),
 399                                          mime.getParameter("terminators"));
 400                                 }
 401                             }
 402 
 403                             // But don't store any of these parameters in the
 404                             // DataFlavor itself for any text natives (even
 405                             // non-charset ones). The SystemFlavorMap will
 406                             // synthesize the appropriate mappings later.
 407                             mime.removeParameter("charset");
 408                             mime.removeParameter("class");
 409                             mime.removeParameter("eoln");
 410                             mime.removeParameter("terminators");
 411                             value = mime.toString();
 412                         }
 413                     } catch (MimeTypeParseException e) {
 414                         e.printStackTrace();
 415                         continue;
 416                     }
 417 
 418                     DataFlavor flavor;
 419                     try {
 420                         flavor = new DataFlavor(value);
 421                     } catch (Exception e) {
 422                         try {
 423                             flavor = new DataFlavor(value, null);
 424                         } catch (Exception ee) {
 425                             ee.printStackTrace();
 426                             continue;
 427                         }
 428                     }
 429 
 430                     final LinkedHashSet<DataFlavor> dfs = new LinkedHashSet<>();
 431                     dfs.add(flavor);
 432 
 433                     if ("text".equals(flavor.getPrimaryType())) {
 434                         dfs.addAll(convertMimeTypeToDataFlavors(value));
 435                         store(flavor.mimeType.getBaseType(), key, getTextTypeToNative());
 436                     }
 437 
 438                     for (DataFlavor df : dfs) {
 439                         store(df, key, getFlavorToNative());
 440                         store(key, df, getNativeToFlavor());
 441                     }
 442                 }
 443             }
 444         }
 445     }

 450     private boolean continueLine (String line) {
 451         int slashCount = 0;
 452         int index = line.length() - 1;
 453         while((index >= 0) && (line.charAt(index--) == '\\')) {
 454             slashCount++;
 455         }
 456         return (slashCount % 2 == 1);
 457     }

 462     private String loadConvert(String theString) {
 463         char aChar;
 464         int len = theString.length();
 465         StringBuilder outBuffer = new StringBuilder(len);
 466 
 467         for (int x = 0; x < len; ) {
 468             aChar = theString.charAt(x++);
 469             if (aChar == '\\') {
 470                 aChar = theString.charAt(x++);
 471                 if (aChar == 'u') {
 472                     // Read the xxxx
 473                     int value = 0;
 474                     for (int i = 0; i < 4; i++) {
 475                         aChar = theString.charAt(x++);
 476                         switch (aChar) {
 477                           case '0': case '1': case '2': case '3': case '4':
 478                           case '5': case '6': case '7': case '8': case '9': {
 479                              value = (value << 4) + aChar - '0';
 480                              break;
 481                           }
 482                           case 'a': case 'b': case 'c':
 483                           case 'd': case 'e': case 'f': {
 484                              value = (value << 4) + 10 + aChar - 'a';
 485                              break;
 486                           }
 487                           case 'A': case 'B': case 'C':
 488                           case 'D': case 'E': case 'F': {
 489                              value = (value << 4) + 10 + aChar - 'A';
 490                              break;
 491                           }
 492                           default: {
 493                               throw new IllegalArgumentException(
 494                                            "Malformed \\uxxxx encoding.");
 495                           }
 496                         }
 497                     }
 498                     outBuffer.append((char)value);
 499                 } else {
 500                     if (aChar == 't') {
 501                         aChar = '\t';
 502                     } else if (aChar == 'r') {
 503                         aChar = '\r';
 504                     } else if (aChar == 'n') {
 505                         aChar = '\n';
 506                     } else if (aChar == 'f') {
 507                         aChar = '\f';
 508                     }
 509                     outBuffer.append(aChar);
 510                 }
 511             } else {
 512                 outBuffer.append(aChar);
 513             }
 514         }
 515         return outBuffer.toString();
 516     }

 520      * standard map, the listed object will not replace any object already at
 521      * the appropriate Map location, but rather will be appended to a List
 522      * stored in that location.

 524     private <H, L> void store(H hashed, L listed, Map<H, LinkedHashSet<L>> map) {

 526         if (list == null) {
 527             list = new LinkedHashSet<>(1);

 529         }
 530         if (!list.contains(listed)) {
 531             list.add(listed);

 535     /**
 536      * Semantically equivalent to 'nativeToFlavor.get(nat)'. This method
 537      * handles the case where 'nat' is not found in 'nativeToFlavor'. In that
 538      * case, a new DataFlavor is synthesized, stored, and returned, if and
 539      * only if the specified native is encoded as a Java MIME type.
 540      */
 541     private LinkedHashSet<DataFlavor> nativeToFlavorLookup(String nat) {
 542         LinkedHashSet<DataFlavor> flavors = getNativeToFlavor().get(nat);
 543 
 544 
 545         if (nat != null && !disabledMappingGenerationKeys.contains(nat)) {
 546             DataTransferer transferer = DataTransferer.getInstance();
 547             if (transferer != null) {
 548                 LinkedHashSet<DataFlavor> platformFlavors =
 549                     transferer.getPlatformMappingsForNative(nat);
 550                 if (!platformFlavors.isEmpty()) {
 551                     if (flavors != null) {
 552                         // Prepending the platform-specific mappings ensures
 553                         // that the flavors added with
 554                         // addFlavorForUnencodedNative() are at the end of
 555                         // list.
 556                         platformFlavors.addAll(flavors);
 557                     }
 558                     flavors = platformFlavors;
 559                 }
 560             }
 561         }
 562 
 563         if (flavors == null && isJavaMIMEType(nat)) {
 564             String decoded = decodeJavaMIMEType(nat);
 565             DataFlavor flavor = null;
 566 
 567             try {
 568                 flavor = new DataFlavor(decoded);
 569             } catch (Exception e) {
 570                 System.err.println("Exception \"" + e.getClass().getName() +
 571                                    ": " + e.getMessage()  +
 572                                    "\"while constructing DataFlavor for: " +
 573                                    decoded);
 574             }
 575 
 576             if (flavor != null) {
 577                 flavors = new LinkedHashSet<>(1);
 578                 getNativeToFlavor().put(nat, flavors);
 579                 flavors.add(flavor);
 580                 flavorsForNativeCache.remove(nat);
 581 
 582                 LinkedHashSet<String> natives = getFlavorToNative().get(flavor);
 583                 if (natives == null) {
 584                     natives = new LinkedHashSet<>(1);
 585                     getFlavorToNative().put(flavor, natives);
 586                 }
 587                 natives.add(nat);
 588                 nativesForFlavorCache.remove(flavor);
 589             }
 590         }
 591 
 592         return (flavors != null) ? flavors : new LinkedHashSet<>(0);
 593     }

 595     /**
 596      * Semantically equivalent to 'flavorToNative.get(flav)'. This method
 597      * handles the case where 'flav' is not found in 'flavorToNative' depending
 598      * on the value of passes 'synthesize' parameter. If 'synthesize' is
 599      * SYNTHESIZE_IF_NOT_FOUND a native is synthesized, stored, and returned by
 600      * encoding the DataFlavor's MIME type. Otherwise an empty List is returned
 601      * and 'flavorToNative' remains unaffected.
 602      */
 603     private LinkedHashSet<String> flavorToNativeLookup(final DataFlavor flav,
 604                                                        final boolean synthesize) {
 605 
 606         LinkedHashSet<String> natives = getFlavorToNative().get(flav);
 607 
 608         if (flav != null && !disabledMappingGenerationKeys.contains(flav)) {
 609             DataTransferer transferer = DataTransferer.getInstance();
 610             if (transferer != null) {
 611                 LinkedHashSet<String> platformNatives =
 612                     transferer.getPlatformMappingsForFlavor(flav);
 613                 if (!platformNatives.isEmpty()) {
 614                     if (natives != null) {
 615                         // Prepend the platform-specific mappings to ensure
 616                         // that the natives added with
 617                         // addUnencodedNativeForFlavor() are at the end of
 618                         // list.
 619                         platformNatives.addAll(natives);
 620                     }
 621                     natives = platformNatives;
 622                 }
 623             }
 624         }
 625 
 626         if (natives == null) {
 627             if (synthesize) {
 628                 String encoded = encodeDataFlavor(flav);
 629                 natives = new LinkedHashSet<>(1);
 630                 getFlavorToNative().put(flav, natives);
 631                 natives.add(encoded);
 632 
 633                 LinkedHashSet<DataFlavor> flavors = getNativeToFlavor().get(encoded);
 634                 if (flavors == null) {
 635                     flavors = new LinkedHashSet<>(1);
 636                     getNativeToFlavor().put(encoded, flavors);
 637                 }
 638                 flavors.add(flav);
 639 
 640                 nativesForFlavorCache.remove(flav);
 641                 flavorsForNativeCache.remove(encoded);
 642             } else {
 643                 natives = new LinkedHashSet<>(0);
 644             }
 645         }
 646 
 647         return new LinkedHashSet<>(natives);
 648     }

 650     /**
 651      * Returns a <code>List</code> of <code>String</code> natives to which the
 652      * specified <code>DataFlavor</code> can be translated by the data transfer
 653      * subsystem. The <code>List</code> will be sorted from best native to
 654      * worst. That is, the first native will best reflect data in the specified
 655      * flavor to the underlying native platform.
 656      * <p>
 657      * If the specified <code>DataFlavor</code> is previously unknown to the
 658      * data transfer subsystem and the data transfer subsystem is unable to
 659      * translate this <code>DataFlavor</code> to any existing native, then
 660      * invoking this method will establish a
 661      * mapping in both directions between the specified <code>DataFlavor</code>
 662      * and an encoded version of its MIME type as its native.
 663      *
 664      * @param flav the <code>DataFlavor</code> whose corresponding natives
 665      *        should be returned. If <code>null</code> is specified, all
 666      *        natives currently known to the data transfer subsystem are
 667      *        returned in a non-deterministic order.
 668      * @return a <code>java.util.List</code> of <code>java.lang.String</code>
 669      *         objects which are platform-specific representations of platform-
 670      *         specific data formats
 671      *
 672      * @see #encodeDataFlavor
 673      * @since 1.4
 674      */
 675     @Override
 676     public synchronized List<String> getNativesForFlavor(DataFlavor flav) {
 677         LinkedHashSet<String> retval = nativesForFlavorCache.check(flav);
 678         if (retval != null) {
 679             return new ArrayList<>(retval);
 680         }
 681 
 682         if (flav == null) {
 683             retval = new LinkedHashSet<>(getNativeToFlavor().keySet());
 684         } else if (disabledMappingGenerationKeys.contains(flav)) {
 685             // In this case we shouldn't synthesize a native for this flavor,
 686             // since its mappings were explicitly specified.
 687             retval = flavorToNativeLookup(flav, false);
 688         } else if (DataTransferer.isFlavorCharsetTextType(flav)) {
 689             retval = new LinkedHashSet<>(0);
 690 
 691             // For text/* flavors, flavor-to-native mappings specified in
 692             // flavormap.properties are stored per flavor's base type.
 693             if ("text".equals(flav.getPrimaryType())) {
 694                 LinkedHashSet<String> textTypeNatives =
 695                         getTextTypeToNative().get(flav.mimeType.getBaseType());
 696                 if (textTypeNatives != null) {
 697                     retval.addAll(textTypeNatives);
 698                 }
 699             }
 700 
 701             // Also include text/plain natives, but don't duplicate Strings
 702             LinkedHashSet<String> textTypeNatives =
 703                     getTextTypeToNative().get(TEXT_PLAIN_BASE_TYPE);
 704             if (textTypeNatives != null) {
 705                 retval.addAll(textTypeNatives);
 706             }
 707 
 708             if (retval.isEmpty()) {
 709                 retval = flavorToNativeLookup(flav, true);
 710             } else {
 711                 // In this branch it is guaranteed that natives explicitly
 712                 // listed for flav's MIME type were added with
 713                 // addUnencodedNativeForFlavor(), so they have lower priority.
 714                 retval.addAll(flavorToNativeLookup(flav, false));
 715             }
 716         } else if (DataTransferer.isFlavorNoncharsetTextType(flav)) {
 717             retval = getTextTypeToNative().get(flav.mimeType.getBaseType());
 718 
 719             if (retval == null || retval.isEmpty()) {
 720                 retval = flavorToNativeLookup(flav, true);
 721             } else {
 722                 // In this branch it is guaranteed that natives explicitly
 723                 // listed for flav's MIME type were added with
 724                 // addUnencodedNativeForFlavor(), so they have lower priority.
 725                 retval.addAll(flavorToNativeLookup(flav, false));
 726             }
 727         } else {
 728             retval = flavorToNativeLookup(flav, true);
 729         }
 730 
 731         nativesForFlavorCache.put(flav, retval);
 732         // Create a copy, because client code can modify the returned list.
 733         return new ArrayList<>(retval);
 734     }

 736     /**
 737      * Returns a <code>List</code> of <code>DataFlavor</code>s to which the
 738      * specified <code>String</code> native can be translated by the data
 739      * transfer subsystem. The <code>List</code> will be sorted from best
 740      * <code>DataFlavor</code> to worst. That is, the first
 741      * <code>DataFlavor</code> will best reflect data in the specified
 742      * native to a Java application.
 743      * <p>
 744      * If the specified native is previously unknown to the data transfer
 745      * subsystem, and that native has been properly encoded, then invoking this
 746      * method will establish a mapping in both directions between the specified
 747      * native and a <code>DataFlavor</code> whose MIME type is a decoded
 748      * version of the native.
 749      * <p>
 750      * If the specified native is not a properly encoded native and the
 751      * mappings for this native have not been altered with
 752      * <code>setFlavorsForNative</code>, then the contents of the
 753      * <code>List</code> is platform dependent, but <code>null</code>
 754      * cannot be returned.
 755      *
 756      * @param nat the native whose corresponding <code>DataFlavor</code>s
 757      *        should be returned. If <code>null</code> is specified, all
 758      *        <code>DataFlavor</code>s currently known to the data transfer
 759      *        subsystem are returned in a non-deterministic order.
 760      * @return a <code>java.util.List</code> of <code>DataFlavor</code>
 761      *         objects into which platform-specific data in the specified,
 762      *         platform-specific native can be translated
 763      *
 764      * @see #encodeJavaMIMEType
 765      * @since 1.4
 766      */
 767     @Override
 768     public synchronized List<DataFlavor> getFlavorsForNative(String nat) {
 769         LinkedHashSet<DataFlavor> returnValue = flavorsForNativeCache.check(nat);
 770         if (returnValue != null) {
 771             return new ArrayList<>(returnValue);
 772         } else {
 773             returnValue = new LinkedHashSet<>();
 774         }
 775 
 776         if (nat == null) {
 777             for (String n : getNativesForFlavor(null)) {
 778                 returnValue.addAll(getFlavorsForNative(n));
 779             }
 780         } else {
 781             final LinkedHashSet<DataFlavor> flavors = nativeToFlavorLookup(nat);
 782             if (disabledMappingGenerationKeys.contains(nat)) {
 783                 return new ArrayList<>(flavors);
 784             }
 785 
 786             final LinkedHashSet<DataFlavor> flavorsWithSynthesized =
 787                     nativeToFlavorLookup(nat);
 788 
 789             for (DataFlavor df : flavorsWithSynthesized) {
 790                 returnValue.add(df);
 791                 if ("text".equals(df.getPrimaryType())) {
 792                     String baseType = df.mimeType.getBaseType();
 793                     returnValue.addAll(convertMimeTypeToDataFlavors(baseType));
 794                 }
 795             }
 796         }
 797         flavorsForNativeCache.put(nat, returnValue);
 798         return new ArrayList<>(returnValue);
 799     }

 801     private static Set<DataFlavor> convertMimeTypeToDataFlavors(
 802         final String baseType) {
 803 
 804         final Set<DataFlavor> returnValue = new LinkedHashSet<>();
 805 
 806         String subType = null;
 807 
 808         try {
 809             final MimeType mimeType = new MimeType(baseType);
 810             subType = mimeType.getSubType();
 811         } catch (MimeTypeParseException mtpe) {
 812             // Cannot happen, since we checked all mappings
 813             // on load from flavormap.properties.
 814         }
 815 
 816         if (DataTransferer.doesSubtypeSupportCharset(subType, null)) {
 817             if (TEXT_PLAIN_BASE_TYPE.equals(baseType))
 818             {
 819                 returnValue.add(DataFlavor.stringFlavor);
 820             }
 821 
 822             for (String unicodeClassName : UNICODE_TEXT_CLASSES) {
 823                 final String mimeType = baseType + ";charset=Unicode;class=" +
 824                                             unicodeClassName;
 825 
 826                 final LinkedHashSet<String> mimeTypes =
 827                     handleHtmlMimeTypes(baseType, mimeType);
 828                 for (String mt : mimeTypes) {
 829                     DataFlavor toAdd = null;
 830                     try {
 831                         toAdd = new DataFlavor(mt);
 832                     } catch (ClassNotFoundException cannotHappen) {
 833                     }
 834                     returnValue.add(toAdd);
 835                 }
 836             }
 837 
 838             for (String charset : DataTransferer.standardEncodings()) {
 839 
 840                 for (String encodedTextClass : ENCODED_TEXT_CLASSES) {
 841                     final String mimeType =
 842                             baseType + ";charset=" + charset +
 843                             ";class=" + encodedTextClass;
 844 
 845                     final LinkedHashSet<String> mimeTypes =
 846                         handleHtmlMimeTypes(baseType, mimeType);
 847 
 848                     for (String mt : mimeTypes) {
 849 
 850                         DataFlavor df = null;
 851 
 852                         try {
 853                             df = new DataFlavor(mt);
 854                             // Check for equality to plainTextFlavor so
 855                             // that we can ensure that the exact charset of
 856                             // plainTextFlavor, not the canonical charset
 857                             // or another equivalent charset with a
 858                             // different name, is used.
 859                             if (df.equals(DataFlavor.plainTextFlavor)) {
 860                                 df = DataFlavor.plainTextFlavor;
 861                             }
 862                         } catch (ClassNotFoundException cannotHappen) {
 863                         }
 864 
 865                         returnValue.add(df);
 866                     }
 867                 }
 868             }
 869 
 870             if (TEXT_PLAIN_BASE_TYPE.equals(baseType))
 871             {
 872                 returnValue.add(DataFlavor.plainTextFlavor);
 873             }
 874         } else {
 875             // Non-charset text natives should be treated as
 876             // opaque, 8-bit data in any of its various
 877             // representations.
 878             for (String encodedTextClassName : ENCODED_TEXT_CLASSES) {
 879                 DataFlavor toAdd = null;
 880                 try {
 881                     toAdd = new DataFlavor(baseType +
 882                          ";class=" + encodedTextClassName);
 883                 } catch (ClassNotFoundException cannotHappen) {
 884                 }
 885                 returnValue.add(toAdd);
 886             }
 887         }
 888         return returnValue;
 889     }

 892             new String [] {"all", "selection", "fragment"};

 894     private static LinkedHashSet<String> handleHtmlMimeTypes(String baseType,
 895                                                              String mimeType) {
 896 
 897         LinkedHashSet<String> returnValues = new LinkedHashSet<>();

 901                 returnValues.add(mimeType + ";document=" + documentType);

 903         } else {
 904             returnValues.add(mimeType);
 905         }
 906 
 907         return returnValues;
