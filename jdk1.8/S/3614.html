  28 import java.io.IOException;

  33 /**
  34  * Result of a query for buffered notifications.  Notifications in
  35  * a notification buffer have positive, monotonically increasing
  36  * sequence numbers.  The result of a notification query contains the
  37  * following elements:
  38  *
  39  * 
    
  40  *
  41  * 
  • The sequence number of the earliest notification still in
    • 
  42  * the buffer.
  43  *
  44  * 
  • The sequence number of the next notification available for
    • 
  45  * querying.  This will be the starting sequence number for the next
  46  * notification query.
  47  *
  48  * 
  • An array of (Notification,listenerID) pairs corresponding to
    • 
  49  * the returned notifications and the listeners they correspond to.
  50  *
  51  * 
  52  *
  53  * It is possible for the nextSequenceNumber to be less
  54  * than the earliestSequenceNumber.  This signifies that
  55  * notifications between the two might have been lost.
  56  *
  57  * @since 1.5
  58  */
  59 public class NotificationResult implements Serializable {
  60 
  61     private static final long serialVersionUID = 1191800228721395279L;
  62 
  63     /**
  64      * Constructs a notification query result.
  65      *
  66      * @param earliestSequenceNumber the sequence number of the
  67      * earliest notification still in the buffer.
  68      * @param nextSequenceNumber the sequence number of the next
  69      * notification available for querying.
  70      * @param targetedNotifications the notifications resulting from
  71      * the query, and the listeners they correspond to.  This array
  72      * can be empty.
  73      *
  74      * @exception IllegalArgumentException if
  75      * targetedNotifications is null or if
  76      * earliestSequenceNumber or
  77      * nextSequenceNumber is negative.
  78      */
  79     public NotificationResult(long earliestSequenceNumber,
  80                               long nextSequenceNumber,
  81                               TargetedNotification[] targetedNotifications) {
  82         validate(targetedNotifications, earliestSequenceNumber, nextSequenceNumber);
  83         this.earliestSequenceNumber = earliestSequenceNumber;
  84         this.nextSequenceNumber = nextSequenceNumber;
  85         this.targetedNotifications = (targetedNotifications.length == 0 ? targetedNotifications : targetedNotifications.clone());
  86     }
  87 
  88     /**
  89      * Returns the sequence number of the earliest notification still
  90      * in the buffer.
  91      *
  92      * @return the sequence number of the earliest notification still
  93      * in the buffer.
  94      */
  95     public long getEarliestSequenceNumber() {
  96         return earliestSequenceNumber;
  97     }
  98 
  99     /**
 100      * Returns the sequence number of the next notification available
 101      * for querying.
 102      *
 103      * @return the sequence number of the next notification available
 104      * for querying.
 105      */
 106     public long getNextSequenceNumber() {
 107         return nextSequenceNumber;
 108     }
 109 
 110     /**
 111      * Returns the notifications resulting from the query, and the
 112      * listeners they correspond to.
 113      *
 114      * @return the notifications resulting from the query, and the
 115      * listeners they correspond to.  This array can be empty.
 116      */
 117     public TargetedNotification[] getTargetedNotifications() {
 118         return targetedNotifications.length == 0 ? targetedNotifications : targetedNotifications.clone();
 119     }
 120 
 121     /**
 122      * Returns a string representation of the object.  The result
 123      * should be a concise but informative representation that is easy
 124      * for a person to read.
 125      *
 126      * @return a string representation of the object.
 127      */
 128     public String toString() {
 129         return "NotificationResult: earliest=" + getEarliestSequenceNumber() +
 130             "; next=" + getNextSequenceNumber() + "; nnotifs=" +
 131             getTargetedNotifications().length;
 132     }
 133 
 134     private void readObject(ObjectInputStream ois) throws IOException, ClassNotFoundException {
 135         ois.defaultReadObject();
 136         try {
 137             validate(
 138                 this.targetedNotifications,
 139                 this.earliestSequenceNumber,
 140                 this.nextSequenceNumber
 141             );
 142 
 143             this.targetedNotifications = this.targetedNotifications.length == 0 ?
 144                                             this.targetedNotifications :
 145                                             this.targetedNotifications.clone();
 146         } catch (IllegalArgumentException e) {
 147             throw new InvalidObjectException(e.getMessage());
 148         }
 149     }
 150 
 151     private long earliestSequenceNumber;
 152     private long nextSequenceNumber;
 153     private TargetedNotification[] targetedNotifications;
 154 
 155     private static void validate(TargetedNotification[] targetedNotifications,
 156                                  long earliestSequenceNumber,
 157                                  long nextSequenceNumber)
 158         throws IllegalArgumentException {
 159         if (targetedNotifications == null) {
 160             final String msg = "Notifications null";
 161             throw new IllegalArgumentException(msg);
 162         }
 163 
 164         if (earliestSequenceNumber < 0 || nextSequenceNumber < 0)
 165             throw new IllegalArgumentException("Bad sequence numbers");
 166         /* We used to check nextSequenceNumber >= earliestSequenceNumber
 167            here.  But in fact the opposite can legitimately be true if
 168            notifications have been lost.  */
 169     }
 170 }
