  83     @Override
  84     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  85         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  86         componentColors = extendedCacheKeys;
  87         //generate this entire method. Each state/bg/fg/border combo that has
  88         //been painted gets its own KEY and paint method.
  89         switch(state) {
  90             case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
  91             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
  92             case BACKGROUND_MOUSEOVER_FOCUSED: paintBackgroundMouseOverAndFocused(g); break;
  93             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
  94             case BACKGROUND_PRESSED_FOCUSED: paintBackgroundPressedAndFocused(g); break;
  95 
  96         }
  97     }
