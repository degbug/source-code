  28 /**
  29  * Containts the map between qnames and XSObject's.

  39     // the type of component stored here: complex or simple type

  42     /**

  57      * @param namespaces the namespaces to which the components belong

  63         super(namespaces, maps, num);
  64         fType = type;

  67     /**
  68      * The number of XSObjects in the XSObjectList. The
  69      * range of valid child node indices is 0 to length-1
  70      * inclusive.
  71      */
  72     public synchronized int getLength() {
  73         if (fLength == -1) {
  74             // first get the number of components for all types
  75             int length = 0;
  76             for (int i = 0; i < fNSNum; i++) {
  77                 length += fMaps[i].getLength();
  78             }
  79             // then copy all types to an temporary array
  80             int pos = 0;
  81             XSObject[] array = new XSObject[length];
  82             for (int i = 0; i < fNSNum; i++) {
  83                 pos += fMaps[i].getValues(array, pos);
  84             }
  85             // then copy either simple or complex types to fArray,
  86             // depending on which kind is required
  87             fLength = 0;
  88             fArray = new XSObject[length];
  89             XSTypeDefinition type;
  90             for (int i = 0; i < length; i++) {
  91                 type = (XSTypeDefinition)array[i];
  92                 if (type.getTypeCategory() == fType) {
  93                     fArray[fLength++] = type;
  94                 }
  95             }
  96         }
  97         return fLength;
  98     }

 100     /**
 101      * Retrieves an XSObject specified by local name and namespace
 102      * URI.
 103      * @param namespace The namespace URI of the XSObject to
 104      *   retrieve.
 105      * @param localName The local name of the XSObject to retrieve.
 106      * @return A XSObject (of any type) with the specified local
 107      *   name and namespace URI, or null if they do not
 108      *   identify any XSObject in this map.
 109      */
 110     public XSObject itemByName(String namespace, String localName) {
 111         for (int i = 0; i < fNSNum; i++) {
 112             if (isEqual(namespace, fNamespaces[i])) {
 113                 XSTypeDefinition type = (XSTypeDefinition)fMaps[i].get(localName);
 114                 // only return it if it matches the required type
 115                 if (type != null && type.getTypeCategory() == fType) {
 116                     return type;
 117                 }
 118                 return null;
 119             }
 120         }
 121         return null;
 122     }
 123 
 124     /**
 125      * Returns the indexth item in the map. The index starts at
 126      * 0. If index is greater than or equal to the number of
 127      * nodes in the list, this returns null.
 128      * @param index The position in the map from which the item is to be
 129      *   retrieved.
 130      * @return The XSObject at the indexth position
 131      *   in the XSNamedMap, or null if that is
 132      *   not a valid index.
 133      */
 134     public synchronized XSObject item(int index) {
 135         if (fArray == null) {
 136             getLength();
 137         }
 138         if (index < 0 || index >= fLength) {
 139             return null;
 140         }
 141         return fArray[index];
 142     }
