  82     @Override
  83     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  84         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  85         componentColors = extendedCacheKeys;
  86         //generate this entire method. Each state/bg/fg/border combo that has
  87         //been painted gets its own KEY and paint method.
  88         switch(state) {
  89             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
  90             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  91             case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
  92             case BORDER_DISABLED: paintBorderDisabled(g); break;
  93             case BORDER_FOCUSED: paintBorderFocused(g); break;
  94             case BORDER_ENABLED: paintBorderEnabled(g); break;
  95 
  96         }
  97     }

  99     protected Object[] getExtendedCacheKeys(JComponent c) {
 100         Object[] extendedCacheKeys = null;
 101         switch(state) {
 102             case BACKGROUND_ENABLED:
 103                 extendedCacheKeys = new Object[] {
 104                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
 105                 break;
 106             case BORDER_FOCUSED:
 107                 extendedCacheKeys = new Object[] {
 108                      getComponentColor(c, "background", color9, 0.004901961f, -0.19999999f, 0),
 109                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
 110                 break;
 111             case BORDER_ENABLED:
 112                 extendedCacheKeys = new Object[] {
 113                      getComponentColor(c, "background", color9, 0.004901961f, -0.19999999f, 0),
 114                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
 115                 break;
 116         }
 117         return extendedCacheKeys;
 118     }
