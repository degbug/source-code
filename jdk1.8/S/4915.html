  46   /**
  47    * This function is used to fixup variables from QNames to stack frame
  48    * indexes at stylesheet build time.
  49    * @param vars List of QNames that correspond to variables.  This list
  50    * should be searched backwards for the first qualified name that
  51    * corresponds to the variable reference qname.  The position of the
  52    * QName in the vector from the start of the vector will be its position
  53    * in the stack frame (but variables above the globalsTop value will need
  54    * to be offset to the current stack frame).
  55    */
  56   public void fixupVariables(java.util.Vector vars, int globalsSize)
  57   {
  58     m_left.fixupVariables(vars, globalsSize);
  59     m_right.fixupVariables(vars, globalsSize);
  60   }

  63   /**
  64    * Tell if this expression or it's subexpressions can traverse outside
  65    * the current subtree.
  66    *
  67    * @return true if traversal outside the context node's subtree can occur.
  68    */
  69   public boolean canTraverseOutsideSubtree()
  70   {
  71 
  72     if (null != m_left && m_left.canTraverseOutsideSubtree())
  73       return true;
  74 
  75     if (null != m_right && m_right.canTraverseOutsideSubtree())
  76       return true;
  77 
  78     return false;
  79   }

  96   /**
  97    * Execute a binary operation by calling execute on each of the operands,
  98    * and then calling the operate method on the derived class.
  99    *
 100    *
 101    * @param xctxt The runtime execution context.
 102    *
 103    * @return The XObject result of the operation.
 104    *
 105    * @throws javax.xml.transform.TransformerException
 106    */
 107   public XObject execute(XPathContext xctxt)
 108           throws javax.xml.transform.TransformerException
 109   {
 110 
 111     XObject left = m_left.execute(xctxt, true);
 112     XObject right = m_right.execute(xctxt, true);
 113 
 114     XObject result = operate(left, right);
 115     left.detach();
 116     right.detach();
 117     return result;
 118   }

 172   public void callVisitors(ExpressionOwner owner, XPathVisitor visitor)
 173   {
 174         if(visitor.visitBinaryOperation(owner, this))

 176                 m_left.callVisitors(new LeftExprOwner(), visitor);

 198   /**
 199    * @see Expression#deepEquals(Expression)
 200    */
 201   public boolean deepEquals(Expression expr)
 202   {
 203         if(!isSameClass(expr))
 204                 return false;
 205 
 206         if(!m_left.deepEquals(((Operation)expr).m_left))
 207                 return false;
 208 
 209         if(!m_right.deepEquals(((Operation)expr).m_right))
 210                 return false;
 211
