  37 import java.util.concurrent.locks.AbstractQueuedSynchronizer;

  39 /**
  40  * A synchronization aid that allows one or more threads to wait until
  41  * a set of operations being performed in other threads completes.
  42  *
  43  * A {@code CountDownLatch} is initialized with a given count.
  44  * The {@link #await await} methods block until the current count reaches
  45  * zero due to invocations of the {@link #countDown} method, after which
  46  * all waiting threads are released and any subsequent invocations of
  47  * {@link #await await} return immediately.  This is a one-shot phenomenon
  48  * -- the count cannot be reset.  If you need a version that resets the
  49  * count, consider using a {@link CyclicBarrier}.
  50  *
  51  * A {@code CountDownLatch} is a versatile synchronization tool
  52  * and can be used for a number of purposes.  A
  53  * {@code CountDownLatch} initialized with a count of one serves as a
  54  * simple on/off latch, or gate: all threads invoking {@link #await await}
  55  * wait at the gate until it is opened by a thread invoking {@link
  56  * #countDown}.  A {@code CountDownLatch} initialized to N
  57  * can be used to make one thread wait until N threads have
  58  * completed some action, or some action has been completed N times.
  59  *
  60  * A useful property of a {@code CountDownLatch} is that it
  61  * doesn't require that threads calling {@code countDown} wait for
  62  * the count to reach zero before proceeding, it simply prevents any
  63  * thread from proceeding past an {@link #await await} until all
  64  * threads could pass.
  65  *
  66  * Sample usage: Here is a pair of classes in which a group
  67  * of worker threads use two countdown latches:
  68  * 
    
  69  * 
  • The first is a start signal that prevents any worker from proceeding
    • 
  70  * until the driver is ready for them to proceed;
  71  * 
  • The second is a completion signal that allows the driver to wait
    • 
  72  * until all workers have completed.
  73  * 
  74  *
  75  *  
     {@code
    
  76  * class Driver { // ...
  77  *   void main() throws InterruptedException {
  78  *     CountDownLatch startSignal = new CountDownLatch(1);
  79  *     CountDownLatch doneSignal = new CountDownLatch(N);
  80  *
  81  *     for (int i = 0; i < N; ++i) // create and start threads
  82  *       new Thread(new Worker(startSignal, doneSignal)).start();
  83  *
  84  *     doSomethingElse();            // don't let run yet
  85  *     startSignal.countDown();      // let all threads proceed
  86  *     doSomethingElse();
  87  *     doneSignal.await();           // wait for all to finish
  88  *   }
  89  * }
  90  *
  91  * class Worker implements Runnable {
  92  *   private final CountDownLatch startSignal;
  93  *   private final CountDownLatch doneSignal;
  94  *   Worker(CountDownLatch startSignal, CountDownLatch doneSignal) {
  95  *     this.startSignal = startSignal;
  96  *     this.doneSignal = doneSignal;
  97  *   }
  98  *   public void run() {
  99  *     try {
 100  *       startSignal.await();
 101  *       doWork();
 102  *       doneSignal.countDown();
 103  *     } catch (InterruptedException ex) {} // return;
 104  *   }
 105  *
 106  *   void doWork() { ... }
 107  * }}
 108  *
 109  * Another typical usage would be to divide a problem into N parts,
 110  * describe each part with a Runnable that executes that portion and
 111  * counts down on the latch, and queue all the Runnables to an
 112  * Executor.  When all sub-parts are complete, the coordinating thread
 113  * will be able to pass through await. (When threads must repeatedly
 114  * count down in this way, instead use a {@link CyclicBarrier}.)
 115  *
 116  *  
     {@code
    
 117  * class Driver2 { // ...
 118  *   void main() throws InterruptedException {
 119  *     CountDownLatch doneSignal = new CountDownLatch(N);
 120  *     Executor e = ...
 121  *
 122  *     for (int i = 0; i < N; ++i) // create and start threads
 123  *       e.execute(new WorkerRunnable(doneSignal, i));
 124  *
 125  *     doneSignal.await();           // wait for all to finish
 126  *   }
 127  * }
 128  *
 129  * class WorkerRunnable implements Runnable {
 130  *   private final CountDownLatch doneSignal;
 131  *   private final int i;
 132  *   WorkerRunnable(CountDownLatch doneSignal, int i) {
 133  *     this.doneSignal = doneSignal;
 134  *     this.i = i;
 135  *   }
 136  *   public void run() {
 137  *     try {
 138  *       doWork(i);
 139  *       doneSignal.countDown();
 140  *     } catch (InterruptedException ex) {} // return;
 141  *   }
 142  *
 143  *   void doWork() { ... }
 144  * }}
 145  *
 146  * Memory consistency effects: Until the count reaches
 147  * zero, actions in a thread prior to calling
 148  * {@code countDown()}
 149  * happen-before
 150  * actions following a successful return from a corresponding
 151  * {@code await()} in another thread.
 152  *
 153  * @since 1.5
 154  * @author Doug Lea
 155  */
 156 public class CountDownLatch {
 157     /**
 158      * Synchronization control For CountDownLatch.
 159      * Uses AQS state to represent count.
 160      */
 161     private static final class Sync extends AbstractQueuedSynchronizer {
 162         private static final long serialVersionUID = 4982264981922014374L;
 163 
 164         Sync(int count) {
 165             setState(count);
 166         }
 167 
 168         int getCount() {
 169             return getState();
 170         }
 171 
 172         protected int tryAcquireShared(int acquires) {
 173             return (getState() == 0) ? 1 : -1;
 174         }
 175 
 176         protected boolean tryReleaseShared(int releases) {
 177             // Decrement count; signal when transition to zero
 178             for (;;) {
 179                 int c = getState();
 180                 if (c == 0)
 181                     return false;
 182                 int nextc = c-1;
 183                 if (compareAndSetState(c, nextc))
 184                     return nextc == 0;
 185             }
 186         }
 187     }
 188 
 189     private final Sync sync;
 190 
 191     /**
 192      * Constructs a {@code CountDownLatch} initialized with the given count.
 193      *
 194      * @param count the number of times {@link #countDown} must be invoked
 195      *        before threads can pass through {@link #await}
 196      * @throws IllegalArgumentException if {@code count} is negative
 197      */
 198     public CountDownLatch(int count) {
 199         if (count < 0) throw new IllegalArgumentException("count < 0");
 200         this.sync = new Sync(count);
 201     }
 202 
 203     /**
 204      * Causes the current thread to wait until the latch has counted down to
 205      * zero, unless the thread is {@linkplain Thread#interrupt interrupted}.
 206      *
 207      * If the current count is zero then this method returns immediately.
 208      *
 209      * If the current count is greater than zero then the current
 210      * thread becomes disabled for thread scheduling purposes and lies
 211      * dormant until one of two things happen:
 212      * 
      
 213      * 
    • The count reaches zero due to invocations of the
      • 
 214      * {@link #countDown} method; or
 215      * 
    • Some other thread {@linkplain Thread#interrupt interrupts}
      • 
 216      * the current thread.
 217      * 
 218      *
 219      * If the current thread:
 220      * 
        
 221      * 
      • has its interrupted status set on entry to this method; or
        • 
 222      * 
      • is {@linkplain Thread#interrupt interrupted} while waiting,
        • 
 223      * 
 224      * then {@link InterruptedException} is thrown and the current thread's
 225      * interrupted status is cleared.
 226      *
 227      * @throws InterruptedException if the current thread is interrupted
 228      *         while waiting
 229      */
 230     public void await() throws InterruptedException {
 231         sync.acquireSharedInterruptibly(1);
 232     }
 233 
 234     /**
 235      * Causes the current thread to wait until the latch has counted down to
 236      * zero, unless the thread is {@linkplain Thread#interrupt interrupted},
 237      * or the specified waiting time elapses.
 238      *
 239      * If the current count is zero then this method returns immediately
 240      * with the value {@code true}.
 241      *
 242      * If the current count is greater than zero then the current
 243      * thread becomes disabled for thread scheduling purposes and lies
 244      * dormant until one of three things happen:
 245      * 
          
 246      * 
        • The count reaches zero due to invocations of the
          • 
 247      * {@link #countDown} method; or
 248      * 
        • Some other thread {@linkplain Thread#interrupt interrupts}
          • 
 249      * the current thread; or
 250      * 
        • The specified waiting time elapses.
          • 
 251      * 
 252      *
 253      * If the count reaches zero then the method returns with the
 254      * value {@code true}.
 255      *
 256      * If the current thread:
 257      * 
            
 258      * 
          • has its interrupted status set on entry to this method; or
            • 
 259      * 
          • is {@linkplain Thread#interrupt interrupted} while waiting,
            • 
 260      * 
 261      * then {@link InterruptedException} is thrown and the current thread's
 262      * interrupted status is cleared.
 263      *
 264      * If the specified waiting time elapses then the value {@code false}
 265      * is returned.  If the time is less than or equal to zero, the method
 266      * will not wait at all.
 267      *
 268      * @param timeout the maximum time to wait
 269      * @param unit the time unit of the {@code timeout} argument
 270      * @return {@code true} if the count reached zero and {@code false}
 271      *         if the waiting time elapsed before the count reached zero
 272      * @throws InterruptedException if the current thread is interrupted
 273      *         while waiting
 274      */
 275     public boolean await(long timeout, TimeUnit unit)
 276         throws InterruptedException {
 277         return sync.tryAcquireSharedNanos(1, unit.toNanos(timeout));
 278     }
 279 
 280     /**
 281      * Decrements the count of the latch, releasing all waiting threads if
 282      * the count reaches zero.
 283      *
 284      * If the current count is greater than zero then it is decremented.
 285      * If the new count is zero then all waiting threads are re-enabled for
 286      * thread scheduling purposes.
 287      *
 288      * If the current count equals zero then nothing happens.
 289      */
 290     public void countDown() {
 291         sync.releaseShared(1);
 292     }
 293 
 294     /**
 295      * Returns the current count.
 296      *
 297      * This method is typically used for debugging and testing purposes.
 298      *
 299      * @return the current count
 300      */
 301     public long getCount() {
 302         return sync.getCount();
 303     }
 304 
 305     /**
 306      * Returns a string identifying this latch, as well as its state.
 307      * The state, in brackets, includes the String {@code "Count ="}
 308      * followed by the current count.
 309      *
 310      * @return a string identifying this latch, as well as its state
 311      */
 312     public String toString() {
 313         return super.toString() + "[Count = " + sync.getCount() + "]";
 314     }
