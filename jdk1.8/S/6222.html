 305     public String getClassNameFromUri(String uri)
 306     {
 307         String className = (String)_extensionNamespaceTable.get(uri);
 308 
 309         if (className != null)
 310             return className;
 311         else {
 312             if (uri.startsWith(JAVA_EXT_XSLTC)) {
 313                 int length = JAVA_EXT_XSLTC.length() + 1;
 314                 return (uri.length() > length) ? uri.substring(length) : EMPTYSTRING;
 315             }
 316             else if (uri.startsWith(JAVA_EXT_XALAN)) {
 317                 int length = JAVA_EXT_XALAN.length() + 1;
 318                 return (uri.length() > length) ? uri.substring(length) : EMPTYSTRING;
 319             }
 320             else if (uri.startsWith(JAVA_EXT_XALAN_OLD)) {
 321                 int length = JAVA_EXT_XALAN_OLD.length() + 1;
 322                 return (uri.length() > length) ? uri.substring(length) : EMPTYSTRING;
 323             }
 324             else {
 325                 int index = uri.lastIndexOf('/');
 326                 return (index > 0) ? uri.substring(index+1) : uri;
 327             }
 328         }

 331     /**
 332      * Type check a function call. Since different type conversions apply,
 333      * type checking is different for standard and external (Java) functions.
 334      */
 335     @Override
 336     public Type typeCheck(SymbolTable stable)
 337         throws TypeCheckError
 338     {
 339         if (_type != null) return _type;
 340 
 341         final String namespace = _fname.getNamespace();
 342         String local = _fname.getLocalPart();
 343 
 344         if (isExtension()) {
 345             _fname = new QName(null, null, local);
 346             return typeCheckStandard(stable);
 347         }
 348         else if (isStandard()) {
 349             return typeCheckStandard(stable);
 350         }
 351         // Handle extension functions (they all have a namespace)
 352         else {
 353             try {
 354                 _className = getClassNameFromUri(namespace);
 355 
 356                 final int pos = local.lastIndexOf('.');
 357                 if (pos > 0) {
 358                     _isStatic = true;
 359                     if (_className != null && _className.length() > 0) {
 360                         _namespace_format = NAMESPACE_FORMAT_PACKAGE;
 361                         _className = _className + "." + local.substring(0, pos);
 362                     }
 363                     else {
 364                         _namespace_format = NAMESPACE_FORMAT_JAVA;
 365                         _className = local.substring(0, pos);
 366                     }
 367 
 368                     _fname = new QName(namespace, null, local.substring(pos + 1));
 369                 }
 370                 else {
 371                     if (_className != null && _className.length() > 0) {
 372                         try {
 373                             _clazz = ObjectFactory.findProviderClass(_className, true);
 374                             _namespace_format = NAMESPACE_FORMAT_CLASS;
 375                         }
 376                         catch (ClassNotFoundException e) {
 377                             _namespace_format = NAMESPACE_FORMAT_PACKAGE;
 378                         }
 379                     }
 380                     else
 381                         _namespace_format = NAMESPACE_FORMAT_JAVA;
 382 
 383                     if (local.indexOf('-') > 0) {
 384                         local = replaceDash(local);
 385                     }
 386 
 387                     String extFunction = (String)_extensionFunctionTable.get(namespace + ":" + local);
 388                     if (extFunction != null) {
 389                         _fname = new QName(null, null, extFunction);
 390                         return typeCheckStandard(stable);
 391                     }
 392                     else
 393                         _fname = new QName(namespace, null, local);
 394                 }
 395 
 396                 return typeCheckExternal(stable);
 397             }
 398             catch (TypeCheckError e) {
 399                 ErrorMsg errorMsg = e.getErrorMsg();
 400                 if (errorMsg == null) {
 401                     final String name = _fname.getLocalPart();
 402                     errorMsg = new ErrorMsg(ErrorMsg.METHOD_NOT_FOUND_ERR, name);
 403                 }
 404                 getParser().reportError(ERROR, errorMsg);
 405                 return _type = Type.Void;
 406             }
 407           }
 408     }

 411      * Type check a call to a standard function. Insert CastExprs when needed.
 412      * If as a result of the insertion of a CastExpr a type check error is
 413      * thrown, then catch it and re-throw it with a new "this".
 414      */
 415     public Type typeCheckStandard(SymbolTable stable) throws TypeCheckError {
 416         _fname.clearNamespace();        // HACK!!!
 417 
 418         final int n = _arguments.size();
 419         final Vector argsType = typeCheckArgs(stable);
 420         final MethodType args = new MethodType(Type.Void, argsType);
 421         final MethodType ptype =
 422             lookupPrimop(stable, _fname.getLocalPart(), args);
 423 
 424         if (ptype != null) {
 425             for (int i = 0; i < n; i++) {
 426                 final Type argType = (Type) ptype.argsType().elementAt(i);
 427                 final Expression exp = (Expression)_arguments.elementAt(i);
 428                 if (!argType.identicalTo(exp.getType())) {
 429                     try {
 430                         _arguments.setElementAt(new CastExpr(exp, argType), i);
 431                     }
 432                     catch (TypeCheckError e) {

 434                     }
 435                 }
 436             }
 437             _chosenMethodType = ptype;
 438             return _type = ptype.resultType();
 439         }
 440         throw new TypeCheckError(this);

 445     public Type typeCheckConstructor(SymbolTable stable) throws TypeCheckError{
 446         final Vector constructors = findConstructors();
 447         if (constructors == null) {
 448             // Constructor not found in this class
 449             throw new TypeCheckError(ErrorMsg.CONSTRUCTOR_NOT_FOUND,
 450                 _className);
 451 
 452         }
 453 
 454         final int nConstructors = constructors.size();
 455         final int nArgs = _arguments.size();
 456         final Vector argsType = typeCheckArgs(stable);
 457 
 458         // Try all constructors
 459         int bestConstrDistance = Integer.MAX_VALUE;
 460         _type = null;                   // reset
 461         for (int j, i = 0; i < nConstructors; i++) {
 462             // Check if all parameters to this constructor can be converted
 463             final Constructor constructor =
 464                 (Constructor)constructors.elementAt(i);
 465             final Class[] paramTypes = constructor.getParameterTypes();
 466 
 467             Class extType = null;
 468             int currConstrDistance = 0;
 469             for (j = 0; j < nArgs; j++) {
 470                 // Convert from internal (translet) type to external (Java) type
 471                 extType = paramTypes[j];
 472                 final Type intType = (Type)argsType.elementAt(j);
 473                 Object match = _internal2Java.maps(intType, extType);
 474                 if (match != null) {
 475                     currConstrDistance += ((JavaType)match).distance;
 476                 }
 477                 else if (intType instanceof ObjectType) {
 478                     ObjectType objectType = (ObjectType)intType;
 479                     if (objectType.getJavaClass() == extType)
 480                         continue;
 481                     else if (extType.isAssignableFrom(objectType.getJavaClass()))
 482                         currConstrDistance += 1;
 483                     else {
 484                         currConstrDistance = Integer.MAX_VALUE;
 485                         break;
 486                     }
 487                 }
 488                 else {
 489                     // no mapping available
 490                     currConstrDistance = Integer.MAX_VALUE;
 491                     break;
 492                 }
 493             }
 494 
 495             if (j == nArgs && currConstrDistance < bestConstrDistance ) {
 496                 _chosenConstructor = constructor;
 497                 _isExtConstructor = true;
 498                 bestConstrDistance = currConstrDistance;
 499 
 500                 _type = (_clazz != null) ? Type.newObjectType(_clazz)
 501                     : Type.newObjectType(_className);
 502             }
 503         }
 504 
 505         if (_type != null) {
 506             return _type;
 507         }
 508 
 509         throw new TypeCheckError(ErrorMsg.ARGUMENT_CONVERSION_ERR, getMethodSignature(argsType));
 510     }

 513     /**
 514      * Type check a call to an external (Java) method.
 515      * The method must be static an public, and a legal type conversion
 516      * must exist for all its arguments and its return type.
 517      * Every method of name <code>_fname</code> is inspected
 518      * as a possible candidate.
 519      */
 520     public Type typeCheckExternal(SymbolTable stable) throws TypeCheckError {
 521         int nArgs = _arguments.size();
 522         final String name = _fname.getLocalPart();
 523 
 524         // check if function is a contructor 'new'
 525         if (_fname.getLocalPart().equals("new")) {
 526             return typeCheckConstructor(stable);
 527         }
 528         // check if we are calling an instance method
 529         else {
 530             boolean hasThisArgument = false;
 531 
 532             if (nArgs == 0)
 533                 _isStatic = true;
 534 
 535             if (!_isStatic) {
 536                 if (_namespace_format == NAMESPACE_FORMAT_JAVA
 537                     || _namespace_format == NAMESPACE_FORMAT_PACKAGE)
 538                     hasThisArgument = true;
 539 
 540                 Expression firstArg = (Expression)_arguments.elementAt(0);
 541                 Type firstArgType = (Type)firstArg.typeCheck(stable);
 542 
 543                 if (_namespace_format == NAMESPACE_FORMAT_CLASS
 544                     && firstArgType instanceof ObjectType
 545                     && _clazz != null
 546                     && _clazz.isAssignableFrom(((ObjectType)firstArgType).getJavaClass()))
 547                     hasThisArgument = true;
 548 
 549                 if (hasThisArgument) {
 550                     _thisArgument = (Expression) _arguments.elementAt(0);
 551                     _arguments.remove(0); nArgs--;
 552                     if (firstArgType instanceof ObjectType) {
 553                         _className = ((ObjectType) firstArgType).getJavaClassName();
 554                     }
 555                     else
 556                         throw new TypeCheckError(ErrorMsg.NO_JAVA_FUNCT_THIS_REF, name);
 557                 }
 558             }
 559             else if (_className.length() == 0) {
 560                 /*
 561                  * Warn user if external function could not be resolved.
 562                  * Warning will _NOT_ be issued is the call is properly
 563                  * wrapped in an <xsl:if> or <xsl:when> element. For details
 564                  * see If.parserContents() and When.parserContents()
 565                  */
 566                 final Parser parser = getParser();
 567                 if (parser != null) {
 568                     reportWarning(this, parser, ErrorMsg.FUNCTION_RESOLVE_ERR,
 569                                   _fname.toString());
 570                 }
 571                 unresolvedExternal = true;
 572                 return _type = Type.Int;        // use "Int" as "unknown"
 573             }
 574         }
 575 
 576         final Vector methods = findMethods();
 577 
 578         if (methods == null) {
 579             // Method not found in this class
 580             throw new TypeCheckError(ErrorMsg.METHOD_NOT_FOUND_ERR, _className + "." + name);
 581         }
 582 
 583         Class extType = null;
 584         final int nMethods = methods.size();
 585         final Vector argsType = typeCheckArgs(stable);
 586 
 587         // Try all methods to identify the best fit
 588         int bestMethodDistance  = Integer.MAX_VALUE;
 589         _type = null;                       // reset internal type
 590         for (int j, i = 0; i < nMethods; i++) {
 591             // Check if all paramteters to this method can be converted
 592             final Method method = (Method)methods.elementAt(i);
 593             final Class[] paramTypes = method.getParameterTypes();
 594 
 595             int currMethodDistance = 0;
 596             for (j = 0; j < nArgs; j++) {
 597                 // Convert from internal (translet) type to external (Java) type
 598                 extType = paramTypes[j];
 599                 final Type intType = (Type)argsType.elementAt(j);
 600                 Object match = _internal2Java.maps(intType, extType);
 601                 if (match != null) {
 602                     currMethodDistance += ((JavaType)match).distance;
 603                 }
 604                 else {
 605                     // no mapping available
 606                     //
 607                     // Allow a Reference type to match any external (Java) type at
 608                     // the moment. The real type checking is performed at runtime.
 609                     if (intType instanceof ReferenceType) {
 610                        currMethodDistance += 1;
 611                     }
 612                     else if (intType instanceof ObjectType) {
 613                         ObjectType object = (ObjectType)intType;
 614                         if (extType.getName().equals(object.getJavaClassName()))
 615                             currMethodDistance += 0;
 616                         else if (extType.isAssignableFrom(object.getJavaClass()))
 617                             currMethodDistance += 1;
 618                         else {
 619                             currMethodDistance = Integer.MAX_VALUE;
 620                             break;
 621                         }
 622                     }
 623                     else {
 624                         currMethodDistance = Integer.MAX_VALUE;
 625                         break;
 626                     }
 627                 }
 628             }
 629 
 630             if (j == nArgs) {
 631                   // Check if the return type can be converted
 632                   extType = method.getReturnType();
 633 
 634                   _type = (Type) _java2Internal.get(extType);
 635                   if (_type == null) {
 636                       _type = Type.newObjectType(extType);
 637                   }
 638 
 639                   // Use this method if all parameters & return type match
 640                   if (_type != null && currMethodDistance < bestMethodDistance) {
 641                       _chosenMethod = method;
 642                       bestMethodDistance = currMethodDistance;
 643                   }
 644             }
 645         }
 646 
 647         // It is an error if the chosen method is an instance menthod but we don't
 648         // have a this argument.
 649         if (_chosenMethod != null && _thisArgument == null &&
 650             !Modifier.isStatic(_chosenMethod.getModifiers())) {
 651             throw new TypeCheckError(ErrorMsg.NO_JAVA_FUNCT_THIS_REF, getMethodSignature(argsType));
 652         }
 653 
 654         if (_type != null) {
 655             if (_type == Type.NodeSet) {
 656                 getXSLTC().setMultiDocument(true);
 657             }
 658             return _type;
 659         }
 660 
 661         throw new TypeCheckError(ErrorMsg.ARGUMENT_CONVERSION_ERR, getMethodSignature(argsType));
 662     }

 714     /**
 715      * Translate a function call. The compiled code will leave the function's
 716      * return value on the JVM's stack.
 717      */
 718     @Override
 719     public void translate(ClassGenerator classGen, MethodGenerator methodGen) {
 720         final int n = argumentCount();
 721         final ConstantPoolGen cpg = classGen.getConstantPool();
 722         final InstructionList il = methodGen.getInstructionList();
 723         final boolean isSecureProcessing = classGen.getParser().getXSLTC().isSecureProcessing();
 724         final boolean isExtensionFunctionEnabled = classGen.getParser().getXSLTC()
 725                 .getFeature(FeatureManager.Feature.ORACLE_ENABLE_EXTENSION_FUNCTION);
 726         int index;
 727 
 728         // Translate calls to methods in the BasisLibrary
 729         if (isStandard() || isExtension()) {
 730             for (int i = 0; i < n; i++) {
 731                 final Expression exp = argument(i);
 732                 exp.translate(classGen, methodGen);
 733                 exp.startIterator(classGen, methodGen);
 734             }
 735 
 736             // append "F" to the function's name
 737             final String name = _fname.toString().replace('-', '_') + "F";
 738             String args = Constants.EMPTYSTRING;
 739 
 740             // Special precautions for some method calls
 741             if (name.equals("sumF")) {
 742                 args = DOM_INTF_SIG;
 743                 il.append(methodGen.loadDOM());
 744             }
 745             else if (name.equals("normalize_spaceF")) {
 746                 if (_chosenMethodType.toSignature(args).
 747                     equals("()Ljava/lang/String;")) {
 748                     args = "I"+DOM_INTF_SIG;
 749                     il.append(methodGen.loadContextNode());
 750                     il.append(methodGen.loadDOM());
 751                 }
 752             }
 753 
 754             // Invoke the method in the basis library
 755             index = cpg.addMethodref(BASIS_LIBRARY_CLASS, name,
 756                                      _chosenMethodType.toSignature(args));
 757             il.append(new INVOKESTATIC(index));
 758         }
 759         // Add call to BasisLibrary.unresolved_externalF() to generate
 760         // run-time error message for unsupported external functions
 761         else if (unresolvedExternal) {
 762             index = cpg.addMethodref(BASIS_LIBRARY_CLASS,
 763                                      "unresolved_externalF",
 764                                      "(Ljava/lang/String;)V");
 765             il.append(new PUSH(cpg, _fname.toString()));
 766             il.append(new INVOKESTATIC(index));
 767         }
 768         else if (_isExtConstructor) {
 769             if (isSecureProcessing && !isExtensionFunctionEnabled)
 770                 translateUnallowedExtension(cpg, il);
 771 
 772             final String clazz =
 773                 _chosenConstructor.getDeclaringClass().getName();
 774             Class[] paramTypes = _chosenConstructor.getParameterTypes();
 775             LocalVariableGen[] paramTemp = new LocalVariableGen[n];
 776 
 777             // Backwards branches are prohibited if an uninitialized object is
 778             // on the stack by section 4.9.4 of the JVM Specification, 2nd Ed.
 779             // We don't know whether this code might contain backwards branches
 780             // so we mustn't create the new object until after we've created
 781             // the suspect arguments to its constructor.  Instead we calculate
 782             // the values of the arguments to the constructor first, store them
 783             // in temporary variables, create the object and reload the
 784             // arguments from the temporaries to avoid the problem.
 785 
 786             for (int i = 0; i < n; i++) {
 787                 final Expression exp = argument(i);
 788                 Type expType = exp.getType();
 789                 exp.translate(classGen, methodGen);
 790                 // Convert the argument to its Java type
 791                 exp.startIterator(classGen, methodGen);
 792                 expType.translateTo(classGen, methodGen, paramTypes[i]);
 793                 paramTemp[i] =
 794                     methodGen.addLocalVariable("function_call_tmp"+i,
 795                                                expType.toJCType(),
 796                                                null, null);
 797                 paramTemp[i].setStart(
 798                         il.append(expType.STORE(paramTemp[i].getIndex())));
 799 
 800             }
 801 
 802             il.append(new NEW(cpg.addClass(_className)));
 803             il.append(InstructionConstants.DUP);
 804 
 805             for (int i = 0; i < n; i++) {
 806                 final Expression arg = argument(i);
 807                 paramTemp[i].setEnd(
 808                         il.append(arg.getType().LOAD(paramTemp[i].getIndex())));
 809             }
 810 
 811             final StringBuffer buffer = new StringBuffer();
 812             buffer.append('(');
 813             for (int i = 0; i < paramTypes.length; i++) {
 814                 buffer.append(getSignature(paramTypes[i]));
 815             }
 816             buffer.append(')');
 817             buffer.append("V");
 818 
 819             index = cpg.addMethodref(clazz,
 820                                      "<init>",
 821                                      buffer.toString());
 822             il.append(new INVOKESPECIAL(index));
 823 
 824             // Convert the return type back to our internal type
 825             (Type.Object).translateFrom(classGen, methodGen,
 826                                 _chosenConstructor.getDeclaringClass());
 827 
 828         }
 829         // Invoke function calls that are handled in separate classes
 830         else {
 831             if (isSecureProcessing && !isExtensionFunctionEnabled)
 832                 translateUnallowedExtension(cpg, il);
 833 
 834             final String clazz = _chosenMethod.getDeclaringClass().getName();
 835             Class[] paramTypes = _chosenMethod.getParameterTypes();
 836 
 837             // Push "this" if it is an instance method
 838             if (_thisArgument != null) {
 839                 _thisArgument.translate(classGen, methodGen);
 840             }
 841 
 842             for (int i = 0; i < n; i++) {
 843                 final Expression exp = argument(i);
 844                 exp.translate(classGen, methodGen);
 845                 // Convert the argument to its Java type
 846                 exp.startIterator(classGen, methodGen);
 847                 exp.getType().translateTo(classGen, methodGen, paramTypes[i]);
 848             }
 849 
 850             final StringBuffer buffer = new StringBuffer();
 851             buffer.append('(');
 852             for (int i = 0; i < paramTypes.length; i++) {
 853                 buffer.append(getSignature(paramTypes[i]));
 854             }
 855             buffer.append(')');
 856             buffer.append(getSignature(_chosenMethod.getReturnType()));
 857 
 858             if (_thisArgument != null && _clazz.isInterface()) {
 859                 index = cpg.addInterfaceMethodref(clazz,
 860                                      _fname.getLocalPart(),
 861                                      buffer.toString());
 862                 il.append(new INVOKEINTERFACE(index, n+1));
 863             }
 864             else {
 865                 index = cpg.addMethodref(clazz,
 866                                      _fname.getLocalPart(),
 867                                      buffer.toString());
 868                 il.append(_thisArgument != null ? (InvokeInstruction) new INVOKEVIRTUAL(index) :
 869                           (InvokeInstruction) new INVOKESTATIC(index));
 870             }
 871 
 872             // Convert the return type back to our internal type
 873             _type.translateFrom(classGen, methodGen,
 874                                 _chosenMethod.getReturnType());
 875         }
 876     }

 893     /**
 894      * Returns a vector with all methods named <code>_fname</code>
 895      * after stripping its namespace or <code>null</code>
 896      * if no such methods exist.
 897      */
 898     private Vector findMethods() {
 899 
 900           Vector result = null;
 901           final String namespace = _fname.getNamespace();
 902 
 903           if (_className != null && _className.length() > 0) {
 904             final int nArgs = _arguments.size();
 905             try {
 906                 if (_clazz == null) {
 907                     final boolean isSecureProcessing = getXSLTC().isSecureProcessing();
 908                     final boolean isExtensionFunctionEnabled = getXSLTC()
 909                             .getFeature(FeatureManager.Feature.ORACLE_ENABLE_EXTENSION_FUNCTION);
 910 
 911                     //Check if FSP and SM - only then proceed with loading
 912                     if (namespace != null && isSecureProcessing
 913                             && isExtensionFunctionEnabled
 914                             && (namespace.equals(JAVA_EXT_XALAN)
 915                             || namespace.equals(JAVA_EXT_XSLTC)
 916                             || namespace.equals(JAVA_EXT_XALAN_OLD)
 917                             || namespace.startsWith(XALAN_CLASSPACKAGE_NAMESPACE))) {
 918                         _clazz = getXSLTC().loadExternalFunction(_className);
 919                     } else {
 920                         _clazz = ObjectFactory.findProviderClass(_className, true);
 921                     }
 922 
 923                 if (_clazz == null) {
 924                   final ErrorMsg msg =
 925                         new ErrorMsg(ErrorMsg.CLASS_NOT_FOUND_ERR, _className);
 926                   getParser().reportError(Constants.ERROR, msg);
 927                 }
 928               }
 929 
 930               final String methodName = _fname.getLocalPart();
 931               final Method[] methods = _clazz.getMethods();
 932 
 933               for (int i = 0; i < methods.length; i++) {
 934                 final int mods = methods[i].getModifiers();
 935                 // Is it public and same number of args ?
 936                 if (Modifier.isPublic(mods)
 937                     && methods[i].getName().equals(methodName)
 938                     && methods[i].getParameterTypes().length == nArgs)
 939                 {
 940                   if (result == null) {
 941                     result = new Vector();
 942                   }
 943                   result.addElement(methods[i]);
 944                 }
 945               }
 946             }
 947             catch (ClassNotFoundException e) {
 948                   final ErrorMsg msg = new ErrorMsg(ErrorMsg.CLASS_NOT_FOUND_ERR, _className);
 949                   getParser().reportError(Constants.ERROR, msg);
 950             }
 951           }
 952           return result;
 953     }

 999     /**
1000      * Compute the JVM signature for the class.
1001      */
1002     static final String getSignature(Class clazz) {
1003         if (clazz.isArray()) {
1004             final StringBuffer sb = new StringBuffer();
1005             Class cl = clazz;
1006             while (cl.isArray()) {
1007                 sb.append("[");
1008                 cl = cl.getComponentType();
1009             }
1010             sb.append(getSignature(cl));
1011             return sb.toString();
1012         }
1013         else if (clazz.isPrimitive()) {
1014             if (clazz == Integer.TYPE) {
1015                 return "I";
1016             }
1017             else if (clazz == Byte.TYPE) {
1018                 return "B";
1019             }
1020             else if (clazz == Long.TYPE) {
1021                 return "J";
1022             }
1023             else if (clazz == Float.TYPE) {
1024                 return "F";
1025             }
1026             else if (clazz == Double.TYPE) {
1027                 return "D";
1028             }
1029             else if (clazz == Short.TYPE) {
1030                 return "S";
1031             }
1032             else if (clazz == Character.TYPE) {
1033                 return "C";
1034             }
1035             else if (clazz == Boolean.TYPE) {
1036                 return "Z";
1037             }
1038             else if (clazz == Void.TYPE) {
1039                 return "V";
1040             }
1041             else {
1042                 final String name = clazz.toString();
1043                 ErrorMsg err = new ErrorMsg(ErrorMsg.UNKNOWN_SIG_TYPE_ERR,name);
1044                 throw new Error(err.toString());
1045             }
1046         }
1047         else {
1048             return "L" + clazz.getName().replace('.', '/') + ';';
1049         }
1050     }

1097     /**
1098      * To support EXSLT extensions, convert names with dash to allowable Java names:
1099      * e.g., convert abc-xyz to abcXyz.
1100      * Note: dashes only appear in middle of an EXSLT function or element name.
1101      */
1102     protected static String replaceDash(String name)
1103     {
1104         char dash = '-';
1105         final StringBuilder buff = new StringBuilder("");
1106         for (int i = 0; i < name.length(); i++) {
1107         if (i > 0 && name.charAt(i-1) == dash)
1108             buff.append(Character.toUpperCase(name.charAt(i)));
1109         else if (name.charAt(i) != dash)
1110             buff.append(name.charAt(i));
1111         }
1112         return buff.toString();
1113     }
