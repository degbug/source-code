 106         /**
 107          * Construct a Duration of type xdt:yearMonthDuration using the specified milliseconds as defined in
 108          * 
 109          *   XQuery 1.0 and XPath 2.0 Data Model, xdt:yearMonthDuration.
 110          *
 111          * The datatype xdt:yearMonthDuration is a subtype of xs:duration
 112          * whose lexical representation contains only year and month components.
 113          * This datatype resides in the namespace {@link javax.xml.XMLConstants#W3C_XPATH_DATATYPE_NS_URI}.
 114          *
 115      * Both values are set by computing their values from the specified milliseconds
 116      * and are availabe using the get methods of  the created {@link Duration}.
 117      * The values conform to and are defined by:
 118      * 
    
 119      *   
  • ISO 8601:2000(E) Section 5.5.3.2 Alternative format
    • 
 120      *   
    • 
 121      *     W3C XML Schema 1.0 Part 2, Appendix D, ISO 8601 Date and Time Formats
 122      *   
 123      *   
  • {@link XMLGregorianCalendar}  Date/Time Datatype Field Mapping Between XML Schema 1.0 and Java Representation
    • 
 124      * 
 125      *
 126          * The default start instance is defined by {@link GregorianCalendar}'s use of the start of the epoch: i.e.,
 127          * {@link java.util.Calendar#YEAR} = 1970,
 128          * {@link java.util.Calendar#MONTH} = {@link java.util.Calendar#JANUARY},
 129          * {@link java.util.Calendar#DATE} = 1, etc.
 130          * This is important as there are variations in the Gregorian Calendar,
 131          * e.g. leap years have different days in the month = {@link java.util.Calendar#FEBRUARY}
 132          * so the result of {@link Duration#getMonths()} can be influenced.
 133          *
 134      * Any remaining milliseconds after determining the year and month are discarded.
 135          *
 136          * @param durationInMilliseconds Milliseconds of Duration to create.
 137          */
 138     protected DurationYearMonthImpl(long durationInMilliseconds) {
 139 
 140         super(durationInMilliseconds);
 141         convertToCanonicalYearMonth();
 142         //Any remaining milliseconds after determining the year and month are discarded.
 143         days = null;
 144         hours = null;
 145         minutes = null;
 146         seconds = null;
 147         signum = calcSignum((signum<0)?false:true);
 148     }

 151         /**
 152          * Construct a Duration of type xdt:yearMonthDuration by parsing its String representation,
 153          * "PnYnM", 
 154          *   XQuery 1.0 and XPath 2.0 Data Model, xdt:yearMonthDuration.
 155          *
 156          * The datatype xdt:yearMonthDuration is a subtype of xs:duration
 157          * whose lexical representation contains only year and month components.
 158          * This datatype resides in the namespace {@link javax.xml.XMLConstants#W3C_XPATH_DATATYPE_NS_URI}.
 159          *
 160      * Both values are set and availabe from the created {@link Duration}
 161          *
 162      * The XML Schema specification states that values can be of an arbitrary size.
 163      * Implementations may chose not to or be incapable of supporting arbitrarily large and/or small values.
 164      * An {@link UnsupportedOperationException} will be thrown with a message indicating implementation limits
 165      * if implementation capacities are exceeded.
 166      *
 167          * @param lexicalRepresentation Lexical representation of a duration.
 168          *
 169          * @throws IllegalArgumentException If lexicalRepresentation is not a valid representation of a Duration expressed only in terms of years and months.
 170          * @throws UnsupportedOperationException If implementation cannot support requested values.
 171          * @throws NullPointerException If lexicalRepresentation is null.
 172          */
 173     protected DurationYearMonthImpl(String lexicalRepresentation) {
 174         super(lexicalRepresentation);
 175         if (getDays() > 0 || getHours() > 0
 176                 || getMinutes() > 0 || getSeconds() > 0) {
 177             throw new IllegalArgumentException(
 178                     "Trying to create an xdt:yearMonthDuration with an invalid"
 179                     + " lexical representation of \"" + lexicalRepresentation
 180                     + "\", data model requires PnYnM.");
 181         }
 182         convertToCanonicalYearMonth();
 183     }

 185     /**
 186      * The value space of xs:yearMonthDuration is the set of xs:integer month values.
 187      * @return the value of yearMonthDuration
 188      */
 189     public int getValue() {
 190         return getYears() * 12 + getMonths();
 191     }
 192 
 193     private void convertToCanonicalYearMonth() {
 194         while (getMonths() >= 12)
 195         {
 196             months = months.subtract(BigInteger.valueOf(12));
 197             years = BigInteger.valueOf((long) getYears()).add(BigInteger.ONE);
 198         }
 199     }
