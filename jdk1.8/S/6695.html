 147     public String getFileName() {
 148         JFileChooser fc = getFileChooser();
 149         String typedInName = fileNameTextField != null ?
 150             fileNameTextField.getText() : null;
 151 
 152         if (!fc.isMultiSelectionEnabled()) {
 153             return typedInName;
 154         }
 155 
 156         int mode = fc.getFileSelectionMode();
 157         JList list = mode == JFileChooser.DIRECTORIES_ONLY ?
 158             directoryList : fileList;
 159         Object[] files = list.getSelectedValues();
 160         int len = files.length;
 161         Vector<String> result = new Vector<String>(len + 1);
 162 
 163         // we return all selected file names
 164         for (int i = 0; i < len; i++) {
 165             File file = (File)files[i];
 166             result.add(file.getName());
 167         }
 168         // plus the file name typed into the text field, if not already there
 169         if (typedInName != null && !result.contains(typedInName)) {
 170             result.add(typedInName);
 171         }
 172 
 173         StringBuffer buf = new StringBuffer();
 174         len = result.size();
 175 
 176         // construct the resulting string
 177         for (int i=0; i<len; i++) {
 178             if (i > 0) {
 179                 buf.append(" ");
 180             }
 181             if (len > 1) {
 182                 buf.append("\"");
 183             }
 184             buf.append(result.get(i));
 185             if (len > 1) {
 186                 buf.append("\"");
 187             }
 188         }
 189         return buf.toString();
 190     }

 274     protected void doDirectoryChanged(PropertyChangeEvent e) {
 275         directoryList.clearSelection();
 276         ListSelectionModel sm = directoryList.getSelectionModel();
 277         if (sm instanceof DefaultListSelectionModel) {
 278             ((DefaultListSelectionModel)sm).moveLeadSelectionIndex(0);
 279             sm.setAnchorSelectionIndex(0);
 280         }
 281         fileList.clearSelection();
 282         sm = fileList.getSelectionModel();
 283         if (sm instanceof DefaultListSelectionModel) {
 284             ((DefaultListSelectionModel)sm).moveLeadSelectionIndex(0);
 285             sm.setAnchorSelectionIndex(0);
 286         }
 287 
 288         File currentDirectory = getFileChooser().getCurrentDirectory();
 289         if (currentDirectory != null) {
 290             try {
 291                 setDirectoryName(ShellFolder.getNormalizedFile((File)e.getNewValue()).getPath());
 292             } catch (IOException ioe) {
 293                 setDirectoryName(((File)e.getNewValue()).getAbsolutePath());
 294             }
 295             if ((getFileChooser().getFileSelectionMode() == JFileChooser.DIRECTORIES_ONLY) && !getFileChooser().isMultiSelectionEnabled()) {
 296                 setFileName(pathField.getText());
 297             }
 298             directoryComboBoxModel.addItem(currentDirectory);
 299             directoryListModel.directoryChanged();
 300         }
 301         super.doDirectoryChanged(e);

 372     class DoubleClickListener extends MouseAdapter {
 373         JList list;
 374         public  DoubleClickListener(JList list) {
 375             this.list = list;
 376         }

 378         public void mouseClicked(MouseEvent e) {
 379             if (SwingUtilities.isLeftMouseButton(e) && e.getClickCount() == 2) {
 380                 int index = list.locationToIndex(e.getPoint());
 381                 if (index >= 0) {
 382                     File f = (File) list.getModel().getElementAt(index);
 383                     try {
 384                         // Strip trailing ".."
 385                         f = ShellFolder.getNormalizedFile(f);
 386                     } catch (IOException ex) {
 387                         // That's ok, we'll use f as is
 388                     }
 389                     if (getFileChooser().isTraversable(f)) {
 390                         list.clearSelection();
 391                         if (getFileChooser().getCurrentDirectory().equals(f)){
 392                             rescanCurrentDirectory(getFileChooser());
 393                         } else {
 394                             getFileChooser().setCurrentDirectory(f);
 395                         }
 396                     } else {
 397                         getFileChooser().approveSelection();
 398                     }
 399                 }
 400             }
 401         }
 402 
 403         public void mouseEntered(MouseEvent evt) {
 404             if (list != null) {
 405                 TransferHandler th1 = getFileChooser().getTransferHandler();
 406                 TransferHandler th2 = list.getTransferHandler();
 407                 if (th1 != th2) {
 408                     list.setTransferHandler(th1);
 409                 }
 410                 if (getFileChooser().getDragEnabled() != list.getDragEnabled()) {
 411                     list.setDragEnabled(getFileChooser().getDragEnabled());
 412                 }
 413             }
 414         }
 415     }

 423     protected class SelectionListener implements ListSelectionListener {
 424         public void valueChanged(ListSelectionEvent e) {
 425             if (!e.getValueIsAdjusting()) {
 426                 JFileChooser chooser = getFileChooser();
 427                 JList list = (JList) e.getSource();
 428 
 429                 if (chooser.isMultiSelectionEnabled()) {
 430                     File[] files = null;
 431                     Object[] objects = list.getSelectedValues();
 432                     if (objects != null) {
 433                         if (objects.length == 1
 434                             && ((File)objects[0]).isDirectory()
 435                             && chooser.isTraversable(((File)objects[0]))
 436                             && (chooser.getFileSelectionMode() != chooser.DIRECTORIES_ONLY
 437                                 || !chooser.getFileSystemView().isFileSystem(((File)objects[0])))) {
 438                             setDirectorySelected(true);
 439                             setDirectory(((File)objects[0]));
 440                         } else {
 441                             ArrayList<File> fList = new ArrayList<File>(objects.length);
 442                             for (Object object : objects) {
 443                                 File f = (File) object;
 444                                 if ((chooser.isFileSelectionEnabled() && f.isFile())
 445                                     || (chooser.isDirectorySelectionEnabled() && f.isDirectory())) {
 446                                     fList.add(f);
 447                                 }
 448                             }
 449                             if (fList.size() > 0) {
 450                                 files = fList.toArray(new File[fList.size()]);
 451                             }
 452                             setDirectorySelected(false);
 453                         }
 454                     }
 455                     chooser.setSelectedFiles(files);
 456                 } else {
 457                     File file = (File)list.getSelectedValue();
 458                     if (file != null
 459                         && file.isDirectory()
 460                         && chooser.isTraversable(file)
 461                         && (chooser.getFileSelectionMode() == chooser.FILES_ONLY
 462                             || !chooser.getFileSystemView().isFileSystem(file))) {
 463 
 464                         setDirectorySelected(true);
 465                         setDirectory(file);
 466                     } else {
 467                         setDirectorySelected(false);
 468                         if (file != null) {
 469                             chooser.setSelectedFile(file);
 470                         }
 471                     }
 472                 }
 473             }
 474         }
 475     }

1088     /**
1089      * Data model for a type-face selection combo-box.
1090      */
1091     protected class DirectoryComboBoxModel extends AbstractListModel implements ComboBoxModel {
1092         Vector<File> directories = new Vector<File>();
1093         File selectedDirectory = null;
1094         JFileChooser chooser = getFileChooser();
1095         FileSystemView fsv = chooser.getFileSystemView();
1096 
1097         public DirectoryComboBoxModel() {
1098             // Add the current directory to the model, and make it the
1099             // selectedDirectory
1100             File dir = getFileChooser().getCurrentDirectory();
1101             if (dir != null) {
1102                 addItem(dir);
1103             }
1104         }
1105 
1106         /**
1107          * Adds the directory to the model and sets it to be selected,
1108          * additionally clears out the previous selected directory and
1109          * the paths leading up to it, if any.
1110          */
1111         private void addItem(File directory) {
1112 
1113             if (directory == null) {
1114                 return;
1115             }
1116 
1117             int oldSize = directories.size();
1118             directories.clear();
1119             if (oldSize > 0) {
1120                 fireIntervalRemoved(this, 0, oldSize);
1121             }
1122 
1123             // Get the canonical (full) path. This has the side
1124             // benefit of removing extraneous chars from the path,
1125             // for example /foo/bar/ becomes /foo/bar
1126             File canonical;
1127             try {
1128                 canonical = fsv.createFileObject(ShellFolder.getNormalizedFile(directory).getPath());
1129             } catch (IOException e) {
1130                 // Maybe drive is not ready. Can't abort here.
1131                 canonical = directory;
1132             }
1133 
1134             // create File instances of each directory leading up to the top
1135             File f = canonical;
1136             do {
1137                 directories.add(f);
1138             } while ((f = f.getParentFile()) != null);
1139             int newSize = directories.size();
1140             if (newSize > 0) {
1141                 fireIntervalAdded(this, 0, newSize);
1142             }
1143             setSelectedItem(canonical);
1144         }
1145 
1146         public void setSelectedItem(Object selectedDirectory) {
1147             this.selectedDirectory = (File)selectedDirectory;
1148             fireContentsChanged(this, -1, -1);
1149         }
1150 
1151         public Object getSelectedItem() {
1152             return selectedDirectory;
1153         }
1154 
1155         public int getSize() {
1156             return directories.size();
1157         }
1158 
1159         public Object getElementAt(int index) {
1160             return directories.elementAt(index);
1161         }
1162     }

1181     private class NewFolderAction extends AbstractAction {
1182         protected NewFolderAction() {
1183             super(FilePane.ACTION_NEW_FOLDER);
1184         }
1185         public void actionPerformed(ActionEvent e) {
1186             if (readOnly) {
1187                 return;
1188             }
1189             JFileChooser fc = getFileChooser();
1190             File currentDirectory = fc.getCurrentDirectory();
1191             String dirName = JOptionPane.showInputDialog(fc,
1192                     newFolderDialogText, newFolderButtonText,
1193                     JOptionPane.PLAIN_MESSAGE);
1194 
1195             if (dirName != null) {
1196                 if (!currentDirectory.exists()) {
1197                     JOptionPane.showMessageDialog(fc,
1198                             MessageFormat.format(newFolderNoDirectoryErrorText, dirName),
1199                             newFolderNoDirectoryErrorTitleText, JOptionPane.ERROR_MESSAGE);
1200                     return;
1201                 }
1202 
1203                 File newDir = fc.getFileSystemView().createFileObject
1204                         (currentDirectory, dirName);
1205                 if (newDir == null || !newDir.mkdir()) {
1206                     JOptionPane.showMessageDialog(fc,
1207                             newFolderErrorText + newFolderErrorSeparator + " \"" +
1208                             dirName + "\"",
1209                             newFolderErrorText, JOptionPane.ERROR_MESSAGE);
1210                 }
1211                 fc.rescanCurrentDirectory();
1212             }
1213         }
1214     }

1251         public void actionPerformed(ActionEvent e) {

1266                 File newFile = fc.getFileSystemView().createFileObject
1267                         (currentDirectory, newFileName);
1268                 if (oldFile == null || newFile == null ||
1269                         !getModel().renameFile(oldFile, newFile)) {
1270                     JOptionPane.showMessageDialog(fc,
1271                             new MessageFormat(renameFileErrorText).
1272                             format(new Object[] { getFileName(), newFileName}),
1273                             renameFileErrorTitle, JOptionPane.ERROR_MESSAGE);
1274                 } else {
1275                     setFileName(getFileChooser().getName(newFile));
1276                     fc.rescanCurrentDirectory();
1277                 }
1278             }
1279         }

1332     /**
1333      * Data model for filter combo-box.
1334      */
1335     protected class FilterComboBoxModel extends AbstractListModel
1336             implements ComboBoxModel, PropertyChangeListener {
1337         protected FileFilter[] filters;
1338 
1339         protected FilterComboBoxModel() {
1340             super();
1341             filters = getFileChooser().getChoosableFileFilters();
1342         }
1343 
1344         public void propertyChange(PropertyChangeEvent e) {
1345             String prop = e.getPropertyName();
1346             if (prop == JFileChooser.CHOOSABLE_FILE_FILTER_CHANGED_PROPERTY) {
1347                 filters = (FileFilter[]) e.getNewValue();
1348                 fireContentsChanged(this, -1, -1);
1349             } else if (prop == JFileChooser.FILE_FILTER_CHANGED_PROPERTY) {
1350                 fireContentsChanged(this, -1, -1);
1351             }
1352         }
1353 
1354         public void setSelectedItem(Object filter) {
1355             if (filter != null) {
1356                 getFileChooser().setFileFilter((FileFilter) filter);
1357                 fireContentsChanged(this, -1, -1);
1358             }
1359         }
1360 
1361         public Object getSelectedItem() {
1362             // Ensure that the current filter is in the list.
1363             // NOTE: we shouldnt' have to do this, since JFileChooser adds
1364             // the filter to the choosable filters list when the filter
1365             // is set. Lets be paranoid just in case someone overrides
1366             // setFileFilter in JFileChooser.
1367             FileFilter currentFilter = getFileChooser().getFileFilter();
1368             boolean found = false;
1369             if (currentFilter != null) {
1370                 for (FileFilter filter : filters) {
1371                     if (filter == currentFilter) {
1372                         found = true;
1373                     }
1374                 }
1375                 if (found == false) {
1376                     getFileChooser().addChoosableFileFilter(currentFilter);
1377                 }
1378             }
1379             return getFileChooser().getFileFilter();
1380         }
1381 
1382         public int getSize() {
1383             if (filters != null) {
1384                 return filters.length;
1385             } else {
1386                 return 0;
1387             }
1388         }
1389 
1390         public Object getElementAt(int index) {
1391             if (index > getSize() - 1) {
1392                 // This shouldn't happen. Try to recover gracefully.
1393                 return getFileChooser().getFileFilter();
1394             }
1395             if (filters != null) {
1396                 return filters[index];
1397             } else {
1398                 return null;
1399             }
1400         }
