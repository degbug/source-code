  77     @Override
  78     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  79         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  80         componentColors = extendedCacheKeys;
  81         //generate this entire method. Each state/bg/fg/border combo that has
  82         //been painted gets its own KEY and paint method.
  83         switch(state) {
  84             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
  85             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  86 
  87         }
  88     }
