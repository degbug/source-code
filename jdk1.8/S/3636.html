  33 /**
  34  * This permission represents "trust" in a signer or codebase.
  35  * <p>
  36  * MBeanTrustPermission contains a target name but no actions list.
  37  * A single target name, "register", is defined for this permission.
  38  * The target "*" is also allowed, permitting "register" and any future
  39  * targets that may be defined.
  40  * Only the null value or the empty string are allowed for the action
  41  * to allow the policy object to create the permissions specified in
  42  * the policy file.
  43  * <p>
  44  * If a signer, or codesource is granted this permission, then it is
  45  * considered a trusted source for MBeans. Only MBeans from trusted
  46  * sources may be registered in the MBeanServer.
  47  *
  48  * @since 1.5
  49  */
  50 public class MBeanTrustPermission extends BasicPermission {
  51 
  52     private static final long serialVersionUID = -2952178077029018140L;
  53 
  54     /** <p>Create a new MBeanTrustPermission with the given name.</p>
  55         <p>This constructor is equivalent to
  56         <code>MBeanTrustPermission(name,null)</code>.</p>
  57         @param name the name of the permission. It must be
  58         "register" or "*" for this permission.
  59      *
  60      * @throws NullPointerException if <code>name</code> is <code>null</code>.
  61      * @throws IllegalArgumentException if <code>name</code> is neither
  62      * "register" nor "*".
  63      */
  64     public MBeanTrustPermission(String name) {
  65         this(name, null);
  66     }
  67 
  68     /** <p>Create a new MBeanTrustPermission with the given name.</p>
  69         @param name the name of the permission. It must be
  70         "register" or "*" for this permission.
  71         @param actions the actions for the permission.  It must be
  72         null or <code>""</code>.
  73      *
  74      * @throws NullPointerException if <code>name</code> is <code>null</code>.
  75      * @throws IllegalArgumentException if <code>name</code> is neither
  76      * "register" nor "*"; or if <code>actions</code> is a non-null
  77      * non-empty string.
  78      */
  79     public MBeanTrustPermission(String name, String actions) {
  80         super(name, actions);
  81         validate(name,actions);
  82     }
  83 
  84     private static void validate(String name, String actions) {
  85         /* Check that actions is a null empty string */
  86         if (actions != null && actions.length() > 0) {
  87             throw new IllegalArgumentException("MBeanTrustPermission actions must be null: " +
  88                                                actions);
  89         }
  90 
  91         if (!name.equals("register") && !name.equals("*")) {
  92             throw new IllegalArgumentException("MBeanTrustPermission: Unknown target name " +
  93                                                "[" + name + "]");
  94         }
  95     }

 105             throw new InvalidObjectException(e.getMessage());
 106         }
 107     }
