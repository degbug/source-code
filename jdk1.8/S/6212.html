  42 final class LangCall extends FunctionCall {
  43     private Expression _lang;
  44     private Type _langType;
  45 
  46     /**
  47      * Get the parameters passed to function:
  48      *   lang(string)
  49      */
  50     public LangCall(QName fname, Vector arguments) {
  51         super(fname, arguments);
  52         _lang = argument(0);
  53     }
  54 
  55     /**
  56      *
  57      */
  58     public Type typeCheck(SymbolTable stable) throws TypeCheckError {
  59         _langType = _lang.typeCheck(stable);
  60         if (!(_langType instanceof StringType)) {
  61             _lang = new CastExpr(_lang, Type.String);
  62         }
  63         return Type.Boolean;
  64     }
  65 
  66     /**
  67      *
  68      */
  69     public Type getType() {
  70         return(Type.Boolean);
  71     }
  72 
  73     /**
  74      * This method is called when the constructor is compiled in
  75      * Stylesheet.compileConstructor() and not as the syntax tree is traversed.
  76      */
  77     public void translate(ClassGenerator classGen,
  78                           MethodGenerator methodGen) {
  79         final ConstantPoolGen cpg = classGen.getConstantPool();
  80         final InstructionList il = methodGen.getInstructionList();
  81 
  82         final int tst = cpg.addMethodref(BASIS_LIBRARY_CLASS,
  83                                          "testLanguage",
  84                                          "("+STRING_SIG+DOM_INTF_SIG+"I)Z");
  85         _lang.translate(classGen,methodGen);
  86         il.append(methodGen.loadDOM());
  87         if (classGen instanceof FilterGenerator)
  88             il.append(new ILOAD(1));
  89         else
  90             il.append(methodGen.loadContextNode());
  91         il.append(new INVOKESTATIC(tst));
  92     }
  93 }
