  90     private void restart(IOException ioe) throws IOException {
  91         // check state
  92         synchronized(lock) {
  93             if (state == TERMINATED) {
  94                 throw new IOException("The client has been closed.");
  95             } else if (state == FAILED) { // already failed to re-start by another thread
  96                 throw ioe;
  97             } else if (state == RE_CONNECTING) {
  98                 // restart process has been called by another thread
  99                 // we need to wait
 100                 while(state == RE_CONNECTING) {
 101                     try {
 102                         lock.wait();
 103                     } catch (InterruptedException ire) {
 104                         // be asked to give up
 105                         InterruptedIOException iioe = new InterruptedIOException(ire.toString());
 106                         EnvHelp.initCause(iioe, ire);
 107 
 108                         throw iioe;
 109                     }
 110                 }
 111 
 112                 if (state == TERMINATED) {
 113                     throw new IOException("The client has been closed.");
 114                 } else if (state != CONNECTED) {
 115                     // restarted is failed by another thread
 116                     throw ioe;
 117                 }
 118                 return;
 119             } else {
 120                 state = RE_CONNECTING;
 121                 lock.notifyAll();
 122             }
 123         }
 124 
 125         // re-starting
 126         try {
 127             doStart();
 128             synchronized(lock) {
 129                 if (state == TERMINATED) {
 130                     throw new IOException("The client has been closed.");
 131                 }
 132 
 133                 state = CONNECTED;
 134 
 135                 lock.notifyAll();
 136             }
 137 
 138             return;
 139         } catch (Exception e) {
 140             logger.warning("restart", "Failed to restart: " + e);
 141             logger.debug("restart",e);
 142 
 143             synchronized(lock) {
 144                 if (state == TERMINATED) {
 145                     throw new IOException("The client has been closed.");
 146                 }
 147 
 148                 state = FAILED;
 149 
 150                 lock.notifyAll();
 151             }
 152 
 153             try {
 154                 doStop();
 155             } catch (Exception eee) {
 156                 // OK.
 157                 // We know there is a problem.
 158             }
 159 
 160             terminate();
 161 
 162             throw ioe;
 163         }
 164     }

 172 
 173             while (state != TERMINATED && !myThread.isInterrupted()) {
 174                 try {

 182                     break;
 183                 }

 192                     }

 217                 }
 218             }
