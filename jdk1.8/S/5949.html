  26 import org.w3c.dom.*;

  28 /**
  29  * Static Namespace query methods.
  30  *
  31  * <p>This class defines a set of static methods that can be called
  32  * to analyze the namespace properties of DOM nodes.</p>
  33  *
  34  * @author Norman Walsh
  35  * <a href="mailto:Norman.Walsh@Sun.COM">Norman.Walsh@Sun.COM</a>
  36  *
  37  */
  38 public class Namespaces {
  39     /**
  40      * Returns the "prefix" part of a QName or the empty string (not
  41      * null) if the name has no prefix.
  42      *
  43      * @param element The QName of an element.
  44      * @return The prefix part of the element name.
  45      */
  46     public static String getPrefix(Element element) {
  47         String name = element.getTagName();
  48         String prefix = "";
  49 
  50         if (name.indexOf(':') > 0) {
  51             prefix = name.substring(0, name.indexOf(':'));
  52         }
  53 
  54         return prefix;
  55     }
  56 
  57     /**
  58      * Returns the "localname" part of a QName, which is the whole
  59      * name if it has no prefix.
  60      *
  61      * @param element The QName of an element.
  62      * @return The local part of a QName.
  63      */
  64     public static String getLocalName(Element element) {
  65         String name = element.getTagName();
  66 
  67         if (name.indexOf(':') > 0) {
  68             name = name.substring(name.indexOf(':')+1);
  69         }
  70 
  71         return name;
  72     }
  73 
  74     /**
  75      * Returns the namespace URI for the specified prefix at the
  76      * specified context node.
  77      *
  78      * @param node The context node.
  79      * @param prefix The prefix.
  80      * @return The namespace URI associated with the prefix, or
  81      * null if no namespace declaration exists for the prefix.
  82      */
  83     public static String getNamespaceURI(Node node, String prefix) {
  84         if (node == null || node.getNodeType() != Node.ELEMENT_NODE) {
  85             return null;
  86         }
  87 
  88         if (prefix.equals("")) {
  89             if (((Element) node).hasAttribute("xmlns")) {
  90                 return ((Element) node).getAttribute("xmlns");
  91             }
  92         } else {
  93             String nsattr = "xmlns:" + prefix;
  94             if (((Element) node).hasAttribute(nsattr)) {
  95                 return ((Element) node).getAttribute(nsattr);
  96             }
  97         }
  98 
  99         return getNamespaceURI(node.getParentNode(), prefix);
 100     }
 101 
 102     /**
 103      * Returns the namespace URI for the namespace to which the
 104      * element belongs.
 105      *
 106      * @param element The element.
 107      * @return The namespace URI associated with the namespace of the
 108      * element, or null if no namespace declaration exists for it.
 109      */
 110     public static String getNamespaceURI(Element element) {
 111         String prefix = getPrefix(element);
 112         return getNamespaceURI(element, prefix);
 113     }
 114 }
