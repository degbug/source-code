 214      * Implementation of ComboPopup.hide().
 215      */
 216     public void hide() {
 217         MenuSelectionManager manager = MenuSelectionManager.defaultManager();
 218         MenuElement [] selection = manager.getSelectedPath();
 219         for ( int i = 0 ; i < selection.length ; i++ ) {
 220             if ( selection[i] == this ) {
 221                 manager.clearSelectedPath();
 222                 break;
 223             }
 224         }
 225         if (selection.length > 0) {
 226             comboBox.repaint();
 227         }
 228     }

 276     /**

 281     public void uninstallingUI() {
 282         if (propertyChangeListener != null) {
 283             comboBox.removePropertyChangeListener( propertyChangeListener );
 284         }
 285         if (itemListener != null) {
 286             comboBox.removeItemListener( itemListener );
 287         }
 288         uninstallComboBoxModelListeners(comboBox.getModel());
 289         uninstallKeyboardActions();
 290         uninstallListListeners();
 291         uninstallScrollerListeners();
 292         // We do this, otherwise the listener the ui installs on

 479     /**
 480      * Creates the JList used in the popup to display

 482      * is created.

 484      * @return a JList used to display the combo box items
 485      */
 486     protected JList createList() {
 487         return new JList( comboBox.getModel() ) {
 488             public void processMouseEvent(MouseEvent e)  {
 489                 if (BasicGraphicsUtils.isMenuShortcutKeyDown(e))  {
 490                     // Fix for 4234053. Filter out the Control Key from the list.
 491                     // ie., don't allow CTRL key deselection.
 492                     Toolkit toolkit = Toolkit.getDefaultToolkit();
 493                     e = new MouseEvent((Component)e.getSource(), e.getID(), e.getWhen(),
 494                                        e.getModifiers() ^ toolkit.getMenuShortcutKeyMask(),
 495                                        e.getX(), e.getY(),
 496                                        e.getXOnScreen(), e.getYOnScreen(),
 497                                        e.getClickCount(),
 498                                        e.isPopupTrigger(),
 499                                        MouseEvent.NOBUTTON);
 500                 }
 501                 super.processMouseEvent(e);
 502             }
 503         };
 504     }

 816     private class Handler implements ItemListener, MouseListener,
 817                           MouseMotionListener, MouseWheelListener,
 818                           PropertyChangeListener, Serializable {
 819         //
 820         // MouseListener
 821         // NOTE: this is added to both the JList and JComboBox
 822         //
 823         public void mouseClicked(MouseEvent e) {
 824         }
 825 
 826         public void mousePressed(MouseEvent e) {
 827             if (e.getSource() == list) {
 828                 return;
 829             }
 830             if (!SwingUtilities.isLeftMouseButton(e) || !comboBox.isEnabled())
 831                 return;
 832 
 833             if ( comboBox.isEditable() ) {
 834                 Component comp = comboBox.getEditor().getEditorComponent();
 835                 if ((!(comp instanceof JComponent)) || ((JComponent)comp).isRequestFocusEnabled()) {
 836                     comp.requestFocus();
 837                 }
 838             }
 839             else if (comboBox.isRequestFocusEnabled()) {
 840                 comboBox.requestFocus();
 841             }
 842             togglePopup();
 843         }
 844 
 845         public void mouseReleased(MouseEvent e) {
 846             if (e.getSource() == list) {
 847                 if (list.getModel().getSize() > 0) {
 848                     // JList mouse listener
 849                     if (comboBox.getSelectedIndex() == list.getSelectedIndex()) {
 850                         comboBox.getEditor().setItem(list.getSelectedValue());
 851                     }
 852                     comboBox.setSelectedIndex(list.getSelectedIndex());
 853                 }
 854                 comboBox.setPopupVisible(false);
 855                 // workaround for cancelling an edited item (bug 4530953)
 856                 if (comboBox.isEditable() && comboBox.getEditor() != null) {
 857                     comboBox.configureEditor(comboBox.getEditor(),
 858                                              comboBox.getSelectedItem());
 859                 }
 860                 return;
 861             }
 862             // JComboBox mouse listener
 863             Component source = (Component)e.getSource();
 864             Dimension size = source.getSize();
 865             Rectangle bounds = new Rectangle( 0, 0, size.width - 1, size.height - 1 );
 866             if ( !bounds.contains( e.getPoint() ) ) {
 867                 MouseEvent newEvent = convertMouseEvent( e );
 868                 Point location = newEvent.getPoint();
 869                 Rectangle r = new Rectangle();
 870                 list.computeVisibleRect( r );
 871                 if ( r.contains( location ) ) {
 872                     if (comboBox.getSelectedIndex() == list.getSelectedIndex()) {
 873                         comboBox.getEditor().setItem(list.getSelectedValue());
 874                     }
 875                     comboBox.setSelectedIndex(list.getSelectedIndex());
 876                 }
 877                 comboBox.setPopupVisible(false);
 878             }
 879             hasEntered = false;
 880             stopAutoScrolling();
 881         }

 883         public void mouseEntered(MouseEvent e) {
 884         }
 885 
 886         public void mouseExited(MouseEvent e) {
 887         }

 889         //
 890         // MouseMotionListener:
 891         // NOTE: this is added to both the List and ComboBox
 892         //
 893         public void mouseMoved(MouseEvent anEvent) {
 894             if (anEvent.getSource() == list) {
 895                 Point location = anEvent.getPoint();
 896                 Rectangle r = new Rectangle();
 897                 list.computeVisibleRect( r );
 898                 if ( r.contains( location ) ) {
 899                     updateListBoxSelectionForEvent( anEvent, false );
 900                 }
 901             }
 902         }
 903 
 904         public void mouseDragged( MouseEvent e ) {
 905             if (e.getSource() == list) {
 906                 return;
 907             }
 908             if ( isVisible() ) {
 909                 MouseEvent newEvent = convertMouseEvent( e );
 910                 Rectangle r = new Rectangle();
 911                 list.computeVisibleRect( r );
 912 
 913                 if ( newEvent.getPoint().y >= r.y && newEvent.getPoint().y <= r.y + r.height - 1 ) {
 914                     hasEntered = true;
 915                     if ( isAutoScrolling ) {
 916                         stopAutoScrolling();
 917                     }
 918                     Point location = newEvent.getPoint();
 919                     if ( r.contains( location ) ) {
 920                         updateListBoxSelectionForEvent( newEvent, false );
 921                     }
 922                 }
 923                 else {
 924                     if ( hasEntered ) {
 925                         int directionToScroll = newEvent.getPoint().y < r.y ? SCROLL_UP : SCROLL_DOWN;
 926                         if ( isAutoScrolling && scrollDirection != directionToScroll ) {
 927                             stopAutoScrolling();
 928                             startAutoScrolling( directionToScroll );
 929                         }
 930                         else if ( !isAutoScrolling ) {
 931                             startAutoScrolling( directionToScroll );
 932                         }
 933                     }
 934                     else {
 935                         if ( e.getPoint().y < 0 ) {
 936                             hasEntered = true;
 937                             startAutoScrolling( SCROLL_UP );
 938                         }
 939                     }
 940                 }
 941             }
 942         }

 944         //
 945         // PropertyChangeListener
 946         //
 947         public void propertyChange(PropertyChangeEvent e) {
 948             JComboBox comboBox = (JComboBox)e.getSource();
 949             String propertyName = e.getPropertyName();
 950 
 951             if ( propertyName == "model" ) {
 952                 ComboBoxModel oldModel = (ComboBoxModel)e.getOldValue();
 953                 ComboBoxModel newModel = (ComboBoxModel)e.getNewValue();
 954                 uninstallComboBoxModelListeners(oldModel);
 955                 installComboBoxModelListeners(newModel);
 956 
 957                 list.setModel(newModel);
 958 
 959                 if ( isVisible() ) {
 960                     hide();
 961                 }
 962             }
 963             else if ( propertyName == "renderer" ) {
 964                 list.setCellRenderer( comboBox.getRenderer() );
 965                 if ( isVisible() ) {
 966                     hide();
 967                 }
 968             }
 969             else if (propertyName == "componentOrientation") {
 970                 // Pass along the new component orientation
 971                 // to the list and the scroller
 972 
 973                 ComponentOrientation o =(ComponentOrientation)e.getNewValue();
 974 
 975                 JList list = getList();
 976                 if (list!=null && list.getComponentOrientation()!=o) {
 977                     list.setComponentOrientation(o);
 978                 }
 979 
 980                 if (scroller!=null && scroller.getComponentOrientation()!=o) {
 981                     scroller.setComponentOrientation(o);
 982                 }
 983 
 984                 if (o!=getComponentOrientation()) {
 985                     setComponentOrientation(o);
 986                 }
 987             }
 988             else if (propertyName == "lightWeightPopupEnabled") {
 989                 setLightWeightPopupEnabled(comboBox.isLightWeightPopupEnabled());
 990             }
 991         }

1027     /**
1028      * This protected method is implementation specific and should be private.
1029      * do not call or override.
1030      */
1031     protected void startAutoScrolling( int direction ) {
1032         // XXX - should be a private method within InvocationMouseMotionHandler
1033         // if possible.
1034         if ( isAutoScrolling ) {
1035             autoscrollTimer.stop();
1036         }
1037 
1038         isAutoScrolling = true;
1039 
1040         if ( direction == SCROLL_UP ) {
1041             scrollDirection = SCROLL_UP;
1042             Point convertedPoint = SwingUtilities.convertPoint( scroller, new Point( 1, 1 ), list );
1043             int top = list.locationToIndex( convertedPoint );
1044             list.setSelectedIndex( top );
1045 
1046             autoscrollTimer = new Timer( 100, new AutoScrollActionHandler(
1047                                              SCROLL_UP) );
1048         }
1049         else if ( direction == SCROLL_DOWN ) {
1050             scrollDirection = SCROLL_DOWN;
1051             Dimension size = scroller.getSize();
1052             Point convertedPoint = SwingUtilities.convertPoint( scroller,
1053                                                                 new Point( 1, (size.height - 1) - 2 ),
1054                                                                 list );
1055             int bottom = list.locationToIndex( convertedPoint );
1056             list.setSelectedIndex( bottom );
1057 
1058             autoscrollTimer = new Timer(100, new AutoScrollActionHandler(
1059                                             SCROLL_DOWN));
1060         }
1061         autoscrollTimer.start();
1062     }

1079      * do not call or override.

1081     protected void autoScrollUp() {
1082         int index = list.getSelectedIndex();
1083         if ( index > 0 ) {
1084             list.setSelectedIndex( index - 1 );
1085             list.ensureIndexIsVisible( index - 1 );
1086         }
1087     }

1089     /**
1090      * This protected method is implementation specific and should be private.
1091      * do not call or override.
1092      */
1093     protected void autoScrollDown() {
1094         int index = list.getSelectedIndex();
1095         int lastItem = list.getModel().getSize() - 1;
1096         if ( index < lastItem ) {
1097             list.setSelectedIndex( index + 1 );
1098             list.ensureIndexIsVisible( index + 1 );
1099         }

1191     /**
1192      * Retrieves the height of the popup based on the current
1193      * ListCellRenderer and the maximum row count.
1194      */
1195     protected int getPopupHeightForRowCount(int maxRowCount) {
1196         // Set the cached value of the minimum row count
1197         int minRowCount = Math.min( maxRowCount, comboBox.getItemCount() );
1198         int height = 0;
1199         ListCellRenderer renderer = list.getCellRenderer();
1200         Object value = null;
1201 
1202         for ( int i = 0; i < minRowCount; ++i ) {
1203             value = list.getModel().getElementAt( i );
1204             Component c = renderer.getListCellRendererComponent( list, value, i, false, false );
1205             height += c.getPreferredSize().height;
1206         }
1207 
1208         if (height == 0) {
1209             height = comboBox.getHeight();
1210         }
1211 
1212         Border border = scroller.getViewportBorder();
1213         if (border != null) {
1214             Insets insets = border.getBorderInsets(null);
1215             height += insets.top + insets.bottom;
1216         }
1217 
1218         border = scroller.getBorder();
1219         if (border != null) {
1220             Insets insets = border.getBorderInsets(null);
1221             height += insets.top + insets.bottom;
1222         }
1223 
1224         return height;
1225     }

1227     /**
1228      * Calculate the placement and size of the popup portion of the combo box based
1229      * on the combo box location and the enclosing screen bounds. If
1230      * no transformations are required, then the returned rectangle will
1231      * have the same values as the parameters.
1232      *
1233      * @param px starting x location
1234      * @param py starting y location
1235      * @param pw starting width
1236      * @param ph starting height
1237      * @return a rectangle which represents the placement and size of the popup
1238      */
1239     protected Rectangle computePopupBounds(int px,int py,int pw,int ph) {
1240         Toolkit toolkit = Toolkit.getDefaultToolkit();
1241         Rectangle screenBounds;
1242 
1243         // Calculate the desktop dimensions relative to the combo box.
1244         GraphicsConfiguration gc = comboBox.getGraphicsConfiguration();
1245         Point p = new Point();
1246         SwingUtilities.convertPointFromScreen(p, comboBox);
1247         if (gc != null) {
1248             Insets screenInsets = toolkit.getScreenInsets(gc);
1249             screenBounds = gc.getBounds();
1250             screenBounds.width -= (screenInsets.left + screenInsets.right);
1251             screenBounds.height -= (screenInsets.top + screenInsets.bottom);
1252             screenBounds.x += (p.x + screenInsets.left);
1253             screenBounds.y += (p.y + screenInsets.top);
1254         }
1255         else {
1256             screenBounds = new Rectangle(p, toolkit.getScreenSize());
1257         }
1258 
1259         Rectangle rect = new Rectangle(px,py,pw,ph);
1260         if (py+ph > screenBounds.y+screenBounds.height
1261             && ph < screenBounds.height) {
1262             rect.y = -rect.height;
1263         }
1264         return rect;
1265     }

1267     /**
1268      * Calculates the upper left location of the Popup.
1269      */
1270     private Point getPopupLocation() {
1271         Dimension popupSize = comboBox.getSize();
1272         Insets insets = getInsets();
1273 
1274         // reduce the width of the scrollpane by the insets so that the popup
1275         // is the same width as the combo box.
1276         popupSize.setSize(popupSize.width - (insets.right + insets.left),
1277                           getPopupHeightForRowCount( comboBox.getMaximumRowCount()));
1278         Rectangle popupBounds = computePopupBounds( 0, comboBox.getBounds().height,
1279                                                     popupSize.width, popupSize.height);
1280         Dimension scrollSize = popupBounds.getSize();
1281         Point popupLocation = popupBounds.getLocation();
1282 
1283         scroller.setMaximumSize( scrollSize );
1284         scroller.setPreferredSize( scrollSize );
1285         scroller.setMinimumSize( scrollSize );
1286 
1287         list.revalidate();
1288 
1289         return popupLocation;
1290     }

1292     /**
1293      * A utility method used by the event listeners.  Given a mouse event, it changes
1294      * the list selection to the list item below the mouse.
1295      */
1296     protected void updateListBoxSelectionForEvent(MouseEvent anEvent,boolean shouldScroll) {
1297         // XXX - only seems to be called from this class. shouldScroll flag is
1298         // never true
1299         Point location = anEvent.getPoint();
1300         if ( list == null )
1301             return;
1302         int index = list.locationToIndex(location);
1303         if ( index == -1 ) {
1304             if ( location.y < 0 )
1305                 index = 0;
1306             else
1307                 index = comboBox.getModel().getSize() - 1;
1308         }
1309         if ( list.getSelectedIndex() != index ) {
1310             list.setSelectedIndex(index);
1311             if ( shouldScroll )
1312                 list.ensureIndexIsVisible(index);
1313         }
1314     }
