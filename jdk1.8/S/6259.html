  66   /**
  67    * This method is an extension that implements as a Xalan extension
  68    * the node-set function also found in xt and saxon.
  69    * If the argument is a Result Tree Fragment, then nodeset
  70    * returns a node-set consisting of a single root node as described in
  71    * section 11.1 of the XSLT 1.0 Recommendation.  If the argument is a
  72    * node-set, nodeset returns a node-set.  If the argument
  73    * is a string, number, or boolean, then nodeset returns
  74    * a node-set consisting of a single root node with a single text node
  75    * child that is the result of calling the XPath string() function on the
  76    * passed parameter.  If the argument is anything else, then a node-set
  77    * is returned consisting of a single root node with a single text node
  78    * child that is the result of calling the java toString()
  79    * method on the passed argument.
  80    * Most of the
  81    * actual work here is done in MethodResolver and
  82    * XRTreeFrag.
  83    * @param myProcessor Context passed by the extension processor
  84    * @param rtf Argument in the stylesheet to the nodeset extension function
  85    *
  86    * NEEDSDOC ($objectName$) @return
  87    */
  88   public static NodeSet nodeset(ExpressionContext myProcessor, Object rtf)
  89   {
  90 
  91     String textNodeValue;
  92 
  93     if (rtf instanceof NodeIterator)
  94     {
  95       return new NodeSet((NodeIterator) rtf);
  96     }
  97     else
  98     {
  99       if (rtf instanceof String)
 100       {
 101         textNodeValue = (String) rtf;
 102       }
 103       else if (rtf instanceof Boolean)
 104       {
 105         textNodeValue = new XBoolean(((Boolean) rtf).booleanValue()).str();
 106       }
 107       else if (rtf instanceof Double)
 108       {
 109         textNodeValue = new XNumber(((Double) rtf).doubleValue()).str();
 110       }
 111       else
 112       {
 113         textNodeValue = rtf.toString();
 114       }
 115 
 116       // This no longer will work right since the DTM.
 117       // Document myDoc = myProcessor.getContextNode().getOwnerDocument();
 118       Document myDoc = getDocument();
 119 
 120         Text textNode = myDoc.createTextNode(textNodeValue);
 121         DocumentFragment docFrag = myDoc.createDocumentFragment();
 122 
 123         docFrag.appendChild(textNode);
 124 
 125       return new NodeSet(docFrag);
 126     }
 127   }

 178   /**
 179    * Returns true if both node-sets contain the same set of nodes.
 180    *
 181    * @param nl1 NodeList for first node-set
 182    * @param nl2 NodeList for second node-set
 183    * @return true if nl1 and nl2 contain exactly the same set of nodes.
 184    */
 185   public static boolean hasSameNodes(NodeList nl1, NodeList nl2)
 186   {
 187 
 188     NodeSet ns1 = new NodeSet(nl1);
 189     NodeSet ns2 = new NodeSet(nl2);
 190 
 191     if (ns1.getLength() != ns2.getLength())
 192       return false;
 193 
 194     for (int i = 0; i < ns1.getLength(); i++)
 195     {
 196       Node n = ns1.elementAt(i);
 197 
 198       if (!ns2.contains(n))
 199         return false;
 200     }
 201 
 202     return true;
 203   }

 228   /**
 229    * Returns a NodeSet containing one text node for each token in the first argument.
 230    * Delimiters are specified in the second argument.
 231    * Tokens are determined by a call to StringTokenizer.
 232    * If the first argument is an empty string or contains only delimiters, the result
 233    * will be an empty NodeSet.
 234    *
 235    * Contributed to XalanJ1 by Benoit Cerrina.
 236    *
 237    * @param toTokenize The string to be split into text tokens.
 238    * @param delims The delimiters to use.
 239    * @return a NodeSet as described above.
 240    */
 241   public static NodeList tokenize(String toTokenize, String delims)
 242   {
 243 
 244     Document doc = getDocument();
 245 
 246     StringTokenizer lTokenizer = new StringTokenizer(toTokenize, delims);
 247     NodeSet resultSet = new NodeSet();
 248 
 249     synchronized (doc)
 250     {
 251       while (lTokenizer.hasMoreTokens())
 252       {
 253         resultSet.addNode(doc.createTextNode(lTokenizer.nextToken()));
 254       }
 255     }
 256 
 257     return resultSet;
 258   }

 278   /**
 279    * Return a Node of basic debugging information from the
 280    * EnvironmentCheck utility about the Java environment.
 281    *
 282    * Simply calls the {@link com.sun.org.apache.xalan.internal.xslt.EnvironmentCheck}
 283    * utility to grab info about the Java environment and CLASSPATH,
 284    * etc., and then returns the resulting Node.  Stylesheets can
 285    * then maniuplate this data or simply xsl:copy-of the Node.  Note
 286    * that we first attempt to load the more advanced
 287    * org.apache.env.Which utility by reflection; only if that fails
 288    * to we still use the internal version.  Which is available from
 289    * http://xml.apache.org/commons/.
 290    *
 291    * We throw a WrappedRuntimeException in the unlikely case
 292    * that reading information from the environment throws us an
 293    * exception. (Is this really the best thing to do?)
 294    *
 295    * @param myContext an ExpressionContext passed in by the
 296    *                  extension mechanism.  This must be an XPathContext.
 297    * @return a Node as described above.
 298    */
 299   public static Node checkEnvironment(ExpressionContext myContext)
 300   {
 301 
 302     Document factoryDocument = getDocument();
 303 
 304     Node resultNode = null;
 305     try
 306     {
 307       // First use reflection to try to load Which, which is a
 308       //  better version of EnvironmentCheck
 309       resultNode = checkEnvironmentUsingWhich(myContext, factoryDocument);
 310 
 311       if (null != resultNode)
 312         return resultNode;
 313 
 314       // If reflection failed, fallback to our internal EnvironmentCheck
 315       EnvironmentCheck envChecker = new EnvironmentCheck();
 316       Hashtable h = envChecker.getEnvironmentHash();
 317       resultNode = factoryDocument.createElement("checkEnvironmentExtension");
 318       envChecker.appendEnvironmentReport(resultNode, factoryDocument, h);
 319       envChecker = null;
 320     }
 321     catch(Exception e)
 322     {
 323       throw new com.sun.org.apache.xml.internal.utils.WrappedRuntimeException(e);
 324     }
 325 
 326     return resultNode;
 327   }

 329   /**
 330    * Private worker method to attempt to use org.apache.env.Which.
 331    *
 332    * @param myContext an ExpressionContext passed in by the
 333    *                  extension mechanism.  This must be an XPathContext.
 334    * @param factoryDocument providing createElement services, etc.
 335    * @return a Node with environment info; null if any error
 336    */
 337   private static Node checkEnvironmentUsingWhich(ExpressionContext myContext,
 338         Document factoryDocument)
 339   {
 340     final String WHICH_CLASSNAME = "org.apache.env.Which";
 341     final String WHICH_METHODNAME = "which";
 342     final Class WHICH_METHOD_ARGS[] = { java.util.Hashtable.class,
 343                                         java.lang.String.class,
 344                                         java.lang.String.class };
 345     try
 346     {
 347       // Use reflection to try to find xml-commons utility 'Which'
 348       Class clazz = ObjectFactory.findProviderClass(WHICH_CLASSNAME, true);
 349       if (null == clazz)
 350         return null;
 351 
 352       // Fully qualify names since this is the only method they're used in
 353       java.lang.reflect.Method method = clazz.getMethod(WHICH_METHODNAME, WHICH_METHOD_ARGS);
 354       Hashtable report = new Hashtable();
 355 
 356       // Call the method with our Hashtable, common options, and ignore return value
 357       Object[] methodArgs = { report, "XmlCommons;Xalan;Xerces;Crimson;Ant", "" };
 358       Object returnValue = method.invoke(null, methodArgs);
 359 
 360       // Create a parent to hold the report and append hash to it
 361       Node resultNode = factoryDocument.createElement("checkEnvironmentExtension");
 362       com.sun.org.apache.xml.internal.utils.Hashtree2Node.appendHashToNode(report, "whichReport",
 363             resultNode, factoryDocument);
 364 
 365       return resultNode;
 366     }
 367     catch (Throwable t)
 368     {
 369       // Simply return null; no need to report error
 370       return null;
 371     }
 372   }
