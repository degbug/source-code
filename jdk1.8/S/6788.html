  30 /**
  31  * A dummy ColorSpace to enable ColorModel
  32  * for image data which do not have an innate color representation.
  33  */
  34 public class BogusColorSpace extends ColorSpace {
  35     /**
  36      * Return the type given the number of components.
  37      *
  38      * @param numComponents The number of components in the
  39      * ColorSpace.
  40      * @exception IllegalArgumentException if numComponents
  41      * is less than 1.
  42      */
  43     private static int getType(int numComponents) {
  44         if(numComponents < 1) {
  45             throw new IllegalArgumentException("numComponents < 1!");
  46         }
  47 
  48         int type;
  49         switch(numComponents) {
  50         case 1:
  51             type = ColorSpace.TYPE_GRAY;
  52             break;
  53         default:
  54             // Based on the constant definitions TYPE_2CLR=12 through
  55             // TYPE_FCLR=25. This will return unknown types for
  56             // numComponents > 15.
  57             type = numComponents + 10;
  58         }
  59 
  60         return type;
  61     }
  62 
  63     /**
  64      * Constructs a bogus ColorSpace.
  65      *
  66      * @param numComponents The number of components in the
  67      * ColorSpace.
  68      * @exception IllegalArgumentException if numComponents
  69      * is less than 1.
  70      */
  71     public BogusColorSpace(int numComponents) {
  72         super(getType(numComponents), numComponents);
  73     }
