  30 /**
  31  * Extended by classes that store results of compilations.  State
  32  * might be stored in the form of Java classes, Java class files or scripting
  33  * language opcodes.  The script may be executed repeatedly
  34  * without reparsing.
  35  * 


  36  * Each CompiledScript is associated with a ScriptEngine -- A call to an  eval
  37  * method of the CompiledScript causes the execution of the script by the
  38  * ScriptEngine.  Changes in the state of the ScriptEngine caused by execution
  39  * of the CompiledScript  may visible during subsequent executions of scripts by the engine.
  40  *
  41  * @author Mike Grogan
  42  * @since 1.6
  43  */

  49      * @param context A ScriptContext that is used in the same way as
  50      * the ScriptContext passed to the eval methods of
  51      * ScriptEngine.
  52      *
  53      * @return The value returned by the script execution, if any.  Should return null
  54      * if no value is returned by the script execution.
  55      *
  56      * @throws ScriptException if an error occurs.
  57      * @throws NullPointerException if context is null.
  58      */
  59 
  60     public abstract Object eval(ScriptContext context) throws ScriptException;

  62     /**
  63      * Executes the program stored in the CompiledScript object using
  64      * the supplied Bindings of attributes as the ENGINE_SCOPE of the
  65      * associated ScriptEngine during script execution.  If bindings is null,
  66      * then the effect of calling this method is same as that of eval(getEngine().getContext()).
  67      * .
  68      * The GLOBAL_SCOPE Bindings, Reader and Writer
  69      * associated with the default ScriptContext of the associated ScriptEngine are used.
  70      *
  71      * @param bindings The bindings of attributes used for the ENGINE_SCOPE.
  72      *
  73      * @return The return value from the script execution
  74      *
  75      * @throws ScriptException if an error occurs.
  76      */
  77     public Object eval(Bindings bindings) throws ScriptException {
  78 
  79         ScriptContext ctxt = getEngine().getContext();
  80 
  81         if (bindings != null) {
  82             SimpleScriptContext tempctxt = new SimpleScriptContext();
  83             tempctxt.setBindings(bindings, ScriptContext.ENGINE_SCOPE);
  84             tempctxt.setBindings(ctxt.getBindings(ScriptContext.GLOBAL_SCOPE),
  85                     ScriptContext.GLOBAL_SCOPE);
  86             tempctxt.setWriter(ctxt.getWriter());
  87             tempctxt.setReader(ctxt.getReader());
  88             tempctxt.setErrorWriter(ctxt.getErrorWriter());
