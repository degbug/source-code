  Basic UI implementation for JComboBox.
  

  The combo box is a compound component which means that it is an aggregate of
  many simpler components. This class creates and manages the listeners
  on the combo box and the combo box model. These listeners update the user
  interface in response to changes in the properties and state of the combo box.
  

  All event handling is handled by listener classes created with the
  createxxxListener() methods and internal classes.
  You can change the behavior of this class by overriding the
  createxxxListener() methods and supplying your own
  event listeners or subclassing from the ones supplied in this class.
  

  For adding specific actions,
  overide installKeyboardActions to add actions in response to
  KeyStroke bindings. See the article How to Use Key Bindings
  @author Arnaud Weber
  @author Tom Santos
  @author Mark Davidson

      portion of the combo box in which the selected item is displayed. This
      method will use the prototype display value if it has been set.

      use a prototype display value to significantly speed up the display
      size calculation.
      @return the size of the display area calculated from the combo box items

1332     protected Dimension getDisplaySize() {
1333         if (!isDisplaySizeDirty)  {
1334             return new Dimension(cachedDisplaySize);
1335         }

1337

1339         if (renderer == null)  {
1340             renderer = new DefaultListCellRenderer();
1341         }
1342 
1343         sameBaseline = true;
1344 
1345         Object prototypeValue = comboBox.getPrototypeDisplayValue();
1346         if (prototypeValue != null)  {
1347             // Calculates the dimension based on the prototype value
1348             result = getSizeForComponent(renderer.getListCellRendererComponent(listBox,
1349                                                                                prototypeValue,
1350                                                                                -1, false, false));
1351         } else {
1352             // Calculate the dimension by iterating over all the elements in the combo
1353             // box list.
1354             ComboBoxModel model = comboBox.getModel();
1355             int modelSize = model.getSize();
1356             int baseline = -1;
1357             Dimension d;
1358 
1359             Component cpn;
1360

1362                 for (int i = 0; i < modelSize ; i++ ) {
1363                     // Calculates the maximum height and width based on the largest

1365                     Object value = model.getElementAt(i);
1366                     Component c = renderer.getListCellRendererComponent(
1367                             listBox, value, -1, false, false);
1368                     d = getSizeForComponent(c);
1369                     if (sameBaseline && value != null &&
1370                             (!(value instanceof String) || !"".equals(value))) {
1371                         int newBaseline = c.getBaseline(d.width, d.height);
1372                         if (newBaseline == -1) {

1374                         }
1375                         else if (baseline == -1) {
1376                             baseline = newBaseline;
1377                         }
1378                         else if (baseline != newBaseline) {
1379                             sameBaseline = false;
1380                         }
1381                     }
1382                     result.width = Math.max(result.width,d.width);
1383                     result.height = Math.max(result.height,d.height);

1403         }

1608                             }

1612                 }

1616                         comboBox.setSelectedItem(comboBox.getSelectedItem());

1636                     }

1642             int listHeight = comboBox.getMaximumRowCount();
1643 
1644             int selectedIndex = comboBox.getSelectedIndex();
1645             if (UIManager.getBoolean("ComboBox.noActionOnKeyNavigation")
1646                     && (comboBox.getUI() instanceof BasicComboBoxUI)) {
1647                 selectedIndex = ((BasicComboBoxUI) comboBox.getUI()).listBox.getSelectedIndex();
1648             }
1649

1651                 int index = selectedIndex - listHeight;
1652                 return (index < 0 ? 0: index);
1653             }
1654             else if (key == PAGE_DOWN) {
1655                 int index = selectedIndex + listHeight;
1656                 int max = comboBox.getItemCount();
1657                 return (index < max ? index: max-1);
1658             }
1659             else if (key == HOME) {
1660                 return 0;
1661             }
1662             else if (key == END) {
1663                 return comboBox.getItemCount() - 1;
1664             }
1665             return comboBox.getSelectedIndex();

1976     class DefaultKeySelectionManager implements JComboBox.KeySelectionManager, UIResource {
1977         private String prefix = "";
1978         private String typedString = "";
1979 
1980         public int selectionForKey(char aKey,ComboBoxModel aModel) {
1981             if (lastTime == 0L) {
1982                 prefix = "";
1983                 typedString = "";
1984             }
1985             boolean startingFromSelection = true;
1986 
1987             int startIndex = comboBox.getSelectedIndex();
1988             if (time - lastTime < timeFactor) {
1989                 typedString += aKey;
1990                 if((prefix.length() == 1) && (aKey == prefix.charAt(0))) {
1991                     // Subsequent same key presses move the keyboard focus to the next
1992                     // object that starts with the same letter.
1993                     startIndex++;
1994                 } else {
1995                     prefix = typedString;
1996                 }
1997             } else {
1998                 startIndex++;
1999                 typedString = "" + aKey;
2000                 prefix = typedString;
2001             }
2002             lastTime = time;
2003 
2004             if (startIndex < 0 || startIndex >= aModel.getSize()) {
2005                 startingFromSelection = false;
2006                 startIndex = 0;
2007             }
2008             int index = listBox.getNextMatch(prefix, startIndex,
2009                                              Position.Bias.Forward);
2010             if (index < 0 && startingFromSelection) { // wrap
2011                 index = listBox.getNextMatch(prefix, 0,
2012                                              Position.Bias.Forward);
2013             }
2014             return index;
2015         }
