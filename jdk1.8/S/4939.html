 165     private DocumentBuilder getParser() {
 166         try {
 167             // we'd really like to cache those DocumentBuilders, but we can't because:
 168             // 1. thread safety. parsers are not thread-safe, so at least
 169             //    we need one instance per a thread.
 170             // 2. parsers are non-reentrant, so now we are looking at having a
 171             // pool of parsers.
 172             // 3. then the class loading issue. The look-up procedure of
 173             //    DocumentBuilderFactory.newInstance() depends on context class loader
 174             //    and system properties, which may change during the execution of JVM.
 175             //
 176             // so we really have to create a fresh DocumentBuilder every time we need one
 177             // - KK
 178             DocumentBuilderFactory dbf = FactoryImpl.getDOMFactory(useServiceMechanism);
 179             dbf.setNamespaceAware( true );
 180             dbf.setValidating( false );
 181             return dbf.newDocumentBuilder();
 182         } catch (ParserConfigurationException e) {
 183             // this should never happen with a well-behaving JAXP implementation.
 184             throw new Error(e);
 185         }
 186     }

 189     private XObject eval(String expression, Object contextItem)
 190         throws javax.xml.transform.TransformerException {
 191         com.sun.org.apache.xpath.internal.XPath xpath = new com.sun.org.apache.xpath.internal.XPath( expression,
 192             null, prefixResolver, com.sun.org.apache.xpath.internal.XPath.SELECT );
 193         com.sun.org.apache.xpath.internal.XPathContext xpathSupport = null;
 194         if ( functionResolver != null ) {
 195             JAXPExtensionsProvider jep = new JAXPExtensionsProvider(
 196                     functionResolver, featureSecureProcessing, featureManager );
 197             xpathSupport = new com.sun.org.apache.xpath.internal.XPathContext( jep );
 198         } else {
 199             xpathSupport = new com.sun.org.apache.xpath.internal.XPathContext();
 200         }
 201 
 202         XObject xobj = null;
 203 
 204         xpathSupport.setVarStack(new JAXPVariableStack(variableResolver));
 205 
 206         // If item is null, then we will create a a Dummy contextNode
 207         if ( contextItem instanceof Node ) {
 208             xobj = xpath.execute (xpathSupport, (Node)contextItem,
 209                     prefixResolver );
 210         } else {
 211             xobj = xpath.execute ( xpathSupport, DTM.NULL, prefixResolver );
 212         }
 213 
 214         return xobj;
 215     }

 217     /**
 218      * <p>Evaluate an <code>XPath</code> expression in the specified context and return the result as the specified type.</p>
 219      *
 220      * <p>See "Evaluation of XPath Expressions" section of JAXP 1.3 spec
 221      * for context item evaluation,
 222      * variable, function and <code>QName</code> resolution and return type conversion.</p>
 223      *
 224      * <p>If <code>returnType</code> is not one of the types defined in {@link XPathConstants} (
 225      * {@link XPathConstants#NUMBER NUMBER},
 226      * {@link XPathConstants#STRING STRING},
 227      * {@link XPathConstants#BOOLEAN BOOLEAN},
 228      * {@link XPathConstants#NODE NODE} or
 229      * {@link XPathConstants#NODESET NODESET})
 230      * then an <code>IllegalArgumentException</code> is thrown.</p>
 231      *
 232      * <p>If a <code>null</code> value is provided for
 233      * <code>item</code>, an empty document will be used for the
 234      * context.
 235      * If <code>expression</code> or <code>returnType</code> is <code>null</code>, then a
 236      * <code>NullPointerException</code> is thrown.</p>
 237      *
 238      * @param expression The XPath expression.
 239      * @param item The starting context (node or node list, for example).
 240      * @param returnType The desired return type.
 241      *
 242      * @return Result of evaluating an XPath expression as an <code>Object</code> of <code>returnType</code>.
 243      *
 244      * @throws XPathExpressionException If <code>expression</code> cannot be evaluated.
 245      * @throws IllegalArgumentException If <code>returnType</code> is not one of the types defined in {@link XPathConstants}.
 246      * @throws NullPointerException If <code>expression</code> or <code>returnType</code> is <code>null</code>.
 247      */
 248     public Object evaluate(String expression, Object item, QName returnType)
 249             throws XPathExpressionException {
 250         if ( expression == null ) {
 251             String fmsg = XSLMessages.createXPATHMessage(
 252                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 253                     new Object[] {"XPath expression"} );
 254             throw new NullPointerException ( fmsg );
 255         }
 256         if ( returnType == null ) {
 257             String fmsg = XSLMessages.createXPATHMessage(
 258                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 259                     new Object[] {"returnType"} );
 260             throw new NullPointerException ( fmsg );
 261         }
 262         // Checking if requested returnType is supported. returnType need to
 263         // be defined in XPathConstants
 264         if ( !isSupported ( returnType ) ) {
 265             String fmsg = XSLMessages.createXPATHMessage(
 266                     XPATHErrorResources.ER_UNSUPPORTED_RETURN_TYPE,
 267                     new Object[] { returnType.toString() } );
 268             throw new IllegalArgumentException ( fmsg );
 269         }
 270 
 271         try {
 272 
 273             XObject resultObject = eval( expression, item );
 274             return getResultAsType( resultObject, returnType );
 275         } catch ( java.lang.NullPointerException npe ) {
 276             // If VariableResolver returns null Or if we get
 277             // NullPointerException at this stage for some other reason
 278             // then we have to reurn XPathException
 279             throw new XPathExpressionException ( npe );
 280         } catch ( javax.xml.transform.TransformerException te ) {
 281             Throwable nestedException = te.getException();
 282             if ( nestedException instanceof javax.xml.xpath.XPathFunctionException ) {
 283                 throw (javax.xml.xpath.XPathFunctionException)nestedException;
 284             } else {
 285                 // For any other exceptions we need to throw
 286                 // XPathExpressionException ( as per spec )
 287                 throw new XPathExpressionException ( te );
 288             }
 289         }
 290 
 291     }

 293     private boolean isSupported( QName returnType ) {
 294         if ( ( returnType.equals( XPathConstants.STRING ) ) ||
 295              ( returnType.equals( XPathConstants.NUMBER ) ) ||
 296              ( returnType.equals( XPathConstants.BOOLEAN ) ) ||
 297              ( returnType.equals( XPathConstants.NODE ) ) ||
 298              ( returnType.equals( XPathConstants.NODESET ) )  ) {
 299 
 300             return true;
 301         }
 302         return false;
 303      }

 305     private Object getResultAsType( XObject resultObject, QName returnType )
 306         throws javax.xml.transform.TransformerException {
 307         // XPathConstants.STRING
 308         if ( returnType.equals( XPathConstants.STRING ) ) {
 309             return resultObject.str();
 310         }
 311         // XPathConstants.NUMBER
 312         if ( returnType.equals( XPathConstants.NUMBER ) ) {
 313             return new Double ( resultObject.num());
 314         }
 315         // XPathConstants.BOOLEAN
 316         if ( returnType.equals( XPathConstants.BOOLEAN ) ) {
 317             return new Boolean( resultObject.bool());
 318         }
 319         // XPathConstants.NODESET ---ORdered, UNOrdered???
 320         if ( returnType.equals( XPathConstants.NODESET ) ) {
 321             return resultObject.nodelist();
 322         }
 323         // XPathConstants.NODE
 324         if ( returnType.equals( XPathConstants.NODE ) ) {
 325             NodeIterator ni = resultObject.nodeset();
 326             //Return the first node, or null
 327             return ni.nextNode();
 328         }
 329         String fmsg = XSLMessages.createXPATHMessage(
 330                 XPATHErrorResources.ER_UNSUPPORTED_RETURN_TYPE,
 331                 new Object[] { returnType.toString()});
 332         throw new IllegalArgumentException( fmsg );
 333     }

 384         throws XPathExpressionException {
 385         if ( expression == null ) {
 386             String fmsg = XSLMessages.createXPATHMessage(
 387                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 388                     new Object[] {"XPath expression"} );
 389             throw new NullPointerException ( fmsg );
 390         }
 391         try {
 392             com.sun.org.apache.xpath.internal.XPath xpath = new XPath (expression, null,

 394             // Can have errorListener
 395             XPathExpressionImpl ximpl = new XPathExpressionImpl (xpath,
 396                     prefixResolver, functionResolver, variableResolver,
 397                     featureSecureProcessing, useServiceMechanism, featureManager );
 398             return ximpl;
 399         } catch ( javax.xml.transform.TransformerException te ) {
 400             throw new XPathExpressionException ( te ) ;
 401         }

 405     /**
 406      * <p>Evaluate an XPath expression in the context of the specified <code>InputSource</code>
 407      * and return the result as the specified type.</p>
 408      *
 409      * <p>This method builds a data model for the {@link InputSource} and calls
 410      * {@link #evaluate(String expression, Object item, QName returnType)} on the resulting document object.</p>
 411      *
 412      * <p>See "Evaluation of XPath Expressions" section of JAXP 1.3 spec
 413      * for context item evaluation,
 414      * variable, function and QName resolution and return type conversion.</p>
 415      *
 416      * <p>If <code>returnType</code> is not one of the types defined in {@link XPathConstants},
 417      * then an <code>IllegalArgumentException</code> is thrown.</p>
 418      *
 419      * <p>If <code>expression</code>, <code>source</code> or <code>returnType</code> is <code>null</code>,
 420      * then a <code>NullPointerException</code> is thrown.</p>
 421      *
 422      * @param expression The XPath expression.
 423      * @param source The input source of the document to evaluate over.
 424      * @param returnType The desired return type.
 425      *
 426      * @return The <code>Object</code> that encapsulates the result of evaluating the expression.
 427      *
 428      * @throws XPathExpressionException If expression cannot be evaluated.
 429      * @throws IllegalArgumentException If <code>returnType</code> is not one of the types defined in {@link XPathConstants}.
 430      * @throws NullPointerException If <code>expression</code>, <code>source</code> or <code>returnType</code>
 431      *   is <code>null</code>.
 432      */
 433     public Object evaluate(String expression, InputSource source,
 434             QName returnType) throws XPathExpressionException {
 435         // Checking validity of different parameters
 436         if( source== null ) {
 437             String fmsg = XSLMessages.createXPATHMessage(
 438                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 439                     new Object[] {"source"} );
 440             throw new NullPointerException ( fmsg );
 441         }
 442         if ( expression == null ) {
 443             String fmsg = XSLMessages.createXPATHMessage(
 444                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 445                     new Object[] {"XPath expression"} );
 446             throw new NullPointerException ( fmsg );
 447         }
 448         if ( returnType == null ) {
 449             String fmsg = XSLMessages.createXPATHMessage(
 450                     XPATHErrorResources.ER_ARG_CANNOT_BE_NULL,
 451                     new Object[] {"returnType"} );
 452             throw new NullPointerException ( fmsg );
 453         }
 454 
 455         //Checking if requested returnType is supported.
 456         //returnType need to be defined in XPathConstants
 457         if ( !isSupported ( returnType ) ) {
 458             String fmsg = XSLMessages.createXPATHMessage(
 459                     XPATHErrorResources.ER_UNSUPPORTED_RETURN_TYPE,
 460                     new Object[] { returnType.toString() } );
 461             throw new IllegalArgumentException ( fmsg );
 462         }
 463 
 464         try {
 465 
 466             Document document = getParser().parse( source );
 467 
 468             XObject resultObject = eval( expression, document );
 469             return getResultAsType( resultObject, returnType );
 470         } catch ( SAXException e ) {
 471             throw new XPathExpressionException ( e );
 472         } catch( IOException e ) {
 473             throw new XPathExpressionException ( e );
 474         } catch ( javax.xml.transform.TransformerException te ) {
 475             Throwable nestedException = te.getException();
 476             if ( nestedException instanceof javax.xml.xpath.XPathFunctionException ) {
 477                 throw (javax.xml.xpath.XPathFunctionException)nestedException;
 478             } else {
 479                 throw new XPathExpressionException ( te );
 480             }
 481         }
 482 
 483     }
