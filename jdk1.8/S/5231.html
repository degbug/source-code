 100      * Returns the next character on the input.
 101      * 
 102      * Note: The character is not consumed.
 103      *
 104      * @throws IOException  Thrown if i/o error occurs.
 105      * @throws EOFException Thrown on end of file.
 106      */
 107     public int peekChar() throws IOException {
 108 
 109         // load more characters, if needed
 110         if (fCurrentEntity.position == fCurrentEntity.count) {
 111             load(0, true, true);
 112         }
 113 
 114         // peek at character
 115         int c = fCurrentEntity.ch[fCurrentEntity.position];
 116 
 117         // return peeked character
 118         if (fCurrentEntity.isExternal()) {
 119             return (c != '\r' && c != 0x85 && c != 0x2028) ? c : '\n';
 120         }
 121         else {
 122             return c;
 123         }
 124

 127     /**
 128      * Returns the next character on the input.
 129      * 
 130      * Note: The character is consumed.
 131      *
 132      * @throws IOException  Thrown if i/o error occurs.
 133      * @throws EOFException Thrown on end of file.
 134      */
 135     public int scanChar() throws IOException {
 136 
 137         // load more characters, if needed
 138         if (fCurrentEntity.position == fCurrentEntity.count) {
 139             load(0, true, true);
 140         }
 141 
 142         // scan character
 143         int c = fCurrentEntity.ch[fCurrentEntity.position++];
 144         boolean external = false;
 145         if (c == '\n' ||
 146             ((c == '\r' || c == 0x85 || c == 0x2028) && (external = fCurrentEntity.isExternal()))) {
 147             fCurrentEntity.lineNumber++;
 148             fCurrentEntity.columnNumber = 1;
 149             if (fCurrentEntity.position == fCurrentEntity.count) {
 150                 invokeListeners(1);
 151                 fCurrentEntity.ch[0] = (char)c;
 152                 load(1, false, false);
 153             }
 154             if (c == '\r' && external) {
 155                 int cc = fCurrentEntity.ch[fCurrentEntity.position++];
 156                 if (cc != '\n' && cc != 0x85) {
 157                     fCurrentEntity.position--;
 158                 }
 159             }
 160             c = '\n';
 161         }
 162 
 163         // return character that was scanned
 164         fCurrentEntity.columnNumber++;
 165         return c;
 166 
 167     } // scanChar():int

 185         // load more characters, if needed
 186         if (fCurrentEntity.position == fCurrentEntity.count) {
 187             load(0, true, true);
 188         }
 189 
 190         // scan nmtoken
 191         int offset = fCurrentEntity.position;
 192 
 193         do {
 194             char ch = fCurrentEntity.ch[fCurrentEntity.position];
 195             if (XML11Char.isXML11Name(ch)) {
 196                 if (++fCurrentEntity.position == fCurrentEntity.count) {
 197                     int length = fCurrentEntity.position - offset;
 198                     invokeListeners(length);
 199                     if (length == fCurrentEntity.ch.length) {
 200                         // bad luck we have to resize our buffer
 201                         char[] tmp = new char[fCurrentEntity.ch.length << 1];
 202                         System.arraycopy(fCurrentEntity.ch, offset,
 203                                          tmp, 0, length);
 204                         fCurrentEntity.ch = tmp;
 205                     }
 206                     else {
 207                         System.arraycopy(fCurrentEntity.ch, offset,
 208                                          fCurrentEntity.ch, 0, length);
 209                     }
 210                     offset = 0;
 211                     if (load(length, false, false)) {
 212                         break;
 213                     }
 214                 }
 215             }
 216             else if (XML11Char.isXML11NameHighSurrogate(ch)) {
 217                 if (++fCurrentEntity.position == fCurrentEntity.count) {
 218                     int length = fCurrentEntity.position - offset;
 219                     invokeListeners(length);
 220                     if (length == fCurrentEntity.ch.length) {
 221                         // bad luck we have to resize our buffer
 222                         char[] tmp = new char[fCurrentEntity.ch.length << 1];
 223                         System.arraycopy(fCurrentEntity.ch, offset,
 224                                          tmp, 0, length);
 225                         fCurrentEntity.ch = tmp;
 226                     }
 227                     else {
 228                         System.arraycopy(fCurrentEntity.ch, offset,
 229                                          fCurrentEntity.ch, 0, length);
 230                     }
 231                     offset = 0;
 232                     if (load(length, false, false)) {
 233                         --fCurrentEntity.startPosition;
 234                         --fCurrentEntity.position;
 235                         break;
 236                     }
 237                 }
 238                 char ch2 = fCurrentEntity.ch[fCurrentEntity.position];
 239                 if ( !XMLChar.isLowSurrogate(ch2) ||
 240                      !XML11Char.isXML11Name(XMLChar.supplemental(ch, ch2)) ) {
 241                     --fCurrentEntity.position;
 242                     break;
 243                 }
 244                 if (++fCurrentEntity.position == fCurrentEntity.count) {
 245                     int length = fCurrentEntity.position - offset;
 246                     invokeListeners(length);
 247                     if (length == fCurrentEntity.ch.length) {
 248                         // bad luck we have to resize our buffer
 249                         char[] tmp = new char[fCurrentEntity.ch.length << 1];
 250                         System.arraycopy(fCurrentEntity.ch, offset,
 251                                          tmp, 0, length);
 252                         fCurrentEntity.ch = tmp;
 253                     }
 254                     else {
 255                         System.arraycopy(fCurrentEntity.ch, offset,
 256                                          fCurrentEntity.ch, 0, length);
 257                     }
 258                     offset = 0;
 259                     if (load(length, false, false)) {
 260                         break;
 261                     }
 262                 }
 263             }
 264             else {
 265                 break;
 266             }
 267         }
 268         while (true);
 269 
 270         int length = fCurrentEntity.position - offset;
 271         fCurrentEntity.columnNumber += length;
 272 
 273         // return nmtoken
 274         String symbol = null;
 275         if (length > 0) {
 276             symbol = fSymbolTable.addSymbol(fCurrentEntity.ch, offset, length);
 277         }
 278         return symbol;

1319     /**
1320      * Skips a character appearing immediately on the input.
1321      * 
1322      * Note: The character is consumed only if it matches
1323      * the specified character.
1324      *
1325      * @param c The character to skip.
1326      *
1327      * @return Returns true if the character was skipped.
1328      *
1329      * @throws IOException  Thrown if i/o error occurs.
1330      * @throws EOFException Thrown on end of file.
1331      */
1332     public boolean skipChar(int c) throws IOException {
1333 
1334         // load more characters, if needed
1335         if (fCurrentEntity.position == fCurrentEntity.count) {
1336             load(0, true, true);
1337         }
1338 
1339         // skip character
1340         int cc = fCurrentEntity.ch[fCurrentEntity.position];
1341         if (cc == c) {
1342             fCurrentEntity.position++;
1343             if (c == '\n') {
1344                 fCurrentEntity.lineNumber++;
1345                 fCurrentEntity.columnNumber = 1;
1346             }
1347             else {
1348                 fCurrentEntity.columnNumber++;
1349             }
1350             return true;
1351         }
1352         else if (c == '\n' && ((cc == 0x2028 || cc == 0x85) && fCurrentEntity.isExternal())) {
1353             fCurrentEntity.position++;
1354             fCurrentEntity.lineNumber++;
1355             fCurrentEntity.columnNumber = 1;
1356             return true;
1357         }
1358         else if (c == '\n' && (cc == '\r' ) && fCurrentEntity.isExternal()) {
1359             // handle newlines
1360             if (fCurrentEntity.position == fCurrentEntity.count) {
1361                 invokeListeners(1);
1362                 fCurrentEntity.ch[0] = (char)cc;
1363                 load(1, false, false);
1364             }
1365             int ccc = fCurrentEntity.ch[++fCurrentEntity.position];
1366             if (ccc == '\n' || ccc == 0x85) {
1367                 fCurrentEntity.position++;
1368             }
1369             fCurrentEntity.lineNumber++;
1370             fCurrentEntity.columnNumber = 1;
1371             return true;
1372         }
1373 
1374         // character was not skipped
1375         return false;
1376

1380      * Skips space characters appearing immediately on the input.
1381      * 
1382      * Note: The characters are consumed only if they are
1383      * space characters.

1391      * @see com.sun.org.apache.xerces.internal.util.XML11Char#isXML11Space
1392      */
1393     public boolean skipSpaces() throws IOException {
1394 
1395         // load more characters, if needed
1396         if (fCurrentEntity.position == fCurrentEntity.count) {
1397             load(0, true, true);
1398         }
1399 
1400 
1401         //we are doing this check only in skipSpace() because it is called by
1402         //fMiscDispatcher and we want the parser to exit gracefully when document
1403         //is well-formed.
1404         //it is possible that end of document is reached and
1405         //fCurrentEntity becomes null
1406         //nothing was read so entity changed  'false' should be returned.
1407         if(fCurrentEntity == null){
1408             return false ;
1409         }
1410 
1411         // skip spaces
1412         int c = fCurrentEntity.ch[fCurrentEntity.position];
1413 
1414         // External --  Match: S + 0x85 + 0x2028, and perform end of line normalization
1415         if (fCurrentEntity.isExternal()) {
1416             if (XML11Char.isXML11Space(c)) {
1417                 do {
1418                     boolean entityChanged = false;
1419                     // handle newlines
1420                     if (c == '\n' || c == '\r' || c == 0x85 || c == 0x2028) {
1421                         fCurrentEntity.lineNumber++;
1422                         fCurrentEntity.columnNumber = 1;
1423                         if (fCurrentEntity.position == fCurrentEntity.count - 1) {
1424                             invokeListeners(0);
1425                             fCurrentEntity.ch[0] = (char)c;
1426                             entityChanged = load(1, true, false);
1427                             if (!entityChanged) {
1428                                 // the load change the position to be 1,
1429                                 // need to restore it when entity not changed
1430                                 fCurrentEntity.startPosition = 0;
1431                                 fCurrentEntity.position = 0;
1432                             } else if(fCurrentEntity == null){
1433                                 return true ;
1434                             }
1435 
1436                         }
1437                         if (c == '\r') {
1438                             // REVISIT: Does this need to be updated to fix the
1439                             //          #x0D ^#x0A newline normalization problem? -Ac
1440                             int cc = fCurrentEntity.ch[++fCurrentEntity.position];
1441                             if (cc != '\n' && cc != 0x85 ) {
1442                                 fCurrentEntity.position--;
1443                             }
1444                         }
1445                     }
1446                     else {
1447                         fCurrentEntity.columnNumber++;
1448                     }
1449                     // load more characters, if needed
1450                     if (!entityChanged)
1451                         fCurrentEntity.position++;
1452                     if (fCurrentEntity.position == fCurrentEntity.count) {
1453                         load(0, true, true);
1454 
1455                         if(fCurrentEntity == null){
1456                         return true ;
1457                         }
1458 
1459                     }
1460                 } while (XML11Char.isXML11Space(c = fCurrentEntity.ch[fCurrentEntity.position]));
1461                 return true;
1462             }
1463         }
1464         // Internal -- Match: S (only)
1465         else if (XMLChar.isSpace(c)) {
1466             do {
1467                 boolean entityChanged = false;
1468                 // handle newlines
1469                 if (c == '\n') {
1470                     fCurrentEntity.lineNumber++;
1471                     fCurrentEntity.columnNumber = 1;
1472                     if (fCurrentEntity.position == fCurrentEntity.count - 1) {
1473                         fCurrentEntity.ch[0] = (char)c;
1474                         entityChanged = load(1, true, true);
1475                         if (!entityChanged) {
1476                             // the load change the position to be 1,
1477                             // need to restore it when entity not changed
1478                             fCurrentEntity.startPosition = 0;
1479                             fCurrentEntity.position = 0;
1480                         } else if(fCurrentEntity == null){
1481                         return true ;
1482                         }
1483                     }
1484                 }
1485                 else {
1486                     fCurrentEntity.columnNumber++;
1487                 }
1488                 // load more characters, if needed
1489                 if (!entityChanged)
1490                     fCurrentEntity.position++;
1491                 if (fCurrentEntity.position == fCurrentEntity.count) {
1492                     load(0, true, true);
1493 
1494                     if(fCurrentEntity == null){
1495                         return true ;
1496                     }
1497 
1498                 }
1499             } while (XMLChar.isSpace(c = fCurrentEntity.ch[fCurrentEntity.position]));
1500             return true;
1501         }
1502 
1503         // no spaces were found
1504         return false;

1508     /**
1509      * Skips the specified string appearing immediately on the input.
1510      * 
1511      * Note: The characters are consumed only if they are
1512      * space characters.
1513      *
1514      * @param s The string to skip.
1515      *
1516      * @return Returns true if the string was skipped.
1517      *
1518      * @throws IOException  Thrown if i/o error occurs.
1519      * @throws EOFException Thrown on end of file.
1520      */
1521     public boolean skipString(String s) throws IOException {
1522 
1523         // load more characters, if needed
1524         if (fCurrentEntity.position == fCurrentEntity.count) {
1525             load(0, true, true);
1526         }
1527 
1528         // skip string
1529         final int length = s.length();
1530         for (int i = 0; i < length; i++) {
1531             char c = fCurrentEntity.ch[fCurrentEntity.position++];
1532             if (c != s.charAt(i)) {
1533                 fCurrentEntity.position -= i + 1;
1534                 return false;
1535             }
1536             if (i < length - 1 && fCurrentEntity.position == fCurrentEntity.count) {
1537                 invokeListeners(0);
1538                 System.arraycopy(fCurrentEntity.ch, fCurrentEntity.count - i - 1, fCurrentEntity.ch, 0, i + 1);
1539                 // REVISIT: Can a string to be skipped cross an
1540                 //          entity boundary? -Ac
1541                 if (load(i + 1, false, false)) {
1542                     fCurrentEntity.startPosition -= i + 1;
1543                     fCurrentEntity.position -= i + 1;
1544                     return false;
1545                 }
1546             }
1547         }
1548         fCurrentEntity.columnNumber += length;
1549         return true;
1550
