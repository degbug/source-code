  86     /**
  87      * Constructs an ASCII reader from the specified input stream
  88      * and buffer size.
  89      *
  90      * @param inputStream The input stream.
  91      * @param size        The initial buffer size.
  92      * @param messageFormatter  the MessageFormatter to use to message reporting.
  93      * @param locale    the Locale for which messages are to be reported
  94      */
  95     public ASCIIReader(InputStream inputStream, int size,
  96             MessageFormatter messageFormatter, Locale locale) {
  97         fInputStream = inputStream;
  98         BufferAllocator ba = ThreadLocalBufferAllocator.getBufferAllocator();
  99         fBuffer = ba.getByteBuffer(size);
 100         if (fBuffer == null) {
 101             fBuffer = new byte[size];
 102         }
 103         fFormatter = messageFormatter;
 104         fLocale = locale;
 105     } // (InputStream,int, MessageFormatter, Locale)

 111     /**
 112      * Read a single character.  This method will block until a character is
 113      * available, an I/O error occurs, or the end of the stream is reached.
 114      *
 115      *  Subclasses that intend to support efficient single-character input
 116      * should override this method.
 117      *
 118      * @return     The character read, as an integer in the range 0 to 127
 119      *             (0x00-0x7f), or -1 if the end of the stream has
 120      *             been reached
 121      *
 122      * @exception  IOException  If an I/O error occurs
 123      */
 124     public int read() throws IOException {
 125         int b0 = fInputStream.read();
 126         if (b0 >= 0x80) {
 127             throw new MalformedByteSequenceException(fFormatter,
 128                 fLocale, XMLMessageFormatter.XML_DOMAIN,
 129                 "InvalidASCII", new Object [] {Integer.toString(b0)});
 130         }
 131         return b0;
 132     } // read():int
 133 
 134     /**
 135      * Read characters into a portion of an array.  This method will block
 136      * until some input is available, an I/O error occurs, or the end of the
 137      * stream is reached.
 138      *
 139      * @param      ch     Destination buffer
 140      * @param      offset Offset at which to start storing characters
 141      * @param      length Maximum number of characters to read
 142      *
 143      * @return     The number of characters read, or -1 if the end of the
 144      *             stream has been reached
 145      *
 146      * @exception  IOException  If an I/O error occurs
 147      */
 148     public int read(char ch[], int offset, int length) throws IOException {
 149         if (length > fBuffer.length) {
 150             length = fBuffer.length;
 151         }
 152         int count = fInputStream.read(fBuffer, 0, length);
 153         for (int i = 0; i < count; i++) {
 154             int b0 = fBuffer[i];
 155             if (b0 < 0) {
 156                 throw new MalformedByteSequenceException(fFormatter,
 157                     fLocale, XMLMessageFormatter.XML_DOMAIN,
 158                     "InvalidASCII", new Object [] {Integer.toString(b0 & 0x0FF)});
 159             }
 160             ch[offset + i] = (char)b0;
 161         }
 162         return count;
 163     } // read(char[],int,int)

 233     /**
 234      * Close the stream.  Once a stream has been closed, further read(),
 235      * ready(), mark(), or reset() invocations will throw an IOException.
 236      * Closing a previously-closed stream, however, has no effect.

 238      * @exception  IOException  If an I/O error occurs

 240      public void close() throws IOException {
 241          BufferAllocator ba = ThreadLocalBufferAllocator.getBufferAllocator();
