  23 package com.sun.org.apache.xpath.internal;
  24 
  25 
  26 /**
  27  * This object represents a Source Tree, and any associated
  28  * information.
  29  * @xsl.usage internal
  30  */
  31 public class SourceTree
  32 {
  33 
  34   /**
  35    * Constructor SourceTree
  36    *
  37    *
  38    * @param root The root of the source tree, which may or may not be a
  39    * {@link org.w3c.dom.Document} node.
  40    * @param url The URI of the source tree.
  41    */
  42   public SourceTree(int root, String url)
  43   {
  44     m_root = root;
  45     m_url = url;
  46   }
  47 
  48   /** The URI of the source tree.   */
  49   public String m_url;
  50 
  51   /** The root of the source tree, which may or may not be a
  52    * {@link org.w3c.dom.Document} node.  */
  53   public int m_root;
  54 }
