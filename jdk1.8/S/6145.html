  43     public void translate(ClassGenerator classGen, MethodGenerator methodGen) {
  44         final InstructionList il = methodGen.getInstructionList();
  45         argument().translate(classGen, methodGen);
  46         il.append(ICONST_1);
  47         il.append(IXOR);
  48     }
  49 
  50     public void translateDesynthesized(ClassGenerator classGen,
  51                                        MethodGenerator methodGen) {
  52         final InstructionList il = methodGen.getInstructionList();
  53         final Expression exp = argument();
  54         exp.translateDesynthesized(classGen, methodGen);
  55         final BranchHandle gotoh = il.append(new GOTO(null));
  56         _trueList = exp._falseList;     // swap flow lists
  57         _falseList = exp._trueList;
  58         _falseList.add(gotoh);
  59     }
