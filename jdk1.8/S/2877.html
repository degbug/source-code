 316     public void paintComponent(Graphics g)  {
 317         paintTitleBackground(g);
 318 
 319         if(frame.getTitle() != null) {
 320             boolean isSelected = frame.isSelected();
 321             Font f = g.getFont();
 322             g.setFont(getFont());
 323             if(isSelected)
 324                 g.setColor(selectedTextColor);
 325             else
 326                 g.setColor(notSelectedTextColor);
 327 
 328             // Center text vertically.
 329             FontMetrics fm = SwingUtilities2.getFontMetrics(frame, g);
 330             int baseline = (getHeight() + fm.getAscent() - fm.getLeading() -
 331                     fm.getDescent()) / 2;
 332 
 333             int titleX;
 334             Rectangle r = new Rectangle(0, 0, 0, 0);
 335             if (frame.isIconifiable())  r = iconButton.getBounds();
 336             else if (frame.isMaximizable())  r = maxButton.getBounds();
 337             else if (frame.isClosable())  r = closeButton.getBounds();
 338             int titleW;
 339 
 340             String title = frame.getTitle();
 341             if( BasicGraphicsUtils.isLeftToRight(frame) ) {
 342               if (r.x == 0)  r.x = frame.getWidth()-frame.getInsets().right;
 343               titleX = menuBar.getX() + menuBar.getWidth() + 2;
 344               titleW = r.x - titleX - 3;
 345               title = getTitle(frame.getTitle(), fm, titleW);
 346             } else {
 347                 titleX = menuBar.getX() - 2
 348                          - SwingUtilities2.stringWidth(frame,fm,title);
 349             }
 350 
 351             SwingUtilities2.drawString(frame, g, title, titleX, baseline);
 352             g.setFont(f);
 353         }
 354     }

 421     private class Handler implements LayoutManager, PropertyChangeListener {
 422         //
 423         // PropertyChangeListener
 424         //
 425         public void propertyChange(PropertyChangeEvent evt) {
 426             String prop = evt.getPropertyName();
 427 
 428             if (prop == JInternalFrame.IS_SELECTED_PROPERTY) {
 429                 repaint();
 430                 return;
 431             }
 432 
 433             if (prop == JInternalFrame.IS_ICON_PROPERTY ||
 434                     prop == JInternalFrame.IS_MAXIMUM_PROPERTY) {
 435                 setButtonIcons();
 436                 enableActions();
 437                 return;
 438             }
 439 
 440             if ("closable" == prop) {
 441                 if (evt.getNewValue() == Boolean.TRUE) {
 442                     add(closeButton);
 443                 } else {
 444                     remove(closeButton);
 445                 }
 446             } else if ("maximizable" == prop) {
 447                 if (evt.getNewValue() == Boolean.TRUE) {
 448                     add(maxButton);
 449                 } else {
 450                     remove(maxButton);
 451                 }
 452             } else if ("iconable" == prop) {
 453                 if (evt.getNewValue() == Boolean.TRUE) {
 454                     add(iconButton);
 455                 } else {
 456                     remove(iconButton);
 457                 }
 458             }
 459             enableActions();
 460 
 461             revalidate();
 462             repaint();
 463         }

 475         public Dimension minimumLayoutSize(Container c) {
 476             // Calculate width.
 477             int width = 22;
 478 
 479             if (frame.isClosable()) {
 480                 width += 19;
 481             }
 482             if (frame.isMaximizable()) {
 483                 width += 19;
 484             }
 485             if (frame.isIconifiable()) {
 486                 width += 19;
 487             }
 488 
 489             FontMetrics fm = frame.getFontMetrics(getFont());
 490             String frameTitle = frame.getTitle();
 491             int title_w = frameTitle != null ? SwingUtilities2.stringWidth(
 492                                frame, fm, frameTitle) : 0;
 493             int title_length = frameTitle != null ? frameTitle.length() : 0;
 494 
 495             // Leave room for three characters in the title.
 496             if (title_length > 3) {
 497                 int subtitle_w = SwingUtilities2.stringWidth(
 498                     frame, fm, frameTitle.substring(0, 3) + "...");
 499                 width += (title_w < subtitle_w) ? title_w : subtitle_w;
 500             } else {
 501                 width += title_w;
 502             }
 503 
 504             // Calculate height.
 505             Icon icon = frame.getFrameIcon();
 506             int fontHeight = fm.getHeight();
 507             fontHeight += 2;
 508             int iconHeight = 0;
 509             if (icon != null) {
 510                 // SystemMenuBar forces the icon to be 16x16 or less.
 511                 iconHeight = Math.min(icon.getIconHeight(), 16);
 512             }
 513             iconHeight += 2;
 514 
 515             int height = Math.max( fontHeight, iconHeight );
 516 
 517             Dimension dim = new Dimension(width, height);
 518 
 519             // Take into account the border insets if any.
 520             if (getBorder() != null) {
 521                 Insets insets = getBorder().getBorderInsets(c);
 522                 dim.height += insets.top + insets.bottom;
 523                 dim.width += insets.left + insets.right;
 524             }
 525             return dim;
 526         }

 528         public void layoutContainer(Container c) {
 529             boolean leftToRight = BasicGraphicsUtils.isLeftToRight(frame);
 530 
 531             int w = getWidth();
 532             int h = getHeight();
 533             int x;
 534 
 535             int buttonHeight = closeButton.getIcon().getIconHeight();
 536 
 537             Icon icon = frame.getFrameIcon();
 538             int iconHeight = 0;
 539             if (icon != null) {
 540                 iconHeight = icon.getIconHeight();
 541             }
 542             x = (leftToRight) ? 2 : w - 16 - 2;
 543             menuBar.setBounds(x, (h - iconHeight) / 2, 16, 16);
 544 
 545             x = (leftToRight) ? w - 16 - 2 : 2;
 546 
 547             if (frame.isClosable()) {
 548                 closeButton.setBounds(x, (h - buttonHeight) / 2, 16, 14);
 549                 x += (leftToRight) ? -(16 + 2) : 16 + 2;
 550             }
 551 
 552             if (frame.isMaximizable()) {
 553                 maxButton.setBounds(x, (h - buttonHeight) / 2, 16, 14);
 554                 x += (leftToRight) ? -(16 + 2) : 16 + 2;
 555             }
 556 
 557             if (frame.isIconifiable()) {
 558                 iconButton.setBounds(x, (h - buttonHeight) / 2, 16, 14);
 559             }
 560         }

 755     public class SystemMenuBar extends JMenuBar {

 757         public void requestFocus() {}
 758         public void paint(Graphics g) {
 759             Icon icon = frame.getFrameIcon();
 760             if (icon == null) {
 761               icon = (Icon)DefaultLookup.get(frame, frame.getUI(),
 762                       "InternalFrame.icon");
 763             }
 764             if (icon != null) {
 765                 // Resize to 16x16 if necessary.
 766                 if (icon instanceof ImageIcon && (icon.getIconWidth() > 16 || icon.getIconHeight() > 16)) {
 767                     Image img = ((ImageIcon)icon).getImage();
 768                     ((ImageIcon)icon).setImage(img.getScaledInstance(16, 16, Image.SCALE_SMOOTH));
 769                 }
 770                 icon.paintIcon(this, g, 0, 0);
 771             }
 772         }
 773 
 774         public boolean isOpaque() {
 775             return true;
 776         }
 777     } // end SystemMenuBar

 784             setMargin(new Insets(0,0,0,0));
 785             this.uiKey = uiKey;
 786

 789                 setOpaque(((Boolean)opacity).booleanValue());
 790             }
 791         }

 793         public void requestFocus() {}
 794         public AccessibleContext getAccessibleContext() {
 795             AccessibleContext ac = super.getAccessibleContext();
 796             if (uiKey != null) {
 797                 ac.setAccessibleName(UIManager.getString(uiKey));
 798                 uiKey = null;
 799             }
