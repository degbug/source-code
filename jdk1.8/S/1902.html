  28 /**
  29  * File attributes associated with a file in a file system that supports
  30  * legacy "DOS" attributes.
  31  *
  32  *  Usage Example:
  33  * 

  34  *    Path file = ...
  35  *    DosFileAttributes attrs = Files.readAttributes(file, DosFileAttributes.class);
  36  * 
  37  *
  38  * @since 1.7
  39  */
  40 
  41 public interface DosFileAttributes
  42     extends BasicFileAttributes
  43 {
  44     /**
  45      * Returns the value of the read-only attribute.
  46      *
  47      *  This attribute is often used as a simple access control mechanism
  48      * to prevent files from being deleted or updated. Whether the file system
  49      * or platform does any enforcement to prevent read-only files
  50      * from being updated is implementation specific.
  51      *
  52      * @return  the value of the read-only attribute
  53      */
  54     boolean isReadOnly();
  55 
  56     /**
  57      * Returns the value of the hidden attribute.
  58      *
  59      *  This attribute is often used to indicate if the file is visible to
  60      * users.
  61      *
  62      * @return  the value of the hidden attribute
  63      */
  64     boolean isHidden();
  65 
  66     /**
  67      * Returns the value of the archive attribute.
  68      *
  69      *  This attribute is typically used by backup programs.
  70      *
  71      * @return  the value of the archive attribute
  72      */
  73     boolean isArchive();
  74 
  75     /**
  76      * Returns the value of the system attribute.
  77      *
  78      *  This attribute is often used to indicate that the file is a component
  79      * of the operating system.
  80      *
  81      * @return  the value of the system attribute
  82      */
  83     boolean isSystem();
  84 }
