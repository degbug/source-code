  29 /**
  30  * This class provides a skeletal implementation of the <tt>Map</tt>
  31  * interface, to minimize the effort required to implement this interface.
  32  *
  33  * <p>To implement an unmodifiable map, the programmer needs only to extend this
  34  * class and provide an implementation for the <tt>entrySet</tt> method, which
  35  * returns a set-view of the map's mappings.  Typically, the returned set
  36  * will, in turn, be implemented atop <tt>AbstractSet</tt>.  This set should
  37  * not support the <tt>add</tt> or <tt>remove</tt> methods, and its iterator
  38  * should not support the <tt>remove</tt> method.
  39  *
  40  * <p>To implement a modifiable map, the programmer must additionally override
  41  * this class's <tt>put</tt> method (which otherwise throws an
  42  * <tt>UnsupportedOperationException</tt>), and the iterator returned by
  43  * <tt>entrySet().iterator()</tt> must additionally implement its
  44  * <tt>remove</tt> method.
  45  *
  46  * <p>The programmer should generally provide a void (no argument) and map
  47  * constructor, as per the recommendation in the <tt>Map</tt> interface
  48  * specification.

  51  * implementation in detail.  Each of these methods may be overridden if the
  52  * map being implemented admits a more efficient implementation.

 212     /**
 213      * {@inheritDoc}
 214      *
 215      * @implSpec
 216      * This implementation iterates over <tt>entrySet()</tt> searching for an
 217      * entry with the specified key.  If such an entry is found, its value is
 218      * obtained with its <tt>getValue</tt> operation, the entry is removed
 219      * from the collection (and the backing map) with the iterator's
 220      * <tt>remove</tt> operation, and the saved value is returned.  If the
 221      * iteration terminates without finding such an entry, <tt>null</tt> is
 222      * returned.  Note that this implementation requires linear time in the
 223      * size of the map; many implementations will override this method.
 224      *
 225      * <p>Note that this implementation throws an
 226      * <tt>UnsupportedOperationException</tt> if the <tt>entrySet</tt>
 227      * iterator does not support the <tt>remove</tt> method and this map
 228      * contains a mapping for the specified key.
 229      *
 230      * @throws UnsupportedOperationException {@inheritDoc}
 231      * @throws ClassCastException            {@inheritDoc}
 232      * @throws NullPointerException          {@inheritDoc}
 233      */
 234     public V remove(Object key) {
 235         Iterator<Entry<K,V>> i = entrySet().iterator();
 236         Entry<K,V> correctEntry = null;
 237         if (key==null) {
 238             while (correctEntry==null && i.hasNext()) {
 239                 Entry<K,V> e = i.next();
 240                 if (e.getKey()==null)
 241                     correctEntry = e;
 242             }
 243         } else {
 244             while (correctEntry==null && i.hasNext()) {
 245                 Entry<K,V> e = i.next();
 246                 if (key.equals(e.getKey()))
 247                     correctEntry = e;
 248             }
 249         }
 250 
 251         V oldValue = null;
 252         if (correctEntry !=null) {
 253             oldValue = correctEntry.getValue();
 254             i.remove();
 255         }
 256         return oldValue;
 257     }

 430     /**
 431      * Compares the specified object with this map for equality.  Returns
 432      * <tt>true</tt> if the given object is also a map and the two maps
 433      * represent the same mappings.  More formally, two maps <tt>m1</tt> and
 434      * <tt>m2</tt> represent the same mappings if
 435      * <tt>m1.entrySet().equals(m2.entrySet())</tt>.  This ensures that the
 436      * <tt>equals</tt> method works properly across different implementations
 437      * of the <tt>Map</tt> interface.
 438      *
 439      * @implSpec
 440      * This implementation first checks if the specified object is this map;
 441      * if so it returns <tt>true</tt>.  Then, it checks if the specified
 442      * object is a map whose size is identical to the size of this map; if
 443      * not, it returns <tt>false</tt>.  If so, it iterates over this map's
 444      * <tt>entrySet</tt> collection, and checks that the specified map
 445      * contains each mapping that this map contains.  If the specified map
 446      * fails to contain such a mapping, <tt>false</tt> is returned.  If the
 447      * iteration completes, <tt>true</tt> is returned.
 448      *
 449      * @param o object to be compared for equality with this map
 450      * @return <tt>true</tt> if the specified object is equal to this map
 451      */
 452     public boolean equals(Object o) {
 453         if (o == this)
 454             return true;
 455 
 456         if (!(o instanceof Map))
 457             return false;
 458         Map<?,?> m = (Map<?,?>) o;
 459         if (m.size() != size())
 460             return false;
 461 
 462         try {
 463             Iterator<Entry<K,V>> i = entrySet().iterator();
 464             while (i.hasNext()) {
 465                 Entry<K,V> e = i.next();
 466                 K key = e.getKey();
 467                 V value = e.getValue();
 468                 if (value == null) {
 469                     if (!(m.get(key)==null && m.containsKey(key)))
 470                         return false;
 471                 } else {
 472                     if (!value.equals(m.get(key)))
 473                         return false;
 474                 }
 475             }
 476         } catch (ClassCastException unused) {
 477             return false;
 478         } catch (NullPointerException unused) {
 479             return false;
 480         }
 481 
 482         return true;
 483     }

 485     /**
 486      * Returns the hash code value for this map.  The hash code of a map is
 487      * defined to be the sum of the hash codes of each entry in the map's
 488      * <tt>entrySet()</tt> view.  This ensures that <tt>m1.equals(m2)</tt>
 489      * implies that <tt>m1.hashCode()==m2.hashCode()</tt> for any two maps
 490      * <tt>m1</tt> and <tt>m2</tt>, as required by the general contract of
 491      * {@link Object#hashCode}.

 493      * @implSpec
 494      * This implementation iterates over <tt>entrySet()</tt>, calling
 495      * {@link Map.Entry#hashCode hashCode()} on each element (entry) in the
 496      * set, and adding up the results.
 497      *
 498      * @return the hash code value for this map
 499      * @see Map.Entry#hashCode()

 511     /**
 512      * Returns a string representation of this map.  The string representation
 513      * consists of a list of key-value mappings in the order returned by the
 514      * map's <tt>entrySet</tt> view's iterator, enclosed in braces
 515      * (<tt>"{}"</tt>).  Adjacent mappings are separated by the characters
 516      * <tt>", "</tt> (comma and space).  Each key-value mapping is rendered as
 517      * the key followed by an equals sign (<tt>"="</tt>) followed by the
 518      * associated value.  Keys and values are converted to strings as by
 519      * {@link String#valueOf(Object)}.
 520      *
 521      * @return a string representation of this map
 522      */
 523     public String toString() {
 524         Iterator<Entry<K,V>> i = entrySet().iterator();
 525         if (! i.hasNext())
 526             return "{}";
 527 
 528         StringBuilder sb = new StringBuilder();
 529         sb.append('{');
 530         for (;;) {
 531             Entry<K,V> e = i.next();
 532             K key = e.getKey();
 533             V value = e.getValue();
 534             sb.append(key   == this ? "(this Map)" : key);
 535             sb.append('=');
 536             sb.append(value == this ? "(this Map)" : value);
 537             if (! i.hasNext())
 538                 return sb.append('}').toString();
 539             sb.append(',').append(' ');
 540         }
 541     }

 701             return key + "=" + value;
 702         }

 704     }

 706     /**
 707      * An Entry maintaining an immutable key and value.  This class
 708      * does not support method <tt>setValue</tt>.  This class may be
 709      * convenient in methods that return thread-safe snapshots of
 710      * key-value mappings.
 711      *
 712      * @since 1.6
 713      */
 714     public static class SimpleImmutableEntry<K,V>
 715         implements Entry<K,V>, java.io.Serializable
 716     {
 717         private static final long serialVersionUID = 7138329143949025153L;
 718 
 719         private final K key;
 720         private final V value;
 721 
 722         /**
 723          * Creates an entry representing a mapping from the specified
 724          * key to the specified value.
 725          *
 726          * @param key the key represented by this entry
 727          * @param value the value represented by this entry
 728          */
 729         public SimpleImmutableEntry(K key, V value) {
 730             this.key   = key;
 731             this.value = value;
 732         }
 733 
 734         /**
 735          * Creates an entry representing the same mapping as the
 736          * specified entry.
 737          *
 738          * @param entry the entry to copy
 739          */
 740         public SimpleImmutableEntry(Entry<? extends K, ? extends V> entry) {
 741             this.key   = entry.getKey();
 742             this.value = entry.getValue();
 743         }
 744 
 745         /**
 746          * Returns the key corresponding to this entry.
 747          *
 748          * @return the key corresponding to this entry
 749          */
 750         public K getKey() {
 751             return key;
 752         }
 753 
 754         /**
 755          * Returns the value corresponding to this entry.
 756          *
 757          * @return the value corresponding to this entry
 758          */
 759         public V getValue() {
 760             return value;
 761         }
 762 
 763         /**
 764          * Replaces the value corresponding to this entry with the specified
 765          * value (optional operation).  This implementation simply throws
 766          * <tt>UnsupportedOperationException</tt>, as this class implements
 767          * an <i>immutable</i> map entry.
 768          *
 769          * @param value new value to be stored in this entry
 770          * @return (Does not return)
 771          * @throws UnsupportedOperationException always
 772          */
 773         public V setValue(V value) {
 774             throw new UnsupportedOperationException();
 775         }
 776 
 777         /**
 778          * Compares the specified object with this entry for equality.
 779          * Returns {@code true} if the given object is also a map entry and
 780          * the two entries represent the same mapping.  More formally, two
 781          * entries {@code e1} and {@code e2} represent the same mapping
 782          * if<pre>
 783          *   (e1.getKey()==null ?
 784          *    e2.getKey()==null :
 785          *    e1.getKey().equals(e2.getKey()))
 786          *   &amp;&amp;
 787          *   (e1.getValue()==null ?
 788          *    e2.getValue()==null :
 789          *    e1.getValue().equals(e2.getValue()))</pre>
 790          * This ensures that the {@code equals} method works properly across
 791          * different implementations of the {@code Map.Entry} interface.
 792          *
 793          * @param o object to be compared for equality with this map entry
 794          * @return {@code true} if the specified object is equal to this map
 795          *         entry
 796          * @see    #hashCode
 797          */
 798         public boolean equals(Object o) {
 799             if (!(o instanceof Map.Entry))
 800                 return false;
 801             Map.Entry<?,?> e = (Map.Entry<?,?>)o;
 802             return eq(key, e.getKey()) && eq(value, e.getValue());
 803         }
 804 
 805         /**
 806          * Returns the hash code value for this map entry.  The hash code
 807          * of a map entry {@code e} is defined to be: <pre>
 808          *   (e.getKey()==null   ? 0 : e.getKey().hashCode()) ^
 809          *   (e.getValue()==null ? 0 : e.getValue().hashCode())</pre>
 810          * This ensures that {@code e1.equals(e2)} implies that
 811          * {@code e1.hashCode()==e2.hashCode()} for any two Entries
 812          * {@code e1} and {@code e2}, as required by the general
 813          * contract of {@link Object#hashCode}.
 814          *
 815          * @return the hash code value for this map entry
 816          * @see    #equals
 817          */
 818         public int hashCode() {
 819             return (key   == null ? 0 :   key.hashCode()) ^
 820                    (value == null ? 0 : value.hashCode());
 821         }
 822 
 823         /**
 824          * Returns a String representation of this map entry.  This
 825          * implementation returns the string representation of this
 826          * entry's key followed by the equals character ("<tt>=</tt>")
 827          * followed by the string representation of this entry's value.
 828          *
 829          * @return a String representation of this map entry
 830          */
 831         public String toString() {
 832             return key + "=" + value;
 833         }
 834 
 835     }
