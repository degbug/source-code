  23 package com.sun.org.apache.xpath.internal.axes;
  24 
  25 import org.w3c.dom.Node;
  26 import org.w3c.dom.traversal.NodeIterator;
  27 
  28 /**
  29  * Classes who implement this interface can be a
  30  * current node list,
  31  * also refered to here as a context node list.
  32  * @xsl.usage advanced
  33  */
  34 public interface ContextNodeList
  35 {
  36 
  37   /**
  38    * Get the current node.
  39    *
  40    *
  41    * @return The current node, or null.
  42    */
  43   public Node getCurrentNode();
  44 
  45   /**
  46    * Get the current position, which is one less than
  47    * the next nextNode() call will retrieve.  i.e. if
  48    * you call getCurrentPos() and the return is 0, the next
  49    * fetch will take place at index 1.
  50    *
  51    * @return The position of the
  52    * current node
  53    * in the  current node list.
  54    */
  55   public int getCurrentPos();
  56 
  57   /**
  58    * Reset the iterator.
  59    */
  60   public void reset();
  61 
  62   /**
  63    * If setShouldCacheNodes(true) is called, then nodes will
  64    * be cached.  They are not cached by default.
  65    *
  66    * @param b true if the nodes should be cached.
  67    */
  68   public void setShouldCacheNodes(boolean b);
  69 
  70   /**
  71    * If an index is requested, NodeSetDTM will call this method
  72    * to run the iterator to the index.  By default this sets
  73    * m_next to the index.  If the index argument is -1, this
  74    * signals that the iterator should be run to the end.
  75    *
  76    * @param index The index to run to, or -1 if the iterator should be run
  77    *              to the end.
  78    */
  79   public void runTo(int index);
  80 
  81   /**
  82    * Set the current position in the node set.
  83    * @param i Must be a valid index.
  84    */
  85   public void setCurrentPos(int i);
  86 
  87   /**
  88    * Get the length of the list.
  89    *
  90    * @return The number of nodes in this node list.
  91    */
  92   public int size();
  93 
  94   /**
  95    * Tells if this NodeSetDTM is "fresh", in other words, if
  96    * the first nextNode() that is called will return the
  97    * first node in the set.
  98    *
  99    * @return true if the iteration of this list has not yet begun.
 100    */
 101   public boolean isFresh();
 102 
 103   /**
 104    * Get a cloned Iterator that is reset to the start of the iteration.
 105    *
 106    * @return A clone of this iteration that has been reset.
 107    *
 108    * @throws CloneNotSupportedException
 109    */
 110   public NodeIterator cloneWithReset() throws CloneNotSupportedException;
 111 
 112   /**
 113    * Get a clone of this iterator.  Be aware that this operation may be
 114    * somewhat expensive.
 115    *
 116    *
 117    * @return A clone of this object.
 118    *
 119    * @throws CloneNotSupportedException
 120    */
 121   public Object clone() throws CloneNotSupportedException;
 122 
 123   /**
 124    * Get the index of the last node in this list.
 125    *
 126    *
 127    * @return the index of the last node in this list.
 128    */
 129   public int getLast();
 130 
 131   /**
 132    * Set the index of the last node in this list.
 133    *
 134    *
 135    * @param last the index of the last node in this list.
 136    */
 137   public void setLast(int last);
 138 }
