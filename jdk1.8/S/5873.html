  25 /**
  26  * A very simple table that stores a list of byte. Very similar API to our
  27  * IntVector class (same API); different internal storage.
  28  *
  29  * This version uses an array-of-arrays solution. Read/write access is thus
  30  * a bit slower than the simple IntVector, and basic storage is a trifle
  31  * higher due to the top-level array -- but appending is O(1) fast rather
  32  * than O(N**2) slow, which will swamp those costs in situations where
  33  * long vectors are being built up.
  34  *
  35  * Known issues:
  36  *
  37  * Some methods are private because they haven't yet been tested properly.
  38  *
  39  * If an element has not been set (because we skipped it), its value will
  40  * initially be 0. Shortening the vector does not clear old storage; if you
  41  * then skip values and setElementAt a higher index again, you may see old data
  42  * reappear in the truncated-and-restored section. Doing anything else would
  43  * have performance costs.
  44  * @xsl.usage internal
  45  */
  46 public class SuballocatedByteVector
  47 {
  48   /** Size of blocks to allocate          */
  49   protected int m_blocksize;
  50 
  51   /** Number of blocks to (over)allocate by */
  52   protected  int m_numblocks=32;
  53 
  54   /** Array of arrays of bytes          */
  55   protected byte m_map[][];
  56 
  57   /** Number of bytes in array          */
  58   protected int m_firstFree = 0;
  59 
  60   /** "Shortcut" handle to m_map[0] */
  61   protected byte m_map0[];
  62 
  63   /**
  64    * Default constructor.  Note that the default
  65    * block size is very small, for small lists.
  66    */
  67   public SuballocatedByteVector()

  70   }

  72   /**
  73    * Construct a ByteVector, using the given block size.
  74    *
  75    * @param blocksize Size of block to allocate
  76    */
  77   public SuballocatedByteVector(int blocksize)
  78   {
  79     m_blocksize = blocksize;
  80     m_map0=new byte[blocksize];
  81     m_map = new byte[m_numblocks][];
  82     m_map[0]=m_map0;
  83   }

 118   /**
 119    * Append a byte onto the vector.
 120    *
 121    * @param value Byte to add to the list
 122    */
 123   public  void addElement(byte value)
 124   {
 125     if(m_firstFree<m_blocksize)
 126       m_map0[m_firstFree++]=value;
 127     else
 128     {
 129       int index=m_firstFree/m_blocksize;
 130       int offset=m_firstFree%m_blocksize;
 131       ++m_firstFree;
 132 
 133       if(index>=m_map.length)
 134       {
 135         int newsize=index+m_numblocks;
 136         byte[][] newMap=new byte[newsize][];
 137         System.arraycopy(m_map, 0, newMap, 0, m_map.length);
 138         m_map=newMap;
 139       }
 140       byte[] block=m_map[index];
 141       if(null==block)
 142         block=m_map[index]=new byte[m_blocksize];
 143       block[offset]=value;
 144     }
 145   }

 147   /**
 148    * Append several byte values onto the vector.
 149    *
 150    * @param value Byte to add to the list
 151    */
 152   private  void addElements(byte value, int numberOfElements)
 153   {
 154     if(m_firstFree+numberOfElements<m_blocksize)
 155       for (int i = 0; i < numberOfElements; i++)
 156       {
 157         m_map0[m_firstFree++]=value;
 158       }
 159     else
 160     {
 161       int index=m_firstFree/m_blocksize;
 162       int offset=m_firstFree%m_blocksize;
 163       m_firstFree+=numberOfElements;
 164       while( numberOfElements>0)
 165       {
 166         if(index>=m_map.length)
 167         {
 168           int newsize=index+m_numblocks;
 169           byte[][] newMap=new byte[newsize][];
 170           System.arraycopy(m_map, 0, newMap, 0, m_map.length);
 171           m_map=newMap;
 172         }
 173         byte[] block=m_map[index];
 174         if(null==block)
 175           block=m_map[index]=new byte[m_blocksize];
 176         int copied=(m_blocksize-offset < numberOfElements)
 177           ? m_blocksize-offset : numberOfElements;
 178         numberOfElements-=copied;
 179         while(copied-- > 0)
 180           block[offset++]=value;
 181 
 182         ++index;offset=0;
 183       }
 184     }
 185   }

 187   /**
 188    * Append several slots onto the vector, but do not set the values.
 189    * Note: "Not Set" means the value is unspecified.
 190    *
 191    * @param numberOfElements
 192    */
 193   private  void addElements(int numberOfElements)
 194   {
 195     int newlen=m_firstFree+numberOfElements;
 196     if(newlen>m_blocksize)
 197     {
 198       int index=m_firstFree%m_blocksize;
 199       int newindex=(m_firstFree+numberOfElements)%m_blocksize;
 200       for(int i=index+1;i<=newindex;++i)
 201         m_map[i]=new byte[m_blocksize];
 202     }
 203     m_firstFree=newlen;
 204   }

 206   /**
 207    * Inserts the specified node in this vector at the specified index.
 208    * Each component in this vector with an index greater or equal to
 209    * the specified index is shifted upward to have an index one greater
 210    * than the value it had previously.
 211    *
 212    * Insertion may be an EXPENSIVE operation!
 213    *
 214    * @param value Byte to insert
 215    * @param at Index of where to insert
 216    */
 217   private  void insertElementAt(byte value, int at)
 218   {
 219     if(at==m_firstFree)
 220       addElement(value);
 221     else if (at>m_firstFree)
 222     {
 223       int index=at/m_blocksize;
 224       if(index>=m_map.length)
 225       {
 226         int newsize=index+m_numblocks;
 227         byte[][] newMap=new byte[newsize][];
 228         System.arraycopy(m_map, 0, newMap, 0, m_map.length);
 229         m_map=newMap;
 230       }
 231       byte[] block=m_map[index];
 232       if(null==block)
 233         block=m_map[index]=new byte[m_blocksize];
 234       int offset=at%m_blocksize;
 235       block[offset]=value;
 236       m_firstFree=offset+1;
 237     }
 238     else
 239     {
 240       int index=at/m_blocksize;
 241       int maxindex=m_firstFree+1/m_blocksize;
 242       ++m_firstFree;
 243       int offset=at%m_blocksize;
 244       byte push;
 245 
 246       // ***** Easier to work down from top?
 247       while(index<=maxindex)
 248       {
 249         int copylen=m_blocksize-offset-1;
 250         byte[] block=m_map[index];
 251         if(null==block)
 252         {
 253           push=0;
 254           block=m_map[index]=new byte[m_blocksize];
 255         }
 256         else
 257         {
 258           push=block[m_blocksize-1];
 259           System.arraycopy(block, offset , block, offset+1, copylen);
 260         }
 261         block[offset]=value;
 262         value=push;
 263         offset=0;
 264         ++index;
 265       }
 266     }
 267   }

 277   /**
 278    * Removes the first occurrence of the argument from this vector.
 279    * If the object is found in this vector, each component in the vector
 280    * with an index greater or equal to the object's index is shifted
 281    * downward to have an index one smaller than the value it had
 282    * previously.
 283    *
 284    * @param s Byte to remove from array
 285    *
 286    * @return True if the byte was removed, false if it was not found
 287    */
 288   private  boolean removeElement(byte s)
 289   {
 290     int at=indexOf(s,0);
 291     if(at<0)
 292       return false;
 293     removeElementAt(at);
 294     return true;
 295   }

 297   /**
 298    * Deletes the component at the specified index. Each component in
 299    * this vector with an index greater or equal to the specified
 300    * index is shifted downward to have an index one smaller than
 301    * the value it had previously.
 302    *
 303    * @param at index of where to remove a byte
 304    */
 305   private  void removeElementAt(int at)
 306   {
 307     // No point in removing elements that "don't exist"...
 308     if(at<m_firstFree)
 309     {
 310       int index=at/m_blocksize;
 311       int maxindex=m_firstFree/m_blocksize;
 312       int offset=at%m_blocksize;
 313 
 314       while(index<=maxindex)
 315       {
 316         int copylen=m_blocksize-offset-1;
 317         byte[] block=m_map[index];
 318         if(null==block)
 319           block=m_map[index]=new byte[m_blocksize];
 320         else
 321           System.arraycopy(block, offset+1, block, offset, copylen);
 322         if(index<maxindex)
 323         {
 324           byte[] next=m_map[index+1];
 325           if(next!=null)
 326             block[m_blocksize-1]=(next!=null) ? next[0] : 0;
 327         }
 328         else
 329           block[m_blocksize-1]=0;
 330         offset=0;
 331         ++index;
 332       }
 333     }
 334     --m_firstFree;
 335   }

 337   /**
 338    * Sets the component at the specified index of this vector to be the
 339    * specified object. The previous component at that position is discarded.
 340    *
 341    * The index must be a value greater than or equal to 0 and less
 342    * than the current size of the vector.
 343    *
 344    * @param value
 345    * @param at     Index of where to set the object
 346    */
 347   public void setElementAt(byte value, int at)
 348   {
 349     if(at<m_blocksize)
 350     {
 351       m_map0[at]=value;
 352       return;
 353     }
 354 
 355     int index=at/m_blocksize;
 356     int offset=at%m_blocksize;
 357 
 358     if(index>=m_map.length)
 359     {
 360       int newsize=index+m_numblocks;
 361       byte[][] newMap=new byte[newsize][];
 362       System.arraycopy(m_map, 0, newMap, 0, m_map.length);
 363       m_map=newMap;
 364     }
 365 
 366     byte[] block=m_map[index];
 367     if(null==block)
 368       block=m_map[index]=new byte[m_blocksize];
 369     block[offset]=value;
 370 
 371     if(at>=m_firstFree)
 372       m_firstFree=at+1;
 373   }

 375   /**
 376    * Get the nth element. This is often at the innermost loop of an
 377    * application, so performance is critical.
 378    *
 379    * @param i index of value to get
 380    *
 381    * @return value at given index. If that value wasn't previously set,
 382    * the result is undefined for performance reasons. It may throw an
 383    * exception (see below), may return zero, or (if setSize has previously
 384    * been used) may return stale data.
 385    *
 386    * @throws ArrayIndexOutOfBoundsException if the index was _clearly_
 387    * unreasonable (negative, or past the highest block).
 388    *
 389    * @throws NullPointerException if the index points to a block that could
 390    * have existed (based on the highest index used) but has never had anything
 391    * set into it.
 392    * %REVIEW% Could add a catch to create the block in that case, or return 0.
 393    * Try/Catch is _supposed_ to be nearly free when not thrown to. Do we
 394    * believe that? Should we have a separate safeElementAt?
 395    */
 396   public byte elementAt(int i)
 397   {
 398     // %OPT% Does this really buy us anything? Test versus division for small,
 399     // test _plus_ division for big docs.
 400     if(i<m_blocksize)
 401       return m_map0[i];
 402 
 403     return m_map[i/m_blocksize][i%m_blocksize];
 404   }

 418   /**
 419    * Searches for the first occurence of the given argument,
 420    * beginning the search at index, and testing for equality
 421    * using the equals method.
 422    *
 423    * @param elem object to look for
 424    * @param index Index of where to begin search
 425    * @return the index of the first occurrence of the object
 426    * argument in this vector at position index or later in the
 427    * vector; returns -1 if the object is not found.
 428    */
 429   public int indexOf(byte elem, int index)
 430   {
 431     if(index>=m_firstFree)
 432       return -1;
 433 
 434     int bindex=index/m_blocksize;
 435     int boffset=index%m_blocksize;
 436     int maxindex=m_firstFree/m_blocksize;
 437     byte[] block;
 438 
 439     for(;bindex<maxindex;++bindex)
 440     {
 441       block=m_map[bindex];
 442       if(block!=null)
 443         for(int offset=boffset;offset<m_blocksize;++offset)
 444           if(block[offset]==elem)
 445             return offset+bindex*m_blocksize;
 446       boffset=0; // after first
 447     }
 448     // Last block may need to stop before end
 449     int maxoffset=m_firstFree%m_blocksize;
 450     block=m_map[maxindex];
 451     for(int offset=boffset;offset<maxoffset;++offset)
 452       if(block[offset]==elem)
 453         return offset+maxindex*m_blocksize;
 454 
 455     return -1;
 456   }

 473   /**
 474    * Searches for the first occurence of the given argument,
 475    * beginning the search at index, and testing for equality
 476    * using the equals method.
 477    *
 478    * @param elem Object to look for
 479    * @return the index of the first occurrence of the object
 480    * argument in this vector at position index or later in the
 481    * vector; returns -1 if the object is not found.
 482    */
 483   private  int lastIndexOf(byte elem)
 484   {
 485     int boffset=m_firstFree%m_blocksize;
 486     for(int index=m_firstFree/m_blocksize;
 487         index>=0;
 488         --index)
 489     {
 490       byte[] block=m_map[index];
 491       if(block!=null)
 492         for(int offset=boffset; offset>=0; --offset)
 493           if(block[offset]==elem)
 494             return offset+index*m_blocksize;
 495       boffset=0; // after first
 496     }
 497     return -1;
 498   }
