 148     private void parseExpression(final NamespaceContext context)
 149         throws XPathException {
 150 
 151         // tokens
 152         final XPath.Tokens xtokens = new XPath.Tokens(fSymbolTable);
 153 
 154         // scanner
 155         XPath.Scanner scanner = new XPath.Scanner(fSymbolTable) {
 156             protected void addToken(XPath.Tokens tokens, int token)
 157                 throws XPathException {
 158                 if (
 159                     token == XPath.Tokens.EXPRTOKEN_ATSIGN ||
 160                     token == XPath.Tokens.EXPRTOKEN_AXISNAME_ATTRIBUTE ||
 161                     token == XPath.Tokens.EXPRTOKEN_NAMETEST_QNAME ||
 162                     token == XPath.Tokens.EXPRTOKEN_OPERATOR_SLASH ||
 163                     token == XPath.Tokens.EXPRTOKEN_PERIOD ||
 164                     token == XPath.Tokens.EXPRTOKEN_NAMETEST_ANY ||
 165                     token == XPath.Tokens.EXPRTOKEN_NAMETEST_NAMESPACE ||
 166                     token == XPath.Tokens.EXPRTOKEN_OPERATOR_DOUBLE_SLASH ||
 167                     token == XPath.Tokens.EXPRTOKEN_OPERATOR_UNION ||
 168                     token == XPath.Tokens.EXPRTOKEN_AXISNAME_CHILD ||
 169                     token == XPath.Tokens.EXPRTOKEN_DOUBLE_COLON
 170                     ) {
 171                     super.addToken(tokens, token);
 172                     return;
 173                 }
 174                 throw new XPathException("c-general-xpath");
 175             }
 176         };
 177 
 178         int length = fExpression.length();
 179 
 180         boolean success = scanner.scanExpr(fSymbolTable,
 181                                            xtokens, fExpression, 0, length);
 182         if(!success)
 183             throw new XPathException("c-general-xpath");
 184 
 185         //fTokens.dumpTokens();
 186         Vector stepsVector = new Vector();
 187         Vector locationPathsVector= new Vector();
 188 
 189         // true when the next token should be 'Step' (as defined in
 190         // the production rule [3] of XML Schema P1 section 3.11.6
 191         // if false, we are expecting either '|' or '/'.
 192         //
 193         // this is to make sure we can detect a token list like
 194         // 'abc' '/' '/' 'def' 'ghi'
 195         boolean expectingStep = true;
 196         boolean expectingDoubleColon = false;
 197 
 198         while(xtokens.hasMore()) {
 199             final int token = xtokens.nextToken();
 200 
 201             switch (token) {
 202                 case  XPath.Tokens.EXPRTOKEN_OPERATOR_UNION :{
 203                     check(!expectingStep);
 204                     locationPathsVector.addElement(buildLocationPath(stepsVector));
 205                     expectingStep=true;
 206                     break;
 207                 }
 208 
 209                 case XPath.Tokens.EXPRTOKEN_ATSIGN: {
 210                     check(expectingStep);
 211                     Step step = new Step(
 212                             new Axis(Axis.ATTRIBUTE),
 213                             parseNodeTest(xtokens.nextToken(),xtokens,context));
 214                     stepsVector.addElement(step);
 215                     expectingStep=false;
 216                     break;
 217                 }
 218                 case XPath.Tokens.EXPRTOKEN_NAMETEST_ANY:
 219                 case XPath.Tokens.EXPRTOKEN_NAMETEST_NAMESPACE:
 220                 case XPath.Tokens.EXPRTOKEN_NAMETEST_QNAME: {
 221                     check(expectingStep);
 222                     Step step = new Step(
 223                             new Axis(Axis.CHILD),
 224                             parseNodeTest(token,xtokens,context));
 225                     stepsVector.addElement(step);
 226                     expectingStep=false;
 227                     break;
 228                 }
 229 
 230                 case XPath.Tokens.EXPRTOKEN_PERIOD: {
 231                     check(expectingStep);
 232                     expectingStep=false;
 233 
 234                     // unless this is the first step in this location path,
 235                     // there's really no reason to keep them in LocationPath.
 236                     // This amounts to shorten "a/././b/./c" to "a/b/c".
 237                     // Also, the matcher fails to work correctly if XPath
 238                     // has those redundant dots.
 239                     if (stepsVector.size()==0) {
 240                         // build step
 241                         Axis axis = new Axis(Axis.SELF);
 242                         NodeTest nodeTest = new NodeTest(NodeTest.NODE);
 243                         Step step = new Step(axis, nodeTest);
 244                         stepsVector.addElement(step);
 245 
 246                         if( xtokens.hasMore()
 247                          && xtokens.peekToken() == XPath.Tokens.EXPRTOKEN_OPERATOR_DOUBLE_SLASH){
 248                             // consume '//'
 249                             xtokens.nextToken();
 250 
 251                             // build step
 252                             axis = new Axis(Axis.DESCENDANT);
 253                             nodeTest = new NodeTest(NodeTest.NODE);
 254                             step = new Step(axis, nodeTest);
 255                             stepsVector.addElement(step);
 256                             expectingStep=true;
 257                         }
 258                     }
 259                     break;
 260                 }
 261 
 262                 case XPath.Tokens.EXPRTOKEN_OPERATOR_DOUBLE_SLASH:{
 263                     // this cannot appear in arbitrary position.
 264                     // it is only allowed right after '.' when
 265                     // '.' is the first token of a location path.
 266                     throw new XPathException("c-general-xpath");
 267                 }
 268                 case XPath.Tokens.EXPRTOKEN_OPERATOR_SLASH: {
 269                     check(!expectingStep);
 270                     expectingStep=true;
 271                     break;
 272                 }
 273                 case XPath.Tokens.EXPRTOKEN_AXISNAME_ATTRIBUTE: {
 274                      check(expectingStep);
 275                      expectingDoubleColon = true;
 276 
 277                      if (xtokens.nextToken() == XPath.Tokens.EXPRTOKEN_DOUBLE_COLON){
 278                          Step step = new Step(
 279                          new Axis(Axis.ATTRIBUTE),
 280                                  parseNodeTest(xtokens.nextToken(),xtokens,context));
 281                          stepsVector.addElement(step);
 282                          expectingStep=false;
 283                          expectingDoubleColon = false;
 284                      }
 285                      break;
 286                 }
 287                 case XPath.Tokens.EXPRTOKEN_AXISNAME_CHILD:{
 288                     check(expectingStep);
 289                     expectingDoubleColon = true;
 290                     break;
 291                 }
 292                 case XPath.Tokens.EXPRTOKEN_DOUBLE_COLON :{
 293                     check(expectingStep);
 294                     check(expectingDoubleColon);
 295                     expectingDoubleColon = false;
 296                     break;
 297                 }
 298                 default:
 299                     // we should have covered all the tokens that we can possibly see.
 300                     throw new XPathException("c-general-xpath");
 301            }
 302         }
 303 
 304         check(!expectingStep);
 305 
 306         locationPathsVector.addElement(buildLocationPath(stepsVector));
 307 
 308         // save location path
 309         fLocationPaths=new LocationPath[locationPathsVector.size()];
 310         locationPathsVector.copyInto(fLocationPaths);
 311 
 312 
 313         if (DEBUG_XPATH_PARSE) {
 314             System.out.println(">>> "+fLocationPaths);
 315         }
 316 
 317     } // parseExpression(SymbolTable,NamespaceContext)

 323     private NodeTest parseNodeTest( int typeToken, Tokens xtokens, NamespaceContext context )
 324         throws XPathException {
 325         switch(typeToken) {
 326         case XPath.Tokens.EXPRTOKEN_NAMETEST_ANY:
 327             return new NodeTest(NodeTest.WILDCARD);
 328 
 329         case XPath.Tokens.EXPRTOKEN_NAMETEST_NAMESPACE:
 330         case XPath.Tokens.EXPRTOKEN_NAMETEST_QNAME:
 331             // consume QName token
 332             String prefix = xtokens.nextTokenAsString();
 333             String uri = null;
 334             if (context != null && prefix != XMLSymbols.EMPTY_STRING) {
 335                 uri = context.getURI(prefix);
 336             }
 337             if (prefix != XMLSymbols.EMPTY_STRING && context != null && uri == null) {
 338                 throw new XPathException("c-general-xpath-ns");
 339             }
 340 
 341             if (typeToken==XPath.Tokens.EXPRTOKEN_NAMETEST_NAMESPACE)
 342                 return new NodeTest(prefix,uri);
 343 
 344             String localpart = xtokens.nextTokenAsString();
 345             String rawname = prefix != XMLSymbols.EMPTY_STRING
 346             ? fSymbolTable.addSymbol(prefix+':'+localpart)
 347             : localpart;
 348 
 349             return new NodeTest(new QName(prefix, localpart, rawname, uri));
 350 
 351         default:
 352             // assertion error
 353             throw new XPathException("c-general-xpath");
 354 
 355         }
