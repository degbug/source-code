  67     /* An Executor that allows at most one executing and one pending
  68        Runnable.  It uses at most one thread -- as soon as there is
  69        no pending Runnable the thread can exit.  Another thread is
  70        created as soon as there is a new pending Runnable.  This
  71        Executor is adapted for use in a situation where each Runnable
  72        usually schedules up another Runnable.  On return from the
  73        first one, the second one is immediately executed.  So this
  74        just becomes a complicated way to write a while loop, but with
  75        the advantage that you can replace it with another Executor,
  76        for instance one that you are using to execute a bunch of other
  77        unrelated work.
  78 
  79        You might expect that a java.util.concurrent.ThreadPoolExecutor
  80        with corePoolSize=0 and maximumPoolSize=1 would have the same
  81        behavior, but it does not.  A ThreadPoolExecutor only creates
  82        a new thread when a new task is submitted and the number of
  83        existing threads is < corePoolSize.  This can never happen when
  84        corePoolSize=0, so new threads are never created.  Surprising,
  85        but there you are.
  86     */
  87     private static class LinearExecutor implements Executor {
  88         public synchronized void execute(Runnable command) {
  89             if (this.command != null)
  90                 throw new IllegalArgumentException("More than one command");
  91             this.command = command;
  92             if (thread == null) {
  93                 thread = new Thread() {
  94 
  95                     @Override
  96                     public void run() {
  97                         while (true) {
  98                             Runnable r;
  99                             synchronized (LinearExecutor.this) {
 100                                 if (LinearExecutor.this.command == null) {
 101                                     thread = null;
 102                                     return;
 103                                 } else {
 104                                     r = LinearExecutor.this.command;
 105                                     LinearExecutor.this.command = null;
 106                                 }
 107                             }
 108                             r.run();
 109                         }
 110                     }
 111                 };
 112                 thread.setDaemon(true);
 113                 thread.setName("ClientNotifForwarder-" + ++threadId);
 114                 thread.start();
 115             }
 116         }

 118         private Runnable command;
 119         private Thread thread;
 120     }

 305      * This method is intended to be called only by a client connector:
 306      * RMIConnector and ClientIntermediary.

 308     public synchronized void postReconnection(ClientListenerInfo[] listenerInfos)
 309         throws IOException {
 310 
 311         if (state == TERMINATED) {
 312             return;
 313         }
 314 
 315         while (state == STOPPING) {
 316             try {
 317                 wait();
 318             } catch (InterruptedException ire) {
 319                 IOException ioe = new IOException(ire.toString());
 320                 EnvHelp.initCause(ioe, ire);
 321                 throw ioe;
 322             }
 323         }
 324 
 325         final boolean trace = logger.traceOn();
 326         final int len   = listenerInfos.length;
 327 
 328         for (int i=0; i<len; i++) {
 329             if (trace) {
 330                 logger.trace("addNotificationListeners",
 331                              "Add a listener at "+
 332                              listenerInfos[i].getListenerID());
 333             }
 334 
 335             infoList.put(listenerInfos[i].getListenerID(), listenerInfos[i]);
 336         }
 337 
 338         beingReconnected = false;
 339         notifyAll();
 340 
 341         if (currentFetchThread == Thread.currentThread() ||
 342               state == STARTING || state == STARTED) { // doing or waiting reconnection
 343               // only update mbeanRemovedNotifID
 344             try {
 345                 mbeanRemovedNotifID = addListenerForMBeanRemovedNotif();
 346             } catch (Exception e) {
 347                 final String msg =
 348                     "Failed to register a listener to the mbean " +
 349                     "server: the client will not do clean when an MBean " +
 350                     "is unregistered";
 351                 if (logger.traceOn()) {
 352                     logger.trace("init", msg, e);
 353                 }
 354             }
 355         } else {
 356               while (state == STOPPING) {
 357                   try {
 358                       wait();
 359                   } catch (InterruptedException ire) {
 360                       IOException ioe = new IOException(ire.toString());
 361                       EnvHelp.initCause(ioe, ire);
 362                       throw ioe;
 363                   }
 364               }
 365 
 366               if (listenerInfos.length > 0) { // old listeners are re-added
 367                   init(true); // not update clientSequenceNumber
 368               } else if (infoList.size() > 0) { // only new listeners added during reconnection
 369                   init(false); // need update clientSequenceNumber
 370               }
 371           }
 372     }

 408         // Set new context class loader, returns previous one.
 409         private final ClassLoader setContextClassLoader(final ClassLoader loader) {
 410             final AccessControlContext ctxt = ClientNotifForwarder.this.acc;
 411             // if ctxt is null, log a config message and throw a
 412             // SecurityException.
 413             if (ctxt == null) {
 414                 logOnce("AccessControlContext must not be null.",null);
 415                 throw new SecurityException("AccessControlContext must not be null");
 416             }
 417             return AccessController.doPrivileged(
 418                 new PrivilegedAction<ClassLoader>() {
 419                     public ClassLoader run() {
 420                         try {
 421                             // get context class loader - may throw
 422                             // SecurityException - though unlikely.
 423                             final ClassLoader previous =
 424                                 Thread.currentThread().getContextClassLoader();
 425 
 426                             // if nothing needs to be done, break here...
 427                             if (loader == previous) return previous;
 428 
 429                             // reset context class loader - may throw
 430                             // SecurityException
 431                             Thread.currentThread().setContextClassLoader(loader);
 432                             return previous;
 433                         } catch (SecurityException x) {
 434                             logOnce("Permission to set ContextClassLoader missing. " +
 435                                     "Notifications will not be dispatched. " +
 436                                     "Please check your Java policy configuration: " +
 437                                     x, x);
 438                             throw x;
 439                         }
 440                     }
 441                 }, ctxt);
 442         }

 444         public void run() {
 445             final ClassLoader previous;
 446             if (defaultClassLoader != null) {
 447                 previous = setContextClassLoader(defaultClassLoader);
 448             } else {
 449                 previous = null;
 450             }
 451             try {
 452                 doRun();
 453             } finally {
 454                 if (defaultClassLoader != null) {
 455                     setContextClassLoader(previous);
 456                 }
 457             }
 458         }

 460         private void doRun() {
 461             synchronized (ClientNotifForwarder.this) {
 462                 currentFetchThread = Thread.currentThread();
 463 
 464                 if (state == STARTING) {
 465                     setState(STARTED);
 466                 }
 467             }
 468 
 469 
 470             NotificationResult nr = null;
 471             if (!shouldStop() && (nr = fetchNotifs()) != null) {
 472                 // nr == null means got exception
 473 
 474                 final TargetedNotification[] notifs =
 475                     nr.getTargetedNotifications();
 476                 final int len = notifs.length;
 477                 final Map<Integer, ClientListenerInfo> listeners;
 478                 final Integer myListenerID;
 479 
 480                 long missed = 0;
 481 
 482                 synchronized(ClientNotifForwarder.this) {
 483                     // check sequence number.
 484                     //
 485                     if (clientSequenceNumber >= 0) {
 486                         missed = nr.getEarliestSequenceNumber() -
 487                             clientSequenceNumber;
 488                     }
 489 
 490                     clientSequenceNumber = nr.getNextSequenceNumber();
 491 
 492                     listeners = new HashMap<Integer, ClientListenerInfo>();
 493 
 494                     for (int i = 0 ; i < len ; i++) {
 495                         final TargetedNotification tn = notifs[i];
 496                         final Integer listenerID = tn.getListenerID();
 497 
 498                         // check if an mbean unregistration notif
 499                         if (!listenerID.equals(mbeanRemovedNotifID)) {
 500                             final ClientListenerInfo li = infoList.get(listenerID);
 501                             if (li != null) {
 502                                 listeners.put(listenerID, li);
 503                             }
 504                             continue;
 505                         }
 506                         final Notification notif = tn.getNotification();
 507                         final String unreg =
 508                             MBeanServerNotification.UNREGISTRATION_NOTIFICATION;
 509                         if (notif instanceof MBeanServerNotification &&
 510                             notif.getType().equals(unreg)) {
 511 
 512                             MBeanServerNotification mbsn =
 513                                 (MBeanServerNotification) notif;
 514                             ObjectName name = mbsn.getMBeanName();
 515 
 516                             removeNotificationListener(name);
 517                         }
 518                     }
 519                     myListenerID = mbeanRemovedNotifID;
 520                 }
 521 
 522                 if (missed > 0) {
 523                     final String msg =
 524                         "May have lost up to " + missed +
 525                         " notification" + (missed == 1 ? "" : "s");
 526                     lostNotifs(msg, missed);
 527                     logger.trace("NotifFetcher.run", msg);
 528                 }
 529 
 530                 // forward
 531                 for (int i = 0 ; i < len ; i++) {
 532                     final TargetedNotification tn = notifs[i];
 533                     dispatchNotification(tn,myListenerID,listeners);
 534                 }
 535             }
 536 
 537             synchronized (ClientNotifForwarder.this) {
 538                 currentFetchThread = null;
 539             }
 540 
 541             if (nr == null) {
 542                 if (logger.traceOn()) {
 543                     logger.trace("NotifFetcher-run",
 544                             "Recieved null object as notifs, stops fetching because the "
 545                                     + "notification server is terminated.");
 546                 }
 547             }
 548             if (nr == null || shouldStop()) {
 549                 // tell that the thread is REALLY stopped
 550                 setState(STOPPED);
 551 
 552                 try {
 553                       removeListenerForMBeanRemovedNotif(mbeanRemovedNotifID);
 554                 } catch (Exception e) {
 555                     if (logger.traceOn()) {
 556                         logger.trace("NotifFetcher-run",
 557                                 "removeListenerForMBeanRemovedNotif", e);
 558                     }
 559                 }
 560             } else {
 561                 executor.execute(this);
 562             }
 563         }

 565         void dispatchNotification(TargetedNotification tn,
 566                                   Integer myListenerID,
 567                                   Map<Integer, ClientListenerInfo> listeners) {
 568             final Notification notif = tn.getNotification();
 569             final Integer listenerID = tn.getListenerID();
 570 
 571             if (listenerID.equals(myListenerID)) return;
 572             final ClientListenerInfo li = listeners.get(listenerID);
 573 
 574             if (li == null) {
 575                 logger.trace("NotifFetcher.dispatch",
 576                              "Listener ID not in map");
 577                 return;
 578             }
 579 
 580             NotificationListener l = li.getListener();
 581             Object h = li.getHandback();
 582             try {
 583                 l.handleNotification(notif, h);
 584             } catch (RuntimeException e) {
 585                 final String msg =
 586                     "Failed to forward a notification " +
 587                     "to a listener";
 588                 logger.trace("NotifFetcher-run", msg, e);
 589             }

 593         private NotificationResult fetchNotifs() {
 594             try {
 595                 NotificationResult nr = ClientNotifForwarder.this.
 596                     fetchNotifs(clientSequenceNumber,maxNotifications,
 597                                 timeout);
 598 
 599                 if (logger.traceOn()) {
 600                     logger.trace("NotifFetcher-run",
 601                                  "Got notifications from the server: "+nr);
 602                 }
 603 
 604                 return nr;
 605             } catch (ClassNotFoundException | NotSerializableException | UnmarshalException e) {
 606                 logger.trace("NotifFetcher.fetchNotifs", e);
 607                 return fetchOneNotif();
 608             } catch (IOException ioe) {
 609                 if (!shouldStop()) {
 610                     logger.error("NotifFetcher-run",
 611                                  "Failed to fetch notification, " +
 612                                  "stopping thread. Error is: " + ioe, ioe);
 613                     logger.debug("NotifFetcher-run",ioe);
 614                 }
 615 
 616                 // no more fetching
 617                 return null;
 618             }
 619         }

 621         /* Fetch one notification when we suspect that it might be a
 622            notification that we can't deserialize (because of a
 623            missing class).  First we ask for 0 notifications with 0
 624            timeout.  This allows us to skip sequence numbers for
 625            notifications that don't match our filters.  Then we ask
 626            for one notification.  If that produces a
 627            ClassNotFoundException, NotSerializableException or
 628            UnmarshalException, we increase our sequence number and ask again.
 629            Eventually we will either get a successful notification, or a
 630            return with 0 notifications.  In either case we can return a
 631            NotificationResult.  This algorithm works (albeit less
 632            well) even if the server implementation doesn't optimize a
 633            request for 0 notifications to skip sequence numbers for
 634            notifications that don't match our filters.
 635 
 636            If we had at least one
 637            ClassNotFoundException/NotSerializableException/UnmarshalException,
 638            then we must emit a JMXConnectionNotification.LOST_NOTIFS.
 639         */
 640         private NotificationResult fetchOneNotif() {
 641             ClientNotifForwarder cnf = ClientNotifForwarder.this;
 642 
 643             long startSequenceNumber = clientSequenceNumber;
 644 
 645             int notFoundCount = 0;
 646 
 647             NotificationResult result = null;
 648             long firstEarliest = -1;
 649 
 650             while (result == null && !shouldStop()) {
 651                 NotificationResult nr;
 652 
 653                 try {
 654                     // 0 notifs to update startSequenceNumber
 655                     nr = cnf.fetchNotifs(startSequenceNumber, 0, 0L);
 656                 } catch (ClassNotFoundException e) {
 657                     logger.warning("NotifFetcher.fetchOneNotif",
 658                                    "Impossible exception: " + e);
 659                     logger.debug("NotifFetcher.fetchOneNotif",e);
 660                     return null;
 661                 } catch (IOException e) {
 662                     if (!shouldStop())
 663                         logger.trace("NotifFetcher.fetchOneNotif", e);
 664                     return null;
 665                 }
 666 
 667                 if (shouldStop() || nr == null)
 668                     return null;
 669 
 670                 startSequenceNumber = nr.getNextSequenceNumber();
 671                 if (firstEarliest < 0)
 672                     firstEarliest = nr.getEarliestSequenceNumber();
 673 
 674                 try {
 675                     // 1 notif to skip possible missing class
 676                     result = cnf.fetchNotifs(startSequenceNumber, 1, 0L);
 677                 } catch (ClassNotFoundException | NotSerializableException | UnmarshalException e) {
 678                     logger.warning("NotifFetcher.fetchOneNotif",
 679                                    "Failed to deserialize a notification: "+e.toString());
 680                     if (logger.traceOn()) {
 681                         logger.trace("NotifFetcher.fetchOneNotif",
 682                                      "Failed to deserialize a notification.", e);
 683                     }
 684 
 685                     notFoundCount++;
 686                     startSequenceNumber++;
 687                 } catch (Exception e) {
 688                     if (!shouldStop())
 689                         logger.trace("NotifFetcher.fetchOneNotif", e);
 690                     return null;
 691                 }
 692             }
 693 
 694             if (notFoundCount > 0) {
 695                 final String msg =
 696                     "Dropped " + notFoundCount + " notification" +
 697                     (notFoundCount == 1 ? "" : "s") +
 698                     " because classes were missing locally or incompatible";
 699                 lostNotifs(msg, notFoundCount);
 700                 // Even if result.getEarliestSequenceNumber() is now greater than
 701                 // it was initially, meaning some notifs have been dropped
 702                 // from the buffer, we don't want the caller to see that
 703                 // because it is then likely to renotify about the lost notifs.
 704                 // So we put back the first value of earliestSequenceNumber
 705                 // that we saw.
 706                 if (result != null) {
 707                     result = new NotificationResult(
 708                             firstEarliest, result.getNextSequenceNumber(),
 709                             result.getTargetedNotifications());
 710                 }
 711             }
 712 
 713             return result;
 714         }

 745     /*
 746      * Called to decide whether need to start a thread for fetching notifs.
 747      * 
The parameter reconnected will decide whether to initilize the clientSequenceNumber,

 748      * initilaizing the clientSequenceNumber means to ignore all notifications arrived before.
 749      * If it is reconnected, we will not initialize in order to get all notifications arrived
 750      * during the reconnection. It may cause the newly registered listeners to receive some
 751      * notifications arrived before its registray.
 752      */
 753     private synchronized void init(boolean reconnected) throws IOException {
 754         switch (state) {
 755         case STARTED:
 756             return;
 757         case STARTING:
 758             return;
 759         case TERMINATED:
 760             throw new IOException("The ClientNotifForwarder has been terminated.");
 761         case STOPPING:
 762             if (beingReconnected == true) {
 763                 // wait for another thread to do, which is doing reconnection
 764                 return;
 765             }
 766 
 767             while (state == STOPPING) { // make sure only one fetching thread.
 768                 try {
 769                     wait();
 770                 } catch (InterruptedException ire) {
 771                     IOException ioe = new IOException(ire.toString());
 772                     EnvHelp.initCause(ioe, ire);
 773 
 774                     throw ioe;
 775                 }
 776             }
 777 
 778             // re-call this method to check the state again,
 779             // the state can be other value like TERMINATED.
 780             init(reconnected);
 781 
 782             return;
 783         case STOPPED:
 784             if (beingReconnected == true) {
 785                 // wait for another thread to do, which is doing reconnection
 786                 return;
 787             }
 788 
 789             if (logger.traceOn()) {
 790                 logger.trace("init", "Initializing...");
 791             }
 792 
 793             // init the clientSequenceNumber if not reconnected.
 794             if (!reconnected) {
 795                 try {
 796                     NotificationResult nr = fetchNotifs(-1, 0, 0);
 797 
 798                     if (state != STOPPED) { // JDK-8038940
 799                                             // reconnection must happen during
 800                                             // fetchNotifs(-1, 0, 0), and a new
 801                                             // thread takes over the fetching job
 802                         return;
 803                     }
 804 
 805                     clientSequenceNumber = nr.getNextSequenceNumber();
 806                 } catch (ClassNotFoundException e) {
 807                     // can't happen
 808                     logger.warning("init", "Impossible exception: "+ e);
 809                     logger.debug("init",e);
 810                 }
 811             }
 812 
 813             // for cleaning
 814             try {
 815                 mbeanRemovedNotifID = addListenerForMBeanRemovedNotif();
 816             } catch (Exception e) {
 817                 final String msg =
 818                     "Failed to register a listener to the mbean " +
 819                     "server: the client will not do clean when an MBean " +
 820                     "is unregistered";
 821                 if (logger.traceOn()) {
 822                     logger.trace("init", msg, e);
 823                 }
 824             }
 825 
 826             setState(STARTING);
 827 
 828             // start fetching
 829             executor.execute(new NotifFetcher());
 830 
 831             return;
 832         default:
 833             // should not
 834             throw new IOException("Unknown state.");
 835         }
 836     }

 838     /**
 839      * Import: should not remove a listener during reconnection, the reconnection
 840      * needs to change the listener list and that will possibly make removal fail.
 841      */
 842     private synchronized void beforeRemove() throws IOException {
 843         while (beingReconnected) {
 844             if (state == TERMINATED) {
 845                 throw new IOException("Terminated.");
 846             }
 847 
 848             try {
 849                 wait();
 850             } catch (InterruptedException ire) {
 851                 IOException ioe = new IOException(ire.toString());
 852                 EnvHelp.initCause(ioe, ire);
 853 
 854                 throw ioe;
 855             }
 856         }
 857 
 858         if (state == TERMINATED) {
 859             throw new IOException("Terminated.");
 860         }
 861     }
