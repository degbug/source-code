 104     /**
 105      * [schema default]

 107      * @return The canonical lexical representation of the declaration's {value constraint} value.

 112     }

 115      * [schema normalized value]

 117      *
 118      * @see

 120      */
 121     public String getSchemaNormalizedValue() {
 122         return fNormalizedValue;
 123     }
 124 
 125     /**
