 221      *
 222      * @param algorithmURI
 223      * @return the JCE standard name corresponding to the given URI
 224      */
 225     public static String translateURItoJCEID(String algorithmURI) {
 226         if (log.isLoggable(java.util.logging.Level.FINE)) {
 227             log.log(java.util.logging.Level.FINE, "Request for URI " + algorithmURI);
 228         }
 229 
 230         Algorithm algorithm = algorithmsMap.get(algorithmURI);
 231         if (algorithm != null) {
 232             return algorithm.jceName;
 233         }
 234         return null;
 235     }

 307     /**
 308      * Represents the Algorithm xml element
 309      */
 310     public static class Algorithm {
 311 
 312         final String requiredKey;
 313         final String jceName;
 314         final String algorithmClass;
 315         final int keyLength;
 316 
 317         /**
 318          * Gets data from element
 319          * @param el
 320          */
 321         public Algorithm(Element el) {
 322             requiredKey = el.getAttribute("RequiredKey");
 323             jceName = el.getAttribute("JCEName");
 324             algorithmClass = el.getAttribute("AlgorithmClass");
 325             if (el.hasAttribute("KeyLength")) {
 326                 keyLength = Integer.parseInt(el.getAttribute("KeyLength"));
 327             } else {
 328                 keyLength = 0;
 329             }
 330         }
 331 
 332         public Algorithm(String requiredKey, String jceName) {
 333             this(requiredKey, jceName, null, 0);
 334         }
 335 
 336         public Algorithm(String requiredKey, String jceName, String algorithmClass) {
 337             this(requiredKey, jceName, algorithmClass, 0);
 338         }
 339 
 340         public Algorithm(String requiredKey, String jceName, int keyLength) {
 341             this(requiredKey, jceName, null, keyLength);
 342         }
 343 
 344         public Algorithm(String requiredKey, String jceName, String algorithmClass, int keyLength) {
 345             this.requiredKey = requiredKey;
 346             this.jceName = jceName;
 347             this.algorithmClass = algorithmClass;
 348             this.keyLength = keyLength;
 349         }
 350     }
