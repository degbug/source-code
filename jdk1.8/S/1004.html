  28 /**
  29  * This class provides support for general purpose decompression using the
  30  * popular ZLIB compression library. The ZLIB compression library was
  31  * initially developed as part of the PNG graphics standard and is not
  32  * protected by patents. It is fully described in the specifications at
  33  * the <a href="package-summary.html#package_description">java.util.zip
  34  * package description</a>.
  35  *
  36  * <p>The following code fragment demonstrates a trivial compression
  37  * and decompression of a string using <tt>Deflater</tt> and
  38  * <tt>Inflater</tt>.
  39  *
  40  * <blockquote><pre>
  41  * try {
  42  *     // Encode a String into bytes
  43  *     String inputString = "blahblahblah\u20AC\u20AC";
  44  *     byte[] input = inputString.getBytes("UTF-8");
  45  *
  46  *     // Compress the bytes
  47  *     byte[] output = new byte[100];
  48  *     Deflater compresser = new Deflater();
  49  *     compresser.setInput(input);
  50  *     compresser.finish();
  51  *     int compressedDataLength = compresser.deflate(output);
  52  *
  53  *     // Decompress the bytes
  54  *     Inflater decompresser = new Inflater();
  55  *     decompresser.setInput(output, 0, compressedDataLength);
  56  *     byte[] result = new byte[100];
  57  *     int resultLength = decompresser.inflate(result);
  58  *     decompresser.end();
  59  *
  60  *     // Decode the bytes into a String
  61  *     String outputString = new String(result, 0, resultLength, "UTF-8");
  62  * } catch(java.io.UnsupportedEncodingException ex) {
  63  *     // handle
  64  * } catch (java.util.zip.DataFormatException ex) {
  65  *     // handle
  66  * }
  67  * </pre></blockquote>
  68  *
  69  * @see         Deflater
  70  * @author      David Connelly
  71  *
  72  */
  73 public

  91     /**
  92      * Creates a new decompressor. If the parameter 'nowrap' is true then
  93      * the ZLIB header and checksum fields will not be used. This provides
  94      * compatibility with the compression format used by both GZIP and PKZIP.
  95      * <p>
  96      * Note: When using the 'nowrap' option it is also necessary to provide
  97      * an extra "dummy" byte as input. This is required by the ZLIB native
  98      * library in order to support certain optimizations.
  99      *
 100      * @param nowrap if true then support GZIP compatible compression
 101      */
 102     public Inflater(boolean nowrap) {
 103         zsRef = new ZStreamRef(init(nowrap));
 104     }

 113     /**
 114      * Sets input data for decompression. Should be called whenever
 115      * needsInput() returns true indicating that more input data is
 116      * required.
 117      * @param b the input data bytes
 118      * @param off the start offset of the input data
 119      * @param len the length of the input data
 120      * @see Inflater#needsInput
 121      */
 122     public void setInput(byte[] b, int off, int len) {
 123         if (b == null) {
 124             throw new NullPointerException();
 125         }
 126         if (off < 0 || len < 0 || off > b.length - len) {
 127             throw new ArrayIndexOutOfBoundsException();
 128         }
 129         synchronized (zsRef) {
 130             this.buf = b;
 131             this.off = off;
 132             this.len = len;
 133         }
 134     }

 232     /**
 233      * Uncompresses bytes into specified buffer. Returns actual number
 234      * of bytes uncompressed. A return value of 0 indicates that
 235      * needsInput() or needsDictionary() should be called in order to
 236      * determine if more input data or a preset dictionary is required.
 237      * In the latter case, getAdler() can be used to get the Adler-32
 238      * value of the dictionary required.
 239      * @param b the buffer for the uncompressed data
 240      * @param off the start offset of the data
 241      * @param len the maximum number of uncompressed bytes
 242      * @return the actual number of uncompressed bytes
 243      * @exception DataFormatException if the compressed data format is invalid
 244      * @see Inflater#needsInput
 245      * @see Inflater#needsDictionary
 246      */
 247     public int inflate(byte[] b, int off, int len)
 248         throws DataFormatException
 249     {
 250         if (b == null) {
 251             throw new NullPointerException();
 252         }
 253         if (off < 0 || len < 0 || off > b.length - len) {
 254             throw new ArrayIndexOutOfBoundsException();
 255         }
 256         synchronized (zsRef) {
 257             ensureOpen();
 258             int thisLen = this.len;
 259             int n = inflateBytes(zsRef.address(), b, off, len);
 260             bytesWritten += n;
 261             bytesRead += (thisLen - this.len);
 262             return n;
 263         }
 264     }

 346     /**
 347      * Resets inflater so that a new set of input data can be processed.
 348      */
 349     public void reset() {
 350         synchronized (zsRef) {
 351             ensureOpen();
 352             reset(zsRef.address());
 353             buf = defaultBuf;
 354             finished = false;
 355             needDict = false;
 356             off = len = 0;
 357             bytesRead = bytesWritten = 0;
 358         }
 359     }

 361     /**
 362      * Closes the decompressor and discards any unprocessed input.
 363      * This method should be called when the decompressor is no longer
 364      * being used, but will also be called automatically by the finalize()
 365      * method. Once this method is called, the behavior of the Inflater
 366      * object is undefined.
 367      */
 368     public void end() {
 369         synchronized (zsRef) {
 370             long addr = zsRef.address();
 371             zsRef.clear();
 372             if (addr != 0) {
 373                 end(addr);
 374                 buf = null;
 375             }
 376         }
 377     }

 379     /**
 380      * Closes the decompressor when garbage is collected.
 381      */
 382     protected void finalize() {
 383         end();
 384     }

 386     private void ensureOpen () {
 387         assert Thread.holdsLock(zsRef);
 388         if (zsRef.address() == 0)
 389             throw new NullPointerException("Inflater has been closed");
 390     }
 391 
 392     boolean ended() {
 393         synchronized (zsRef) {
 394             return zsRef.address() == 0;
 395         }
 396     }
