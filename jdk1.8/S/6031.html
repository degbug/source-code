  30 /**
  31  * This iterator is a wrapper that always returns the position of
  32  * a node in document order. It is needed for the case where
  33  * a call to position() occurs in the context of an XSLT element
  34  * such as xsl:for-each, xsl:apply-templates, etc.
  35  *
  36  * The getPosition() methods in DTMAxisIterators defined
  37  * in DTMDefaultBaseIterators always return the position
  38  * in document order, which is backwards for XPath in the
  39  * case of the ancestor, ancestor-or-self, previous and
  40  * previous-sibling.
  41  *
  42  * XSLTC implements position() with the
  43  * BasisLibrary.positionF() method, and uses the
  44  * DTMAxisIterator.isReverse() method to determine
  45  * whether the result of getPosition() should be
  46  * interpreted as being equal to position().
  47  * But when the expression appears in apply-templates of
  48  * for-each, the position() function operates in document
  49  * order.
  50  *
  51  * The only effect of the ForwardPositionIterator is to force
  52  * the result of isReverse() to false, so that
  53  * BasisLibrary.positionF() calculates position() in a way
  54  * that's consistent with the context in which the
  55  * iterator is being used."
  56  *
  57  * (Apparently the correction of isReverse() occurs
  58  * implicitly, by inheritance. This class also appears
  59  * to maintain its own position counter, which seems
  60  * redundant.)
  61  *
  62  * @deprecated This class exists only for backwards compatibility with old
  63  *             translets.  New code should not reference it.
  64  */
  65 public final class ForwardPositionIterator extends DTMAxisIteratorBase {
  66 
  67     private DTMAxisIterator _source;
  68 
  69     public ForwardPositionIterator(DTMAxisIterator source) {
  70         _source = source;
  71     }
  72 
  73     public DTMAxisIterator cloneIterator() {
  74         try {
  75             final ForwardPositionIterator clone =
  76                 (ForwardPositionIterator) super.clone();
  77             clone._source = _source.cloneIterator();
  78             clone._isRestartable = false;
  79             return clone.reset();
  80         }
  81         catch (CloneNotSupportedException e) {
  82             BasisLibrary.runTimeError(BasisLibrary.ITERATOR_CLONE_ERR,
  83                                       e.toString());
  84             return null;
  85         }
  86     }
  87 
  88     public int next() {
  89         return returnNode(_source.next());
  90     }
  91 
  92     public DTMAxisIterator setStartNode(int node) {
  93         _source.setStartNode(node);
  94         return this;
  95     }
  96 
  97     public DTMAxisIterator reset() {
  98         _source.reset();
  99         return resetPosition();
 100     }
 101 
 102     public void setMark() {
 103         _source.setMark();
 104     }
 105 
 106     public void gotoMark() {
 107         _source.gotoMark();
 108     }
 109 }
