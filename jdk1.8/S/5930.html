 106   /**
 107    * Implements the guts of the <code>resolveEntity</code> method
 108    * for the SAX interface.
 109    *
 110    * <p>Presented with an optional public identifier and a system
 111    * identifier, this function attempts to locate a mapping in the
 112    * catalogs.</p>
 113    *
 114    * <p>If such a mapping is found, it is returned.  If no mapping is
 115    * found, null is returned.</p>
 116    *
 117    * @param publicId  The public identifier for the entity in question.
 118    * This may be null.
 119    *
 120    * @param systemId  The system identifier for the entity in question.
 121    * XML requires a system identifier on all external entities, so this
 122    * value is always specified.
 123    *
 124    * @return The resolved identifier (a URI reference).
 125    */
 126   public String getResolvedEntity (String publicId, String systemId) {
 127     String resolved = null;
 128 
 129     if (catalog == null) {
 130       catalogManager.debug.message(1, "Catalog resolution attempted with null catalog; ignored");
 131       return null;
 132     }
 133 
 134     if (systemId != null) {
 135       try {
 136         resolved = catalog.resolveSystem(systemId);
 137       } catch (MalformedURLException me) {
 138         catalogManager.debug.message(1, "Malformed URL exception trying to resolve",
 139                       publicId);
 140         resolved = null;
 141       } catch (IOException ie) {
 142         catalogManager.debug.message(1, "I/O exception trying to resolve", publicId);
 143         resolved = null;
 144       }
 145     }
 146 
 147     if (resolved == null) {
 148       if (publicId != null) {
 149         try {
 150           resolved = catalog.resolvePublic(publicId, systemId);
 151         } catch (MalformedURLException me) {
 152           catalogManager.debug.message(1, "Malformed URL exception trying to resolve",
 153                         publicId);
 154         } catch (IOException ie) {
 155           catalogManager.debug.message(1, "I/O exception trying to resolve", publicId);
 156         }
 157       }
 158 
 159       if (resolved != null) {
 160         catalogManager.debug.message(2, "Resolved public", publicId, resolved);
 161       }
 162     } else {
 163       catalogManager.debug.message(2, "Resolved system", systemId, resolved);
 164     }
 165 
 166     return resolved;
 167   }

 169   /**
 170    * Implements the <code>resolveEntity</code> method
 171    * for the SAX interface.
 172    *
 173    * <p>Presented with an optional public identifier and a system
 174    * identifier, this function attempts to locate a mapping in the
 175    * catalogs.</p>
 176    *
 177    * <p>If such a mapping is found, the resolver attempts to open
 178    * the mapped value as an InputSource and return it. Exceptions are
 179    * ignored and null is returned if the mapped value cannot be opened
 180    * as an input source.</p>
 181    *
 182    * <p>If no mapping is found (or an error occurs attempting to open
 183    * the mapped value as an input source), null is returned and the system
 184    * will use the specified system identifier as if no entityResolver
 185    * was specified.</p>
 186    *
 187    * @param publicId  The public identifier for the entity in question.
 188    * This may be null.
 189    *
 190    * @param systemId  The system identifier for the entity in question.
 191    * XML requires a system identifier on all external entities, so this
 192    * value is always specified.
 193    *
 194    * @return An InputSource for the mapped identifier, or null.
 195    */
 196   public InputSource resolveEntity (String publicId, String systemId) {
 197     String resolved = getResolvedEntity(publicId, systemId);
 198 
 199     if (resolved != null) {
 200       try {
 201         InputSource iSource = new InputSource(resolved);
 202         iSource.setPublicId(publicId);
 203 
 204         // Ideally this method would not attempt to open the
 205         // InputStream, but there is a bug (in Xerces, at least)
 206         // that causes the parser to mistakenly open the wrong
 207         // system identifier if the returned InputSource does
 208         // not have a byteStream.
 209         //
 210         // It could be argued that we still shouldn't do this here,
 211         // but since the purpose of calling the entityResolver is
 212         // almost certainly to open the input stream, it seems to
 213         // do little harm.
 214         //
 215         URL url = new URL(resolved);
 216         InputStream iStream = url.openStream();
 217         iSource.setByteStream(iStream);
 218 
 219         return iSource;
 220       } catch (Exception e) {
 221         catalogManager.debug.message(1, "Failed to create InputSource", resolved);
 222         return null;
 223       }
 224     }
 225 
 226     return null;
 227   }

 229   /** JAXP URIResolver API */
 230   public Source resolve(String href, String base)
 231     throws TransformerException {
 232 
 233     String uri = href;
 234     String fragment = null;
 235     int hashPos = href.indexOf("#");
 236     if (hashPos >= 0) {
 237       uri = href.substring(0, hashPos);
 238       fragment = href.substring(hashPos+1);
 239     }
 240 
 241     String result = null;
 242 
 243     try {
 244       result = catalog.resolveURI(href);
 245     } catch (Exception e) {
 246       // nop;
 247     }
 248 
 249     if (result == null) {
 250       try {
 251         URL url = null;
 252 
 253         if (base==null) {
 254           url = new URL(uri);
 255           result = url.toString();
 256         } else {
 257           URL baseURL = new URL(base);
 258           url = (href.length()==0 ? baseURL : new URL(baseURL, uri));
 259           result = url.toString();
 260         }
 261       } catch (java.net.MalformedURLException mue) {
 262         // try to make an absolute URI from the current base
 263         String absBase = makeAbsolute(base);
 264         if (!absBase.equals(base)) {
 265           // don't bother if the absBase isn't different!
 266           return resolve(href, absBase);
 267         } else {
 268           throw new TransformerException("Malformed URL "
 269                                          + href + "(base " + base + ")",
 270                                          mue);
 271         }
 272       }
 273     }
 274 
 275     catalogManager.debug.message(2, "Resolved URI", href, result);
 276 
 277     SAXSource source = new SAXSource();
 278     source.setInputSource(new InputSource(result));
 279     setEntityResolver(source);
 280     return source;
 281   }

 283   /**
 284    * <p>Establish an entityResolver for newly resolved URIs.</p>
 285    *
 286    * <p>This is called from the URIResolver to set an EntityResolver
 287    * on the SAX parser to be used for new XML documents that are
 288    * encountered as a result of the document() function, xsl:import,
 289    * or xsl:include.  This is done because the XSLT processor calls
 290    * out to the SAXParserFactory itself to create a new SAXParser to
 291    * parse the new document.  The new parser does not automatically
 292    * inherit the EntityResolver of the original (although arguably
 293    * it should).  See below:</p>
 294    *
 295    * <tt>"If an application wants to set the ErrorHandler or
 296    * EntityResolver for an XMLReader used during a transformation,
 297    * it should use a URIResolver to return the SAXSource which
 298    * provides (with getXMLReader) a reference to the XMLReader"</tt>
 299    *
 300    * <p>...quoted from page 118 of the Java API for XML
 301    * Processing 1.1 specification</p>
 302    *
 303    */
 304   private void setEntityResolver(SAXSource source) throws TransformerException {
 305     XMLReader reader = source.getXMLReader();
 306     if (reader == null) {
 307       SAXParserFactory spFactory = catalogManager.useServicesMechanism() ?
 308                     SAXParserFactory.newInstance() : new SAXParserFactoryImpl();
 309       spFactory.setNamespaceAware(true);
 310       try {
 311         reader = spFactory.newSAXParser().getXMLReader();
 312       }
 313       catch (ParserConfigurationException ex) {
 314         throw new TransformerException(ex);
 315       }
 316       catch (SAXException ex) {
 317         throw new TransformerException(ex);
 318       }
 319     }
 320     reader.setEntityResolver(this);
 321     source.setXMLReader(reader);
 322   }

 324   /** Attempt to construct an absolute URI */
 325   private String makeAbsolute(String uri) {
 326     if (uri == null) {
 327       uri = "";
 328     }
 329 
 330     try {
 331       URL url = new URL(uri);
 332       return url.toString();
 333     } catch (MalformedURLException mue) {
 334       try {
 335         URL fileURL = FileURL.makeURL(uri);
 336         return fileURL.toString();
 337       } catch (MalformedURLException mue2) {
 338         // bail
 339         return uri;
 340       }
 341     }
 342   }
