  24 /**
  25  * Interface for a field activator. The field activator is responsible
  26  * for activating fields within a specific scope; the caller merely
  27  * requests the fields to be activated.
  28  *
  29  * @xerces.internal
  30  *
  31  * @author Andy Clark, IBM
  32  *
  33  */
  34 public interface FieldActivator {
  35 
  36     //
  37     // FieldActivator methods
  38     //
  39 
  40     /**
  41      * Start the value scope for the specified identity constraint. This
  42      * method is called when the selector matches in order to initialize
  43      * the value store.
  44      *
  45      * @param identityConstraint The identity constraint.
  46      * @param initialDepth  the depth at which the selector began matching
  47      */
  48     public void startValueScopeFor(IdentityConstraint identityConstraint,
  49             int initialDepth);
  50 
  51     /**
  52      * Request to activate the specified field. This method returns the
  53      * matcher for the field.
  54      * It's also important for the implementor to ensure that it marks whether a Field
  55      * is permitted to match a value--that is, to call the setMayMatch(Field, Boolean) method.
  56      *
  57      * @param field The field to activate.
  58      * @param initialDepth the 0-indexed depth in the instance document at which the Selector began to match.
  59      */
  60     public XPathMatcher activateField(Field field, int initialDepth);
  61 
  62     /**
  63      * Sets whether the given field is permitted to match a value.
  64      * This should be used to catch instance documents that try
  65      * and match a field several times in the same scope.
  66      *
  67      * @param field The field that may be permitted to be matched.
  68      * @param state Boolean indiciating whether the field may be matched.
  69      */
  70     public void setMayMatch(Field field, Boolean state);
  71 
  72     /**
  73      * Returns whether the given field is permitted to match a value.
  74      *
  75      * @param field The field that may be permitted to be matched.
  76      * @return Boolean indicating whether the field may be matched.
  77      */
  78     public Boolean mayMatch(Field field);
  79 
  80     /**
  81      * Ends the value scope for the specified identity constraint.
  82      *
  83      * @param identityConstraint The identity constraint.
  84      * @param initialDepth  the 0-indexed depth where the Selector began to match.
  85      */
  86     public void endValueScopeFor(IdentityConstraint identityConstraint, int initialDepth);
  87 
  88 } // interface FieldActivator
