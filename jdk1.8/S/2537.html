 101     /**
 102      * Processes a method invocation made on the encapsulating
 103      * proxy instance, proxy, and returns the result.
 104      *
 105      * RemoteObjectInvocationHandler implements this method
 106      * as follows:
 107      *
 108      * If method is one of the following methods, it
 109      * is processed as described below:
 110      *
 111      * 
    
 112      *
 113      * 
  • {@link Object#hashCode Object.hashCode}: Returns the hash
    • 
 114      * code value for the proxy.
 115      *
 116      * 
  • {@link Object#equals Object.equals}: Returns true
    • 
 117      * if the argument (args[0]) is an instance of a dynamic
 118      * proxy class and this invocation handler is equal to the invocation
 119      * handler of that argument, and returns false otherwise.
 120      *
 121      * 
  • {@link Object#toString Object.toString}: Returns a string
    • 
 122      * representation of the proxy.
 123      * 
 124      *
 125      * Otherwise, a remote call is made as follows:
 126      *
 127      * 
      
 128      * 
    • If proxy is not an instance of the interface
      • 
 129      * {@link Remote}, then an {@link IllegalArgumentException} is thrown.
 130      *
 131      * 
    • Otherwise, the {@link RemoteRef#invoke invoke} method is invoked
      • 
 132      * on this invocation handler's RemoteRef, passing
 133      * proxy, method, args, and the
 134      * method hash (defined in section 8.3 of the "Java Remote Method
 135      * Invocation (RMI) Specification") for method, and the
 136      * result is returned.
 137      *
 138      * 
    • If an exception is thrown by RemoteRef.invoke and
      • 
 139      * that exception is a checked exception that is not assignable to any
 140      * exception in the throws clause of the method
 141      * implemented by the proxy's class, then that exception
 142      * is wrapped in an {@link UnexpectedException} and the wrapped
 143      * exception is thrown.  Otherwise, the exception thrown by
 144      * invoke is thrown by this method.
 145      * 
 146      *
 147      * The semantics of this method are unspecified if the
 148      * arguments could not have been produced by an instance of some
 149      * valid dynamic proxy class containing this invocation handler.
 150      *
 151      * @param proxy the proxy instance that the method was invoked on
 152      * @param method the Method instance corresponding to the
 153      * interface method invoked on the proxy instance
 154      * @param args an array of objects containing the values of the
 155      * arguments passed in the method invocation on the proxy instance, or
 156      * null if the method takes no arguments
 157      * @return the value to return from the method invocation on the proxy
 158      * instance
 159      * @throws  Throwable the exception to throw from the method invocation
 160      * on the proxy instance
 161      **/
 162     public Object invoke(Object proxy, Method method, Object[] args)
 163         throws Throwable
 164     {
 165         if (method.getDeclaringClass() == Object.class) {
 166             return invokeObjectMethod(proxy, method, args);
 167         } else if ("finalize".equals(method.getName()) && method.getParameterCount() == 0 &&
 168             !allowFinalizeInvocation) {
 169             return null; // ignore
 170         } else {
 171             return invokeRemoteMethod(proxy, method, args);
 172         }
 173     }

 175     /**
 176      * Handles java.lang.Object methods.
 177      **/
 178     private Object invokeObjectMethod(Object proxy,
 179                                       Method method,
 180                                       Object[] args)
 181     {
 182         String name = method.getName();
 183 
 184         if (name.equals("hashCode")) {
 185             return hashCode();
 186 
 187         } else if (name.equals("equals")) {
 188             Object obj = args[0];
 189             return
 190                 proxy == obj ||
 191                 (obj != null &&
 192                  Proxy.isProxyClass(obj.getClass()) &&
 193                  equals(Proxy.getInvocationHandler(obj)));
 194 
 195         } else if (name.equals("toString")) {
 196             return proxyToString(proxy);
 197 
 198         } else {
 199             throw new IllegalArgumentException(
 200                 "unexpected Object method: " + method);
 201         }
 202     }

 204     /**
 205      * Handles remote methods.
 206      **/
 207     private Object invokeRemoteMethod(Object proxy,
 208                                       Method method,
 209                                       Object[] args)
 210         throws Exception
 211     {
 212         try {
 213             if (!(proxy instanceof Remote)) {
 214                 throw new IllegalArgumentException(
 215                     "proxy not Remote instance");
 216             }
 217             return ref.invoke((Remote) proxy, method, args,
 218                               getMethodHash(method));
 219         } catch (Exception e) {
 220             if (!(e instanceof RuntimeException)) {
 221                 Class<?> cl = proxy.getClass();
 222                 try {
 223                     method = cl.getMethod(method.getName(),
 224                                           method.getParameterTypes());
 225                 } catch (NoSuchMethodException nsme) {
 226                     throw (IllegalArgumentException)
 227                         new IllegalArgumentException().initCause(nsme);
 228                 }
 229                 Class<?> thrownType = e.getClass();
 230                 for (Class<?> declaredType : method.getExceptionTypes()) {
 231                     if (declaredType.isAssignableFrom(thrownType)) {
 232                         throw e;
 233                     }
 234                 }
 235                 e = new UnexpectedException("unexpected exception", e);
 236             }
 237             throw e;
 238         }
 239     }

 241     /**
 242      * Returns a string representation for a proxy that uses this invocation
 243      * handler.
 244      **/
 245     private String proxyToString(Object proxy) {
 246         Class<?>[] interfaces = proxy.getClass().getInterfaces();
 247         if (interfaces.length == 0) {
 248             return "Proxy[" + this + "]";
 249         }
 250         String iface = interfaces[0].getName();
 251         if (iface.equals("java.rmi.Remote") && interfaces.length > 1) {
 252             iface = interfaces[1].getName();
 253         }
 254         int dot = iface.lastIndexOf('.');
 255         if (dot >= 0) {
 256             iface = iface.substring(dot + 1);
 257         }
 258         return "Proxy[" + iface + "," + this + "]";
 259     }

 269     /**
 270      * Returns the method hash for the specified method.  Subsequent calls
 271      * to "getMethodHash" passing the same method argument should be faster
 272      * since this method caches internally the result of the method to
 273      * method hash mapping.  The method hash is calculated using the
 274      * "computeMethodHash" method.
 275      *
 276      * @param method the remote method
 277      * @return the method hash for the specified method
 278      */
 279     private static long getMethodHash(Method method) {
 280         return methodToHash_Maps.get(method.getDeclaringClass()).get(method);
 281     }

 283     /**
 284      * A weak hash map, mapping classes to weak hash maps that map
 285      * method objects to method hashes.
 286      **/
 287     private static class MethodToHash_Maps
 288         extends WeakClassHashMap<Map<Method,Long>>
 289     {
 290         MethodToHash_Maps() {}
 291 
 292         protected Map<Method,Long> computeValue(Class<?> remoteClass) {
 293             return new WeakHashMap<Method,Long>() {
 294                 public synchronized Long get(Object key) {
 295                     Long hash = super.get(key);
 296                     if (hash == null) {
 297                         Method method = (Method) key;
 298                         hash = Util.computeMethodHash(method);
 299                         put(method, hash);
 300                     }
 301                     return hash;
 302                 }
 303             };
 304         }
