 812     /*
 813      * Check the version of the current JDK against that specified in the
 814      * parameter
 815      *
 816      * There is a proposal to change the java version string to:
 817      * MAJOR.MINOR.FU.CPU.PSU-BUILDNUMBER_BUGIDNUMBER_OPTIONAL
 818      * This method would work with both the current format and that proposed
 819      *
 820      * @param compareTo a JDK version to be compared to
 821      * @return true if the current version is the same or above that represented
 822      * by the parameter
 823      */
 824     public static boolean isJavaVersionAtLeast(int compareTo) {
 825         String javaVersion = SecuritySupport.getSystemProperty("java.version");
 826         String versions[] = javaVersion.split("\\.", 3);
 827         if (Integer.parseInt(versions[0]) >= compareTo ||
 828             Integer.parseInt(versions[1]) >= compareTo) {
 829             return true;
 830         }
 831         return false;
 832     }

 861         public ArrayEnumeration(Object[] array) {
 862             this.array = array;
 863         } // (Object[])
 864 
 865         //
 866         // Enumeration methods
 867         //
 868 
 869         /**
 870          * Tests if this enumeration contains more elements.
 871          *
 872          * @return  true if this enumeration contains more elements;
 873          *          false otherwise.
 874          * @since   JDK1.0
 875          */
 876         public boolean hasMoreElements() {
 877             return index < array.length;
 878         } // hasMoreElement():boolean
 879 
 880         /**
 881          * Returns the next element of this enumeration.
 882          *
 883          * @return     the next element of this enumeration.
 884          * @exception  NoSuchElementException  if no more elements exist.
 885          * @since      JDK1.0
 886          */
 887         public Object nextElement() {
 888             if (index < array.length) {
 889                 return array[index++];
 890             }
 891             throw new NoSuchElementException();
 892         } // nextElement():Object
