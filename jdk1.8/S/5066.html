  30 /**
  31  * Contains a list of LSInputs.
  32  *
  33  * @xerces.internal
  34  *
  35  * @author Michael Glavassevich, IBM
  36  *
  37  * @version $Id: LSInputListImpl.java,v 1.2 2010-10-26 23:01:04 joehw Exp $
  38  */
  39 public final class LSInputListImpl extends AbstractList implements LSInputList {
  40 
  41     /**
  42      * An immutable empty list.
  43      */
  44     public static final LSInputListImpl EMPTY_LIST = new LSInputListImpl(new LSInput[0], 0);
  45 
  46     // The array to hold all data
  47     private final LSInput[] fArray;
  48     // Number of elements in this list
  49     private final int fLength;
  50 
  51     /**
  52      * Construct an LSInputList implementation
  53      *
  54      * @param array     the data array
  55      * @param length    the number of elements
  56      */
  57     public LSInputListImpl(LSInput[] array, int length) {
  58         fArray = array;
  59         fLength = length;
  60     }
  61 
  62     /**
  63      * The number of LSInputs in the list. The range of valid
  64      * child object indices is 0 to length-1 inclusive.
  65      */
  66     public int getLength() {
  67         return fLength;
  68     }
  69 
  70     /**
  71      * Returns the indexth item in the collection or
  72      * null if index is greater than or equal to
  73      * the number of objects in the list. The index starts at 0.
  74      * @param index  index into the collection.
  75      * @return  The LSInput at the indexth
  76      *   position in the LSInputList, or null if
  77      *   the index specified is not valid.
  78      */
  79     public LSInput item(int index) {
  80         if (index < 0 || index >= fLength) {
  81             return null;
  82         }
  83         return fArray[index];
  84     }
  85 
  86     /*
  87      * List methods
  88      */
  89 
  90     public Object get(int index) {
  91         if (index >= 0 && index < fLength) {
  92             return fArray[index];
  93         }
  94         throw new IndexOutOfBoundsException("Index: " + index);
  95     }
  96 
  97     public int size() {
  98         return getLength();
  99     }
 100 
 101     public Object[] toArray() {
 102         Object[] a = new Object[fLength];
 103         toArray0(a);
 104         return a;
 105     }
 106 
 107     public Object[] toArray(Object[] a) {
 108         if (a.length < fLength) {
 109             Class arrayClass = a.getClass();
 110             Class componentType = arrayClass.getComponentType();
 111             a = (Object[]) Array.newInstance(componentType, fLength);
 112         }
 113         toArray0(a);
 114         if (a.length > fLength) {
 115             a[fLength] = null;
 116         }
 117         return a;
 118     }
 119 
 120     private void toArray0(Object[] a) {
 121         if (fLength > 0) {
 122             System.arraycopy(fArray, 0, a, 0, fLength);
 123         }
 124     }

 126 } // LSInputListImpl
