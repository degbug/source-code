 137     /**
 138      * Sets the selection model, which must be one of SINGLE_TREE_SELECTION,
 139      * CONTIGUOUS_TREE_SELECTION or DISCONTIGUOUS_TREE_SELECTION. If mode
 140      * is not one of the defined value,
 141      * DISCONTIGUOUS_TREE_SELECTION is assumed.
 142      * This may change the selection if the current selection is not valid
 143      * for the new mode. For example, if three TreePaths are
 144      * selected when the mode is changed to SINGLE_TREE_SELECTION,
 145      * only one TreePath will remain selected. It is up to the particular
 146      * implementation to decide what TreePath remains selected.
 147      * 
 148      * Setting the mode to something other than the defined types will
 149      * result in the mode becoming DISCONTIGUOUS_TREE_SELECTION.
 150      */
 151     public void setSelectionMode(int mode) {
 152         int            oldMode = selectionMode;
 153 
 154         selectionMode = mode;
 155         if(selectionMode != TreeSelectionModel.SINGLE_TREE_SELECTION &&
 156            selectionMode != TreeSelectionModel.CONTIGUOUS_TREE_SELECTION &&
 157            selectionMode != TreeSelectionModel.DISCONTIGUOUS_TREE_SELECTION)
 158             selectionMode = TreeSelectionModel.DISCONTIGUOUS_TREE_SELECTION;
 159         if(oldMode != selectionMode && changeSupport != null)
 160             changeSupport.firePropertyChange(SELECTION_MODE_PROPERTY,
 161                                              Integer.valueOf(oldMode),
 162                                              Integer.valueOf(selectionMode));
 163     }

 192     /**
 193      * Sets the selection. Whether the supplied paths are taken as the
 194      * new selection depends upon the selection mode. If the supplied
 195      * array is {@code null}, or empty, the selection is cleared. If
 196      * the selection mode is {@code SINGLE_TREE_SELECTION}, only the
 197      * first path in {@code pPaths} is used. If the selection
 198      * mode is {@code CONTIGUOUS_TREE_SELECTION} and the supplied paths
 199      * are not contiguous, then only the first path in {@code pPaths} is
 200      * used. If the selection mode is
 201      * {@code DISCONTIGUOUS_TREE_SELECTION}, then all paths are used.
 202      * 
 203      * All {@code null} paths in {@code pPaths} are ignored.
 204      * 
 205      * If this represents a change, all registered {@code
 206      * TreeSelectionListener}s are notified.
 207      * 
 208      * The lead path is set to the last unique path.
 209      * 
 210      * The paths returned from {@code getSelectionPaths} are in the same
 211      * order as those supplied to this method.
 212      *
 213      * @param pPaths the new selection
 214      */
 215     public void setSelectionPaths(TreePath[] pPaths) {
 216         int            newCount, newCounter, oldCount, oldCounter;
 217         TreePath[]     paths = pPaths;
 218 
 219         if(paths == null)
 220             newCount = 0;
 221         else
 222             newCount = paths.length;
 223         if(selection == null)
 224             oldCount = 0;
 225         else
 226             oldCount = selection.length;
 227         if((newCount + oldCount) != 0) {
 228             if(selectionMode == TreeSelectionModel.SINGLE_TREE_SELECTION) {
 229                 /* If single selection and more than one path, only allow
 230                    first. */
 231                 if(newCount > 1) {
 232                     paths = new TreePath[1];
 233                     paths[0] = pPaths[0];
 234                     newCount = 1;
 235                 }
 236             }
 237             else if(selectionMode ==
 238                     TreeSelectionModel.CONTIGUOUS_TREE_SELECTION) {
 239                 /* If contiguous selection and paths aren't contiguous,
 240                    only select the first path item. */
 241                 if(newCount > 0 && !arePathsContiguous(paths)) {
 242                     paths = new TreePath[1];
 243                     paths[0] = pPaths[0];
 244                     newCount = 1;
 245                 }
 246             }
 247 
 248             TreePath         beginLeadPath = leadPath;
 249             Vector<PathPlaceHolder> cPaths = new Vector<PathPlaceHolder>(newCount + oldCount);
 250             List<TreePath> newSelectionAsList =
 251                     new ArrayList<TreePath>(newCount);
 252 
 253             lastPaths.clear();
 254             leadPath = null;
 255             /* Find the paths that are new. */
 256             for(newCounter = 0; newCounter < newCount; newCounter++) {
 257                 TreePath path = paths[newCounter];
 258                 if (path != null && lastPaths.get(path) == null) {
 259                     lastPaths.put(path, Boolean.TRUE);
 260                     if (uniquePaths.get(path) == null) {
 261                         cPaths.addElement(new PathPlaceHolder(path, true));
 262                     }
 263                     leadPath = path;
 264                     newSelectionAsList.add(path);
 265                 }
 266             }
 267 
 268             TreePath[] newSelection = newSelectionAsList.toArray(
 269                     new TreePath[newSelectionAsList.size()]);
 270 
 271             /* Get the paths that were selected but no longer selected. */
 272             for(oldCounter = 0; oldCounter < oldCount; oldCounter++)
 273                 if(selection[oldCounter] != null &&
 274                     lastPaths.get(selection[oldCounter]) == null)
 275                     cPaths.addElement(new PathPlaceHolder
 276                                       (selection[oldCounter], false));
 277 
 278             selection = newSelection;
 279 
 280             Hashtable<TreePath, Boolean>  tempHT = uniquePaths;
 281 
 282             uniquePaths = lastPaths;
 283             lastPaths = tempHT;
 284             lastPaths.clear();
 285 
 286             // No reason to do this now, but will still call it.
 287             insureUniqueness();
 288 
 289             updateLeadIndex();
 290 
 291             resetRowSelection();
 292             /* Notify of the change. */
 293             if(cPaths.size() > 0)
 294                 notifyPathChange(cPaths, beginLeadPath);
 295         }
 296     }

 314     /**
 315       * Adds paths to the current selection. If any of the paths in
 316       * paths are not currently in the selection the TreeSelectionListeners
 317       * are notified. This has
 318       * no effect if paths is null.
 319       * The lead path is set to the last element in paths.
 320       * If the selection mode is CONTIGUOUS_TREE_SELECTION,
 321       * and adding the new paths would make the selection discontiguous.
 322       * Then two things can result: if the TreePaths in paths
 323       * are contiguous, then the selection becomes these TreePaths,
 324       * otherwise the TreePaths aren't contiguous and the selection becomes
 325       * the first TreePath in paths.
 326       *
 327       * @param paths the new path to add to the current selection
 328       */
 329     public void addSelectionPaths(TreePath[] paths) {
 330         int       newPathLength = ((paths == null) ? 0 : paths.length);
 331 
 332         if(newPathLength > 0) {
 333             if(selectionMode == TreeSelectionModel.SINGLE_TREE_SELECTION) {
 334                 setSelectionPaths(paths);
 335             }
 336             else if(selectionMode == TreeSelectionModel.
 337                     CONTIGUOUS_TREE_SELECTION && !canPathsBeAdded(paths)) {
 338                 if(arePathsContiguous(paths)) {
 339                     setSelectionPaths(paths);
 340                 }
 341                 else {
 342                     TreePath[]          newPaths = new TreePath[1];
 343 
 344                     newPaths[0] = paths[0];
 345                     setSelectionPaths(newPaths);
 346                 }
 347             }
 348             else {
 349                 int               counter, validCount;
 350                 int               oldCount;
 351                 TreePath          beginLeadPath = leadPath;
 352                 Vector<PathPlaceHolder>  cPaths = null;
 353 
 354                 if(selection == null)
 355                     oldCount = 0;
 356                 else
 357                     oldCount = selection.length;
 358                 /* Determine the paths that aren't currently in the
 359                    selection. */
 360                 lastPaths.clear();
 361                 for(counter = 0, validCount = 0; counter < newPathLength;
 362                     counter++) {
 363                     if(paths[counter] != null) {
 364                         if (uniquePaths.get(paths[counter]) == null) {
 365                             validCount++;
 366                             if(cPaths == null)
 367                                 cPaths = new Vector<PathPlaceHolder>();
 368                             cPaths.addElement(new PathPlaceHolder
 369                                               (paths[counter], true));
 370                             uniquePaths.put(paths[counter], Boolean.TRUE);
 371                             lastPaths.put(paths[counter], Boolean.TRUE);
 372                         }
 373                         leadPath = paths[counter];
 374                     }
 375                 }
 376 
 377                 if(leadPath == null) {
 378                     leadPath = beginLeadPath;
 379                 }
 380 
 381                 if(validCount > 0) {
 382                     TreePath         newSelection[] = new TreePath[oldCount +
 383                                                                   validCount];
 384 
 385                     /* And build the new selection. */
 386                     if(oldCount > 0)
 387                         System.arraycopy(selection, 0, newSelection, 0,
 388                                          oldCount);
 389                     if(validCount != paths.length) {
 390                         /* Some of the paths in paths are already in
 391                            the selection. */
 392                         Enumeration<TreePath> newPaths = lastPaths.keys();
 393 
 394                         counter = oldCount;
 395                         while (newPaths.hasMoreElements()) {
 396                             newSelection[counter++] = newPaths.nextElement();
 397                         }
 398                     }
 399                     else {
 400                         System.arraycopy(paths, 0, newSelection, oldCount,
 401                                          validCount);
 402                     }
 403 
 404                     selection = newSelection;
 405 
 406                     insureUniqueness();
 407 
 408                     updateLeadIndex();
 409 
 410                     resetRowSelection();
 411 
 412                     notifyPathChange(cPaths, beginLeadPath);
 413                 }
 414                 else
 415                     leadPath = beginLeadPath;
 416                 lastPaths.clear();
 417             }
 418         }
 419     }

 437     /**
 438       * Removes paths from the selection.  If any of the paths in paths
 439       * are in the selection the TreeSelectionListeners are notified.
 440       * This has no effect if paths is null.
 441       *
 442       * @param paths the paths to remove from the selection
 443       */
 444     public void removeSelectionPaths(TreePath[] paths) {
 445         if (paths != null && selection != null && paths.length > 0) {
 446             if(!canPathsBeRemoved(paths)) {
 447                 /* Could probably do something more interesting here! */
 448                 clearSelection();
 449             }
 450             else {
 451                 Vector<PathPlaceHolder> pathsToRemove = null;
 452 
 453                 /* Find the paths that can be removed. */
 454                 for (int removeCounter = paths.length - 1; removeCounter >= 0;
 455                      removeCounter--) {
 456                     if(paths[removeCounter] != null) {
 457                         if (uniquePaths.get(paths[removeCounter]) != null) {
 458                             if(pathsToRemove == null)
 459                                 pathsToRemove = new Vector<PathPlaceHolder>(paths.length);
 460                             uniquePaths.remove(paths[removeCounter]);
 461                             pathsToRemove.addElement(new PathPlaceHolder
 462                                          (paths[removeCounter], false));
 463                         }
 464                     }
 465                 }
 466                 if(pathsToRemove != null) {
 467                     int         removeCount = pathsToRemove.size();
 468                     TreePath    beginLeadPath = leadPath;
 469 
 470                     if(removeCount == selection.length) {
 471                         selection = null;
 472                     }
 473                     else {
 474                         Enumeration<TreePath> pEnum = uniquePaths.keys();
 475                         int                  validCount = 0;
 476 
 477                         selection = new TreePath[selection.length -
 478                                                 removeCount];
 479                         while (pEnum.hasMoreElements()) {
 480                             selection[validCount++] = pEnum.nextElement();
 481                         }
 482                     }
 483                     if (leadPath != null &&
 484                         uniquePaths.get(leadPath) == null) {
 485                         if (selection != null) {
 486                             leadPath = selection[selection.length - 1];
 487                         }
 488                         else {
 489                             leadPath = null;
 490                         }
 491                     }
 492                     else if (selection != null) {
 493                         leadPath = selection[selection.length - 1];
 494                     }
 495                     else {
 496                         leadPath = null;
 497                     }
 498                     updateLeadIndex();
 499 
 500                     resetRowSelection();
 501 
 502                     notifyPathChange(pathsToRemove, beginLeadPath);
 503                 }
 504             }
 505         }
 506     }

 624     protected void fireValueChanged(TreeSelectionEvent e) {
 625         // Guaranteed to return a non-null array
 626         Object[] listeners = listenerList.getListenerList();
 627         // TreeSelectionEvent e = null;
 628         // Process the listeners last to first, notifying
 629         // those that are interested in this event
 630         for (int i = listeners.length-2; i>=0; i-=2) {
 631             if (listeners[i]==TreeSelectionListener.class) {
 632                 // Lazily create the event:
 633                 // if (e == null)
 634                 // e = new ListSelectionEvent(this, firstIndex, lastIndex);
 635                 ((TreeSelectionListener)listeners[i+1]).valueChanged(e);
 636             }
 637         }

 681     /**
 682      * Returns the selection in terms of rows. There is not
 683      * necessarily a one-to-one mapping between the {@code TreePath}s
 684      * returned from {@code getSelectionPaths} and this method. In
 685      * particular, if a {@code TreePath} is not viewable (the {@code
 686      * RowMapper} returns {@code -1} for the row corresponding to the
 687      * {@code TreePath}), then the corresponding row is not included
 688      * in the returned array. For example, if the selection consists
 689      * of two paths, {@code A} and {@code B}, with {@code A} at row
 690      * {@code 10}, and {@code B} not currently viewable, then this method
 691      * returns an array with the single entry {@code 10}.
 692      *
 693      * @return the selection in terms of rows
 694      */
 695     public int[] getSelectionRows() {
 696         // This is currently rather expensive.  Needs
 697         // to be better support from ListSelectionModel to speed this up.
 698         if (rowMapper != null && selection != null && selection.length > 0) {
 699             int[]      rows = rowMapper.getRowsForPaths(selection);
 700 
 701             if (rows != null) {
 702                 int       invisCount = 0;
 703 
 704                 for (int counter = rows.length - 1; counter >= 0; counter--) {
 705                     if (rows[counter] == -1) {
 706                         invisCount++;
 707                     }
 708                 }
 709                 if (invisCount > 0) {
 710                     if (invisCount == rows.length) {
 711                         rows = null;
 712                     }
 713                     else {
 714                         int[]    tempRows = new int[rows.length - invisCount];
 715 
 716                         for (int counter = rows.length - 1, visCounter = 0;
 717                              counter >= 0; counter--) {
 718                             if (rows[counter] != -1) {
 719                                 tempRows[visCounter++] = rows[counter];
 720                             }
 721                         }
 722                         rows = tempRows;
 723                     }
 724                 }
 725             }
 726             return rows;
 727         }
 728         return new int[0];
 729     }

 756     /**
 757      * Updates this object's mapping from TreePath to rows. This should
 758      * be invoked when the mapping from TreePaths to integers has changed
 759      * (for example, a node has been expanded).
 760      * You do not normally have to call this, JTree and its associated
 761      * Listeners will invoke this for you. If you are implementing your own
 762      * View class, then you will have to invoke this.
 763      * This will invoke insureRowContinuity to make sure
 764      * the currently selected TreePaths are still valid based on the
 765      * selection mode.
 766      */
 767     public void resetRowSelection() {
 768         listSelectionModel.clearSelection();
 769         if(selection != null && rowMapper != null) {
 770             int               aRow;
 771             int               validCount = 0;
 772             int[]             rows = rowMapper.getRowsForPaths(selection);
 773 
 774             for(int counter = 0, maxCounter = selection.length;
 775                 counter < maxCounter; counter++) {
 776                 aRow = rows[counter];
 777                 if(aRow != -1) {
 778                     listSelectionModel.addSelectionInterval(aRow, aRow);
 779                 }
 780             }
 781             if(leadIndex != -1 && rows != null) {
 782                 leadRow = rows[leadIndex];
 783             }
 784             else if (leadPath != null) {
 785                 // Lead selection path doesn't have to be in the selection.
 786                 tempPaths[0] = leadPath;
 787                 rows = rowMapper.getRowsForPaths(tempPaths);
 788                 leadRow = (rows != null) ? rows[0] : -1;
 789             }
 790             else {
 791                 leadRow = -1;
 792             }
 793             insureRowContinuity();
 794 
 795         }
 796         else
 797             leadRow = -1;
 798     }

 869     /**
 870      * Makes sure the currently selected TreePaths are valid
 871      * for the current selection mode.
 872      * If the selection mode is CONTIGUOUS_TREE_SELECTION
 873      * and a RowMapper exists, this will make sure all
 874      * the rows are contiguous, that is, when sorted all the rows are
 875      * in order with no gaps.
 876      * If the selection isn't contiguous, the selection is
 877      * reset to contain the first set, when sorted, of contiguous rows.
 878      * 
 879      * If the selection mode is SINGLE_TREE_SELECTION and
 880      * more than one TreePath is selected, the selection is reset to
 881      * contain the first path currently selected.
 882      */
 883     protected void insureRowContinuity() {
 884         if(selectionMode == TreeSelectionModel.CONTIGUOUS_TREE_SELECTION &&
 885            selection != null && rowMapper != null) {
 886             DefaultListSelectionModel lModel = listSelectionModel;
 887             int                       min = lModel.getMinSelectionIndex();
 888 
 889             if(min != -1) {
 890                 for(int counter = min,
 891                         maxCounter = lModel.getMaxSelectionIndex();
 892                         counter <= maxCounter; counter++) {
 893                     if(!lModel.isSelectedIndex(counter)) {
 894                         if(counter == min) {
 895                             clearSelection();
 896                         }
 897                         else {
 898                             TreePath[] newSel = new TreePath[counter - min];
 899                             int selectionIndex[] = rowMapper.getRowsForPaths(selection);
 900                             // find the actual selection pathes corresponded to the
 901                             // rows of the new selection
 902                             for (int i = 0; i < selectionIndex.length; i++) {
 903                                 if (selectionIndex[i]<counter) {
 904                                     newSel[selectionIndex[i]-min] = selection[i];
 905                                 }
 906                             }
 907                             setSelectionPaths(newSel);
 908                             break;
 909                         }
 910                     }
 911                 }
 912             }
 913         }
 914         else if(selectionMode == TreeSelectionModel.SINGLE_TREE_SELECTION &&
 915                 selection != null && selection.length > 1) {
 916             setSelectionPath(selection[0]);
 917         }
 918     }

 920     /**
 921      * Returns true if the paths are contiguous,
 922      * or this object has no RowMapper.
 923      */
 924     protected boolean arePathsContiguous(TreePath[] paths) {
 925         if(rowMapper == null || paths.length < 2)
 926             return true;
 927         else {
 928             BitSet                             bitSet = new BitSet(32);
 929             int                                anIndex, counter, min;
 930             int                                pathCount = paths.length;
 931             int                                validCount = 0;
 932             TreePath[]                         tempPath = new TreePath[1];
 933 
 934             tempPath[0] = paths[0];
 935             min = rowMapper.getRowsForPaths(tempPath)[0];
 936             for(counter = 0; counter < pathCount; counter++) {
 937                 if(paths[counter] != null) {
 938                     tempPath[0] = paths[counter];
 939                     int[] rows = rowMapper.getRowsForPaths(tempPath);
 940                     if (rows == null) {
 941                         return false;
 942                     }
 943                     anIndex = rows[0];
 944                     if(anIndex == -1 || anIndex < (min - pathCount) ||
 945                        anIndex > (min + pathCount))
 946                         return false;
 947                     if(anIndex < min)
 948                         min = anIndex;
 949                     if(!bitSet.get(anIndex)) {
 950                         bitSet.set(anIndex);
 951                         validCount++;
 952                     }
 953                 }
 954             }
 955             int          maxCounter = validCount + min;
 956 
 957             for(counter = min; counter < maxCounter; counter++)
 958                 if(!bitSet.get(counter))
 959                     return false;
 960         }
 961         return true;
 962     }

 964     /**
 965      * Used to test if a particular set of TreePaths can
 966      * be added. This will return true if paths is null (or
 967      * empty), or this object has no RowMapper, or nothing is currently selected,
 968      * or the selection mode is DISCONTIGUOUS_TREE_SELECTION, or
 969      * adding the paths to the current selection still results in a
 970      * contiguous set of TreePaths.
 971      */
 972     protected boolean canPathsBeAdded(TreePath[] paths) {
 973         if(paths == null || paths.length == 0 || rowMapper == null ||
 974            selection == null || selectionMode ==
 975            TreeSelectionModel.DISCONTIGUOUS_TREE_SELECTION)
 976             return true;
 977         else {
 978             BitSet                       bitSet = new BitSet();
 979             DefaultListSelectionModel    lModel = listSelectionModel;
 980             int                          anIndex;
 981             int                          counter;
 982             int                          min = lModel.getMinSelectionIndex();
 983             int                          max = lModel.getMaxSelectionIndex();
 984             TreePath[]                   tempPath = new TreePath[1];
 985 
 986             if(min != -1) {
 987                 for(counter = min; counter <= max; counter++) {
 988                     if(lModel.isSelectedIndex(counter))
 989                         bitSet.set(counter);
 990                 }
 991             }
 992             else {
 993                 tempPath[0] = paths[0];
 994                 min = max = rowMapper.getRowsForPaths(tempPath)[0];
 995             }
 996             for(counter = paths.length - 1; counter >= 0; counter--) {
 997                 if(paths[counter] != null) {
 998                     tempPath[0] = paths[counter];
 999                     int[]   rows = rowMapper.getRowsForPaths(tempPath);
1000                     if (rows == null) {
1001                         return false;
1002                     }
1003                     anIndex = rows[0];
1004                     min = Math.min(anIndex, min);
1005                     max = Math.max(anIndex, max);
1006                     if(anIndex == -1)
1007                         return false;
1008                     bitSet.set(anIndex);
1009                 }
1010             }
1011             for(counter = min; counter <= max; counter++)
1012                 if(!bitSet.get(counter))
1013                     return false;
1014         }
1015         return true;
1016     }

1018     /**
1019      * Returns true if the paths can be removed without breaking the
1020      * continuity of the model.
1021      * This is rather expensive.
1022      */
1023     protected boolean canPathsBeRemoved(TreePath[] paths) {
1024         if(rowMapper == null || selection == null ||
1025            selectionMode == TreeSelectionModel.DISCONTIGUOUS_TREE_SELECTION)
1026             return true;
1027         else {
1028             BitSet               bitSet = new BitSet();
1029             int                  counter;
1030             int                  pathCount = paths.length;
1031             int                  anIndex;
1032             int                  min = -1;
1033             int                  validCount = 0;
1034             TreePath[]           tempPath = new TreePath[1];
1035             int[]                rows;
1036 
1037             /* Determine the rows for the removed entries. */
1038             lastPaths.clear();
1039             for (counter = 0; counter < pathCount; counter++) {
1040                 if (paths[counter] != null) {
1041                     lastPaths.put(paths[counter], Boolean.TRUE);
1042                 }
1043             }
1044             for(counter = selection.length - 1; counter >= 0; counter--) {
1045                 if(lastPaths.get(selection[counter]) == null) {
1046                     tempPath[0] = selection[counter];
1047                     rows = rowMapper.getRowsForPaths(tempPath);
1048                     if(rows != null && rows[0] != -1 && !bitSet.get(rows[0])) {
1049                         validCount++;
1050                         if(min == -1)
1051                             min = rows[0];
1052                         else
1053                             min = Math.min(min, rows[0]);
1054                         bitSet.set(rows[0]);
1055                     }
1056                 }
1057             }
1058             lastPaths.clear();
1059             /* Make sure they are contiguous. */
1060             if(validCount > 1) {
1061                 for(counter = min + validCount - 1; counter >= min;
1062                     counter--)
1063                     if(!bitSet.get(counter))
1064                         return false;
1065             }
1066         }
1067         return true;
1068     }
