  39 /**
  40  * A FocusTraversalPolicy that determines traversal order by sorting the
  41  * Components of a focus traversal cycle based on a given Comparator. Portions
  42  * of the Component hierarchy that are not visible and displayable will not be
  43  * included.
  44  * 
  45  * By default, SortingFocusTraversalPolicy implicitly transfers focus down-
  46  * cycle. That is, during normal focus traversal, the Component
  47  * traversed after a focus cycle root will be the focus-cycle-root's default
  48  * Component to focus. This behavior can be disabled using the
  49  * setImplicitDownCycleTraversal method.
  50  * 
  51  * By default, methods of this class with return a Component only if it is
  52  * visible, displayable, enabled, and focusable. Subclasses can modify this
  53  * behavior by overriding the accept method.
  54  * 
  55  * This policy takes into account 
  56  * href="../../java/awt/doc-files/FocusSpec.html#FocusTraversalPolicyProviders">focus traversal
  57  * policy providers.  When searching for first/last/next/previous Component,
  58  * if a focus traversal policy provider is encountered, its focus traversal
  59  * policy is used to perform the search operation.
  60  *
  61  * @author David Mendenhall
  62  *
  63  * @see java.util.Comparator
  64  * @since 1.4
  65  */
  66 public class SortingFocusTraversalPolicy
  67     extends InternalFrameFocusTraversalPolicy

  69     private Comparator<? super Component> comparator;
  70     private boolean implicitDownCycleTraversal = true;

  74     /**
  75      * Used by getComponentAfter and getComponentBefore for efficiency. In
  76      * order to maintain compliance with the specification of
  77      * FocusTraversalPolicy, if traversal wraps, we should invoke
  78      * getFirstComponent or getLastComponent. These methods may be overriden in
  79      * subclasses to behave in a non-generic way. However, in the generic case,
  80      * these methods will simply return the first or last Components of the
  81      * sorted list, respectively. Since getComponentAfter and
  82      * getComponentBefore have already built the sorted list before determining
  83      * that they need to invoke getFirstComponent or getLastComponent, the
  84      * sorted list should be reused if possible.
  85      */
  86     transient private Container cachedRoot;
  87     transient private List<Component> cachedCycle;

  98      * When true (by default), the legacy merge-sort algo is used to sort an FTP cycle.
  99      * When false, the default (tim-sort) algo is used, which may lead to an exception.

 102     private static final boolean legacySortingFTPEnabled;
 103     private static final Method legacyMergeSortMethod;

 105     static {
 106         legacySortingFTPEnabled = "true".equals(AccessController.doPrivileged(
 107             new GetPropertyAction("swing.legacySortingFTPEnabled", "true")));
 108         legacyMergeSortMethod = legacySortingFTPEnabled ?
 109             AccessController.doPrivileged(new PrivilegedAction<Method>() {
 110                 public Method run() {
 111                     try {
 112                         Class c = Class.forName("java.util.Arrays");
 113                         Method m = c.getDeclaredMethod("legacyMergeSort", new Class[]{Object[].class, Comparator.class});
 114                         m.setAccessible(true);
 115                         return m;
 116                     } catch (ClassNotFoundException | NoSuchMethodException e) {
 117                         // using default sorting algo
 118                         return null;
 119                     }
 120                 }
 121             }) :
 122             null;

 141     private List<Component> getFocusTraversalCycle(Container aContainer) {
 142         List<Component> cycle = new ArrayList<Component>();
 143         enumerateAndSortCycle(aContainer, cycle);
 144         return cycle;
 145     }
 146     private int getComponentIndex(List<Component> cycle, Component aComponent) {
 147         int index;
 148         try {
 149             index = Collections.binarySearch(cycle, aComponent, comparator);
 150         } catch (ClassCastException e) {
 151             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 152                 log.fine("### During the binary search for " + aComponent + " the exception occurred: ", e);
 153             }
 154             return -1;
 155         }
 156         if (index < 0) {
 157             // Fix for 5070991.
 158             // A workaround for a transitivity problem caused by ROW_TOLERANCE,
 159             // because of that the component may be missed in the binary search.
 160             // Try to search it again directly.
 161             index = cycle.indexOf(aComponent);
 162         }
 163         return index;
 164     }

 166     private void enumerateAndSortCycle(Container focusCycleRoot, List<Component> cycle) {
 167         if (focusCycleRoot.isShowing()) {
 168             enumerateCycle(focusCycleRoot, cycle);
 169             if (!legacySortingFTPEnabled ||
 170                 !legacySort(cycle, comparator))
 171             {
 172                 Collections.sort(cycle, comparator);
 173             }
 174         }
 175     }

 177     private boolean legacySort(List<Component> l, Comparator<? super Component> c) {
 178         if (legacyMergeSortMethod == null)
 179             return false;
 180 
 181         Object[] a = l.toArray();
 182         try {
 183             legacyMergeSortMethod.invoke(null, a, c);
 184         } catch (IllegalAccessException | InvocationTargetException e) {
 185             return false;
 186         }
 187         ListIterator<Component> i = l.listIterator();
 188         for (Object e : a) {
 189             i.next();
 190             i.set((Component)e);
 191         }
 192         return true;
 193     }

 195     private void enumerateCycle(Container container, List<Component> cycle) {
 196         if (!(container.isVisible() && container.isDisplayable())) {
 197             return;
 198         }
 199 
 200         cycle.add(container);
 201 
 202         Component[] components = container.getComponents();
 203         for (Component comp : components) {
 204             if (comp instanceof Container) {
 205                 Container cont = (Container)comp;
 206 
 207                 if (!cont.isFocusCycleRoot() &&
 208                     !cont.isFocusTraversalPolicyProvider() &&
 209                     !((cont instanceof JComponent) && ((JComponent)cont).isManagingFocus()))
 210                 {
 211                     enumerateCycle(cont, cycle);
 212                     continue;
 213                 }
 214             }
 215             cycle.add(comp);
 216         }
 217     }

 219     Container getTopmostProvider(Container focusCycleRoot, Component aComponent) {
 220         Container aCont = aComponent.getParent();
 221         Container ftp = null;
 222         while (aCont  != focusCycleRoot && aCont != null) {
 223             if (aCont.isFocusTraversalPolicyProvider()) {
 224                 ftp = aCont;
 225             }
 226             aCont = aCont.getParent();
 227         }
 228         if (aCont == null) {
 229             return null;
 230         }
 231         return ftp;
 232     }

 234     /*
 235      * Checks if a new focus cycle takes place and returns a Component to traverse focus to.
 236      * @param comp a possible focus cycle root or policy provider
 237      * @param traversalDirection the direction of the traversal
 238      * @return a Component to traverse focus to if {@code comp} is a root or provider
 239      *         and implicit down-cycle is set, otherwise {@code null}
 240      */
 241     private Component getComponentDownCycle(Component comp, int traversalDirection) {
 242         Component retComp = null;
 243 
 244         if (comp instanceof Container) {
 245             Container cont = (Container)comp;
 246 
 247             if (cont.isFocusCycleRoot()) {
 248                 if (getImplicitDownCycleTraversal()) {
 249                     retComp = cont.getFocusTraversalPolicy().getDefaultComponent(cont);
 250 
 251                     if (retComp != null && log.isLoggable(PlatformLogger.Level.FINE)) {
 252                         log.fine("### Transfered focus down-cycle to " + retComp +
 253                                  " in the focus cycle root " + cont);
 254                     }
 255                 } else {
 256                     return null;
 257                 }
 258             } else if (cont.isFocusTraversalPolicyProvider()) {
 259                 retComp = (traversalDirection == FORWARD_TRAVERSAL ?
 260                            cont.getFocusTraversalPolicy().getDefaultComponent(cont) :
 261                            cont.getFocusTraversalPolicy().getLastComponent(cont));
 262 
 263                 if (retComp != null && log.isLoggable(PlatformLogger.Level.FINE)) {
 264                     log.fine("### Transfered focus to " + retComp + " in the FTP provider " + cont);
 265                 }
 266             }
 267         }
 268         return retComp;
 269     }

 271     /**
 272      * Returns the Component that should receive the focus after aComponent.
 273      * aContainer must be a focus cycle root of aComponent or a focus traversal policy provider.
 274      * 
 275      * By default, SortingFocusTraversalPolicy implicitly transfers focus down-
 276      * cycle. That is, during normal focus traversal, the Component
 277      * traversed after a focus cycle root will be the focus-cycle-root's
 278      * default Component to focus. This behavior can be disabled using the
 279      * setImplicitDownCycleTraversal method.
 280      * 
 281      * If aContainer is focus
 282      * traversal policy provider, the focus is always transferred down-cycle.
 283      *
 284      * @param aContainer a focus cycle root of aComponent or a focus traversal policy provider
 285      * @param aComponent a (possibly indirect) child of aContainer, or
 286      *        aContainer itself
 287      * @return the Component that should receive the focus after aComponent, or
 288      *         null if no suitable Component can be found
 289      * @throws IllegalArgumentException if aContainer is not a focus cycle
 290      *         root of aComponent or a focus traversal policy provider, or if either aContainer or
 291      *         aComponent is null
 292      */
 293     public Component getComponentAfter(Container aContainer, Component aComponent) {
 294         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 295             log.fine("### Searching in " + aContainer + " for component after " + aComponent);
 296         }
 297 
 298         if (aContainer == null || aComponent == null) {
 299             throw new IllegalArgumentException("aContainer and aComponent cannot be null");
 300         }
 301         if (!aContainer.isFocusTraversalPolicyProvider() && !aContainer.isFocusCycleRoot()) {
 302             throw new IllegalArgumentException("aContainer should be focus cycle root or focus traversal policy provider");
 303 
 304         } else if (aContainer.isFocusCycleRoot() && !aComponent.isFocusCycleRoot(aContainer)) {
 305             throw new IllegalArgumentException("aContainer is not a focus cycle root of aComponent");
 306         }
 307 
 308         // Before all the ckecks below we first see if it's an FTP provider or a focus cycle root.
 309         // If it's the case just go down cycle (if it's set to "implicit").
 310         Component comp = getComponentDownCycle(aComponent, FORWARD_TRAVERSAL);
 311         if (comp != null) {
 312             return comp;
 313         }
 314 
 315         // See if the component is inside of policy provider.
 316         Container provider = getTopmostProvider(aContainer, aComponent);
 317         if (provider != null) {
 318             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 319                 log.fine("### Asking FTP " + provider + " for component after " + aComponent);
 320             }
 321 
 322             // FTP knows how to find component after the given. We don't.
 323             FocusTraversalPolicy policy = provider.getFocusTraversalPolicy();
 324             Component afterComp = policy.getComponentAfter(provider, aComponent);
 325 
 326             // Null result means that we overstepped the limit of the FTP's cycle.
 327             // In that case we must quit the cycle, otherwise return the component found.
 328             if (afterComp != null) {
 329                 if (log.isLoggable(PlatformLogger.Level.FINE)) {
 330                     log.fine("### FTP returned " + afterComp);
 331                 }
 332                 return afterComp;
 333             }
 334             aComponent = provider;
 335         }
 336 
 337         List<Component> cycle = getFocusTraversalCycle(aContainer);
 338 
 339         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 340             log.fine("### Cycle is " + cycle + ", component is " + aComponent);
 341         }
 342 
 343         int index = getComponentIndex(cycle, aComponent);
 344 
 345         if (index < 0) {
 346             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 347                 log.fine("### Didn't find component " + aComponent + " in a cycle " + aContainer);
 348             }
 349             return getFirstComponent(aContainer);
 350         }
 351 
 352         for (index++; index < cycle.size(); index++) {
 353             comp = cycle.get(index);
 354             if (accept(comp)) {
 355                 return comp;
 356             } else if ((comp = getComponentDownCycle(comp, FORWARD_TRAVERSAL)) != null) {
 357                 return comp;
 358             }
 359         }
 360 
 361         if (aContainer.isFocusCycleRoot()) {
 362             this.cachedRoot = aContainer;
 363             this.cachedCycle = cycle;
 364 
 365             comp = getFirstComponent(aContainer);
 366 
 367             this.cachedRoot = null;
 368             this.cachedCycle = null;
 369 
 370             return comp;
 371         }
 372         return null;
 373     }

 375     /**
 376      * Returns the Component that should receive the focus before aComponent.
 377      * aContainer must be a focus cycle root of aComponent or a focus traversal policy provider.
 378      * 
 379      * By default, SortingFocusTraversalPolicy implicitly transfers focus down-
 380      * cycle. That is, during normal focus traversal, the Component
 381      * traversed after a focus cycle root will be the focus-cycle-root's
 382      * default Component to focus. This behavior can be disabled using the
 383      * setImplicitDownCycleTraversal method.
 384      * 
 385      * If aContainer is focus
 386      * traversal policy provider, the focus is always transferred down-cycle.
 387      *
 388      * @param aContainer a focus cycle root of aComponent or a focus traversal policy provider
 389      * @param aComponent a (possibly indirect) child of aContainer, or
 390      *        aContainer itself
 391      * @return the Component that should receive the focus before aComponent,
 392      *         or null if no suitable Component can be found
 393      * @throws IllegalArgumentException if aContainer is not a focus cycle
 394      *         root of aComponent or a focus traversal policy provider, or if either aContainer or
 395      *         aComponent is null
 396      */
 397     public Component getComponentBefore(Container aContainer, Component aComponent) {
 398         if (aContainer == null || aComponent == null) {
 399             throw new IllegalArgumentException("aContainer and aComponent cannot be null");
 400         }
 401         if (!aContainer.isFocusTraversalPolicyProvider() && !aContainer.isFocusCycleRoot()) {
 402             throw new IllegalArgumentException("aContainer should be focus cycle root or focus traversal policy provider");
 403 
 404         } else if (aContainer.isFocusCycleRoot() && !aComponent.isFocusCycleRoot(aContainer)) {
 405             throw new IllegalArgumentException("aContainer is not a focus cycle root of aComponent");
 406         }
 407 
 408         // See if the component is inside of policy provider.
 409         Container provider = getTopmostProvider(aContainer, aComponent);
 410         if (provider != null) {
 411             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 412                 log.fine("### Asking FTP " + provider + " for component after " + aComponent);
 413             }
 414 
 415             // FTP knows how to find component after the given. We don't.
 416             FocusTraversalPolicy policy = provider.getFocusTraversalPolicy();
 417             Component beforeComp = policy.getComponentBefore(provider, aComponent);
 418 
 419             // Null result means that we overstepped the limit of the FTP's cycle.
 420             // In that case we must quit the cycle, otherwise return the component found.
 421             if (beforeComp != null) {
 422                 if (log.isLoggable(PlatformLogger.Level.FINE)) {
 423                     log.fine("### FTP returned " + beforeComp);
 424                 }
 425                 return beforeComp;
 426             }
 427             aComponent = provider;
 428 
 429             // If the provider is traversable it's returned.
 430             if (accept(aComponent)) {
 431                 return aComponent;
 432             }
 433         }
 434 
 435         List<Component> cycle = getFocusTraversalCycle(aContainer);
 436 
 437         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 438             log.fine("### Cycle is " + cycle + ", component is " + aComponent);
 439         }
 440 
 441         int index = getComponentIndex(cycle, aComponent);
 442 
 443         if (index < 0) {
 444             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 445                 log.fine("### Didn't find component " + aComponent + " in a cycle " + aContainer);
 446             }
 447             return getLastComponent(aContainer);
 448         }
 449 
 450         Component comp;
 451         Component tryComp;
 452 
 453         for (index--; index>=0; index--) {
 454             comp = cycle.get(index);
 455             if (comp != aContainer && (tryComp = getComponentDownCycle(comp, BACKWARD_TRAVERSAL)) != null) {
 456                 return tryComp;
 457             } else if (accept(comp)) {
 458                 return comp;
 459             }
 460         }
 461 
 462         if (aContainer.isFocusCycleRoot()) {
 463             this.cachedRoot = aContainer;
 464             this.cachedCycle = cycle;
 465 
 466             comp = getLastComponent(aContainer);
 467 
 468             this.cachedRoot = null;
 469             this.cachedCycle = null;
 470 
 471             return comp;
 472         }
 473         return null;
 474     }

 476     /**
 477      * Returns the first Component in the traversal cycle. This method is used
 478      * to determine the next Component to focus when traversal wraps in the
 479      * forward direction.
 480      *
 481      * @param aContainer a focus cycle root of aComponent or a focus traversal policy provider whose
 482      *        first Component is to be returned
 483      * @return the first Component in the traversal cycle of aContainer,
 484      *         or null if no suitable Component can be found
 485      * @throws IllegalArgumentException if aContainer is null
 486      */
 487     public Component getFirstComponent(Container aContainer) {
 488         List<Component> cycle;
 489 
 490         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 491             log.fine("### Getting first component in " + aContainer);
 492         }
 493         if (aContainer == null) {
 494             throw new IllegalArgumentException("aContainer cannot be null");
 495         }
 496 
 497         if (this.cachedRoot == aContainer) {
 498             cycle = this.cachedCycle;
 499         } else {
 500             cycle = getFocusTraversalCycle(aContainer);
 501         }
 502 
 503         if (cycle.size() == 0) {
 504             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 505                 log.fine("### Cycle is empty");
 506             }
 507             return null;
 508         }
 509         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 510             log.fine("### Cycle is " + cycle);
 511         }
 512 
 513         for (Component comp : cycle) {
 514             if (accept(comp)) {
 515                 return comp;
 516             } else if (comp != aContainer &&
 517                        (comp = getComponentDownCycle(comp, FORWARD_TRAVERSAL)) != null)
 518             {
 519                 return comp;
 520             }
 521         }
 522         return null;
 523     }

 525     /**
 526      * Returns the last Component in the traversal cycle. This method is used
 527      * to determine the next Component to focus when traversal wraps in the
 528      * reverse direction.
 529      *
 530      * @param aContainer a focus cycle root of aComponent or a focus traversal policy provider whose
 531      *        last Component is to be returned
 532      * @return the last Component in the traversal cycle of aContainer,
 533      *         or null if no suitable Component can be found
 534      * @throws IllegalArgumentException if aContainer is null
 535      */
 536     public Component getLastComponent(Container aContainer) {
 537         List<Component> cycle;
 538         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 539             log.fine("### Getting last component in " + aContainer);
 540         }
 541 
 542         if (aContainer == null) {
 543             throw new IllegalArgumentException("aContainer cannot be null");
 544         }
 545 
 546         if (this.cachedRoot == aContainer) {
 547             cycle = this.cachedCycle;
 548         } else {
 549             cycle = getFocusTraversalCycle(aContainer);
 550         }
 551 
 552         if (cycle.size() == 0) {
 553             if (log.isLoggable(PlatformLogger.Level.FINE)) {
 554                 log.fine("### Cycle is empty");
 555             }
 556             return null;
 557         }
 558         if (log.isLoggable(PlatformLogger.Level.FINE)) {
 559             log.fine("### Cycle is " + cycle);
 560         }
 561 
 562         for (int i= cycle.size() - 1; i >= 0; i--) {
 563             Component comp = cycle.get(i);
 564             if (accept(comp)) {
 565                 return comp;
 566             } else if (comp instanceof Container && comp != aContainer) {
 567                 Container cont = (Container)comp;
 568                 if (cont.isFocusTraversalPolicyProvider()) {
 569                     Component retComp = cont.getFocusTraversalPolicy().getLastComponent(cont);
 570                     if (retComp != null) {
 571                         return retComp;
 572                     }
 573                 }
 574             }
 575         }
 576         return null;
 577     }
