 122     @Override
 123     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
 124         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 125         componentColors = extendedCacheKeys;
 126         //generate this entire method. Each state/bg/fg/border combo that has
 127         //been painted gets its own KEY and paint method.
 128         switch(state) {
 129             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
 130             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
 131             case FOREGROUND_ENABLED: paintForegroundEnabled(g); break;
 132             case FOREGROUND_ENABLED_FINISHED: paintForegroundEnabledAndFinished(g); break;
 133             case FOREGROUND_ENABLED_INDETERMINATE: paintForegroundEnabledAndIndeterminate(g); break;
 134             case FOREGROUND_DISABLED: paintForegroundDisabled(g); break;
 135             case FOREGROUND_DISABLED_FINISHED: paintForegroundDisabledAndFinished(g); break;
 136             case FOREGROUND_DISABLED_INDETERMINATE: paintForegroundDisabledAndIndeterminate(g); break;
 137 
 138         }
 139     }
