  27 /**
  28  * This class implements the DOM Level 3 Core interface DOMImplementationList.
  29  *
  30  * @xerces.internal
  31  *
  32  * @author Neil Delima, IBM
  33  * @since DOM Level 3 Core
  34  */
  35 public class DOMImplementationListImpl implements DOMImplementationList {
  36 
  37     //A collection of DOMImplementations
  38     private Vector fImplementations;
  39 
  40     /**
  41      * Construct an empty list of DOMImplementations
  42      */
  43     public DOMImplementationListImpl() {
  44         fImplementations = new Vector();
  45     }
  46 
  47     /**
  48      * Construct an empty list of DOMImplementations
  49      */
  50     public DOMImplementationListImpl(Vector params) {
  51         fImplementations = params;
  52     }
  53 
  54     /**
  55      * Returns the indexth item in the collection.
  56      *
  57      * @param index The index of the DOMImplemetation from the list to return.
  58      */
  59     public DOMImplementation item(int index) {
  60         try {
  61             return (DOMImplementation) fImplementations.elementAt(index);
  62         } catch (ArrayIndexOutOfBoundsException e) {
  63             return null;
  64         }
  65     }
  66 
  67     /**
  68      * Returns the number of DOMImplementations in the list.
  69      *
  70      * @return An integer indicating the number of DOMImplementations.
  71      */
  72     public int getLength() {
  73         return fImplementations.size();
  74     }
  75 }
