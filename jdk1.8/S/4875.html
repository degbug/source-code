  41   /**
  42    * Create a WalkingIterator iterator, including creation
  43    * of step walkers from the opcode list, and call back
  44    * into the Compiler to create predicate expressions.
  45    *
  46    * @param compiler The Compiler which is creating
  47    * this expression.
  48    * @param opPos The position of this iterator in the
  49    * opcode list from the compiler.
  50    * @param shouldLoadWalkers True if walkers should be
  51    * loaded, or false if this is a derived iterator and
  52    * it doesn't wish to load child walkers.
  53    *
  54    * @throws javax.xml.transform.TransformerException
  55    */
  56   WalkingIterator(
  57           Compiler compiler, int opPos, int analysis, boolean shouldLoadWalkers)
  58             throws javax.xml.transform.TransformerException
  59   {
  60     super(compiler, opPos, analysis, shouldLoadWalkers);
  61 
  62     int firstStepPos = OpMap.getFirstChildPos(opPos);
  63 
  64     if (shouldLoadWalkers)
  65     {
  66       m_firstWalker = WalkerFactory.loadWalkers(this, compiler, firstStepPos, 0);
  67       m_lastUsedWalker = m_firstWalker;
  68     }
  69   }

  84   /**
  85    * Get the analysis bits for this walker, as defined in the WalkerFactory.
  86    * @return One of WalkerFactory#BIT_DESCENDANT, etc.
  87    */
  88   public int getAnalysisBits()
  89   {
  90     int bits = 0;
  91     if (null != m_firstWalker)
  92     {
  93       AxesWalker walker = m_firstWalker;
  94 
  95       while (null != walker)
  96       {
  97         int bit = walker.getAnalysisBits();
  98         bits |= bit;
  99         walker = walker.getNextWalker();
 100       }
 101     }
 102     return bits;
 103   }

 164   /**
 165    *  Returns the next node in the set and advances the position of the
 166    * iterator in the set. After a NodeIterator is created, the first call
 167    * to nextNode() returns the first node in the set.
 168    * @return  The next <code>Node</code> in the set being iterated over, or
 169    *   <code>null</code> if there are no more members in that set.
 170    */
 171   public int nextNode()
 172   {
 173         if(m_foundLast)
 174                 return DTM.NULL;
 175 
 176     // If the variable stack position is not -1, we'll have to
 177     // set our position in the variable stack, so our variable access
 178     // will be correct.  Iterators that are at the top level of the
 179     // expression need to reset the variable stack, while iterators
 180     // in predicates do not need to, and should not, since their execution
 181     // may be much later than top-level iterators.
 182     // m_varStackPos is set in setRoot, which is called
 183     // from the execute method.
 184     if (-1 == m_stackFrame)
 185     {
 186       return returnNextNode(m_firstWalker.nextNode());
 187     }
 188     else
 189     {
 190       VariableStack vars = m_execContext.getVarStack();
 191 
 192       // These three statements need to be combined into one operation.
 193       int savedStart = vars.getStackFrame();
 194 
 195       vars.setStackFrame(m_stackFrame);
 196 
 197       int n = returnNextNode(m_firstWalker.nextNode());
 198 
 199       // These two statements need to be combined into one operation.
 200       vars.setStackFrame(savedStart);
 201 
 202       return n;
 203     }
 204   }

 253   /**
 254    *  Detaches the iterator from the set which it iterated over, releasing
 255    * any computational resources and placing the iterator in the INVALID
 256    * state. After<code>detach</code> has been invoked, calls to
 257    * <code>nextNode</code> or<code>previousNode</code> will raise the
 258    * exception INVALID_STATE_ERR.
 259    */
 260   public void detach()
 261   {
 262     if(m_allowDetach)
 263     {
 264                 AxesWalker walker = m_firstWalker;
 265             while (null != walker)
 266             {
 267               walker.detach();
 268               walker = walker.getNextWalker();
 269             }
 270 
 271             m_lastUsedWalker = null;
 272 
 273             // Always call the superclass detach last!
 274             super.detach();
 275     }
 276   }

 278   /**
 279    * This function is used to fixup variables from QNames to stack frame
 280    * indexes at stylesheet build time.
 281    * @param vars List of QNames that correspond to variables.  This list
 282    * should be searched backwards for the first qualified name that
 283    * corresponds to the variable reference qname.  The position of the
 284    * QName in the vector from the start of the vector will be its position
 285    * in the stack frame (but variables above the globalsTop value will need
 286    * to be offset to the current stack frame).
 287    */
 288   public void fixupVariables(java.util.Vector vars, int globalsSize)
 289   {
 290     m_predicateIndex = -1;
 291 
 292     AxesWalker walker = m_firstWalker;
 293 
 294     while (null != walker)
 295     {
 296       walker.fixupVariables(vars, globalsSize);
 297       walker = walker.getNextWalker();
 298     }
 299   }

 341     /**
 342      * @see Expression#deepEquals(Expression)
 343      */
 344     public boolean deepEquals(Expression expr)
 345     {
 346       if (!super.deepEquals(expr))
 347                 return false;
 348 
 349       AxesWalker walker1 = m_firstWalker;
 350       AxesWalker walker2 = ((WalkingIterator)expr).m_firstWalker;
 351       while ((null != walker1) && (null != walker2))
 352       {
 353         if(!walker1.deepEquals(walker2))
 354                 return false;
 355         walker1 = walker1.getNextWalker();
 356         walker2 = walker2.getNextWalker();
 357       }
 358 
 359       if((null != walker1) || (null != walker2))
 360         return false;
 361 
 362       return true;
 363     }
