  26 package org.omg.CORBA;
  27 
  28 /**
  29 * A user exception thrown when a policy error occurs.  A <code>PolicyError</code>
  30 * exception may include one of the following policy error reason codes
  31 * defined in the org.omg.CORBA package: BAD_POLICY, BAD_POLICY_TYPE,
  32 * BAD_POLICY_VALUE, UNSUPPORTED_POLICY, UNSUPPORTED_POLICY_VALUE.
  33 */
  34 
  35 public final class PolicyError extends org.omg.CORBA.UserException {
  36 
  37     /**
  38      * The reason for the <code>PolicyError</code> exception being thrown.
  39      * @serial
  40      */
  41     public short reason;
  42 
  43     /**
  44      * Constructs a default <code>PolicyError</code> user exception
  45      * with no reason code and an empty reason detail message.
  46      */
  47     public PolicyError() {
  48         super();
  49     }
  50 
  51     /**
  52      * Constructs a <code>PolicyError</code> user exception
  53      * initialized with the given reason code and an empty reason detail message.
  54      * @param __reason the reason code.
  55      */
  56     public PolicyError(short __reason) {
  57         super();
  58         reason = __reason;
  59     }
  60 
  61     /**
  62      * Constructs a <code>PolicyError</code> user exception
  63      * initialized with the given reason detail message and reason code.
  64      * @param reason_string the reason detail message.
  65      * @param __reason the reason code.
  66      */
  67     public PolicyError(String reason_string, short __reason) {
  68         super(reason_string);
  69         reason = __reason;
  70     }
  71 }
