  29 /**
  30  * Represents an invocation of a generic class or interface.  For example,
  31  * given the generic interface {@code List}, possible invocations
  32  * include:
  33  * 

  34  *      {@code List}
  35  *      {@code List}
  36  *      {@code List}
  37  * 
  38  * A generic inner class {@code Outer.Inner} might be invoked as:
  39  * 

  40  *      {@code Outer.Inner}
  41  * 
  42  *
  43  * @author Scott Seligman
  44  * @since 1.5
  45  */
  46 public interface ParameterizedType extends Type {
  47 
  48     /**
  49      * Return the generic class or interface that declared this type.
  50      *
  51      * @return the generic class or interface that declared this type.
  52      */
  53     ClassDoc asClassDoc();
  54 
  55     /**
  56      * Return the actual type arguments of this type.
  57      * For a generic type that is nested within some other generic type
  58      * (such as {@code Outer.Inner}),
  59      * only the type arguments of the innermost type are included.
  60      *
  61      * @return the actual type arguments of this type.
  62      */
  63     Type[] typeArguments();
  64 
  65     /**
  66      * Return the class type that is a direct supertype of this one.
  67      * This is the superclass of this type's declaring class,
  68      * with type arguments substituted in.
  69      * Return null if this is an interface type.
  70      *
  71      *  For example, if this parameterized type is
  72      * {@code java.util.ArrayList}, the result will be
  73      * {@code java.util.AbstractList}.
  74      *
  75      * @return the class type that is a direct supertype of this one.
  76      */
  77     Type superclassType();
  78 
  79     /**
  80      * Return the interface types directly implemented by or extended by this
  81      * parameterized type.
  82      * These are the interfaces directly implemented or extended
  83      * by this type's declaring class or interface,
  84      * with type arguments substituted in.
  85      * Return an empty array if there are no interfaces.
  86      *
  87      *  For example, the interface extended by
  88      * {@code java.util.Set} is {@code java.util.Collection}.
  89      *
  90      * @return the interface types directly implemented by or extended by this
  91      * parameterized type.
  92      */
  93     Type[] interfaceTypes();
  94 
  95     /**
  96      * Return the type that contains this type as a member.
  97      * Return null is this is a top-level type.
  98      *
  99      *  For example, the containing type of
 100      * {@code AnInterface.Nested} is the ClassDoc
 101      * representing {@code AnInterface}, and the containing type of
 102      * {@code Outer.Inner} is the
 103      * ParameterizedType representing {@code Outer}.
 104      *
 105      * @return the type that contains this type as a member.
 106      */
 107     Type containingType();
 108 }
