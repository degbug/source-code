  31 /**
  32  * This class is used by the {@link com.sun.jmx.snmp.daemon.SnmpAdaptorServer SNMP adaptor server} to send inform requests
  33  * to an SNMP manager and receive inform responses.
  34  * <P>
  35  * This class provides basic functions that enable you to fire inform requests,
  36  * handle retries, timeouts, and process responses from the manager.
  37  * <BR>
  38  * The SNMP adaptor server specifies the destination of the inform request and controls
  39  * the size of a single inform request/response to fit into its <CODE>bufferSize</CODE>.
  40  * It specifies the maximum number of tries and the timeout to be used for the inform requests.
  41  * It also provides resources such as the authentication mechanism (using its PDU factory),
  42  * controlling all inform requests created by it, and finally the inform response to the user.
  43  * <P>
  44  * Each inform request, when ready to be sent, is assigned a unique identifier which helps
  45  * in identifying the inform request with matching inform responses to the protocol engine
  46  * lying transparently underneath. The engine does the job of retrying the inform requests
  47  * when the timer expires and calls the SNMP adaptor server when a timeout occurs after exhausting
  48  * the maximum number of tries.
  49  * <P>
  50  * The inform request object provides the method, {@link #waitForCompletion waitForCompletion(long time)},
  51  * which enables a user to operate in a synchronous mode with an inform request.
  52  * This is done by blocking the user thread for the desired time interval.
  53  * The user thread gets notified whenever a request reaches completion, independently of the status of the response.
  54  * <P>
  55  * If an {@link com.sun.jmx.snmp.daemon.SnmpInformHandler inform callback} is provided when sending the inform request,
  56  * the user operates in an asynchronous mode with the inform request. The user thread is not blocked
  57  * and the specific inform callback implementation provided by the user is invoked when the inform response is received.
  58  *
  59  * <P>
  60  * <B>Note:</B>
  61  * <BR>From RFC 1905, the SNMP inform request is defined as a request generated and transmitted
  62  * by an SNMPv2 entity acting in a manager role to another SNMPv2 entity also acting in a manager role.
  63  * The mechanisms to implement this behaviour are defined in the SNMP manager API.
  64  * <BR>
  65  * Nevertheless, this feature has derived and in some documentations, the inform request appears
  66  * like an SNMPv2 trap that gets responded.
  67  * <BR>The <CODE>SnmpInformRequest</CODE> class is used to fullfill this latter case.
  68  * <p><b>This API is a Sun Microsystems internal API  and is subject
  69  * to change without notice.</b></p>
  70  */

 370     /**
 371      * Used in synchronous mode only.
 372      * Provides a hook that enables a synchronous operation on a previously sent inform request.
 373      * Only one inform request can be in synchronous mode on a given thread.
 374      * The blocked thread is notified when the inform request state reaches completion.
 375      * If the inform request is not active, the method returns immediately.
 376      * The user must get the error status of the inform request to determine the
 377      * exact status of the request.
 378      *
 379      * @param time The amount of time to wait. Zero means block until complete.
 380      * @return <CODE>true</CODE> if the inform request has completed, <CODE>false</CODE> if it is still active.
 381      */
 382     final public boolean waitForCompletion(long time) {
 383 
 384         if (! inProgress())     // check if request is in progress.
 385             return true;
 386 
 387         if (informSession.thisSessionContext()) {
 388             // We can manipulate callback safely as we are in session thread.
 389             //
 390             SnmpInformHandler savedCallback = callback;
 391             callback = null;
 392             informSession.waitForResponse(this, time);
 393             callback = savedCallback;
 394         } else {
 395             // This is being done from a different thread. So notifyClient will do the notification.
 396             //
 397             synchronized (this) {
 398                 SnmpInformHandler savedCallback = callback ;
 399                 try {
 400                     callback = null ;
 401                     this.wait(time) ;
 402                 } catch (InterruptedException e) {
 403                 }
 404                 callback = savedCallback ;
 405             }
 406         }
 407 
 408         return (! inProgress()); // true if request completed.
 409     }

 579     /**
 580      * This method determines whether the inform request is to be retried. This is used if the
 581      * peer did not respond to a previous request. If the request exceeds
 582      * the maxTries limit, a timeout is signaled.
 583      */
 584     void action() {
 585         if (inProgress() == false)
 586             return;
 587         while (true) {
 588             try {
 589                 if (numTries == 0) {
 590                     invokeOnReady();
 591                 } else if (numTries < getMaxTries()) {
 592                     invokeOnRetry();
 593                 } else {
 594                     invokeOnTimeout();
 595                 }
 596                 return ;
 597             } catch (OutOfMemoryError omerr) {
 598                 // Consider it as a try !
 599                 //
 600                 numTries++;
 601                 if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 602                     SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 603                         "action", "Inform request hit out of memory situation...");
 604                 }
 605                 Thread.yield();
 606             }
 607         }
 608     }

 636     synchronized SnmpPdu constructPduPacket() {

 639         try {

 641             reqpdu.port = port;
 642             reqpdu.type = pduInformRequestPdu;
 643             reqpdu.version = snmpVersionTwo;
 644             reqpdu.community = communityString.getBytes("8859_1");
 645             reqpdu.requestId = getRequestId();
 646             reqpdu.varBindList = internalVarBind;
 647 
 648             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINER)) {

 650                     "constructPduPacket", "Packet built");
 651             }
 652 
 653         } catch (Exception e) {
 654             excep = e;
 655             errorStatus = snmpReqUnknownError;
 656             reason = e.getMessage();
 657         }
 658         if (excep != null) {
 659             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 660                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 661                     "constructPduPacket", "Got unexpected exception", excep);
 662             }
 663             reqpdu = null;
 664             queueResponse();
 665         }
 666         return reqpdu;

 669     boolean sendPdu() {
 670         try {
 671             responsePdu = null;
 672 
 673             SnmpPduFactory pduFactory = adaptor.getPduFactory();
 674             SnmpMessage msg = (SnmpMessage)pduFactory.encodeSnmpPdu((SnmpPduPacket)requestPdu, adaptor.getBufferSize().intValue());
 675 
 676             if (msg == null) {
 677                 if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 678                     SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 679                         "sendPdu", "pdu factory returned a null value");
 680                 }
 681                 throw new SnmpStatusException(snmpReqUnknownError);
 682                 // This exception will caught hereafter and reported as an snmpReqUnknownError
 683                 // FIXME: may be it's not the best behaviour ?
 684             }
 685 
 686             int maxPktSize = adaptor.getBufferSize().intValue();
 687             byte[] encoding = new byte[maxPktSize];
 688             int encodingLength = msg.encodeMessage(encoding);
 689 
 690             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINER)) {
 691                 SNMP_ADAPTOR_LOGGER.logp(Level.FINER, SnmpInformRequest.class.getName(),
 692                     "sendPdu", "Dump : \n" + msg.printMessage());
 693             }
 694 
 695             sendPduPacket(encoding, encodingLength);
 696             return true;
 697         } catch (SnmpTooBigException ar) {
 698 
 699             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 700                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 701                     "sendPdu", "Got unexpected exception", ar);
 702             }
 703 
 704             setErrorStatusAndIndex(snmpReqPacketOverflow, ar.getVarBindCount());
 705             requestPdu = null;
 706             reason = ar.getMessage();
 707             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 708                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 709                     "sendPdu", "Packet Overflow while building inform request");
 710             }
 711         } catch (java.io.IOException ioe) {
 712             setErrorStatusAndIndex(snmpReqSocketIOError, 0);
 713             reason = ioe.getMessage();
 714         } catch (Exception e) {
 715             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 716                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 717                     "sendPdu", "Got unexpected exception", e);
 718             }
 719             setErrorStatusAndIndex(snmpReqUnknownError, 0);
 720             reason = e.getMessage();
 721         }
 722         return false;
 723     }

 725     /**
 726      * Sends the prepared PDU packet to the manager and updates the data structure
 727      * to expect a response. It acquires a lock on the socket to prevent a case
 728      * where a response arrives before this thread could insert the
 729      * request into the wait queue.
 730      * @exception IOException Signals that an I/O exception of some sort has occurred.
 731      */
 732     final void sendPduPacket(byte[] buffer, int length) throws java.io.IOException {
 733 
 734         if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINER)) {
 735             SNMP_ADAPTOR_LOGGER.logp(Level.FINER, SnmpInformRequest.class.getName(),
 736                 "sendPduPacket", "Send to peer. Peer/Port : " + address.getHostName() +
 737                  "/" + port + ". Length = " +  length + "\nDump : \n" +
 738                  SnmpMessage.dumpHexBuffer(buffer,0, length));
 739         }
 740         SnmpSocket theSocket = informSession.getSocket();
 741         synchronized (theSocket) {
 742             theSocket.sendPacket(buffer, length, address, port);
 743             setRequestSentTime(System.currentTimeMillis());
 744         }
 745     }

 747     /**
 748      * For SNMP Runtime internal use only.
 749      */
 750     final void processResponse() {
 751 
 752         if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINER)) {
 753             SNMP_ADAPTOR_LOGGER.logp(Level.FINER, SnmpInformRequest.class.getName(),
 754                 "processResponse", "errstatus = " + errorStatus);
 755         }
 756 
 757         if (inProgress() == false) {  // check if this request is still alive.
 758             responsePdu = null;
 759             return;  // the request may have  cancelled.
 760         }
 761 
 762         if (errorStatus >= snmpReqInternalError) {
 763             handleInternalError("Internal Error...");
 764             return;
 765         }
 766 
 767         try {
 768             parsePduPacket(responsePdu);
 769             //responsePdu = null;
 770 
 771             // At this point the errorIndex is rationalized to start with 0.
 772             switch (errorStatus) {
 773             case snmpRspNoError :
 774                 handleSuccess();
 775                 return;
 776             case snmpReqTimeout :
 777                 handleTimeout();
 778                 return;
 779             case snmpReqInternalError :
 780                 handleInternalError("Unknown internal error.  deal with it later!");
 781                 return;
 782             case snmpReqHandleTooBig :
 783                 setErrorStatusAndIndex(snmpRspTooBig, 0);
 784                 handleError("Cannot handle too-big situation...");
 785                 return;
 786             case snmpReqRefireAfterVbFix :
 787                 // Refire request after fixing varbindlist.
 788                 initializeAndFire();
 789                 return;
 790             default :
 791                 handleError("Error status set in packet...!!");
 792                 return;
 793             }
 794         } catch (Exception e) {
 795             if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINEST)) {
 796                 SNMP_ADAPTOR_LOGGER.logp(Level.FINEST, SnmpInformRequest.class.getName(),
 797                     "processResponse", "Got unexpected exception", e);
 798             }
 799             reason = e.getMessage();
 800         }
 801         handleInternalError(reason);
 802     }

 804     /**
 805      * Parses the inform response packet. If the agent responds with error set,
 806      * it does not parse any further.
 807      */
 808     synchronized void parsePduPacket(SnmpPduRequestType rpdu) {
 809 
 810         if (rpdu == null)
 811             return;
 812 
 813         errorStatus = rpdu.getErrorStatus();
 814         errorIndex = rpdu.getErrorIndex();
 815 
 816         if (errorStatus == snmpRspNoError) {
 817             updateInternalVarBindWithResult(((SnmpPdu)rpdu).varBindList);
 818             return;
 819         }
 820 
 821         if (errorStatus != snmpRspNoError)
 822             --errorIndex;  // rationalize for index to start with 0.
 823 
 824         if (SNMP_ADAPTOR_LOGGER.isLoggable(Level.FINER)) {
 825             SNMP_ADAPTOR_LOGGER.logp(Level.FINER, SnmpInformRequest.class.getName(),
 826                 "parsePduPacket", "received inform response. ErrorStatus/ErrorIndex = "
 827                 + errorStatus + "/" + errorIndex);
 828         }
 829     }

 979     void updateInternalVarBindWithResult(SnmpVarBind[] list) {
 980 
 981         if ((list == null) || (list.length == 0))
 982             return;
 983 
 984         int idx = 0;
 985 
 986         for(int i = 0; i < internalVarBind.length && idx < list.length; i++) {
 987             SnmpVarBind avar = internalVarBind[i];
 988             if (avar == null)
 989                 continue;
 990 
 991             SnmpVarBind res = list[idx];
 992             avar.setSnmpValue(res.getSnmpValue());
 993             idx++;
 994         }
 995     }
