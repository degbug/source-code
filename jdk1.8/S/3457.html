  Defines a monitor MBean designed to observe the values of a string
  attribute.
  

  A string monitor sends notifications as follows:
  
    
  
  •  if the attribute value matches the string to compare value,
       a {@link MonitorNotification#STRING_TO_COMPARE_VALUE_MATCHED
       match notification} is sent.
       The notify match flag must be set to true.
       
    Subsequent matchings of the string to compare values do not
       cause further notifications unless
       the attribute value differs from the string to compare value.
  
  •  if the attribute value differs from the string to compare value,
       a {@link MonitorNotification#STRING_TO_COMPARE_VALUE_DIFFERED
       differ notification} is sent.
       The notify differ flag must be set to true.
       
    Subsequent differences from the string to compare value do
       not cause further notifications unless
       the attribute value matches the string to compare value.

 398     @Override
 399     synchronized MonitorNotification buildAlarmNotification(
 400                                                ObjectName object,
 401                                                String attribute,
 402                                                Comparable<?> value) {
 403         String type = null;
 404         String msg = null;
 405         Object trigger = null;
 406 
 407         final StringMonitorObservedObject o =
 408             (StringMonitorObservedObject) getObservedObject(object);
 409         if (o == null)
 410             return null;
 411 
 412         // Send matching notification if notifyMatch is true.
 413         // Send differing notification if notifyDiffer is true.
 414         //
 415         if (o.getStatus() == MATCHING_OR_DIFFERING) {
 416             if (o.getDerivedGauge().equals(stringToCompare)) {
 417                 if (notifyMatch) {
 418                     type = STRING_TO_COMPARE_VALUE_MATCHED;
 419                     msg = "";
 420                     trigger = stringToCompare;
 421                 }
 422                 o.setStatus(DIFFERING);
 423             } else {
 424                 if (notifyDiffer) {
 425                     type = STRING_TO_COMPARE_VALUE_DIFFERED;
 426                     msg = "";
 427                     trigger = stringToCompare;
 428                 }
 429                 o.setStatus(MATCHING);
 430             }
 431         } else {
 432             if (o.getStatus() == MATCHING) {
 433                 if (o.getDerivedGauge().equals(stringToCompare)) {
 434                     if (notifyMatch) {
 435                         type = STRING_TO_COMPARE_VALUE_MATCHED;
 436                         msg = "";
 437                         trigger = stringToCompare;
 438                     }
 439                     o.setStatus(DIFFERING);
 440                 }
 441             } else if (o.getStatus() == DIFFERING) {
 442                 if (!o.getDerivedGauge().equals(stringToCompare)) {
 443                     if (notifyDiffer) {
 444                         type = STRING_TO_COMPARE_VALUE_DIFFERED;
 445                         msg = "";
 446                         trigger = stringToCompare;
 447                     }
 448                     o.setStatus(MATCHING);
 449                 }
 450             }
 451         }
 452 
 453         return new MonitorNotification(type,
 454                                        this,
 455                                        0,
 456                                        0,
 457                                        msg,
 458                                        null,
 459                                        null,
 460                                        null,
 461                                        trigger);
 462     }
