  24 package com.sun.org.apache.xml.internal.serializer;
  25 
  26 import java.util.Hashtable;
  27 
  28 import org.xml.sax.Attributes;
  29 import org.xml.sax.helpers.AttributesImpl;
  30 
  31 /**
  32  * This class extends org.xml.sax.helpers.AttributesImpl which implements org.
  33  * xml.sax.Attributes. But for optimization this class adds a Hashtable for
  34  * faster lookup of an index by qName, which is commonly done in the stream
  35  * serializer.
  36  *
  37  * @see org.xml.sax.Attributes
  38  *
  39  * @xsl.usage internal
  40  */
  41 public final class AttributesImplSerializer extends AttributesImpl
  42 {
  43     /**
  44      * Hash table of qName/index values to quickly lookup the index
  45      * of an attributes qName.  qNames are in uppercase in the hash table
  46      * to make the search case insensitive.
  47      *
  48      * The keys to the hashtable to find the index are either
  49      * "prefix:localName"  or "{uri}localName".
  50      */
  51     private final Hashtable m_indexFromQName = new Hashtable();
  52 
  53     private final StringBuffer m_buff = new StringBuffer();
  54 
  55     /**
  56      * This is the number of attributes before switching to the hash table,
  57      * and can be tuned, but 12 seems good for now - Brian M.
  58      */
  59     private static final int MAX = 12;
  60 
  61     /**
  62      * One less than the number of attributes before switching to
  63      * the Hashtable.
  64      */
  65     private static final int MAXMinus1 = MAX - 1;
  66 
  67     /**
  68      * This method gets the index of an attribute given its qName.
  69      * @param qname the qualified name of the attribute, e.g. "prefix1:locName1"
  70      * @return the integer index of the attribute.
  71      * @see org.xml.sax.Attributes#getIndex(String)
  72      */
  73     public final int getIndex(String qname)
  74     {
  75         int index;
  76 
  77         if (super.getLength() < MAX)
  78         {
  79             // if we haven't got too many attributes let the
  80             // super class look it up
  81             index = super.getIndex(qname);
  82             return index;
  83         }
  84         // we have too many attributes and the super class is slow
  85         // so find it quickly using our Hashtable.
  86         Integer i = (Integer)m_indexFromQName.get(qname);
  87         if (i == null)
  88             index = -1;
  89         else
  90             index = i.intValue();
  91         return index;
  92     }
  93     /**
  94      * This method adds the attribute, but also records its qName/index pair in
  95      * the hashtable for fast lookup by getIndex(qName).
  96      * @param uri the URI of the attribute
  97      * @param local the local name of the attribute
  98      * @param qname the qualified name of the attribute
  99      * @param type the type of the attribute
 100      * @param val the value of the attribute
 101      *
 102      * @see org.xml.sax.helpers.AttributesImpl#addAttribute(String, String, String, String, String)
 103      * @see #getIndex(String)
 104      */
 105     public final void addAttribute(
 106         String uri,
 107         String local,
 108         String qname,
 109         String type,
 110         String val)
 111     {
 112         int index = super.getLength();
 113         super.addAttribute(uri, local, qname, type, val);
 114         // (index + 1) is now the number of attributes
 115         // so either compare (index+1) to MAX, or compare index to (MAX-1)
 116 
 117         if (index < MAXMinus1)
 118         {
 119             return;
 120         }
 121         else if (index == MAXMinus1)
 122         {
 123             switchOverToHash(MAX);
 124         }
 125         else
 126         {
 127             /* add the key with the format of "prefix:localName" */
 128             /* we have just added the attibute, its index is the old length */
 129             Integer i = new Integer(index);
 130             m_indexFromQName.put(qname, i);
 131 
 132             /* now add with key of the format "{uri}localName" */
 133             m_buff.setLength(0);
 134             m_buff.append('{').append(uri).append('}').append(local);
 135             String key = m_buff.toString();
 136             m_indexFromQName.put(key, i);
 137         }
 138         return;
 139     }
 140 
 141     /**
 142      * We are switching over to having a hash table for quick look
 143      * up of attributes, but up until now we haven't kept any
 144      * information in the Hashtable, so we now update the Hashtable.
 145      * Future additional attributes will update the Hashtable as
 146      * they are added.
 147      * @param numAtts
 148      */
 149     private void switchOverToHash(int numAtts)
 150     {
 151         for (int index = 0; index < numAtts; index++)
 152         {
 153             String qName = super.getQName(index);
 154             Integer i = new Integer(index);
 155             m_indexFromQName.put(qName, i);
 156 
 157             // Add quick look-up to find with uri/local name pair
 158             String uri = super.getURI(index);
 159             String local = super.getLocalName(index);
 160             m_buff.setLength(0);
 161             m_buff.append('{').append(uri).append('}').append(local);
 162             String key = m_buff.toString();
 163             m_indexFromQName.put(key, i);
 164         }
 165     }
 166 
 167     /**
 168      * This method clears the accumulated attributes.
 169      *
 170      * @see org.xml.sax.helpers.AttributesImpl#clear()
 171      */
 172     public final void clear()
 173     {
 174 
 175         int len = super.getLength();
 176         super.clear();
 177         if (MAX <= len)
 178         {
 179             // if we have had enough attributes and are
 180             // using the Hashtable, then clear the Hashtable too.
 181             m_indexFromQName.clear();
 182         }
 183 
 184     }
 185 
 186     /**
 187      * This method sets the attributes, previous attributes are cleared,
 188      * it also keeps the hashtable up to date for quick lookup via
 189      * getIndex(qName).
 190      * @param atts the attributes to copy into these attributes.
 191      * @see org.xml.sax.helpers.AttributesImpl#setAttributes(Attributes)
 192      * @see #getIndex(String)
 193      */
 194     public final void setAttributes(Attributes atts)
 195     {
 196 
 197         super.setAttributes(atts);
 198 
 199         // we've let the super class add the attributes, but
 200         // we need to keep the hash table up to date ourselves for the
 201         // potentially new qName/index pairs for quick lookup.
 202         int numAtts = atts.getLength();
 203         if (MAX <= numAtts)
 204             switchOverToHash(numAtts);
 205 
 206     }
 207 
 208     /**
 209      * This method gets the index of an attribute given its uri and locanName.
 210      * @param uri the URI of the attribute name.
 211      * @param localName the local namer (after the ':' ) of the attribute name.
 212      * @return the integer index of the attribute.
 213      * @see org.xml.sax.Attributes#getIndex(String)
 214      */
 215     public final int getIndex(String uri, String localName)
 216     {
 217         int index;
 218 
 219         if (super.getLength() < MAX)
 220         {
 221             // if we haven't got too many attributes let the
 222             // super class look it up
 223             index = super.getIndex(uri,localName);
 224             return index;
 225         }
 226         // we have too many attributes and the super class is slow
 227         // so find it quickly using our Hashtable.
 228         // Form the key of format "{uri}localName"
 229         m_buff.setLength(0);
 230         m_buff.append('{').append(uri).append('}').append(localName);
 231         String key = m_buff.toString();
 232         Integer i = (Integer)m_indexFromQName.get(key);
 233         if (i == null)
 234             index = -1;
 235         else
 236             index = i.intValue();
 237         return index;
 238     }
