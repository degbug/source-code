  60     /**
  61      * Copies the source tree into the specified place in a destination
  62      * tree. The source node and its children are appended as children
  63      * of the destination node.
  64      * <p>
  65      * <em>Note:</em> This is an iterative implementation.
  66      */
  67     public static void copyInto(Node src, Node dest) throws DOMException {
  68 
  69         // get node factory
  70         Document factory = dest.getOwnerDocument();
  71         boolean domimpl = factory instanceof DocumentImpl;
  72 
  73         // placement variables
  74         Node start  = src;
  75         Node parent = src;
  76         Node place  = src;
  77 
  78         // traverse source tree
  79         while (place != null) {
  80 
  81             // copy this node
  82             Node node = null;
  83             int  type = place.getNodeType();
  84             switch (type) {
  85             case Node.CDATA_SECTION_NODE: {
  86                 node = factory.createCDATASection(place.getNodeValue());
  87                 break;
  88             }
  89             case Node.COMMENT_NODE: {
  90                 node = factory.createComment(place.getNodeValue());
  91                 break;
  92             }
  93             case Node.ELEMENT_NODE: {
  94                 Element element = factory.createElement(place.getNodeName());
  95                 node = element;
  96                 NamedNodeMap attrs  = place.getAttributes();
  97                 int attrCount = attrs.getLength();
  98                 for (int i = 0; i < attrCount; i++) {
  99                     Attr attr = (Attr)attrs.item(i);
 100                     String attrName = attr.getNodeName();
 101                     String attrValue = attr.getNodeValue();
 102                     element.setAttribute(attrName, attrValue);
 103                     if (domimpl && !attr.getSpecified()) {
 104                         ((AttrImpl)element.getAttributeNode(attrName)).setSpecified(false);
 105                     }
 106                 }
 107                 break;
 108             }
 109             case Node.ENTITY_REFERENCE_NODE: {
 110                 node = factory.createEntityReference(place.getNodeName());
 111                 break;
 112             }
 113             case Node.PROCESSING_INSTRUCTION_NODE: {
 114                 node = factory.createProcessingInstruction(place.getNodeName(),
 115                         place.getNodeValue());
 116                 break;
 117             }
 118             case Node.TEXT_NODE: {
 119                 node = factory.createTextNode(place.getNodeValue());
 120                 break;
 121             }
 122             default: {
 123                 throw new IllegalArgumentException("can't copy node type, "+
 124                         type+" ("+
 125                         place.getNodeName()+')');
 126             }
 127             }
 128             dest.appendChild(node);
 129 
 130             // iterate over children
 131             if (place.hasChildNodes()) {
 132                 parent = place;
 133                 place  = place.getFirstChild();
 134                 dest   = node;
 135             }
 136 
 137             // advance
 138             else {
 139                 place = place.getNextSibling();
 140                 while (place == null && parent != start) {
 141                     place  = parent.getNextSibling();
 142                     parent = parent.getParentNode();
 143                     dest   = dest.getParentNode();
 144                 }
 145             }
 146 
 147         }
 148

 708     /**
 709      * Returns the concatenated child text of the specified node.
 710      * This method only looks at the immediate children of type
 711      * <code>Node.TEXT_NODE</code> or the children of any child
 712      * node that is of type <code>Node.CDATA_SECTION_NODE</code>
 713      * for the concatenation.
 714      *
 715      * @param node The node to look at.
 716      */
 717     public static String getChildText(Node node) {
 718 
 719         // is there anything to do?
 720         if (node == null) {
 721             return null;
 722         }
 723 
 724         // concatenate children text
 725         StringBuffer str = new StringBuffer();
 726         Node child = node.getFirstChild();
 727         while (child != null) {
 728             short type = child.getNodeType();
 729             if (type == Node.TEXT_NODE) {
 730                 str.append(child.getNodeValue());
 731             }
 732             else if (type == Node.CDATA_SECTION_NODE) {
 733                 str.append(getChildText(child));
 734             }
 735             child = child.getNextSibling();
 736         }
 737 
 738         // return text value
 739         return str.toString();
 740 
 741     } // getChildText(Node):String

 847     public static DOMException createDOMException(short code, Throwable cause) {
 848         DOMException de = new DOMException(code, cause != null ? cause.getMessage() : null);
 849         if (cause != null && ThrowableMethods.fgThrowableMethodsAvailable) {

 854             catch (Exception e) {}
