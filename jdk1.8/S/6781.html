 222     private void readHeader() throws IIOException {
 223         if (gotHeader) {
 224             return;
 225         }
 226         if (stream == null) {
 227             throw new IllegalStateException("Input source not set!");
 228         }
 229 
 230         try {
 231             byte[] signature = new byte[8];
 232             stream.readFully(signature);
 233 
 234             if (signature[0] != (byte)137 ||
 235                 signature[1] != (byte)80 ||
 236                 signature[2] != (byte)78 ||
 237                 signature[3] != (byte)71 ||
 238                 signature[4] != (byte)13 ||
 239                 signature[5] != (byte)10 ||
 240                 signature[6] != (byte)26 ||
 241                 signature[7] != (byte)10) {
 242                 throw new IIOException("Bad PNG signature!");
 243             }
 244 
 245             int IHDR_length = stream.readInt();
 246             if (IHDR_length != 13) {
 247                 throw new IIOException("Bad length for IHDR chunk!");
 248             }
 249             int IHDR_type = stream.readInt();
 250             if (IHDR_type != IHDR_TYPE) {
 251                 throw new IIOException("Bad type for IHDR chunk!");
 252             }
 253 
 254             this.metadata = new PNGMetadata();
 255 
 256             int width = stream.readInt();
 257             int height = stream.readInt();
 258 
 259             // Re-use signature array to bulk-read these unsigned byte values
 260             stream.readFully(signature, 0, 5);
 261             int bitDepth          = signature[0] & 0xff;
 262             int colorType         = signature[1] & 0xff;
 263             int compressionMethod = signature[2] & 0xff;
 264             int filterMethod      = signature[3] & 0xff;
 265             int interlaceMethod   = signature[4] & 0xff;
 266 
 267             // Skip IHDR CRC
 268             stream.skipBytes(4);
 269 
 270             stream.flushBefore(stream.getStreamPosition());
 271 
 272             if (width == 0) {
 273                 throw new IIOException("Image width == 0!");
 274             }
 275             if (height == 0) {
 276                 throw new IIOException("Image height == 0!");
 277             }
 278             if (bitDepth != 1 && bitDepth != 2 && bitDepth != 4 &&
 279                 bitDepth != 8 && bitDepth != 16) {
 280                 throw new IIOException("Bit depth must be 1, 2, 4, 8, or 16!");
 281             }
 282             if (colorType != 0 && colorType != 2 && colorType != 3 &&
 283                 colorType != 4 && colorType != 6) {
 284                 throw new IIOException("Color type must be 0, 2, 3, 4, or 6!");
 285             }
 286             if (colorType == PNG_COLOR_PALETTE && bitDepth == 16) {
 287                 throw new IIOException("Bad color type/bit depth combination!");
 288             }
 289             if ((colorType == PNG_COLOR_RGB ||
 290                  colorType == PNG_COLOR_RGB_ALPHA ||
 291                  colorType == PNG_COLOR_GRAY_ALPHA) &&
 292                 (bitDepth != 8 && bitDepth != 16)) {
 293                 throw new IIOException("Bad color type/bit depth combination!");
 294             }
 295             if (compressionMethod != 0) {
 296                 throw new IIOException("Unknown compression method (not 0)!");
 297             }
 298             if (filterMethod != 0) {
 299                 throw new IIOException("Unknown filter method (not 0)!");
 300             }
 301             if (interlaceMethod != 0 && interlaceMethod != 1) {
 302                 throw new IIOException("Unknown interlace method (not 0 or 1)!");
 303             }
 304 
 305             metadata.IHDR_present = true;
 306             metadata.IHDR_width = width;
 307             metadata.IHDR_height = height;
 308             metadata.IHDR_bitDepth = bitDepth;
 309             metadata.IHDR_colorType = colorType;
 310             metadata.IHDR_compressionMethod = compressionMethod;
 311             metadata.IHDR_filterMethod = filterMethod;
 312             metadata.IHDR_interlaceMethod = interlaceMethod;
 313             gotHeader = true;
 314         } catch (IOException e) {
 315             throw new IIOException("I/O error reading PNG header!", e);
 316         }
 317     }

 319     private void parse_PLTE_chunk(int chunkLength) throws IOException {
 320         if (metadata.PLTE_present) {
 321             processWarningOccurred(
 322 "A PNG image may not contain more than one PLTE chunk.\n" +
 323 "The chunk wil be ignored.");
 324             return;
 325         } else if (metadata.IHDR_colorType == PNG_COLOR_GRAY ||
 326                    metadata.IHDR_colorType == PNG_COLOR_GRAY_ALPHA) {
 327             processWarningOccurred(
 328 "A PNG gray or gray alpha image cannot have a PLTE chunk.\n" +
 329 "The chunk wil be ignored.");
 330             return;
 331         }
 332 
 333         byte[] palette = new byte[chunkLength];
 334         stream.readFully(palette);
 335 
 336         int numEntries = chunkLength/3;
 337         if (metadata.IHDR_colorType == PNG_COLOR_PALETTE) {
 338             int maxEntries = 1 << metadata.IHDR_bitDepth;
 339             if (numEntries > maxEntries) {
 340                 processWarningOccurred(
 341 "PLTE chunk contains too many entries for bit depth, ignoring extras.");
 342                 numEntries = maxEntries;
 343             }
 344             numEntries = Math.min(numEntries, maxEntries);
 345         }
 346 
 347         // Round array sizes up to 2^2^n
 348         int paletteEntries;
 349         if (numEntries > 16) {
 350             paletteEntries = 256;
 351         } else if (numEntries > 4) {
 352             paletteEntries = 16;
 353         } else if (numEntries > 2) {
 354             paletteEntries = 4;
 355         } else {
 356             paletteEntries = 2;
 357         }
 358 
 359         metadata.PLTE_present = true;
 360         metadata.PLTE_red = new byte[paletteEntries];
 361         metadata.PLTE_green = new byte[paletteEntries];
 362         metadata.PLTE_blue = new byte[paletteEntries];
 363 
 364         int index = 0;
 365         for (int i = 0; i < numEntries; i++) {
 366             metadata.PLTE_red[i] = palette[index++];
 367             metadata.PLTE_green[i] = palette[index++];
 368             metadata.PLTE_blue[i] = palette[index++];
 369         }
 370     }

 571     private void parse_tRNS_chunk(int chunkLength) throws IOException {
 572         int colorType = metadata.IHDR_colorType;
 573         if (colorType == PNG_COLOR_PALETTE) {
 574             if (!metadata.PLTE_present) {
 575                 processWarningOccurred(
 576 "tRNS chunk without prior PLTE chunk, ignoring it.");
 577                 return;
 578             }
 579 
 580             // Alpha table may have fewer entries than RGB palette
 581             int maxEntries = metadata.PLTE_red.length;
 582             int numEntries = chunkLength;
 583             if (numEntries > maxEntries) {
 584                 processWarningOccurred(
 585 "tRNS chunk has more entries than prior PLTE chunk, ignoring extras.");
 586                 numEntries = maxEntries;
 587             }
 588             metadata.tRNS_alpha = new byte[numEntries];
 589             metadata.tRNS_colorType = PNG_COLOR_PALETTE;
 590             stream.read(metadata.tRNS_alpha, 0, numEntries);
 591             stream.skipBytes(chunkLength - numEntries);
 592         } else if (colorType == PNG_COLOR_GRAY) {
 593             if (chunkLength != 2) {
 594                 processWarningOccurred(
 595 "tRNS chunk for gray image must have length 2, ignoring chunk.");
 596                 stream.skipBytes(chunkLength);
 597                 return;
 598             }
 599             metadata.tRNS_gray = stream.readUnsignedShort();
 600             metadata.tRNS_colorType = PNG_COLOR_GRAY;
 601         } else if (colorType == PNG_COLOR_RGB) {
 602             if (chunkLength != 6) {
 603                 processWarningOccurred(
 604 "tRNS chunk for RGB image must have length 6, ignoring chunk.");
 605                 stream.skipBytes(chunkLength);
 606                 return;
 607             }
 608             metadata.tRNS_red = stream.readUnsignedShort();
 609             metadata.tRNS_green = stream.readUnsignedShort();
 610             metadata.tRNS_blue = stream.readUnsignedShort();
 611             metadata.tRNS_colorType = PNG_COLOR_RGB;
 612         } else {
 613             processWarningOccurred(
 614 "Gray+Alpha and RGBS images may not have a tRNS chunk, ignoring it.");
 615             return;
 616         }
 617

 626         int c;

 628             while ((c = iis.read()) != -1) {
 629                 baos.write(c);
 630             }
 631         } finally {
 632             iis.close();
 633         }
 634         return baos.toByteArray();
 635     }

 637     private void parse_zTXt_chunk(int chunkLength) throws IOException {
 638         String keyword = readNullTerminatedString("ISO-8859-1", 80);
 639         metadata.zTXt_keyword.add(keyword);
 640 
 641         int method = stream.readUnsignedByte();
 642         metadata.zTXt_compressionMethod.add(new Integer(method));
 643 
 644         byte[] b = new byte[chunkLength - keyword.length() - 2];
 645         stream.readFully(b);
 646         metadata.zTXt_text.add(new String(inflate(b), "ISO-8859-1"));

 649     private void readMetadata() throws IIOException {
 650         if (gotMetadata) {
 651             return;
 652         }
 653 
 654         readHeader();
 655 
 656         /*
 657          * Optimization: We can skip the remaining metadata if the
 658          * ignoreMetadata flag is set, and only if this is not a palette
 659          * image (in that case, we need to read the metadata to get the
 660          * tRNS chunk, which is needed for the getImageTypes() method).
 661          */
 662         int colorType = metadata.IHDR_colorType;
 663         if (ignoreMetadata && colorType != PNG_COLOR_PALETTE) {
 664             try {
 665                 while (true) {
 666                     int chunkLength = stream.readInt();
 667                     int chunkType = stream.readInt();
 668 
 669                     if (chunkType == IDAT_TYPE) {
 670                         // We've reached the image data
 671                         stream.skipBytes(-8);
 672                         imageStartPosition = stream.getStreamPosition();
 673                         break;
 674                     } else {
 675                         // Skip the chunk plus the 4 CRC bytes that follow
 676                         stream.skipBytes(chunkLength + 4);
 677                     }
 678                 }
 679             } catch (IOException e) {
 680                 throw new IIOException("Error skipping PNG metadata", e);
 681             }
 682 
 683             gotMetadata = true;
 684             return;
 685         }
 686 
 687         try {
 688             loop: while (true) {
 689                 int chunkLength = stream.readInt();
 690                 int chunkType = stream.readInt();
 691                 int chunkCRC;
 692 
 693                 // verify the chunk length
 694                 if (chunkLength < 0) {
 695                     throw new IIOException("Invalid chunk lenght " + chunkLength);
 696                 };
 697 
 698                 try {
 699                     stream.mark();
 700                     stream.seek(stream.getStreamPosition() + chunkLength);
 701                     chunkCRC = stream.readInt();
 702                     stream.reset();
 703                 } catch (IOException e) {
 704                     throw new IIOException("Invalid chunk length " + chunkLength);
 705                 }
 706 
 707                 switch (chunkType) {
 708                 case IDAT_TYPE:
 709                     // If chunk type is 'IDAT', we've reached the image data.
 710                     stream.skipBytes(-8);
 711                     imageStartPosition = stream.getStreamPosition();
 712                     break loop;
 713                 case PLTE_TYPE:
 714                     parse_PLTE_chunk(chunkLength);
 715                     break;
 716                 case bKGD_TYPE:
 717                     parse_bKGD_chunk();
 718                     break;
 719                 case cHRM_TYPE:
 720                     parse_cHRM_chunk();
 721                     break;
 722                 case gAMA_TYPE:
 723                     parse_gAMA_chunk();
 724                     break;
 725                 case hIST_TYPE:
 726                     parse_hIST_chunk(chunkLength);
 727                     break;
 728                 case iCCP_TYPE:
 729                     parse_iCCP_chunk(chunkLength);
 730                     break;
 731                 case iTXt_TYPE:
 732                     parse_iTXt_chunk(chunkLength);
 733                     break;
 734                 case pHYs_TYPE:
 735                     parse_pHYs_chunk();
 736                     break;
 737                 case sBIT_TYPE:
 738                     parse_sBIT_chunk();
 739                     break;
 740                 case sPLT_TYPE:
 741                     parse_sPLT_chunk(chunkLength);
 742                     break;
 743                 case sRGB_TYPE:
 744                     parse_sRGB_chunk();
 745                     break;
 746                 case tEXt_TYPE:
 747                     parse_tEXt_chunk(chunkLength);
 748                     break;
 749                 case tIME_TYPE:
 750                     parse_tIME_chunk();
 751                     break;
 752                 case tRNS_TYPE:
 753                     parse_tRNS_chunk(chunkLength);
 754                     break;
 755                 case zTXt_TYPE:
 756                     parse_zTXt_chunk(chunkLength);
 757                     break;
 758                 default:
 759                     // Read an unknown chunk
 760                     byte[] b = new byte[chunkLength];
 761                     stream.readFully(b);
 762 
 763                     StringBuilder chunkName = new StringBuilder(4);
 764                     chunkName.append((char)(chunkType >>> 24));
 765                     chunkName.append((char)((chunkType >> 16) & 0xff));
 766                     chunkName.append((char)((chunkType >> 8) & 0xff));
 767                     chunkName.append((char)(chunkType & 0xff));
 768 
 769                     int ancillaryBit = chunkType >>> 28;
 770                     if (ancillaryBit == 0) {
 771                         processWarningOccurred(
 772 "Encountered unknown chunk with critical bit set!");
 773                     }
 774 
 775                     metadata.unknownChunkType.add(chunkName.toString());
 776                     metadata.unknownChunkData.add(b);
 777                     break;
 778                 }
 779 
 780                 // double check whether all chunk data were consumed
 781                 if (chunkCRC != stream.readInt()) {
 782                     throw new IIOException("Failed to read a chunk of type " +
 783                             chunkType);
 784                 }
 785                 stream.flushBefore(stream.getStreamPosition());
 786             }
 787         } catch (IOException e) {
 788             throw new IIOException("Error reading PNG metadata", e);
 789         }
 790 
 791         gotMetadata = true;
 792     }

 796     private static void decodeSubFilter(byte[] curr, int coff, int count,
 797                                         int bpp) {
 798         for (int i = bpp; i < count; i++) {
 799             int val;
 800 
 801             val = curr[i + coff] & 0xff;
 802             val += curr[i + coff - bpp] & 0xff;
 803 
 804             curr[i + coff] = (byte)val;
 805         }
 806     }

 808     private static void decodeUpFilter(byte[] curr, int coff,
 809                                        byte[] prev, int poff,
 810                                        int count) {
 811         for (int i = 0; i < count; i++) {
 812             int raw = curr[i + coff] & 0xff;
 813             int prior = prev[i + poff] & 0xff;
 814 
 815             curr[i + coff] = (byte)(raw + prior);
 816         }
 817     }

 819     private static void decodeAverageFilter(byte[] curr, int coff,
 820                                             byte[] prev, int poff,
 821                                             int count, int bpp) {
 822         int raw, priorPixel, priorRow;
 823 
 824         for (int i = 0; i < bpp; i++) {
 825             raw = curr[i + coff] & 0xff;
 826             priorRow = prev[i + poff] & 0xff;
 827 
 828             curr[i + coff] = (byte)(raw + priorRow/2);
 829         }
 830 
 831         for (int i = bpp; i < count; i++) {
 832             raw = curr[i + coff] & 0xff;
 833             priorPixel = curr[i + coff - bpp] & 0xff;
 834             priorRow = prev[i + poff] & 0xff;
 835 
 836             curr[i + coff] = (byte)(raw + (priorPixel + priorRow)/2);
 837         }
 838     }

 840     private static int paethPredictor(int a, int b, int c) {
 841         int p = a + b - c;
 842         int pa = Math.abs(p - a);
 843         int pb = Math.abs(p - b);
 844         int pc = Math.abs(p - c);
 845 
 846         if ((pa <= pb) && (pa <= pc)) {
 847             return a;
 848         } else if (pb <= pc) {
 849             return b;
 850         } else {
 851             return c;
 852         }
 853     }

 855     private static void decodePaethFilter(byte[] curr, int coff,
 856                                           byte[] prev, int poff,
 857                                           int count, int bpp) {
 858         int raw, priorPixel, priorRow, priorRowPixel;
 859 
 860         for (int i = 0; i < bpp; i++) {
 861             raw = curr[i + coff] & 0xff;
 862             priorRow = prev[i + poff] & 0xff;
 863 
 864             curr[i + coff] = (byte)(raw + priorRow);
 865         }
 866 
 867         for (int i = bpp; i < count; i++) {
 868             raw = curr[i + coff] & 0xff;
 869             priorPixel = curr[i + coff - bpp] & 0xff;
 870             priorRow = prev[i + poff] & 0xff;
 871             priorRowPixel = prev[i + poff - bpp] & 0xff;
 872 
 873             curr[i + coff] = (byte)(raw + paethPredictor(priorPixel,
 874                                                          priorRow,
 875                                                          priorRowPixel));
 876         }
 877     }

 922         throws IOException, IIOException  {

 924             return;
 925         }

 927         int inputBands = inputBandsForColorType[metadata.IHDR_colorType];

 934

 936             // processReadAborted will be called later
 937             if (abortRequested()) {
 938                 return;

 948     private void decodePass(int passNum,
 949                             int xStart, int yStart,
 950                             int xStep, int yStep,
 951                             int passWidth, int passHeight) throws IOException {
 952 
 953         if ((passWidth == 0) || (passHeight == 0)) {
 954             return;
 955         }
 956 
 957         WritableRaster imRas = theImage.getWritableTile(0, 0);
 958         int dstMinX = imRas.getMinX();
 959         int dstMaxX = dstMinX + imRas.getWidth() - 1;
 960         int dstMinY = imRas.getMinY();
 961         int dstMaxY = dstMinY + imRas.getHeight() - 1;
 962 
 963         // Determine which pixels will be updated in this pass
 964         int[] vals =
 965           ReaderUtil.computeUpdatedPixels(sourceRegion,
 966                                           destinationOffset,
 967                                           dstMinX, dstMinY,
 968                                           dstMaxX, dstMaxY,
 969                                           sourceXSubsampling,
 970                                           sourceYSubsampling,
 971                                           xStart, yStart,
 972                                           passWidth, passHeight,
 973                                           xStep, yStep);
 974         int updateMinX = vals[0];
 975         int updateMinY = vals[1];
 976         int updateWidth = vals[2];
 977         int updateXStep = vals[4];
 978         int updateYStep = vals[5];
 979 
 980         int bitDepth = metadata.IHDR_bitDepth;
 981         int inputBands = inputBandsForColorType[metadata.IHDR_colorType];
 982         int bytesPerPixel = (bitDepth == 16) ? 2 : 1;
 983         bytesPerPixel *= inputBands;
 984 
 985         int bytesPerRow = (inputBands*passWidth*bitDepth + 7)/8;
 986         int eltsPerRow = (bitDepth == 16) ? bytesPerRow/2 : bytesPerRow;
 987 
 988         // If no pixels need updating, just skip the input data
 989         if (updateWidth == 0) {
 990             for (int srcY = 0; srcY < passHeight; srcY++) {
 991                 // Update count of pixels read
 992                 updateImageProgress(passWidth);
 993                 // Skip filter byte and the remaining row bytes
 994                 pixelStream.skipBytes(1 + bytesPerRow);
 995             }
 996             return;
 997         }
 998 
 999         // Backwards map from destination pixels
1000         // (dstX = updateMinX + k*updateXStep)
1001         // to source pixels (sourceX), and then
1002         // to offset and skip in passRow (srcX and srcXStep)
1003         int sourceX =
1004             (updateMinX - destinationOffset.x)*sourceXSubsampling +
1005             sourceRegion.x;
1006         int srcX = (sourceX - xStart)/xStep;
1007 
1008         // Compute the step factor in the source
1009         int srcXStep = updateXStep*sourceXSubsampling/xStep;
1010 
1011         byte[] byteData = null;
1012         short[] shortData = null;
1013         byte[] curr = new byte[bytesPerRow];
1014         byte[] prior = new byte[bytesPerRow];
1015 
1016         // Create a 1-row tall Raster to hold the data
1017         WritableRaster passRow = createRaster(passWidth, 1, inputBands,
1018                                               eltsPerRow,
1019                                               bitDepth);
1020 
1021         // Create an array suitable for holding one pixel
1022         int[] ps = passRow.getPixel(0, 0, (int[])null);
1023 
1024         DataBuffer dataBuffer = passRow.getDataBuffer();
1025         int type = dataBuffer.getDataType();
1026         if (type == DataBuffer.TYPE_BYTE) {
1027             byteData = ((DataBufferByte)dataBuffer).getData();
1028         } else {
1029             shortData = ((DataBufferUShort)dataBuffer).getData();
1030         }
1031 
1032         processPassStarted(theImage,
1033                            passNum,
1034                            sourceMinProgressivePass,
1035                            sourceMaxProgressivePass,
1036                            updateMinX, updateMinY,
1037                            updateXStep, updateYStep,
1038                            destinationBands);
1039 
1040         // Handle source and destination bands
1041         if (sourceBands != null) {
1042             passRow = passRow.createWritableChild(0, 0,
1043                                                   passRow.getWidth(), 1,
1044                                                   0, 0,
1045                                                   sourceBands);
1046         }
1047         if (destinationBands != null) {
1048             imRas = imRas.createWritableChild(0, 0,
1049                                               imRas.getWidth(),
1050                                               imRas.getHeight(),
1051                                               0, 0,
1052                                               destinationBands);
1053         }
1054 
1055         // Determine if all of the relevant output bands have the
1056         // same bit depth as the source data
1057         boolean adjustBitDepths = false;
1058         int[] outputSampleSize = imRas.getSampleModel().getSampleSize();
1059         int numBands = outputSampleSize.length;
1060         for (int b = 0; b < numBands; b++) {
1061             if (outputSampleSize[b] != bitDepth) {
1062                 adjustBitDepths = true;
1063                 break;
1064             }
1065         }
1066 
1067         // If the bit depths differ, create a lookup table per band to perform
1068         // the conversion
1069         int[][] scale = null;
1070         if (adjustBitDepths) {
1071             int maxInSample = (1 << bitDepth) - 1;
1072             int halfMaxInSample = maxInSample/2;
1073             scale = new int[numBands][];
1074             for (int b = 0; b < numBands; b++) {
1075                 int maxOutSample = (1 << outputSampleSize[b]) - 1;
1076                 scale[b] = new int[maxInSample + 1];
1077                 for (int s = 0; s <= maxInSample; s++) {
1078                     scale[b][s] =
1079                         (s*maxOutSample + halfMaxInSample)/maxInSample;
1080                 }
1081             }
1082         }
1083 
1084         // Limit passRow to relevant area for the case where we
1085         // will can setRect to copy a contiguous span
1086         boolean useSetRect = srcXStep == 1 &&
1087             updateXStep == 1 &&
1088             !adjustBitDepths &&
1089             (imRas instanceof ByteInterleavedRaster);
1090 
1091         if (useSetRect) {
1092             passRow = passRow.createWritableChild(srcX, 0,
1093                                                   updateWidth, 1,
1094                                                   0, 0,
1095                                                   null);
1096         }
1097 
1098         // Decode the (sub)image row-by-row
1099         for (int srcY = 0; srcY < passHeight; srcY++) {
1100             // Update count of pixels read
1101             updateImageProgress(passWidth);
1102 
1103             // Read the filter type byte and a row of data
1104             int filter = pixelStream.read();
1105             try {
1106                 // Swap curr and prior
1107                 byte[] tmp = prior;
1108                 prior = curr;
1109                 curr = tmp;
1110 
1111                 pixelStream.readFully(curr, 0, bytesPerRow);
1112             } catch (java.util.zip.ZipException ze) {
1113                 // TODO - throw a more meaningful exception
1114                 throw ze;
1115             }
1116 
1117             switch (filter) {
1118             case PNG_FILTER_NONE:
1119                 break;
1120             case PNG_FILTER_SUB:
1121                 decodeSubFilter(curr, 0, bytesPerRow, bytesPerPixel);
1122                 break;
1123             case PNG_FILTER_UP:
1124                 decodeUpFilter(curr, 0, prior, 0, bytesPerRow);
1125                 break;
1126             case PNG_FILTER_AVERAGE:
1127                 decodeAverageFilter(curr, 0, prior, 0, bytesPerRow,
1128                                     bytesPerPixel);
1129                 break;
1130             case PNG_FILTER_PAETH:
1131                 decodePaethFilter(curr, 0, prior, 0, bytesPerRow,
1132                                   bytesPerPixel);
1133                 break;
1134             default:
1135                 throw new IIOException("Unknown row filter type (= " +
1136                                        filter + ")!");
1137             }
1138 
1139             // Copy data into passRow byte by byte
1140             if (bitDepth < 16) {
1141                 System.arraycopy(curr, 0, byteData, 0, bytesPerRow);
1142             } else {
1143                 int idx = 0;
1144                 for (int j = 0; j < eltsPerRow; j++) {
1145                     shortData[j] =
1146                         (short)((curr[idx] << 8) | (curr[idx + 1] & 0xff));
1147                     idx += 2;
1148                 }
1149             }
1150 
1151             // True Y position in source
1152             int sourceY = srcY*yStep + yStart;
1153             if ((sourceY >= sourceRegion.y) &&
1154                 (sourceY < sourceRegion.y + sourceRegion.height) &&
1155                 (((sourceY - sourceRegion.y) %
1156                   sourceYSubsampling) == 0)) {
1157 
1158                 int dstY = destinationOffset.y +
1159                     (sourceY - sourceRegion.y)/sourceYSubsampling;
1160                 if (dstY < dstMinY) {
1161                     continue;
1162                 }
1163                 if (dstY > dstMaxY) {
1164                     break;
1165                 }
1166 
1167                 if (useSetRect) {
1168                     imRas.setRect(updateMinX, dstY, passRow);
1169                 } else {
1170                     int newSrcX = srcX;
1171 
1172                     for (int dstX = updateMinX;
1173                          dstX < updateMinX + updateWidth;
1174                          dstX += updateXStep) {
1175 
1176                         passRow.getPixel(newSrcX, 0, ps);
1177                         if (adjustBitDepths) {
1178                             for (int b = 0; b < numBands; b++) {
1179                                 ps[b] = scale[b][ps[b]];
1180                             }
1181                         }
1182                         imRas.setPixel(dstX, dstY, ps);
1183                         newSrcX += srcXStep;
1184                     }
1185                 }
1186 
1187                 processImageUpdate(theImage,
1188                                    updateMinX, dstY,
1189                                    updateWidth, 1,
1190                                    updateXStep, updateYStep,
1191                                    destinationBands);
1192 
1193                 // If read has been aborted, just return
1194                 // processReadAborted will be called later
1195                 if (abortRequested()) {
1196                     return;
1197                 }
1198             }
1199         }
1200 
1201         processPassComplete(theImage);
1202     }

1205         throws IOException, IIOException  {
1206         int width = metadata.IHDR_width;
1207         int height = metadata.IHDR_height;
1208 
1209         this.pixelsDone = 0;
1210         this.totalPixels = width*height;
1211 
1212         clearAbortRequest();
1213 
1214         if (metadata.IHDR_interlaceMethod == 0) {
1215             decodePass(0, 0, 0, 1, 1, width, height);
1216         } else {
1217             for (int i = 0; i <= sourceMaxProgressivePass; i++) {
1218                 int XOffset = adam7XOffset[i];
1219                 int YOffset = adam7YOffset[i];
1220                 int XSubsampling = adam7XSubsampling[i];
1221                 int YSubsampling = adam7YSubsampling[i];
1222                 int xbump = adam7XSubsampling[i + 1] - 1;
1223                 int ybump = adam7YSubsampling[i + 1] - 1;

1226                     decodePass(i,
1227                                XOffset,
1228                                YOffset,
1229                                XSubsampling,
1230                                YSubsampling,
1231                                (width + xbump)/XSubsampling,
1232                                (height + ybump)/YSubsampling);
1233                 } else {
1234                     skipPass((width + xbump)/XSubsampling,
1235                              (height + ybump)/YSubsampling);
1236                 }
1237 
1238                 // If read has been aborted, just return
1239                 // processReadAborted will be called later
1240                 if (abortRequested()) {
1241                     return;
1242                 }
1243             }
1244         }

1247     private void readImage(ImageReadParam param) throws IIOException {
1248         readMetadata();
1249 
1250         int width = metadata.IHDR_width;
1251         int height = metadata.IHDR_height;
1252 
1253         // Init default values
1254         sourceXSubsampling = 1;
1255         sourceYSubsampling = 1;
1256         sourceMinProgressivePass = 0;
1257         sourceMaxProgressivePass = 6;
1258         sourceBands = null;
1259         destinationBands = null;
1260         destinationOffset = new Point(0, 0);
1261 
1262         // If an ImageReadParam is available, get values from it
1263         if (param != null) {
1264             sourceXSubsampling = param.getSourceXSubsampling();
1265             sourceYSubsampling = param.getSourceYSubsampling();
1266 
1267             sourceMinProgressivePass =
1268                 Math.max(param.getSourceMinProgressivePass(), 0);
1269             sourceMaxProgressivePass =
1270                 Math.min(param.getSourceMaxProgressivePass(), 6);
1271 
1272             sourceBands = param.getSourceBands();
1273             destinationBands = param.getDestinationBands();
1274             destinationOffset = param.getDestinationOffset();
1275         }
1276         Inflater inf = null;
1277         try {
1278             stream.seek(imageStartPosition);
1279 
1280             Enumeration<InputStream> e = new PNGImageDataEnumeration(stream);
1281             InputStream is = new SequenceInputStream(e);
1282 
1283            /* InflaterInputStream uses an Inflater instance which consumes
1284             * native (non-GC visible) resources. This is normally implicitly
1285             * freed when the stream is closed. However since the
1286             * InflaterInputStream wraps a client-supplied input stream,
1287             * we cannot close it.
1288             * But the app may depend on GC finalization to close the stream.
1289             * Therefore to ensure timely freeing of native resources we
1290             * explicitly create the Inflater instance and free its resources
1291             * when we are done with the InflaterInputStream by calling
1292             * inf.end();
1293             */
1294             inf = new Inflater();
1295             is = new InflaterInputStream(is, inf);
1296             is = new BufferedInputStream(is);
1297             this.pixelStream = new DataInputStream(is);
1298 
1299             /*
1300              * NB: the PNG spec declares that valid range for width
1301              * and height is [1, 2^31-1], so here we may fail to allocate
1302              * a buffer for destination image due to memory limitation.
1303              *
1304              * However, the recovery strategy for this case should be
1305              * defined on the level of application, so we will not
1306              * try to estimate the required amount of the memory and/or
1307              * handle OOM in any way.
1308              */
1309             theImage = getDestination(param,
1310                                       getImageTypes(0),
1311                                       width,
1312                                       height);
1313 
1314             Rectangle destRegion = new Rectangle(0, 0, 0, 0);
1315             sourceRegion = new Rectangle(0, 0, 0, 0);
1316             computeRegions(param, width, height,
1317                            theImage,
1318                            sourceRegion, destRegion);
1319             destinationOffset.setLocation(destRegion.getLocation());
1320 
1321             // At this point the header has been read and we know
1322             // how many bands are in the image, so perform checking
1323             // of the read param.
1324             int colorType = metadata.IHDR_colorType;
1325             checkReadParamBandSettings(param,
1326                                        inputBandsForColorType[colorType],
1327                                       theImage.getSampleModel().getNumBands());
1328 
1329             processImageStarted(0);
1330             decodeImage();
1331             if (abortRequested()) {
1332                 processReadAborted();
1333             } else {
1334                 processImageComplete();
1335             }
1336         } catch (IOException e) {
1337             throw new IIOException("Error reading PNG image data", e);
1338         } finally {
1339             if (inf != null) {
1340                 inf.end();
1341             }
1342         }

1376     public Iterator<ImageTypeSpecifier> getImageTypes(int imageIndex)
1377       throws IIOException
1378     {
1379         if (imageIndex != 0) {
1380             throw new IndexOutOfBoundsException("imageIndex != 0!");
1381         }
1382 
1383         readHeader();
1384 
1385         ArrayList<ImageTypeSpecifier> l =
1386             new ArrayList<ImageTypeSpecifier>(1);
1387 
1388         ColorSpace rgb;
1389         ColorSpace gray;
1390         int[] bandOffsets;
1391 
1392         int bitDepth = metadata.IHDR_bitDepth;
1393         int colorType = metadata.IHDR_colorType;
1394 
1395         int dataType;
1396         if (bitDepth <= 8) {
1397             dataType = DataBuffer.TYPE_BYTE;
1398         } else {
1399             dataType = DataBuffer.TYPE_USHORT;
1400         }
1401 
1402         switch (colorType) {
1403         case PNG_COLOR_GRAY:
1404             // Packed grayscale
1405             l.add(ImageTypeSpecifier.createGrayscale(bitDepth,
1406                                                      dataType,
1407                                                      false));
1408             break;
1409 
1410         case PNG_COLOR_RGB:
1411             if (bitDepth == 8) {
1412                 // some standard types of buffered images
1413                 // which can be used as destination
1414                 l.add(ImageTypeSpecifier.createFromBufferedImageType(
1415                           BufferedImage.TYPE_3BYTE_BGR));
1416 
1417                 l.add(ImageTypeSpecifier.createFromBufferedImageType(
1418                           BufferedImage.TYPE_INT_RGB));
1419 
1420                 l.add(ImageTypeSpecifier.createFromBufferedImageType(
1421                           BufferedImage.TYPE_INT_BGR));
1422 
1423             }
1424             // Component R, G, B
1425             rgb = ColorSpace.getInstance(ColorSpace.CS_sRGB);
1426             bandOffsets = new int[3];
1427             bandOffsets[0] = 0;
1428             bandOffsets[1] = 1;
1429             bandOffsets[2] = 2;
1430             l.add(ImageTypeSpecifier.createInterleaved(rgb,
1431                                                        bandOffsets,
1432                                                        dataType,
1433                                                        false,
1434                                                        false));
1435             break;
1436 
1437         case PNG_COLOR_PALETTE:
1438             readMetadata(); // Need tRNS chunk
1439 
1440             /*
1441              * The PLTE chunk spec says:
1442              *
1443              * The number of palette entries must not exceed the range that
1444              * can be represented in the image bit depth (for example, 2^4 = 16
1445              * for a bit depth of 4). It is permissible to have fewer entries
1446              * than the bit depth would allow. In that case, any out-of-range
1447              * pixel value found in the image data is an error.
1448              *
1449              * http://www.libpng.org/pub/png/spec/1.2/PNG-Chunks.html#C.PLTE
1450              *
1451              * Consequently, the case when the palette length is smaller than
1452              * 2^bitDepth is legal in the view of PNG spec.
1453              *
1454              * However the spec of createIndexed() method demands the exact
1455              * equality of the palette lengh and number of possible palette
1456              * entries (2^bitDepth).
1457              *
1458              * {@link javax.imageio.ImageTypeSpecifier.html#createIndexed}
1459              *
1460              * In order to avoid this contradiction we need to extend the
1461              * palette arrays to the limit defined by the bitDepth.
1462              */
1463 
1464             int plength = 1 << bitDepth;
1465 
1466             byte[] red = metadata.PLTE_red;
1467             byte[] green = metadata.PLTE_green;
1468             byte[] blue = metadata.PLTE_blue;
1469 
1470             if (metadata.PLTE_red.length < plength) {
1471                 red = Arrays.copyOf(metadata.PLTE_red, plength);
1472                 Arrays.fill(red, metadata.PLTE_red.length, plength,
1473                             metadata.PLTE_red[metadata.PLTE_red.length - 1]);
1474 
1475                 green = Arrays.copyOf(metadata.PLTE_green, plength);
1476                 Arrays.fill(green, metadata.PLTE_green.length, plength,
1477                             metadata.PLTE_green[metadata.PLTE_green.length - 1]);
1478 
1479                 blue = Arrays.copyOf(metadata.PLTE_blue, plength);
1480                 Arrays.fill(blue, metadata.PLTE_blue.length, plength,
1481                             metadata.PLTE_blue[metadata.PLTE_blue.length - 1]);
1482 
1483             }
1484 
1485             // Alpha from tRNS chunk may have fewer entries than
1486             // the RGB LUTs from the PLTE chunk; if so, pad with
1487             // 255.
1488             byte[] alpha = null;
1489             if (metadata.tRNS_present && (metadata.tRNS_alpha != null)) {
1490                 if (metadata.tRNS_alpha.length == red.length) {
1491                     alpha = metadata.tRNS_alpha;
1492                 } else {
1493                     alpha = Arrays.copyOf(metadata.tRNS_alpha, red.length);
1494                     Arrays.fill(alpha,
1495                                 metadata.tRNS_alpha.length,
1496                                 red.length, (byte)255);
1497                 }
1498             }
1499 
1500             l.add(ImageTypeSpecifier.createIndexed(red, green,
1501                                                    blue, alpha,
1502                                                    bitDepth,
1503                                                    DataBuffer.TYPE_BYTE));
1504             break;
1505 
1506         case PNG_COLOR_GRAY_ALPHA:
1507             // Component G, A
1508             gray = ColorSpace.getInstance(ColorSpace.CS_GRAY);
1509             bandOffsets = new int[2];
1510             bandOffsets[0] = 0;
1511             bandOffsets[1] = 1;
1512             l.add(ImageTypeSpecifier.createInterleaved(gray,
1513                                                        bandOffsets,
1514                                                        dataType,
1515                                                        true,
1516                                                        false));
1517             break;
1518 
1519         case PNG_COLOR_RGB_ALPHA:
1520             if (bitDepth == 8) {
1521                 // some standard types of buffered images
1522                 // wich can be used as destination
1523                 l.add(ImageTypeSpecifier.createFromBufferedImageType(
1524                           BufferedImage.TYPE_4BYTE_ABGR));
1525 
1526                 l.add(ImageTypeSpecifier.createFromBufferedImageType(
1527                           BufferedImage.TYPE_INT_ARGB));
1528             }
1529 
1530             // Component R, G, B, A (non-premultiplied)
1531             rgb = ColorSpace.getInstance(ColorSpace.CS_sRGB);
1532             bandOffsets = new int[4];
1533             bandOffsets[0] = 0;
1534             bandOffsets[1] = 1;
1535             bandOffsets[2] = 2;
1536             bandOffsets[3] = 3;
1537 
1538             l.add(ImageTypeSpecifier.createInterleaved(rgb,
1539                                                        bandOffsets,
1540                                                        dataType,
1541                                                        true,
1542                                                        false));
1543             break;
1544 
1545         default:
1546             break;
1547         }
1548 
1549         return l.iterator();
1550     }

1552     /*
1553      * Super class implementation uses first element
1554      * of image types list as raw image type.
1555      *
1556      * Also, super implementation uses first element of this list
1557      * as default destination type image read param does not specify
1558      * anything other.
1559      *
1560      * However, in case of RGB and RGBA color types, raw image type
1561      * produces buffered image of custom type. It causes some
1562      * performance degradation of subsequent rendering operations.
1563      *
1564      * To resolve this contradiction we put standard image types
1565      * at the first positions of image types list (to produce standard
1566      * images by default) and put raw image type (which is custom)
1567      * at the last position of this list.
1568      *
1569      * After this changes we should override getRawImageType()
1570      * to return last element of image types list.
1571      */
1572     public ImageTypeSpecifier getRawImageType(int imageIndex)
1573       throws IOException {
1574 
1575         Iterator<ImageTypeSpecifier> types = getImageTypes(imageIndex);
1576         ImageTypeSpecifier raw = null;
1577         do {
1578             raw = types.next();
1579         } while (types.hasNext());
1580         return raw;
1581     }
