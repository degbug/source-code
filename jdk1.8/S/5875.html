 213   /**
 214    * Figure out whether node2 should be considered as being later
 215    * in the document than node1, in Document Order as defined
 216    * by the XPath model. This may not agree with the ordering defined
 217    * by other XML applications.
 218    * <p>
 219    * There are some cases where ordering isn't defined, and neither are
 220    * the results of this function -- though we'll generally return true.
 221    * <p>
 222    * TODO: Make sure this does the right thing with attribute nodes!!!
 223    *
 224    * @param node1 DOM Node to perform position comparison on.
 225    * @param node2 DOM Node to perform position comparison on .
 226    *
 227    * @return false if node2 comes before node1, otherwise return true.
 228    * You can think of this as
 229    * <code>(node1.documentOrderPosition &lt;= node2.documentOrderPosition)</code>.
 230    */
 231   public static boolean isNodeAfter(Node node1, Node node2)
 232   {
 233 
 234     // Assume first that the nodes are DTM nodes, since discovering node
 235     // order is massivly faster for the DTM.
 236     if(node1 instanceof DOMOrder && node2 instanceof DOMOrder)
 237     {
 238       int index1 = ((DOMOrder) node1).getUid();
 239       int index2 = ((DOMOrder) node2).getUid();
 240 
 241       return index1 <= index2;
 242     }
 243     else
 244     {
 245 
 246       // isNodeAfter will return true if node is after countedNode
 247       // in document order. The base isNodeAfter is sloooow (relatively).
 248       return DOMHelper.isNodeAfter(node1, node2);
 249     }
 250   }

 252   /**
 253    * Get the XPath-model parent of a node.  This version takes advantage
 254    * of the DOM Level 2 Attr.ownerElement() method; the base version we
 255    * would otherwise inherit is prepared to fall back on exhaustively
 256    * walking the document to find an Attr's parent.
 257    *
 258    * @param node Node to be examined
 259    *
 260    * @return the DOM parent of the input node, if there is one, or the
 261    * ownerElement if the input node is an Attr, or null if the node is
 262    * a Document, a DocumentFragment, or an orphan.
 263    */
 264   public static Node getParentOfNode(Node node)
 265   {
 266           Node parent=node.getParentNode();
 267           if(parent==null && (Node.ATTRIBUTE_NODE == node.getNodeType()) )
 268            parent=((Attr) node).getOwnerElement();
 269           return parent;
 270   }
