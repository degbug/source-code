  97     private int getComponentState(JComponent c) {
  98         int state = SynthLookAndFeel.getComponentState(c);
  99         if (SynthLookAndFeel.getSelectedUI() == this &&
 100                         state == SynthConstants.ENABLED) {
 101             state = SynthLookAndFeel.getSelectedUIState() | SynthConstants.ENABLED;
 102         }
 103         return state;
 104     }

 106     /**
 107      * {@inheritDoc}
 108      */
 109     @Override
 110     public int getBaseline(JComponent c, int width, int height) {
 111         if (c == null) {
 112             throw new NullPointerException("Component must be non-null");
 113         }
 114         if (width < 0 || height < 0) {
 115             throw new IllegalArgumentException(
 116                     "Width and height must be >= 0");
 117         }
 118         JLabel label = (JLabel)c;
 119         String text = label.getText();
 120         if (text == null || "".equals(text)) {
 121             return -1;
 122         }
 123         Insets i = label.getInsets();
 124         Rectangle viewRect = new Rectangle();
 125         Rectangle textRect = new Rectangle();
 126         Rectangle iconRect = new Rectangle();
 127         viewRect.x = i.left;
 128         viewRect.y = i.top;
 129         viewRect.width = width - (i.right + viewRect.x);
 130         viewRect.height = height - (i.bottom + viewRect.y);
 131 
 132         // layout the text and icon
 133         SynthContext context = getContext(label);
 134         FontMetrics fm = context.getComponent().getFontMetrics(
 135             context.getStyle().getFont(context));
 136         context.getStyle().getGraphicsUtils(context).layoutText(
 137             context, fm, label.getText(), label.getIcon(),
 138             label.getHorizontalAlignment(), label.getVerticalAlignment(),
 139             label.getHorizontalTextPosition(), label.getVerticalTextPosition(),
 140             viewRect, iconRect, textRect, label.getIconTextGap());
 141         View view = (View)label.getClientProperty(BasicHTML.propertyKey);
 142         int baseline;
 143         if (view != null) {
 144             baseline = BasicHTML.getHTMLBaseline(view, textRect.width,
 145                                                  textRect.height);
 146             if (baseline >= 0) {
 147                 baseline += textRect.y;
 148             }
 149         }
 150         else {
 151             baseline = textRect.y + fm.getAscent();
 152         }
 153         context.dispose();
 154         return baseline;
 155     }

 157     /**
 158      * Notifies this UI delegate to repaint the specified component.
 159      * This method paints the component background, then calls
 160      * the {@link #paint(SynthContext,Graphics)} method.
 161      *
 162      * In general, this method does not need to be overridden by subclasses.
 163      * All Look and Feel rendering code should reside in the {@code paint} method.
 164      *
 165      * @param g the {@code Graphics} object used for painting
 166      * @param c the component being painted
 167      * @see #paint(SynthContext,Graphics)
 168      */
 169     @Override
 170     public void update(Graphics g, JComponent c) {
 171         SynthContext context = getContext(c);
 172 
 173         SynthLookAndFeel.update(context, g);
 174         context.getPainter().paintLabelBackground(context,
 175                           g, 0, 0, c.getWidth(), c.getHeight());
 176         paint(context, g);
 177         context.dispose();
 178     }

 197     /**
 198      * Paints the specified component.
 199      *
 200      * @param context context for the component being painted
 201      * @param g the {@code Graphics} object used for painting
 202      * @see #update(Graphics,JComponent)
 203      */
 204     protected void paint(SynthContext context, Graphics g) {
 205         JLabel label = (JLabel)context.getComponent();
 206         Icon icon = (label.isEnabled()) ? label.getIcon() :
 207                                           label.getDisabledIcon();
 208 
 209         g.setColor(context.getStyle().getColor(context,
 210                                                ColorType.TEXT_FOREGROUND));
 211         g.setFont(style.getFont(context));
 212         context.getStyle().getGraphicsUtils(context).paintText(
 213             context, g, label.getText(), icon,
 214             label.getHorizontalAlignment(), label.getVerticalAlignment(),
 215             label.getHorizontalTextPosition(), label.getVerticalTextPosition(),
 216             label.getIconTextGap(), label.getDisplayedMnemonicIndex(), 0);
 217     }
