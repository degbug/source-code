  76                 MetalUtils.drawFlush3DBorder(g, x, y, controlSize, controlSize);
  77             }
  78             g.setColor( MetalLookAndFeel.getControlInfo() );
  79         } else {
  80             g.setColor( MetalLookAndFeel.getControlShadow() );
  81             g.drawRect( x, y, controlSize-1, controlSize-1);
  82         }
  83 
  84 
  85         if(drawCheck) {
  86             if (cb.isBorderPaintedFlat()) {
