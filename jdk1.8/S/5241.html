  37 /**
  38  *
  39  * @author Joe Wang:
  40  * This is a rewrite of the original class. The focus is on removing caching, and make the filtered
  41  * stream reader more compatible with those in other implementations. Note however, that this version
  42  * will not solve all the issues related to the undefined condition in the spec. The priority is
  43  * to pass the TCK. Issues arising due to the requirement, that is, (1) should it initiate at BEGIN_DOCUMENT
  44  * or an accepted event; (2) should hasNext() advance the underlining stream in order to find an acceptable
  45  * event, would have to wait until 1.1 of StAX in which the filtered stream reader would be defined more clearly.
  46  */

  50     private StreamFilter fStreamFilter = null;
  51     private XMLStreamReader fStreamReader = null;
  52     private int fCurrentEvent;
  53     private boolean fEventAccepted = false;

  55     /**the very issue around a long discussion. but since we must pass the TCK, we have to allow
  56      * hasNext() to advance the underlining stream in order to find the next acceptable event
  57      */
  58     private boolean fStreamAdvancedByHasNext = false;

  61     public XMLStreamFilterImpl(XMLStreamReader reader,StreamFilter filter){
  62         fStreamReader = reader;
  63         this.fStreamFilter = filter;
  64 
  65         //this is debatable to initiate at an acceptable event,
  66         //but it's neccessary in order to pass the TCK and yet avoid skipping element
  67         try {
  68             if (fStreamFilter.accept(fStreamReader)) {
  69                 fEventAccepted = true;
  70             } else {
  71                 findNextEvent();
  72             }
  73         }catch(XMLStreamException xs){
  74             System.err.println("Error while creating a stream Filter"+xs);
  75         }
  76     }

  86     /**
  87      *
  88      * @return
  89      * @throws XMLStreamException
  90      */
  91     public int next() throws XMLStreamException {
  92         if (fStreamAdvancedByHasNext && fEventAccepted) {
  93             fStreamAdvancedByHasNext = false;
  94             return fCurrentEvent;
  95         }
  96         int event = findNextEvent();
  97         if (event != -1) {
  98             return event;
  99         }
 100 
 101         throw new IllegalStateException("The stream reader has reached the end of the document, or there are no more "+
 102                                     " items to return");
 103     }
 104     /**
 105      *
 106      * @throws XMLStreamException
 107      * @return
 108      */
 109     public int nextTag() throws XMLStreamException {
 110         if (fStreamAdvancedByHasNext && fEventAccepted &&
 111                 (fCurrentEvent == XMLEvent.START_ELEMENT || fCurrentEvent == XMLEvent.START_ELEMENT)) {
 112             fStreamAdvancedByHasNext = false;
 113             return fCurrentEvent;
 114         }
 115 
 116         int event = findNextTag();
 117         if (event != -1) {
 118             return event;
 119         }
 120         throw new IllegalStateException("The stream reader has reached the end of the document, or there are no more "+
 121                                     " items to return");
 122     }

 124     /**
 125      *
 126      * @throws XMLStreamException
 127      * @return
 128      */
 129     public boolean hasNext() throws XMLStreamException {
 130         if (fStreamReader.hasNext()) {
 131             if (!fEventAccepted) {
 132                 if ((fCurrentEvent = findNextEvent()) == -1) {
 133                     return false;
 134                 } else {
 135                     fStreamAdvancedByHasNext = true;
 136                 }
 137             }
 138             return true;
 139         }
 140         return false;
 141     }
 142 
 143     private int findNextEvent() throws XMLStreamException {
 144         fStreamAdvancedByHasNext = false;
 145         while(fStreamReader.hasNext()){
 146             fCurrentEvent = fStreamReader.next();
 147             if(fStreamFilter.accept(fStreamReader)){
 148                 fEventAccepted = true;
 149                 return fCurrentEvent;
 150             }
 151         }
 152         //although it seems that IllegalStateException should be thrown when next() is called
 153         //on a stream that has no more items, we have to assume END_DOCUMENT is always accepted
 154         //in order to pass the TCK
 155         if (fCurrentEvent == XMLEvent.END_DOCUMENT)
 156             return fCurrentEvent;
 157         else
 158             return -1;
 159     }
 160     private int findNextTag() throws XMLStreamException {
 161         fStreamAdvancedByHasNext = false;
 162         while(fStreamReader.hasNext()){
 163             fCurrentEvent = fStreamReader.nextTag();
 164             if(fStreamFilter.accept(fStreamReader)){
 165                 fEventAccepted = true;
 166                 return fCurrentEvent;
 167             }
 168         }
 169         if (fCurrentEvent == XMLEvent.END_DOCUMENT)
 170             return fCurrentEvent;
 171         else
 172             return -1;
 173     }
