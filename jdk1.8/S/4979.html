  30 /**
  31  * Base class for functions that accept one argument.
  32  * @xsl.usage advanced
  33  */
  34 public class FunctionOneArg extends Function implements ExpressionOwner
  35 {
  36     static final long serialVersionUID = -5180174180765609758L;
  37 
  38   /** The first argument passed to the function (at index 0).
  39    *  @serial  */
  40   Expression m_arg0;
  41 
  42   /**
  43    * Return the first argument passed to the function (at index 0).
  44    *
  45    * @return An expression that represents the first argument passed to the
  46    *         function.
  47    */
  48   public Expression getArg0()
  49   {
  50     return m_arg0;
  51   }
  52 
  53   /**
  54    * Set an argument expression for a function.  This method is called by the
  55    * XPath compiler.
  56    *
  57    * @param arg non-null expression that represents the argument.
  58    * @param argNum The argument number index.
  59    *
  60    * @throws WrongNumberArgsException If the argNum parameter is greater than 0.
  61    */
  62   public void setArg(Expression arg, int argNum)
  63           throws WrongNumberArgsException
  64   {
  65 
  66     if (0 == argNum)
  67     {
  68       m_arg0 = arg;
  69       arg.exprSetParent(this);
  70     }
  71     else
  72       reportWrongNumberArgs();
  73   }
  74 
  75   /**
  76    * Check that the number of arguments passed to this function is correct.
  77    *
  78    *
  79    * @param argNum The number of arguments that is being passed to the function.
  80    *
  81    * @throws WrongNumberArgsException
  82    */
  83   public void checkNumberArgs(int argNum) throws WrongNumberArgsException
  84   {
  85     if (argNum != 1)
  86       reportWrongNumberArgs();
  87   }
  88 
  89   /**
  90    * Constructs and throws a WrongNumberArgException with the appropriate
  91    * message for this function object.
  92    *
  93    * @throws WrongNumberArgsException
  94    */
  95   protected void reportWrongNumberArgs() throws WrongNumberArgsException {
  96       throw new WrongNumberArgsException(XSLMessages.createXPATHMessage("one", null));
  97   }
  98 
  99   /**
 100    * Tell if this expression or it's subexpressions can traverse outside
 101    * the current subtree.
 102    *
 103    * @return true if traversal outside the context node's subtree can occur.
 104    */
 105    public boolean canTraverseOutsideSubtree()

 107     return m_arg0.canTraverseOutsideSubtree();
 108    }
 109 
 110   /**
 111    * This function is used to fixup variables from QNames to stack frame
 112    * indexes at stylesheet build time.
 113    * @param vars List of QNames that correspond to variables.  This list
 114    * should be searched backwards for the first qualified name that

 120   public void fixupVariables(java.util.Vector vars, int globalsSize)

 122     if(null != m_arg0)

 124   }

 133   }

 153   /**
 154    * @see Expression#deepEquals(Expression)
 155    */
 156   public boolean deepEquals(Expression expr)
 157   {
 158         if(!super.deepEquals(expr))
 159                 return false;
 160 
 161         if(null != m_arg0)
 162         {
 163                 if(null == ((FunctionOneArg)expr).m_arg0)
 164                         return false;
 165 
 166                 if(!m_arg0.deepEquals(((FunctionOneArg)expr).m_arg0))
 167                         return false;
 168         }
 169         else if(null != ((FunctionOneArg)expr).m_arg0)
 170                 return false;
 171 
 172         return true;
 173   }
