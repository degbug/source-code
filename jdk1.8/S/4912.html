  23 package com.sun.org.apache.xpath.internal.operations;
  24 
  25 import com.sun.org.apache.xpath.internal.Expression;
  26 import com.sun.org.apache.xpath.internal.ExpressionOwner;
  27 import com.sun.org.apache.xpath.internal.XPathContext;
  28 import com.sun.org.apache.xpath.internal.XPathVisitor;
  29 import com.sun.org.apache.xpath.internal.objects.XObject;
  30 
  31 /**
  32  * The unary operation base class.
  33  */
  34 public abstract class UnaryOperation extends Expression implements ExpressionOwner
  35 {
  36     static final long serialVersionUID = 6536083808424286166L;
  37 
  38   /** The operand for the operation.
  39    *  @serial */
  40   protected Expression m_right;
  41 
  42   /**
  43    * This function is used to fixup variables from QNames to stack frame
  44    * indexes at stylesheet build time.
  45    * @param vars List of QNames that correspond to variables.  This list
  46    * should be searched backwards for the first qualified name that
  47    * corresponds to the variable reference qname.  The position of the
  48    * QName in the vector from the start of the vector will be its position
  49    * in the stack frame (but variables above the globalsTop value will need
  50    * to be offset to the current stack frame).
  51    */
  52   public void fixupVariables(java.util.Vector vars, int globalsSize)
  53   {
  54     m_right.fixupVariables(vars, globalsSize);
  55   }
  56 
  57   /**
  58    * Tell if this expression or it's subexpressions can traverse outside
  59    * the current subtree.
  60    *
  61    * @return true if traversal outside the context node's subtree can occur.
  62    */
  63   public boolean canTraverseOutsideSubtree()
  64   {
  65 
  66     if (null != m_right && m_right.canTraverseOutsideSubtree())
  67       return true;
  68 
  69     return false;
  70   }
  71 
  72   /**
  73    * Set the expression operand for the operation.
  74    *
  75    *
  76    * @param r The expression operand to which the unary operation will be
  77    *          applied.
  78    */
  79   public void setRight(Expression r)
  80   {
  81     m_right = r;
  82     r.exprSetParent(this);
  83   }
  84 
  85   /**
  86    * Execute the operand and apply the unary operation to the result.
  87    *
  88    *
  89    * @param xctxt The runtime execution context.
  90    *
  91    * @return An XObject that represents the result of applying the unary
  92    *         operation to the evaluated operand.
  93    *
  94    * @throws javax.xml.transform.TransformerException
  95    */
  96   public XObject execute(XPathContext xctxt) throws javax.xml.transform.TransformerException
  97   {
  98 
  99     return operate(m_right.execute(xctxt));
 100   }
 101 
 102   /**
 103    * Apply the operation to two operands, and return the result.
 104    *
 105    *
 106    * @param right non-null reference to the evaluated right operand.
 107    *
 108    * @return non-null reference to the XObject that represents the result of the operation.
 109    *
 110    * @throws javax.xml.transform.TransformerException
 111    */
 112   public abstract XObject operate(XObject right)
 113     throws javax.xml.transform.TransformerException;
 114 
 115   /** @return the operand of unary operation, as an Expression.
 116    */
 117   public Expression getOperand(){
 118     return m_right;
 119   }
 120 
 121   /**
 122    * @see com.sun.org.apache.xpath.internal.XPathVisitable#callVisitors(ExpressionOwner, XPathVisitor)
 123    */
 124   public void callVisitors(ExpressionOwner owner, XPathVisitor visitor)
 125   {
 126         if(visitor.visitUnaryOperation(owner, this))
 127         {
 128                 m_right.callVisitors(this, visitor);
 129         }
 130   }
 131 
 132 
 133   /**
 134    * @see ExpressionOwner#getExpression()
 135    */
 136   public Expression getExpression()
 137   {
 138     return m_right;
 139   }
 140 
 141   /**
 142    * @see ExpressionOwner#setExpression(Expression)
 143    */
 144   public void setExpression(Expression exp)
 145   {
 146         exp.exprSetParent(this);
 147         m_right = exp;
 148   }
 149 
 150   /**
 151    * @see Expression#deepEquals(Expression)
 152    */
 153   public boolean deepEquals(Expression expr)
 154   {
 155         if(!isSameClass(expr))
 156                 return false;
 157 
 158         if(!m_right.deepEquals(((UnaryOperation)expr).m_right))
 159                 return false;
 160 
 161         return true;
 162   }
