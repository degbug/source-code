  42 /**
  43  * Utility class to report simple information about the environment.
  44  * Simplistic reporting about certain classes found in your JVM may
  45  * help answer some FAQs for simple problems.
  46  *
  47  * <p>Usage-command line:
  48  * <code>
  49  * java com.sun.org.apache.xalan.internal.xslt.EnvironmentCheck [-out outFile]
  50  * </code></p>
  51  *
  52  * <p>Usage-from program:
  53  * <code>
  54  * boolean environmentOK =
  55  * (new EnvironmentCheck()).checkEnvironment(yourPrintWriter);
  56  * </code></p>
  57  *
  58  * <p>Usage-from stylesheet:
  59  * <code><pre>
  60  *    &lt;?xml version="1.0"?&gt;
  61  *    &lt;xsl:stylesheet xmlns:xsl="http://www.w3.org/1999/XSL/Transform" version="1.0"
  62  *        xmlns:xalan="http://xml.apache.org/xalan"
  63  *        exclude-result-prefixes="xalan"&gt;
  64  *    &lt;xsl:output indent="yes"/&gt;
  65  *    &lt;xsl:template match="/"&gt;
  66  *      &lt;xsl:copy-of select="xalan:checkEnvironment()"/&gt;
  67  *    &lt;/xsl:template&gt;
  68  *    &lt;/xsl:stylesheet&gt;
  69  * </pre></code></p>
  70  *
  71  * <p>Xalan users reporting problems are encouraged to use this class
  72  * to see if there are potential problems with their actual
  73  * Java environment <b>before</b> reporting a bug.  Note that you
  74  * should both check from the JVM/JRE's command line as well as
  75  * temporarily calling checkEnvironment() directly from your code,
  76  * since the classpath may differ (especially for servlets, etc).</p>
  77  *
  78  * <p>Also see http://xml.apache.org/xalan-j/faq.html</p>
  79  *
  80  * <p>Note: This class is pretty simplistic:
  81  * results are not necessarily definitive nor will it find all
  82  * problems related to environment setup.  Also, you should avoid
  83  * calling this in deployed production code, both because it is
  84  * quite slow and because it forces classes to get loaded.</p>
  85  *
  86  * <p>Note: This class explicitly has very limited compile-time
  87  * dependencies to enable easy compilation and usage even when
  88  * Xalan, DOM/SAX/JAXP, etc. are not present.</p>
  89  *
  90  * <p>Note: for an improved version of this utility, please see
  91  * the xml-commons' project Which utility which does the same kind
  92  * of thing but in a much simpler manner.</p>
  93  *
  94  * @author Shane_Curcuru@us.ibm.com
  95  * @version $Id: EnvironmentCheck.java,v 1.10 2010-11-01 04:34:13 joehw Exp $
  96  */
  97 public class EnvironmentCheck

 103    * {@link #checkEnvironment(PrintWriter)}.</p>
 104    * @param args command line args

 108     // Default to System.out, autoflushing

 111     // Read our simplistic input args, if supplied
 112     for (int i = 0; i < args.length; i++)
 113     {
 114       if ("-out".equalsIgnoreCase(args[i]))
 115       {
 116         i++;
 117

 123           }

 126             System.err.println("# WARNING: -out " + args[i] + " threw "

 128           }
 129         }

 132           System.err.println(
 133             "# WARNING: -out argument should have a filename, output sent to console");
 134         }
 135       }
 136     }
 137 
 138     EnvironmentCheck app = new EnvironmentCheck();
 139     app.checkEnvironment(sendOutputTo);
 140   }

 142   /**
 143    * Programmatic entrypoint: Report on basic Java environment
 144    * and CLASSPATH settings that affect Xalan.
 145    *
 146    * <p>Note that this class is not advanced enough to tell you
 147    * everything about the environment that affects Xalan, and
 148    * sometimes reports errors that will not actually affect
 149    * Xalan's behavior.  Currently, it very simplistically
 150    * checks the JVM's environment for some basic properties and
 151    * logs them out; it will report a problem if it finds a setting
 152    * or .jar file that is <i>likely</i> to cause problems.</p>
 153    *
 154    * <p>Advanced users can peruse the code herein to help them
 155    * investigate potential environment problems found; other users
 156    * may simply send the output from this tool along with any bugs
 157    * they submit to help us in the debugging process.</p>
 158    *
 159    * @param pw PrintWriter to send output to; can be sent to a
 160    * file that will look similar to a Properties file; defaults
 161    * to System.out if null
 162    * @return true if your environment appears to have no major
 163    * problems; false if potential environment problems found
 164    * @see #getEnvironmentHash()
 165    */
 166   public boolean checkEnvironment(PrintWriter pw)
 167   {
 168 
 169     // Use user-specified output writer if non-null
 170     if (null != pw)
 171       outWriter = pw;
 172 
 173     // Setup a hash to store various environment information in
 174     Hashtable hash = getEnvironmentHash();
 175 
 176     // Check for ERROR keys in the hashtable, and print report
 177     boolean environmentHasErrors = writeEnvironmentReport(hash);
 178 
 179     if (environmentHasErrors)
 180     {
 181       // Note: many logMsg calls have # at the start to
 182       //  fake a property-file like output
 183       logMsg("# WARNING: Potential problems found in your environment!");
 184       logMsg("#    Check any 'ERROR' items above against the Xalan FAQs");
 185       logMsg("#    to correct potential problems with your classes/jars");
 186       logMsg("#    http://xml.apache.org/xalan-j/faq.html");
 187       if (null != outWriter)
 188         outWriter.flush();
 189       return false;
 190     }
 191     else
 192     {
 193       logMsg("# YAHOO! Your environment seems to be OK.");
 194       if (null != outWriter)
 195         outWriter.flush();
 196       return true;
 197     }
 198   }

 200   /**
 201    * Fill a hash with basic environment settings that affect Xalan.
 202    *
 203    * <p>Worker method called from various places.</p>
 204    * <p>Various system and CLASSPATH, etc. properties are put into
 205    * the hash as keys with a brief description of the current state
 206    * of that item as the value.  Any serious problems will be put in
 207    * with a key that is prefixed with {@link #ERROR 'ERROR.'} so it
 208    * stands out in any resulting report; also a key with just that
 209    * constant will be set as well for any error.</p>
 210    * <p>Note that some legitimate cases are flaged as potential
 211    * errors - namely when a developer recompiles xalan.jar on their
 212    * own - and even a non-error state doesn't guaruntee that
 213    * everything in the environment is correct.  But this will help
 214    * point out the most common classpath and system property
 215    * problems that we've seen.</p>
 216    *
 217    * @return Hashtable full of useful environment info about Xalan
 218    * and related system properties, etc.
 219    */
 220   public Hashtable getEnvironmentHash()
 221   {
 222     // Setup a hash to store various environment information in
 223     Hashtable hash = new Hashtable();
 224 
 225     // Call various worker methods to fill in the hash
 226     //  These are explicitly separate for maintenance and so
 227     //  advanced users could call them standalone
 228     checkJAXPVersion(hash);
 229     checkProcessorVersion(hash);
 230     checkParserVersion(hash);
 231     checkAntVersion(hash);
 232     if (!checkDOML3(hash)) {
 233     checkDOMVersion(hash);
 234     }
 235     checkSAXVersion(hash);
 236     checkSystemProperties(hash);
 237 
 238     return hash;
 239   }

 241   /**
 242    * Dump a basic Xalan environment report to outWriter.
 243    *
 244    * <p>This dumps a simple header and then each of the entries in
 245    * the Hashtable to our PrintWriter; it does special processing
 246    * for entries that are .jars found in the classpath.</p>
 247    *
 248    * @param h Hashtable of items to report on; presumably
 249    * filled in by our various check*() methods
 250    * @return true if your environment appears to have no major
 251    * problems; false if potential environment problems found
 252    * @see #appendEnvironmentReport(Node, Document, Hashtable)
 253    * for an equivalent that appends to a Node instead
 254    */
 255   protected boolean writeEnvironmentReport(Hashtable h)
 256   {
 257 
 258     if (null == h)
 259     {
 260       logMsg("# ERROR: writeEnvironmentReport called with null Hashtable");
 261       return false;
 262     }
 263 
 264     boolean errors = false;
 265 
 266     logMsg(
 267       "#---- BEGIN writeEnvironmentReport($Revision: 1.10 $): Useful stuff found: ----");
 268 
 269     // Fake the Properties-like output
 270     for (Enumeration keys = h.keys();
 271          keys.hasMoreElements();
 272         /* no increment portion */
 273         )
 274     {
 275       Object key = keys.nextElement();
 276       String keyStr = (String) key;
 277       try
 278       {
 279         // Special processing for classes found..
 280         if (keyStr.startsWith(FOUNDCLASSES))
 281         {
 282           Vector v = (Vector) h.get(keyStr);
 283           errors |= logFoundJars(v, keyStr);
 284         }
 285         // ..normal processing for all other entries
 286         else
 287         {
 288           // Note: we could just check for the ERROR key by itself,
 289           //    since we now set that, but since we have to go
 290           //    through the whole hash anyway, do it this way,
 291           //    which is safer for maintenance
 292           if (keyStr.startsWith(ERROR))
 293           {
 294             errors = true;
 295           }
 296           logMsg(keyStr + "=" + h.get(keyStr));
 297         }
 298       }
 299       catch (Exception e)
 300       {
 301         logMsg("Reading-" + key + "= threw: " + e.toString());
 302       }
 303     }
 304 
 305     logMsg(
 306       "#----- END writeEnvironmentReport: Useful properties found: -----");
 307 
 308     return errors;
 309   }

 347   /**
 348    * Print out report of .jars found in a classpath.
 349    *
 350    * Takes the information encoded from a checkPathForJars()
 351    * call and dumps it out to our PrintWriter.
 352    *
 353    * @param v Vector of Hashtables of .jar file info
 354    * @param desc description to print out in header
 355    *
 356    * @return false if OK, true if any .jars were reported
 357    * as having errors
 358    * @see #checkPathForJars(String, String[])
 359    */
 360   protected boolean logFoundJars(Vector v, String desc)
 361   {
 362 
 363     if ((null == v) || (v.size() < 1))
 364       return false;
 365 
 366     boolean errors = false;
 367 
 368     logMsg("#---- BEGIN Listing XML-related jars in: " + desc + " ----");
 369 
 370     for (int i = 0; i < v.size(); i++)
 371     {
 372       Hashtable subhash = (Hashtable) v.elementAt(i);
 373 
 374       for (Enumeration keys = subhash.keys();
 375            keys.hasMoreElements();
 376            /* no increment portion */
 377           )
 378       {
 379         Object key = keys.nextElement();
 380         String keyStr = (String) key;
 381         try
 382         {
 383           if (keyStr.startsWith(ERROR))
 384           {
 385             errors = true;
 386           }
 387           logMsg(keyStr + "=" + subhash.get(keyStr));
 388 
 389         }
 390         catch (Exception e)
 391         {
 392           errors = true;
 393           logMsg("Reading-" + key + "= threw: " + e.toString());
 394         }
 395       }
 396     }
 397 
 398     logMsg("#----- END Listing XML-related jars in: " + desc + " -----");
 399 
 400     return errors;
 401   }

 403   /**
 404    * Stylesheet extension entrypoint: Dump a basic Xalan
 405    * environment report from getEnvironmentHash() to a Node.
 406    *
 407    * <p>Copy of writeEnvironmentReport that creates a Node suitable
 408    * for other processing instead of a properties-like text output.
 409    * </p>
 410    * @param container Node to append our report to
 411    * @param factory Document providing createElement, etc. services
 412    * @param h Hash presumably from {@link #getEnvironmentHash()}
 413    * @see #writeEnvironmentReport(Hashtable)
 414    * for an equivalent that writes to a PrintWriter instead
 415    */
 416   public void appendEnvironmentReport(Node container, Document factory, Hashtable h)
 417   {
 418     if ((null == container) || (null == factory))
 419     {
 420       return;
 421     }
 422 
 423     try
 424     {
 425       Element envCheckNode = factory.createElement("EnvironmentCheck");
 426       envCheckNode.setAttribute("version", "$Revision: 1.10 $");
 427       container.appendChild(envCheckNode);
 428 
 429       if (null == h)
 430       {
 431         Element statusNode = factory.createElement("status");
 432         statusNode.setAttribute("result", "ERROR");
 433         statusNode.appendChild(factory.createTextNode("appendEnvironmentReport called with null Hashtable!"));
 434         envCheckNode.appendChild(statusNode);
 435         return;
 436       }
 437 
 438       boolean errors = false;
 439 
 440       Element hashNode = factory.createElement("environment");
 441       envCheckNode.appendChild(hashNode);
 442 
 443       for (Enumeration keys = h.keys();
 444            keys.hasMoreElements();
 445           /* no increment portion */
 446           )
 447       {
 448         Object key = keys.nextElement();
 449         String keyStr = (String) key;
 450         try
 451         {
 452           // Special processing for classes found..
 453           if (keyStr.startsWith(FOUNDCLASSES))
 454           {
 455             Vector v = (Vector) h.get(keyStr);
 456             // errors |= logFoundJars(v, keyStr);
 457             errors |= appendFoundJars(hashNode, factory, v, keyStr);
 458           }
 459           // ..normal processing for all other entries
 460           else
 461           {
 462             // Note: we could just check for the ERROR key by itself,
 463             //    since we now set that, but since we have to go
 464             //    through the whole hash anyway, do it this way,
 465             //    which is safer for maintenance
 466             if (keyStr.startsWith(ERROR))
 467             {
 468               errors = true;
 469             }
 470             Element node = factory.createElement("item");
 471             node.setAttribute("key", keyStr);
 472             node.appendChild(factory.createTextNode((String)h.get(keyStr)));
 473             hashNode.appendChild(node);
 474           }
 475         }
 476         catch (Exception e)
 477         {
 478           errors = true;
 479           Element node = factory.createElement("item");
 480           node.setAttribute("key", keyStr);
 481           node.appendChild(factory.createTextNode(ERROR + " Reading " + key + " threw: " + e.toString()));
 482           hashNode.appendChild(node);
 483         }
 484       } // end of for...
 485 
 486       Element statusNode = factory.createElement("status");
 487       statusNode.setAttribute("result", (errors ? "ERROR" : "OK" ));
 488       envCheckNode.appendChild(statusNode);
 489     }
 490     catch (Exception e2)
 491     {
 492       System.err.println("appendEnvironmentReport threw: " + e2.toString());
 493       e2.printStackTrace();
 494     }
 495   }

 498    * Print out report of .jars found in a classpath.
 499    *
 500    * Takes the information encoded from a checkPathForJars()
 501    * call and dumps it out to our PrintWriter.
 502    *
 503    * @param container Node to append our report to
 504    * @param factory Document providing createElement, etc. services

 507    *
 508    * @return false if OK, true if any .jars were reported

 510    * @see #checkPathForJars(String, String[])

 512   protected boolean appendFoundJars(Node container, Document factory,
 513         Vector v, String desc)
 514   {
 515 
 516     if ((null == v) || (v.size() < 1))
 517       return false;
 518 
 519     boolean errors = false;
 520 
 521     for (int i = 0; i < v.size(); i++)
 522     {
 523       Hashtable subhash = (Hashtable) v.elementAt(i);
 524 
 525       for (Enumeration keys = subhash.keys();
 526            keys.hasMoreElements();
 527            /* no increment portion */
 528           )
 529       {
 530         Object key = keys.nextElement();
 531         try
 532         {
 533           String keyStr = (String) key;
 534           if (keyStr.startsWith(ERROR))
 535           {
 536             errors = true;
 537           }
 538           Element node = factory.createElement("foundJar");
 539           node.setAttribute("name", keyStr.substring(0, keyStr.indexOf("-")));
 540           node.setAttribute("desc", keyStr.substring(keyStr.indexOf("-") + 1));
 541           node.appendChild(factory.createTextNode((String)subhash.get(keyStr)));
 542           container.appendChild(node);
 543         }
 544         catch (Exception e)
 545         {
 546           errors = true;
 547           Element node = factory.createElement("foundJar");
 548           node.appendChild(factory.createTextNode(ERROR + " Reading " + key + " threw: " + e.toString()));
 549           container.appendChild(node);
 550         }
 551       }
 552     }
 553     return errors;
 554   }

 558    *
 559    * Logs java.class.path and other likely paths; then attempts
 560    * to search those paths for .jar files with Xalan-related classes.
 561    *
 562    * //@todo NOTE: We don't actually search java.ext.dirs for

 570   {
 571 
 572     if (null == h)
 573       h = new Hashtable();
 574 
 575     // Grab java version for later use
 576     try
 577     {
 578       String javaVersion = SecuritySupport.getSystemProperty("java.version");
 579

 614 
 615         if (null != classpathJars)

 620       //  *.jar files therein! This should be updated

 622 
 623       if (null != othercp)
 624       {

 626 
 627         classpathJars = checkPathForJars(othercp, jarNames);
 628

 635       //  v3 = checkPathForJars(System.getProperty("java.library.path"), jarNames);   // ?? may not be needed
 636     }
 637     catch (SecurityException se2)

 646   /**
 647    * Cheap-o listing of specified .jars found in the classpath.
 648    *
 649    * cp should be separated by the usual File.pathSeparator.  We
 650    * then do a simplistic search of the path for any requested
 651    * .jar filenames, and return a listing of their names and
 652    * where (apparently) they came from.
 653    *
 654    * @param cp classpath to search
 655    * @param jars array of .jar base filenames to look for
 656    *
 657    * @return Vector of Hashtables filled with info about found .jars
 658    * @see #jarNames
 659    * @see #logFoundJars(Vector, String)
 660    * @see #appendFoundJars(Node, Document, Vector, String )
 661    * @see #getApparentVersion(String, long)
 662    */
 663   protected Vector checkPathForJars(String cp, String[] jars)
 664   {
 665 
 666     if ((null == cp) || (null == jars) || (0 == cp.length())
 667             || (0 == jars.length))
 668       return null;
 669 
 670     Vector v = new Vector();
 671     StringTokenizer st = new StringTokenizer(cp, File.pathSeparator);
 672 
 673     while (st.hasMoreTokens())
 674     {
 675 
 676       // Look at each classpath entry for each of our requested jarNames
 677       String filename = st.nextToken();
 678 
 679       for (int i = 0; i < jars.length; i++)
 680       {
 681         if (filename.indexOf(jars[i]) > -1)
 682         {
 683           File f = new File(filename);
 684 
 685           if (f.exists())
 686           {
 687 
 688             // If any requested jarName exists, report on
 689             //  the details of that .jar file
 690             try
 691             {
 692               Hashtable h = new Hashtable(2);
 693               // Note "-" char is looked for in appendFoundJars
 694               h.put(jars[i] + "-path", f.getAbsolutePath());
 695 
 696               // We won't bother reporting on the xalan.jar apparent version
 697               // since this requires knowing the jar size of the xalan.jar
 698               // before we build it.
 699               // For other jars, eg. xml-apis.jar and xercesImpl.jar, we
 700               // report the apparent version of the file we've found
 701               if (!("xalan.jar".equalsIgnoreCase(jars[i]))) {
 702                 h.put(jars[i] + "-apparent.version",
 703                     getApparentVersion(jars[i], f.length()));
 704               }
 705               v.addElement(h);
 706             }
 707             catch (Exception e)
 708             {
 709 
 710               /* no-op, don't add it  */
 711             }
 712           }
 713           else
 714           {
 715             Hashtable h = new Hashtable(2);
 716             // Note "-" char is looked for in appendFoundJars
 717             h.put(jars[i] + "-path", WARNING + " Classpath entry: "
 718                   + filename + " does not exist");
 719             h.put(jars[i] + "-apparent.version", CLASS_NOTPRESENT);
 720             v.addElement(h);
 721           }
 722         }
 723       }
 724     }
 725 
 726     return v;
 727   }

 729   /**
 730    * Cheap-o method to determine the product version of a .jar.
 731    *
 732    * Currently does a lookup into a local table of some recent
 733    * shipped Xalan builds to determine where the .jar probably
 734    * came from.  Note that if you recompile Xalan or Xerces
 735    * yourself this will likely report a potential error, since
 736    * we can't certify builds other than the ones we ship.
 737    * Only reports against selected posted Xalan-J builds.
 738    *
 739    * //@todo actually look up version info in manifests
 740    *
 741    * @param jarName base filename of the .jarfile
 742    * @param jarSize size of the .jarfile
 743    *
 744    * @return String describing where the .jar file probably
 745    * came from
 746    */
 747   protected String getApparentVersion(String jarName, long jarSize)
 748   {
 749     // If we found a matching size and it's for our
 750     //  jar, then return it's description
 751     // Lookup in static jarVersions Hashtable
 752     String foundSize = (String) jarVersions.get(new Long(jarSize));
 753 
 754     if ((null != foundSize) && (foundSize.startsWith(jarName)))
 755     {
 756       return foundSize;
 757     }
 758     else
 759     {
 760       if ("xerces.jar".equalsIgnoreCase(jarName)
 761               || "xercesImpl.jar".equalsIgnoreCase(jarName))
 762 //              || "xalan.jar".equalsIgnoreCase(jarName))
 763       {
 764 
 765         // For xalan.jar and xerces.jar/xercesImpl.jar, which we ship together:
 766         // The jar is not from a shipped copy of xalan-j, so
 767         //  it's up to the user to ensure that it's compatible
 768         return jarName + " " + WARNING + CLASS_PRESENT;
 769       }
 770       else
 771       {
 772 
 773         // Otherwise, it's just a jar we don't have the version info calculated for
 774         return jarName + " " + CLASS_PRESENT;
 775       }
 776     }
 777   }
