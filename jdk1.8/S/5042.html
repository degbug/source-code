 115     public static boolean checkTypeDerivationOk(XSTypeDefinition derived, XSTypeDefinition base, short block) {
 116         // if derived is anyType, then it's valid only if base is anyType too
 117         if (derived == SchemaGrammar.fAnyType)
 118             return derived == base;
 119         // if derived is anySimpleType, then it's valid only if the base
 120         // is ur-type
 121         if (derived == SchemaGrammar.fAnySimpleType) {
 122             return (base == SchemaGrammar.fAnyType ||
 123                     base == SchemaGrammar.fAnySimpleType);
 124         }
 125 
 126         // if derived is simple type
 127         if (derived.getTypeCategory() == XSTypeDefinition.SIMPLE_TYPE) {
 128             // if base is complex type
 129             if (base.getTypeCategory() == XSTypeDefinition.COMPLEX_TYPE) {
 130                 // if base is anyType, change base to anySimpleType,
 131                 // otherwise, not valid
 132                 if (base == SchemaGrammar.fAnyType)
 133                     base = SchemaGrammar.fAnySimpleType;
 134                 else
 135                     return false;
 136             }
 137             return checkSimpleDerivation((XSSimpleType)derived,
 138                     (XSSimpleType)base, block);
 139         }
 140         else {
 141             return checkComplexDerivation((XSComplexTypeDecl)derived, base, block);
 142         }
 143     }

 181     /**
 182      * Note: this will be a private method, and it assumes that derived is not
 183      *       anySimpleType, and base is not anyType. Another method will be
 184      *       introduced for public use, which will call this method.
 185      */
 186     private static boolean checkSimpleDerivation(XSSimpleType derived, XSSimpleType base, short block) {
 187         // 1 They are the same type definition.
 188         if (derived == base)
 189             return true;
 190 
 191         // 2 All of the following must be true:
 192         // 2.1 restriction is not in the subset, or in the {final} of its own {base type definition};
 193         if ((block & XSConstants.DERIVATION_RESTRICTION) != 0 ||
 194                 (derived.getBaseType().getFinal() & XSConstants.DERIVATION_RESTRICTION) != 0) {
 195             return false;
 196         }
 197 
 198         // 2.2 One of the following must be true:
 199         // 2.2.1 D's base type definition is B.
 200         XSSimpleType directBase = (XSSimpleType)derived.getBaseType();
 201         if (directBase == base)
 202             return true;
 203 
 204         // 2.2.2 D's base type definition is not the simple ur-type definition and is validly derived from B given the subset, as defined by this constraint.
 205         if (directBase != SchemaGrammar.fAnySimpleType &&
 206                 checkSimpleDerivation(directBase, base, block)) {
 207             return true;
 208         }
 209 
 210         // 2.2.3 D's {variety} is list or union and B is the simple ur-type definition.
 211         if ((derived.getVariety() == XSSimpleType.VARIETY_LIST ||
 212                 derived.getVariety() == XSSimpleType.VARIETY_UNION) &&
 213                 base == SchemaGrammar.fAnySimpleType) {
 214             return true;
 215         }
 216 
 217         // 2.2.4 B's {variety} is union and D is validly derived from a type definition in B's {member type definitions} given the subset, as defined by this constraint.
 218         if (base.getVariety() == XSSimpleType.VARIETY_UNION) {
 219             XSObjectList subUnionMemberDV = base.getMemberTypes();
 220             int subUnionSize = subUnionMemberDV.getLength();
 221             for (int i=0; i<subUnionSize; i++) {
 222                 base = (XSSimpleType)subUnionMemberDV.item(i);
 223                 if (checkSimpleDerivation(derived, base, block))
 224                     return true;
 225             }
 226         }
 227 
 228         return false;
 229     }

 231     /**
 232      * Note: this will be a private method, and it assumes that derived is not
 233      *       anyType. Another method will be introduced for public use,
 234      *       which will call this method.
 235      */
 236     private static boolean checkComplexDerivation(XSComplexTypeDecl derived, XSTypeDefinition base, short block) {
 237         // 2.1 B and D must be the same type definition.
 238         if (derived == base)
 239             return true;
 240 
 241         // 1 If B and D are not the same type definition, then the {derivation method} of D must not be in the subset.
 242         if ((derived.fDerivedBy & block) != 0)
 243             return false;
 244 
 245         // 2 One of the following must be true:
 246         XSTypeDefinition directBase = derived.fBaseType;
 247         // 2.2 B must be D's {base type definition}.
 248         if (directBase == base)
 249             return true;
 250 
 251         // 2.3 All of the following must be true:
 252         // 2.3.1 D's {base type definition} must not be the ur-type definition.
 253         if (directBase == SchemaGrammar.fAnyType ||
 254                 directBase == SchemaGrammar.fAnySimpleType) {
 255             return false;
 256         }
 257 
 258         // 2.3.2 The appropriate case among the following must be true:
 259         // 2.3.2.1 If D's {base type definition} is complex, then it must be validly derived from B given the subset as defined by this constraint.
 260         if (directBase.getTypeCategory() == XSTypeDefinition.COMPLEX_TYPE)
 261             return checkComplexDerivation((XSComplexTypeDecl)directBase, base, block);
 262 
 263         // 2.3.2.2 If D's {base type definition} is simple, then it must be validly derived from B given the subset as defined in Type Derivation OK (Simple) (3.14.6).
 264         if (directBase.getTypeCategory() == XSTypeDefinition.SIMPLE_TYPE) {
 265             // if base is complex type
 266             if (base.getTypeCategory() == XSTypeDefinition.COMPLEX_TYPE) {
 267                 // if base is anyType, change base to anySimpleType,
 268                 // otherwise, not valid
 269                 if (base == SchemaGrammar.fAnyType)
 270                     base = SchemaGrammar.fAnySimpleType;
 271                 else
 272                     return false;
 273             }
 274             return checkSimpleDerivation((XSSimpleType)directBase,
 275                     (XSSimpleType)base, block);
 276         }
 277 
 278         return false;
 279     }

 282      * check whether a value is a valid default for some type
 283      * returns the compiled form of the value
 284      * The parameter value could be either a String or a ValidatedInfo object

 286     public static Object ElementDefaultValidImmediate(XSTypeDefinition type, String value, ValidationContext context, ValidatedInfo vinfo) {
 287 
 288         XSSimpleType dv = null;
 289 
 290         // e-props-correct
 291         // For a string to be a valid default with respect to a type definition the appropriate case among the following must be true:
 292         // 1 If the type definition is a simple type definition, then the string must be valid with respect to that definition as defined by String Valid (3.14.4).
 293         if (type.getTypeCategory() == XSTypeDefinition.SIMPLE_TYPE) {
 294             dv = (XSSimpleType)type;
 295         }
 296 
 297         // 2 If the type definition is a complex type definition, then all of the following must be true:
 298         else {
 299             // 2.1 its {content type} must be a simple type definition or mixed.
 300             XSComplexTypeDecl ctype = (XSComplexTypeDecl)type;
 301             // 2.2 The appropriate case among the following must be true:
 302             // 2.2.1 If the {content type} is a simple type definition, then the string must be valid with respect to that simple type definition as defined by String Valid (3.14.4).
 303             if (ctype.fContentType == XSComplexTypeDecl.CONTENTTYPE_SIMPLE) {
 304                 dv = ctype.fXSSimpleType;
 305             }
 306             // 2.2.2 If the {content type} is mixed, then the {content type}'s particle must be emptiable as defined by Particle Emptiable (3.9.6).
 307             else if (ctype.fContentType == XSComplexTypeDecl.CONTENTTYPE_MIXED) {
 308                 if (!((XSParticleDecl)ctype.getParticle()).emptiable())
 309                     return null;
 310             }
 311             else {
 312                 return null;
 313             }
 314         }
 315 
 316         // get the simple type declaration, and validate
 317         Object actualValue = null;
 318         if (dv == null) {
 319             // complex type with mixed. to make sure that we store correct
 320             // information in vinfo and return the correct value, we use
 321             // "string" type for validation
 322             dv = STRING_TYPE;
 323         }
 324         try {
 325             // validate the original lexical rep, and set the actual value
 326             actualValue = dv.validate(value, context, vinfo);
 327             // validate the canonical lexical rep
 328             if (vinfo != null)
 329                 actualValue = dv.validate(vinfo.stringValue(), context, vinfo);
 330         } catch (InvalidDatatypeValueException ide) {
 331             return null;
 332         }
 333 
 334         return actualValue;
 335     }

 391                         reportSchemaError(errorReporter, rgLocators[i/2-1],
 392                                 "src-redefine.6.2.2",
 393                                 new Object[]{derivedGrp.fName, "rcase-Recurse.2"});

 396                     try {
 397                         particleValidRestriction(fakeDerived, SGHandler, fakeBase, SGHandler);

 399                         String key = e.getKey();

 528         }

 535     public static void checkElementDeclsConsistent(XSComplexTypeDecl type,
 536             XSParticleDecl particle,
 537             SymbolHash elemDeclHash,
 538             SubstitutionGroupHandler sgHandler)
 539         throws XMLSchemaException {
 540 
 541         // check for elements in the tree with the same name and namespace
 542 
 543         int pType = particle.fType;
 544 
 545         if (pType == XSParticleDecl.PARTICLE_WILDCARD)
 546             return;
 547 
 548         if (pType == XSParticleDecl.PARTICLE_ELEMENT) {
 549             XSElementDecl elem = (XSElementDecl)(particle.fValue);
 550             findElemInTable(type, elem, elemDeclHash);
 551 
 552             if (elem.fScope == XSConstants.SCOPE_GLOBAL) {
 553                 // Check for subsitution groups.
 554                 XSElementDecl[] subGroup = sgHandler.getSubstitutionGroup(elem);
 555                 for (int i = 0; i < subGroup.length; i++) {
 556                     findElemInTable(type, subGroup[i], elemDeclHash);
 557                 }
 558             }
 559             return;
 560         }
 561 
 562         XSModelGroupImpl group = (XSModelGroupImpl)particle.fValue;
 563         for (int i = 0; i < group.fParticleCount; i++)
 564             checkElementDeclsConsistent(type, group.fParticles[i], elemDeclHash, sgHandler);
 565     }

 609             SubstitutionGroupHandler dSGHandler,

 611             SubstitutionGroupHandler bSGHandler,
 612             boolean checkWCOccurrence)
 613         throws XMLSchemaException {

 616         Vector bChildren = null;
 617         int dMinEffectiveTotalRange=OCCURRENCE_UNKNOWN;
 618         int dMaxEffectiveTotalRange=OCCURRENCE_UNKNOWN;
 619 
 620         // By default there has been no expansion
 621         boolean bExpansionHappened = false;
 622 
 623         // Check for empty particles.   If either base or derived particle is empty,
 624         // (and the other isn't) it's an error.
 625         if (dParticle.isEmpty() && !bParticle.emptiable()) {
 626             throw new XMLSchemaException("cos-particle-restrict.a", null);

 628         else if (!dParticle.isEmpty() && bParticle.isEmpty()) {
 629             throw new XMLSchemaException("cos-particle-restrict.b", null);

 634         // This involves:
 635         //   - removing pointless occurrences for groups, and retrieving a vector of
 636         //     non-pointless children
 637         //   - turning top-level elements with substitution groups into CHOICE groups.
 638         //
 639 
 640         short dType = dParticle.fType;

 645             dType = ((XSModelGroupImpl)dParticle.fValue).fCompositor;
 646

 648             // may be none, and the particle of interest trivially becomes an element or

 653                 dParticle = dtmp;

 655                 if (dType == XSParticleDecl.PARTICLE_MODELGROUP)
 656                     dType = ((XSModelGroupImpl)dParticle.fValue).fCompositor;
 657             }
 658 
 659             // Fill in a vector with the children of the particle, removing any
 660             // pointless model groups in the process.
 661             dChildren = removePointlessChildren(dParticle);
 662         }

 665         int dMaxOccurs = dParticle.fMaxOccurs;
 666

 668         // For elements which are the heads of substitution groups, treat as CHOICE
 669         //

 671             XSElementDecl dElement = (XSElementDecl)dParticle.fValue;
 672 
 673             if (dElement.fScope == XSConstants.SCOPE_GLOBAL) {

 702         // Handle pointless groups for the base particle
 703         //

 716                     bType = ((XSModelGroupImpl)bParticle.fValue).fCompositor;
 717             }

 720             // pointless model groups in the process.
 721             bChildren = removePointlessChildren(bParticle);
 722         }

 724         int bMinOccurs = bParticle.fMinOccurs;
 725         int bMaxOccurs = bParticle.fMaxOccurs;
 726 
 727         if (bSGHandler != null && bType == XSParticleDecl.PARTICLE_ELEMENT) {
 728             XSElementDecl bElement = (XSElementDecl)bParticle.fValue;
 729 
 730             if (bElement.fScope == XSConstants.SCOPE_GLOBAL) {

 747                     bSGHandler = null;

1144                     new Object[]{dElement.fName, dElement.fType.getName(), bElement.fType.getName()});

1441     public static boolean overlapUPA(XSElementDecl element1,
1442             XSElementDecl element2,
1443             SubstitutionGroupHandler sgHandler) {
1444         // if the two element have the same name and namespace,
1445         if (element1.fName == element2.fName &&
1446                 element1.fTargetNamespace == element2.fTargetNamespace) {
1447             return true;
1448         }
1449 
1450         // or if there is an element decl in element1's substitution group,
1451         // who has the same name/namespace with element2
1452         XSElementDecl[] subGroup = sgHandler.getSubstitutionGroup(element1);
1453         for (int i = subGroup.length-1; i >= 0; i--) {
1454             if (subGroup[i].fName == element2.fName &&
1455                     subGroup[i].fTargetNamespace == element2.fTargetNamespace) {
1456                 return true;
1457             }
1458         }
1459 
1460         // or if there is an element decl in element2's substitution group,
1461         // who has the same name/namespace with element1
1462         subGroup = sgHandler.getSubstitutionGroup(element2);
1463         for (int i = subGroup.length-1; i >= 0; i--) {
1464             if (subGroup[i].fName == element1.fName &&
1465                     subGroup[i].fTargetNamespace == element1.fTargetNamespace) {
1466                 return true;
1467             }
1468         }
1469 
1470         return false;
