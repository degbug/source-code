 142     public void removeElementAt(int index) {
 143         if ( getElementAt( index ) == selectedObject ) {
 144             if ( index == 0 ) {
 145                 setSelectedItem( getSize() == 1 ? null : getElementAt( index + 1 ) );
 146             }

 149             }

 155     }
