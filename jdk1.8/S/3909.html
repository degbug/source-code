  34 /**
  35  * QName represents a qualified name
  36  * as defined in the XML specifications: 
  37  * href="http://www.w3.org/TR/xmlschema-2/#QName">XML Schema Part2:
  38  * Datatypes specification, 
  39  * href="http://www.w3.org/TR/REC-xml-names/#ns-qualnames">Namespaces
  40  * in XML, 
  41  * href="http://www.w3.org/XML/xml-names-19990114-errata">Namespaces
  42  * in XML Errata.
  43  *
  44  * The value of a QName contains a Namespace
  45  * URI, local part and
  46  * prefix.
  47  *
  48  * The prefix is included in QName to retain lexical
  49  * information when present in an {@link
  50  * javax.xml.transform.Source XML input source}. The prefix is
  51  * NOT used in {@link #equals(Object)
  52  * QName.equals(Object)} or to compute the {@link #hashCode()
  53  * QName.hashCode()}.  Equality and the hash code are defined using
  54  * only the Namespace URI and local part.
  55  *
  56  * If not specified, the Namespace URI is set to {@link
  57  * javax.xml.XMLConstants#NULL_NS_URI XMLConstants.NULL_NS_URI}.
  58  * If not specified, the prefix is set to {@link
  59  * javax.xml.XMLConstants#DEFAULT_NS_PREFIX
  60  * XMLConstants.DEFAULT_NS_PREFIX}.
  61  *
  62  * QName is immutable.
  63  *
  64  * @author Jeff Suttor
  65  * @version $Revision: 1.8 $, $Date: 2010/03/18 03:06:17 $
  66  * @see 
  67  *   XML Schema Part2: Datatypes specification
  68  * @see 
  69  *   Namespaces in XML
  70  * @see 
  71  *   Namespaces in XML Errata
  72  * @since 1.5
  73  */

 336     /**
 337      * Test this QName for equality with another
 338      * Object.
 339      *
 340      * If the Object to be tested is not a
 341      * QName or is null, then this method
 342      * returns false.
 343      *
 344      * Two QNames are considered equal if and only if
 345      * both the Namespace URI and local part are equal. This method
 346      * uses String.equals() to check equality of the
 347      * Namespace URI and local part. The prefix is
 348      * NOT used to determine equality.
 349      *
 350      * This method satisfies the general contract of {@link
 351      * java.lang.Object#equals(Object) Object.equals(Object)}
 352      *
 353      * @param objectToTest the Object to test for
 354      * equality with this QName
 355      * @return true if the given Object is
 356      * equal to this QName else false
 357      */
 358     public final boolean equals(Object objectToTest) {
 359         if (objectToTest == this) {
 360             return true;
 361         }
 362 
 363         if (objectToTest == null || !(objectToTest instanceof QName)) {
 364             return false;
 365         }

 368 
 369         return localPart.equals(qName.localPart)

 422     /**
 423      * QName derived from parsing the formatted
 424      * String.
 425      *
 426      * If the String is null or does not conform to
 427      * {@link #toString() QName.toString()} formatting, an
 428      * IllegalArgumentException is thrown.
 429      *
 430      * The String MUST be in the
 431      * form returned by {@link #toString() QName.toString()}.
 432      *
 433      * The commonly accepted way of representing a QName
 434      * as a String was
 435      * defined
 436      * by James Clark.  Although this is not a standard
 437      * specification, it is in common use, e.g. {@link
 438      * javax.xml.transform.Transformer#setParameter(String name, Object value)}.
 439      * This implementation parses a String formatted
 440      * as: "{" + Namespace URI + "}" + local part.  If the Namespace
 441      * URI .equals(XMLConstants.NULL_NS_URI), only the
 442      * local part should be provided.
 443      *
 444      * The prefix value CANNOT be
 445      * represented in the String and will be set to
 446      * {@link javax.xml.XMLConstants#DEFAULT_NS_PREFIX
 447      * XMLConstants.DEFAULT_NS_PREFIX}.
 448      *
 449      * This method does not do full validation of the resulting
 450      * QName.
 451      * The Namespace URI is not validated as a
 452      * URI reference.
 453      * The local part is not validated as a
 454      * NCName
 455      * as specified in
 456      * Namespaces in XML.
 457      *
 458      * @param qNameAsString String representation
 459      * of the QName
 460      *
 461      * @throws IllegalArgumentException When qNameAsString is
 462      *   null or malformed
 463      *
 464      * @return QName corresponding to the given String
 465      * @see #toString() QName.toString()
 466      */
 467     public static QName valueOf(String qNameAsString) {
 468 
 469         // null is not valid
 470         if (qNameAsString == null) {
 471             throw new IllegalArgumentException(
 472                     "cannot create QName from \"null\" or \"\" String");
 473         }
 474 
 475         // "" local part is valid to preserve compatible behavior with QName 1.0
 476         if (qNameAsString.length() == 0) {
 477             return new QName(
 478                 XMLConstants.NULL_NS_URI,
 479                 qNameAsString,
 480                 XMLConstants.DEFAULT_NS_PREFIX);
 481         }
 482 
 483         // local part only?
 484         if (qNameAsString.charAt(0) != '{') {
 485             return new QName(
 486                 XMLConstants.NULL_NS_URI,
 487                 qNameAsString,
 488                 XMLConstants.DEFAULT_NS_PREFIX);
 489         }
 490 
 491         // Namespace URI improperly specified?
 492         if (qNameAsString.startsWith("{" + XMLConstants.NULL_NS_URI + "}")) {
 493             throw new IllegalArgumentException(
 494                 "Namespace URI .equals(XMLConstants.NULL_NS_URI), "
 495                 + ".equals(\"" + XMLConstants.NULL_NS_URI + "\"), "
 496                 + "only the local part, "
 497                 + "\""
 498                 + qNameAsString.substring(2 + XMLConstants.NULL_NS_URI.length())
 499                 + "\", "
 500                 + "should be provided.");
 501         }
 502 
 503         // Namespace URI and local part specified
 504         int endOfNamespaceURI = qNameAsString.indexOf('}');
 505         if (endOfNamespaceURI == -1) {
 506             throw new IllegalArgumentException(
 507                 "cannot create QName from \""
 508                     + qNameAsString
 509                     + "\", missing closing \"}\"");
 510         }
 511         return new QName(
 512             qNameAsString.substring(1, endOfNamespaceURI),
 513             qNameAsString.substring(endOfNamespaceURI + 1),
 514             XMLConstants.DEFAULT_NS_PREFIX);
 515     }
