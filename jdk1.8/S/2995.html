 150     @Override
 151     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
 152         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 153         componentColors = extendedCacheKeys;
 154         //generate this entire method. Each state/bg/fg/border combo that has
 155         //been painted gets its own KEY and paint method.
 156         switch(state) {
 157             case BACKGROUND_DISABLED_WINDOWMAXIMIZED: paintBackgroundDisabledAndWindowMaximized(g); break;
 158             case BACKGROUND_ENABLED_WINDOWMAXIMIZED: paintBackgroundEnabledAndWindowMaximized(g); break;
 159             case BACKGROUND_MOUSEOVER_WINDOWMAXIMIZED: paintBackgroundMouseOverAndWindowMaximized(g); break;
 160             case BACKGROUND_PRESSED_WINDOWMAXIMIZED: paintBackgroundPressedAndWindowMaximized(g); break;
 161             case BACKGROUND_ENABLED_WINDOWNOTFOCUSED_WINDOWMAXIMIZED: paintBackgroundEnabledAndWindowNotFocusedAndWindowMaximized(g); break;
 162             case BACKGROUND_MOUSEOVER_WINDOWNOTFOCUSED_WINDOWMAXIMIZED: paintBackgroundMouseOverAndWindowNotFocusedAndWindowMaximized(g); break;
 163             case BACKGROUND_PRESSED_WINDOWNOTFOCUSED_WINDOWMAXIMIZED: paintBackgroundPressedAndWindowNotFocusedAndWindowMaximized(g); break;
 164             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
 165             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
 166             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
 167             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
 168             case BACKGROUND_ENABLED_WINDOWNOTFOCUSED: paintBackgroundEnabledAndWindowNotFocused(g); break;
 169             case BACKGROUND_MOUSEOVER_WINDOWNOTFOCUSED: paintBackgroundMouseOverAndWindowNotFocused(g); break;
 170             case BACKGROUND_PRESSED_WINDOWNOTFOCUSED: paintBackgroundPressedAndWindowNotFocused(g); break;
 171 
 172         }
 173     }
