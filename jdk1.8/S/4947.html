  30 /**

  34 public class FuncConcat extends FunctionMultiArgs
  35 {
  36     static final long serialVersionUID = 1737228885202314413L;

  38   /**
  39    * Execute the function.  The function must return
  40    * a valid object.
  41    * @param xctxt The current execution context.
  42    * @return A valid XObject.
  43    *
  44    * @throws javax.xml.transform.TransformerException
  45    */
  46   public XObject execute(XPathContext xctxt) throws javax.xml.transform.TransformerException
  47   {
  48 
  49     StringBuffer sb = new StringBuffer();
  50 
  51     // Compiler says we must have at least two arguments.
  52     sb.append(m_arg0.execute(xctxt).str());
  53     sb.append(m_arg1.execute(xctxt).str());
  54 
  55     if (null != m_arg2)
  56       sb.append(m_arg2.execute(xctxt).str());
  57 
  58     if (null != m_args)
  59     {
  60       for (int i = 0; i < m_args.length; i++)
  61       {
  62         sb.append(m_args[i].execute(xctxt).str());
  63       }
  64     }
  65 
  66     return new XString(sb.toString());
  67   }
  68 
  69   /**
