 769     private ClassLoader getClassLoader(final ObjectName name) {
 770         if(clr == null){
 771             return null;
 772         }
 773         // Restrict to getClassLoader permission only
 774         Permissions permissions = new Permissions();
 775         permissions.add(new MBeanPermission("*", null, name, "getClassLoader"));
 776         ProtectionDomain protectionDomain = new ProtectionDomain(null, permissions);

 778         AccessControlContext ctx = new AccessControlContext(domains);
 779         ClassLoader loader = AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {

 781                 return clr.getClassLoader(name);
 782             }
