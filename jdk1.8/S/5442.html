  28 /**
  29  * Allows to retrieve XSImplementation, DOM Level 3 Core and LS implementations
  30  * and PSVI implementation.
  31  * See also the Document Object Model (DOM) Level 3 Core Specification.
  32  *
  33  * @xerces.internal
  34  *
  35  * @author Elena Litani, IBM
  36  */
  37 public class DOMXSImplementationSourceImpl
  38     extends DOMImplementationSourceImpl {
  39 
  40     /**
  41      * A method to request a DOM implementation.
  42      * @param features A string that specifies which features are required.
  43      *   This is a space separated list in which each feature is specified
  44      *   by its name optionally followed by a space and a version number.
  45      *   This is something like: "XML 1.0 Traversal Events 2.0"
  46      * @return An implementation that has the desired features, or
  47      *   null if this source has none.
  48      */
  49     public DOMImplementation getDOMImplementation(String features) {
  50         DOMImplementation impl = super.getDOMImplementation(features);
  51         if (impl != null){
  52             return impl;
  53         }
  54         // if not try the PSVIDOMImplementation
  55         impl = PSVIDOMImplementationImpl.getDOMImplementation();
  56         if (testImpl(impl, features)) {
  57             return impl;
  58         }
  59         // if not try the XSImplementation
  60         impl = XSImplementationImpl.getDOMImplementation();
  61         if (testImpl(impl, features)) {
  62             return impl;
  63         }
  64 
  65         return null;
  66     }
  67 
  68     /**
  69      * A method to request a list of DOM implementations that support the
  70      * specified features and versions, as specified in .
  71      * @param features A string that specifies which features and versions
  72      *   are required. This is a space separated list in which each feature
  73      *   is specified by its name optionally followed by a space and a
  74      *   version number. This is something like: "XML 3.0 Traversal +Events
  75      *   2.0"
  76      * @return A list of DOM implementations that support the desired
  77      *   features.
  78      */
  79     public DOMImplementationList getDOMImplementationList(String features) {
  80         final Vector implementations = new Vector();
  81 
  82         // first check whether the CoreDOMImplementation would do
  83         DOMImplementationList list = super.getDOMImplementationList(features);
  84         //Add core DOMImplementations
  85         for (int i=0; i < list.getLength(); i++ ) {
  86             implementations.addElement(list.item(i));
  87         }
  88 
  89         DOMImplementation impl = PSVIDOMImplementationImpl.getDOMImplementation();
  90         if (testImpl(impl, features)) {
  91             implementations.addElement(impl);
  92         }
  93 
  94         impl = XSImplementationImpl.getDOMImplementation();
  95         if (testImpl(impl, features)) {
  96             implementations.addElement(impl);
  97         }
  98         return new DOMImplementationListImpl(implementations);
  99     }
