  A class which implements an arbitrary border
  with the addition of a String title in a
  specified position and justification.
  

  If the border, font, or color property values are not
  specified in the constructor or by invoking the appropriate
  set methods, the property values will be defined by the current
  look and feel, using the following property names in the
  Defaults Table:
  
  • "TitledBorder.border"
  
  • "TitledBorder.font"
  
  • "TitledBorder.titleColor"

      Paints the border for the specified component with the
      specified position and size.
      @param c the component for which this border is being painted
      @param g the paint graphics
      @param x the x position of the painted border
      @param y the y position of the painted border
      @param width the width of the painted border
      @param height the height of the painted border

 237     public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
 238         Border border = getBorder();
 239         String title = getTitle();
 240         if ((title != null) && !title.isEmpty()) {
 241             int edge = (border instanceof TitledBorder) ? 0 : EDGE_SPACING;
 242             JLabel label = getLabel(c);
 243             Dimension size = label.getPreferredSize();
 244             Insets insets = getBorderInsets(border, c, new Insets(0, 0, 0, 0));
 245 
 246             int borderX = x + edge;
 247             int borderY = y + edge;
 248             int borderW = width - edge - edge;
 249             int borderH = height - edge - edge;
 250 
 251             int labelY = y;
 252             int labelH = size.height;
 253             int position = getPosition();
 254             switch (position) {
 255                 case ABOVE_TOP:
 256                     insets.left = 0;
 257                     insets.right = 0;
 258                     borderY += labelH - edge;
 259                     borderH -= labelH - edge;
 260                     break;
 261                 case TOP:
 262                     insets.top = edge + insets.top/2 - labelH/2;
 263                     if (insets.top < edge) {
 264                         borderY -= insets.top;
 265                         borderH += insets.top;
 266                     }
 267                     else {
 268                         labelY += insets.top;
 269                     }
 270                     break;
 271                 case BELOW_TOP:
 272                     labelY += insets.top + edge;
 273                     break;
 274                 case ABOVE_BOTTOM:
 275                     labelY += height - labelH - insets.bottom - edge;
 276                     break;
 277                 case BOTTOM:
 278                     labelY += height - labelH;
 279                     insets.bottom = edge + (insets.bottom - labelH) / 2;
 280                     if (insets.bottom < edge) {
 281                         borderH += insets.bottom;
 282                     }
 283                     else {
 284                         labelY -= insets.bottom;
 285                     }
 286                     break;
 287                 case BELOW_BOTTOM:
 288                     insets.left = 0;
 289                     insets.right = 0;
 290                     labelY += height - labelH;
 291                     borderH -= labelH - edge;
 292                     break;
 293             }
 294             insets.left += edge + TEXT_INSET_H;
 295             insets.right += edge + TEXT_INSET_H;
 296 
 297             int labelX = x;
 298             int labelW = width - insets.left - insets.right;
 299             if (labelW > size.width) {
 300                 labelW = size.width;
 301             }
 302             switch (getJustification(c)) {
 303                 case LEFT:
 304                     labelX += insets.left;
 305                     break;
 306                 case RIGHT:
 307                     labelX += width - insets.right - labelW;
 308                     break;
 309                 case CENTER:
 310                     labelX += (width - labelW) / 2;
 311                     break;
 312             }
 313 
 314             if (border != null) {
 315                 if ((position != TOP) && (position != BOTTOM)) {
 316                     border.paintBorder(c, g, borderX, borderY, borderW, borderH);
 317                 }
 318                 else {
 319                     Graphics g2 = g.create();
 320                     if (g2 instanceof Graphics2D) {
 321                         Graphics2D g2d = (Graphics2D) g2;
 322                         Path2D path = new Path2D.Float();
 323                         path.append(new Rectangle(borderX, borderY, borderW, labelY - borderY), false);
 324                         path.append(new Rectangle(borderX, labelY, labelX - borderX - TEXT_SPACING, labelH), false);
 325                         path.append(new Rectangle(labelX + labelW + TEXT_SPACING, labelY, borderX - labelX + borderW - labelW - TEXT_SPACING, labelH), false);
 326                         path.append(new Rectangle(borderX, labelY + labelH, borderW, borderY - labelY + borderH - labelH), false);
 327                         g2d.clip(path);
 328                     }
 329                     border.paintBorder(c, g2, borderX, borderY, borderW, borderH);
 330                     g2.dispose();
 331                 }
 332             }
 333             g.translate(labelX, labelY);
 334             label.setSize(labelW, labelH);
 335             label.paint(g);
 336             g.translate(-labelX, -labelY);
 337         }
 338         else if (border != null) {
 339             border.paintBorder(c, g, x, y, width, height);
 340         }
 341     }

 348     public Insets getBorderInsets(Component c, Insets insets) {
 349         Border border = getBorder();
 350         insets = getBorderInsets(border, c, insets);
 351 
 352         String title = getTitle();
 353         if ((title != null) && !title.isEmpty()) {
 354             int edge = (border instanceof TitledBorder) ? 0 : EDGE_SPACING;
 355             JLabel label = getLabel(c);
 356             Dimension size = label.getPreferredSize();
 357 
 358             switch (getPosition()) {
 359                 case ABOVE_TOP:
 360                     insets.top += size.height - edge;
 361                     break;
 362                 case TOP: {
 363                     if (insets.top < size.height) {
 364                         insets.top = size.height - edge;
 365                     }
 366                     break;
 367                 }
 368                 case BELOW_TOP:
 369                     insets.top += size.height;
 370                     break;
 371                 case ABOVE_BOTTOM:
 372                     insets.bottom += size.height;
 373                     break;
 374                 case BOTTOM: {
 375                     if (insets.bottom < size.height) {
 376                         insets.bottom = size.height - edge;
 377                     }
 378                     break;
 379                 }
 380                 case BELOW_BOTTOM:
 381                     insets.bottom += size.height - edge;
 382                     break;
 383             }
 384             insets.top += edge + TEXT_SPACING;
 385             insets.left += edge + TEXT_SPACING;
 386             insets.right += edge + TEXT_SPACING;
 387             insets.bottom += edge + TEXT_SPACING;
 388         }

      Returns the baseline.
      @throws NullPointerException {@inheritDoc}
      @throws IllegalArgumentException {@inheritDoc}
      @see javax.swing.JComponent#getBaseline(int, int)
      @since 1.6

 565     public int getBaseline(Component c, int width, int height) {
 566         if (c == null) {
 567             throw new NullPointerException("Must supply non-null component");
 568         }
 569         if (width < 0) {
 570             throw new IllegalArgumentException("Width must be >= 0");
 571         }
 572         if (height < 0) {
 573             throw new IllegalArgumentException("Height must be >= 0");
 574         }
 575         Border border = getBorder();
 576         String title = getTitle();
 577         if ((title != null) && !title.isEmpty()) {
 578             int edge = (border instanceof TitledBorder) ? 0 : EDGE_SPACING;
 579             JLabel label = getLabel(c);
 580             Dimension size = label.getPreferredSize();
 581             Insets insets = getBorderInsets(border, c, new Insets(0, 0, 0, 0));
 582 
 583             int baseline = label.getBaseline(size.width, size.height);
 584             switch (getPosition()) {
 585                 case ABOVE_TOP:
 586                     return baseline;
 587                 case TOP:
 588                     insets.top = edge + (insets.top - size.height) / 2;
 589                     return (insets.top < edge)
 590                             ? baseline
 591                             : baseline + insets.top;
 592                 case BELOW_TOP:
 593                     return baseline + insets.top + edge;
 594                 case ABOVE_BOTTOM:
 595                     return baseline + height - size.height - insets.bottom - edge;
 596                 case BOTTOM:
 597                     insets.bottom = edge + (insets.bottom - size.height) / 2;
 598                     return (insets.bottom < edge)
 599                             ? baseline + height - size.height
 600                             : baseline + height - size.height + insets.bottom;
 601                 case BELOW_BOTTOM:
 602                     return baseline + height - size.height;
 603             }
 604         }
 605         return -1;
 606     }

 632     private int getPosition() {
 633         int position = getTitlePosition();
 634         if (position != DEFAULT_POSITION) {
 635             return position;
 636         }
 637         Object value = UIManager.get("TitledBorder.position");
 638         if (value instanceof Integer) {
 639             int i = (Integer) value;
 640             if ((0 < i) && (i <= 6)) {
 641                 return i;
 642             }
 643         }
 644         else if (value instanceof String) {
 645             String s = (String) value;
 646             if (s.equalsIgnoreCase("ABOVE_TOP")) {
 647                 return ABOVE_TOP;
 648             }
 649             if (s.equalsIgnoreCase("TOP")) {
 650                 return TOP;
 651             }
 652             if (s.equalsIgnoreCase("BELOW_TOP")) {
 653                 return BELOW_TOP;
 654             }
 655             if (s.equalsIgnoreCase("ABOVE_BOTTOM")) {
 656                 return ABOVE_BOTTOM;
 657             }
 658             if (s.equalsIgnoreCase("BOTTOM")) {
 659                 return BOTTOM;
 660             }
 661             if (s.equalsIgnoreCase("BELOW_BOTTOM")) {
 662                 return BELOW_BOTTOM;
 663             }
 664         }
 665         return TOP;
 666     }
