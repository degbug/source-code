 162     /**
 163      * Defines an entity.  If the Entity specified
 164      * by name, type, and data
 165      * exists, it is returned; otherwise a new Entity
 166      * is created and is returned.
 167      *
 168      * @param name the name of the Entity as a String
 169      * @param type the type of the Entity
 170      * @param data the Entity's data
 171      * @return the Entity requested or a new Entity
 172      *   if not found
 173      */
 174     public Entity defineEntity(String name, int type, char data[]) {
 175         Entity ent = entityHash.get(name);
 176         if (ent == null) {
 177             ent = new Entity(name, type, data);
 178             entityHash.put(name, ent);
 179             if (((type & GENERAL) != 0) && (data.length == 1)) {
 180                 switch (type & ~GENERAL) {
 181                   case CDATA:
 182                   case SDATA:
 183                       entityHash.put(Integer.valueOf(data[0]), ent);
 184                     break;
 185                 }
 186             }
 187         }
 188         return ent;
 189     }

 253     /**
 254      * Creates and returns an Element.
 255      * @param name the element's name
 256      * @return the new Element
 257      */
 258     protected Element defElement(String name, int type,
 259                        boolean omitStart, boolean omitEnd, ContentModel content,
 260                        String[] exclusions, String[] inclusions, AttributeList atts) {
 261         BitSet excl = null;
 262         if (exclusions != null && exclusions.length > 0) {
 263             excl = new BitSet();
 264             for (String str : exclusions) {
 265                 if (str.length() > 0) {
 266                     excl.set(getElement(str).getIndex());
 267                 }
 268             }
 269         }
 270         BitSet incl = null;
 271         if (inclusions != null && inclusions.length > 0) {
 272             incl = new BitSet();
 273             for (String str : inclusions) {
 274                 if (str.length() > 0) {
 275                     incl.set(getElement(str).getIndex());
 276                 }
 277             }
 278         }
 279         return defineElement(name, type, omitStart, omitEnd, content, excl, incl, atts);
 280     }

 282     /**
 283      * Creates and returns an AttributeList.
 284      * @param name the attribute list's name
 285      * @return the new AttributeList
 286      */
 287     protected AttributeList defAttributeList(String name, int type, int modifier, String value, String values, AttributeList atts) {
 288         Vector<String> vals = null;
 289         if (values != null) {
 290             vals = new Vector<String>();
 291             for (StringTokenizer s = new StringTokenizer(values, "|") ; s.hasMoreTokens() ;) {
 292                 String str = s.nextToken();
 293                 if (str.length() > 0) {
 294                     vals.addElement(str);
 295                 }
 296             }
 297         }
 298         return new AttributeList(name, type, modifier, value, vals, atts);
 299     }

 359     /**
 360      * Recreates a DTD from an archived format.
 361      * @param in  the DataInputStream to read from
 362      */
 363     public void read(DataInputStream in) throws IOException {
 364         if (in.readInt() != FILE_VERSION) {
 365         }
 366 
 367         //
 368         // Read the list of names
 369         //
 370         String[] names = new String[in.readShort()];
 371         for (int i = 0; i < names.length; i++) {
 372             names[i] = in.readUTF();
 373         }
 374 
 375 
 376         //
 377         // Read the entities
 378         //
 379         int num = in.readShort();
 380         for (int i = 0; i < num; i++) {
 381             short nameId = in.readShort();
 382             int type = in.readByte();
 383             String name = in.readUTF();
 384             defEntity(names[nameId], type | GENERAL, name);
 385         }
 386 
 387         // Read the elements
 388         //
 389         num = in.readShort();
 390         for (int i = 0; i < num; i++) {
 391             short nameId = in.readShort();
 392             int type = in.readByte();
 393             byte flags = in.readByte();
 394             ContentModel m = readContentModel(in, names);
 395             String[] exclusions = readNameArray(in, names);
 396             String[] inclusions = readNameArray(in, names);
 397             AttributeList atts = readAttributeList(in, names);
 398             defElement(names[nameId], type,
 399                        ((flags & 0x01) != 0), ((flags & 0x02) != 0),
 400                        m, exclusions, inclusions, atts);
 401         }
 402     }

 404     private ContentModel readContentModel(DataInputStream in, String[] names)
 405                 throws IOException {
 406         byte flag = in.readByte();
 407         switch(flag) {
 408             case 0:             // null
 409                 return null;
 410             case 1: {           // content_c
 411                 int type = in.readByte();
 412                 ContentModel m = readContentModel(in, names);
 413                 ContentModel next = readContentModel(in, names);
 414                 return defContentModel(type, m, next);
 415             }
 416             case 2: {           // content_e
 417                 int type = in.readByte();
 418                 Element el = getElement(names[in.readShort()]);
 419                 ContentModel next = readContentModel(in, names);
 420                 return defContentModel(type, el, next);
 421             }
 422         default:
 423                 throw new IOException("bad bdtd");
 424         }
 425     }

 427     private String[] readNameArray(DataInputStream in, String[] names)
 428                 throws IOException {
 429         int num = in.readShort();
 430         if (num == 0) {
 431             return null;
 432         }
 433         String[] result = new String[num];
 434         for (int i = 0; i < num; i++) {
 435             result[i] = names[in.readShort()];
 436         }
 437         return result;
 438     }

 441     private AttributeList readAttributeList(DataInputStream in, String[] names)
 442                 throws IOException  {
 443         AttributeList result = null;
 444         for (int num = in.readByte(); num > 0; --num) {
 445             short nameId = in.readShort();
 446             int type = in.readByte();
 447             int modifier = in.readByte();
 448             short valueId = in.readShort();
 449             String value = (valueId == -1) ? null : names[valueId];
 450             Vector<String> values = null;
 451             short numValues = in.readShort();
 452             if (numValues > 0) {
 453                 values = new Vector<String>(numValues);
 454                 for (int i = 0; i < numValues; i++) {
 455                     values.addElement(names[in.readShort()]);
 456                 }
 457             }
 458 result = new AttributeList(names[nameId], type, modifier, value,
 459                                        values, result);
 460             // We reverse the order of the linked list by doing this, but
 461             // that order isn't important.
 462         }
 463         return result;
 464     }
