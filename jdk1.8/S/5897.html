  27 /** CustomStringPool is an example of appliction provided data structure
  28  * for a DTM implementation to hold symbol references, e.g. elelment names.
  29  * It will follow the DTMDStringPool interface and use two simple methods
  30  * indexToString(int i) and stringToIndex(Sring s) to map between a set of
  31  * string values and a set of integer index values.  Therefore, an application
  32  * may improve DTM processing speed by substituting the DTM symbol resolution
  33  * tables with application specific quick symbol resolution tables.
  34  *
  35  * %REVIEW% The only difference between this an DTMStringPool seems to be that
  36  * it uses a java.lang.Hashtable full of Integers rather than implementing its
  37  * own hashing. Joe deliberately avoided that approach when writing
  38  * DTMStringPool, since it is both much more memory-hungry and probably slower
  39  * -- especially in JDK 1.1.x, where Hashtable is synchronized. We need to
  40  * either justify this implementation or discard it.
  41  *
  42  * Status: In progress, under discussion.
  43  * */

  61         public void removeAllElements()
  62         {
  63                 m_intToString.removeAllElements();
  64                 if (m_stringToInt != null)
  65                         m_stringToInt.clear();
  66         }

  68         /** @return string whose value is uniquely identified by this integer index.
  69          * @throws java.lang.ArrayIndexOutOfBoundsException
  70          *  if index doesn't map to a string.
  71          * */
  72         public String indexToString(int i)
  73         throws java.lang.ArrayIndexOutOfBoundsException
  74         {
  75                 return(String) m_intToString.elementAt(i);
  76         }
  77 
  78         /** @return integer index uniquely identifying the value of this string. */
  79         public int stringToIndex(String s)
  80         {
  81                 if (s==null) return NULL;
  82                 Integer iobj=(Integer)m_stringToInt.get(s);
  83                 if (iobj==null) {
  84                         m_intToString.addElement(s);
  85                         iobj=new Integer(m_intToString.size());
  86                         m_stringToInt.put(s,iobj);
  87                 }
  88                 return iobj.intValue();
  89         }
