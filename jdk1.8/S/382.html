  26 package org.omg.CORBA;
  27 
  28 
  29 /** An IDL struct in the CORBA module that
  30  *  stores information about a CORBA service available in the
  31  *  ORB implementation and is obtained from the ORB.get_service_information
  32  *  method.
  33  */
  34 public final class ServiceInformation implements org.omg.CORBA.portable.IDLEntity
  35 {
  36     /** Array of ints representing service options.
  37     */
  38     public int[] service_options;
  39 
  40     /** Array of ServiceDetails giving more details about the service.
  41     */
  42     public org.omg.CORBA.ServiceDetail[] service_details;
  43 
  44     /** Constructs a ServiceInformation object with empty service_options
  45     * and service_details.
  46     */
  47     public ServiceInformation() { }
  48 
  49     /** Constructs a ServiceInformation object with the given service_options
  50     * and service_details.
  51     * @param __service_options An array of ints describing the service options.
  52     * @param __service_details An array of ServiceDetails describing the service
  53     * details.
  54     */
  55     public ServiceInformation(int[] __service_options,
  56                               org.omg.CORBA.ServiceDetail[] __service_details)
  57     {
  58         service_options = __service_options;
  59         service_details = __service_details;
  60     }
  61 }
