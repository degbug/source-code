  42 /**
  43  * An unbounded {@linkplain BlockingQueue blocking queue} of
  44  * {@code Delayed} elements, in which an element can only be taken
  45  * when its delay has expired.  The head of the queue is that
  46  * {@code Delayed} element whose delay expired furthest in the
  47  * past.  If no delay has expired there is no head and {@code poll}
  48  * will return {@code null}. Expiration occurs when an element's
  49  * {@code getDelay(TimeUnit.NANOSECONDS)} method returns a value less
  50  * than or equal to zero.  Even though unexpired elements cannot be
  51  * removed using {@code take} or {@code poll}, they are otherwise
  52  * treated as normal elements. For example, the {@code size} method
  53  * returns the count of both expired and unexpired elements.
  54  * This queue does not permit null elements.
  55  *
  56  * This class and its iterator implement all of the
  57  * optional methods of the {@link Collection} and {@link
  58  * Iterator} interfaces.  The Iterator provided in method {@link
  59  * #iterator()} is not guaranteed to traverse the elements of
  60  * the DelayQueue in any particular order.
  61  *
  62  * This class is a member of the
  63  * 
  64  * Java Collections Framework.
  65  *
  66  * @since 1.5
  67  * @author Doug Lea
  68  * @param  the type of elements held in this collection
  69  */
  70 public class DelayQueue<E extends Delayed> extends AbstractQueue<E>
  71     implements BlockingQueue<E> {
  72 
  73     private final transient ReentrantLock lock = new ReentrantLock();
  74     private final PriorityQueue<E> q = new PriorityQueue<E>();

  76     /**
  77      * Thread designated to wait for the element at the head of
  78      * the queue.  This variant of the Leader-Follower pattern
  79      * (http://www.cs.wustl.edu/~schmidt/POSA/POSA2/) serves to
  80      * minimize unnecessary timed waiting.  When a thread becomes
  81      * the leader, it waits only for the next delay to elapse, but
  82      * other threads await indefinitely.  The leader thread must
  83      * signal some other thread before returning from take() or
  84      * poll(...), unless some other thread becomes leader in the
  85      * interim.  Whenever the head of the queue is replaced with
  86      * an element with an earlier expiration time, the leader
  87      * field is invalidated by being reset to null, and some
  88      * waiting thread, but not necessarily the current leader, is
  89      * signalled.  So waiting threads must be prepared to acquire
  90      * and lose leadership while waiting.
  91      */
  92     private Thread leader = null;

  95      * Condition signalled when a newer element becomes available
  96      * at the head of the queue or a new thread may need to
  97      * become leader.
  98      */
  99     private final Condition available = lock.newCondition();

 132      * @param e the element to add
 133      * @return {@code true}
 134      * @throws NullPointerException if the specified element is null
 135      */
 136     public boolean offer(E e) {
 137         final ReentrantLock lock = this.lock;
 138         lock.lock();
 139         try {
 140             q.offer(e);
 141             if (q.peek() == e) {
 142                 leader = null;
 143                 available.signal();
 144             }
 145             return true;
 146         } finally {
 147             lock.unlock();
 148         }

 176     /**
 177      * Retrieves and removes the head of this queue, or returns {@code null}
 178      * if this queue has no elements with an expired delay.
 179      *
 180      * @return the head of this queue, or {@code null} if this
 181      *         queue has no elements with an expired delay
 182      */
 183     public E poll() {
 184         final ReentrantLock lock = this.lock;
 185         lock.lock();
 186         try {
 187             E first = q.peek();
 188             if (first == null || first.getDelay(NANOSECONDS) > 0)
 189                 return null;
 190             else
 191                 return q.poll();
 192         } finally {
 193             lock.unlock();
 194         }
 195     }

 197     /**
 198      * Retrieves and removes the head of this queue, waiting if necessary
 199      * until an element with an expired delay is available on this queue.
 200      *
 201      * @return the head of this queue
 202      * @throws InterruptedException {@inheritDoc}
 203      */
 204     public E take() throws InterruptedException {
 205         final ReentrantLock lock = this.lock;
 206         lock.lockInterruptibly();
 207         try {
 208             for (;;) {
 209                 E first = q.peek();
 210                 if (first == null)
 211                     available.await();
 212                 else {
 213                     long delay = first.getDelay(NANOSECONDS);
 214                     if (delay <= 0)
 215                         return q.poll();
 216                     first = null; // don't retain ref while waiting
 217                     if (leader != null)
 218                         available.await();
 219                     else {
 220                         Thread thisThread = Thread.currentThread();
 221                         leader = thisThread;
 222                         try {
 223                             available.awaitNanos(delay);
 224                         } finally {
 225                             if (leader == thisThread)
 226                                 leader = null;
 227                         }
 228                     }
 229                 }
 230             }
 231         } finally {
 232             if (leader == null && q.peek() != null)
 233                 available.signal();
 234             lock.unlock();
 235         }
 236     }

 238     /**
 239      * Retrieves and removes the head of this queue, waiting if necessary
 240      * until an element with an expired delay is available on this queue,
 241      * or the specified wait time expires.
 242      *
 243      * @return the head of this queue, or {@code null} if the
 244      *         specified waiting time elapses before an element with
 245      *         an expired delay becomes available
 246      * @throws InterruptedException {@inheritDoc}
 247      */
 248     public E poll(long timeout, TimeUnit unit) throws InterruptedException {
 249         long nanos = unit.toNanos(timeout);
 250         final ReentrantLock lock = this.lock;
 251         lock.lockInterruptibly();
 252         try {
 253             for (;;) {
 254                 E first = q.peek();
 255                 if (first == null) {
 256                     if (nanos <= 0)
 257                         return null;
 258                     else
 259                         nanos = available.awaitNanos(nanos);
 260                 } else {
 261                     long delay = first.getDelay(NANOSECONDS);
 262                     if (delay <= 0)
 263                         return q.poll();
 264                     if (nanos <= 0)
 265                         return null;
 266                     first = null; // don't retain ref while waiting
 267                     if (nanos < delay || leader != null)
 268                         nanos = available.awaitNanos(nanos);
 269                     else {
 270                         Thread thisThread = Thread.currentThread();
 271                         leader = thisThread;
 272                         try {
 273                             long timeLeft = available.awaitNanos(delay);
 274                             nanos -= delay - timeLeft;
 275                         } finally {
 276                             if (leader == thisThread)
 277                                 leader = null;
 278                         }
 279                     }
 280                 }
 281             }
 282         } finally {
 283             if (leader == null && q.peek() != null)
 284                 available.signal();
 285             lock.unlock();
 286         }
 287     }

 319     /**
 320      * Returns first element only if it is expired.
 321      * Used only by drainTo.  Call only when holding lock.
 322      */
 323     private E peekExpired() {
 324         // assert lock.isHeldByCurrentThread();
 325         E first = q.peek();
 326         return (first == null || first.getDelay(NANOSECONDS) > 0) ?
 327             null : first;
 328     }

 330     /**
 331      * @throws UnsupportedOperationException {@inheritDoc}
 332      * @throws ClassCastException            {@inheritDoc}
 333      * @throws NullPointerException          {@inheritDoc}
 334      * @throws IllegalArgumentException      {@inheritDoc}
 335      */
 336     public int drainTo(Collection<? super E> c) {
 337         if (c == null)
 338             throw new NullPointerException();
 339         if (c == this)
 340             throw new IllegalArgumentException();
 341         final ReentrantLock lock = this.lock;
 342         lock.lock();
 343         try {
 344             int n = 0;
 345             for (E e; (e = peekExpired()) != null;) {
 346                 c.add(e);       // In this order, in case add() throws.
 347                 q.poll();
 348                 ++n;
 349             }
 350             return n;
 351         } finally {
 352             lock.unlock();
 353         }
 354     }

 356     /**
 357      * @throws UnsupportedOperationException {@inheritDoc}
 358      * @throws ClassCastException            {@inheritDoc}
 359      * @throws NullPointerException          {@inheritDoc}
 360      * @throws IllegalArgumentException      {@inheritDoc}
 361      */
 362     public int drainTo(Collection<? super E> c, int maxElements) {
 363         if (c == null)
 364             throw new NullPointerException();
 365         if (c == this)
 366             throw new IllegalArgumentException();
 367         if (maxElements <= 0)
 368             return 0;
 369         final ReentrantLock lock = this.lock;
 370         lock.lock();
 371         try {
 372             int n = 0;
 373             for (E e; n < maxElements && (e = peekExpired()) != null;) {
 374                 c.add(e);       // In this order, in case add() throws.
 375                 q.poll();
 376                 ++n;
 377             }
 378             return n;
 379         } finally {
 380             lock.unlock();
 381         }
 382     }

 433     /**
 434      * Returns an array containing all of the elements in this queue; the
 435      * runtime type of the returned array is that of the specified array.
 436      * The returned array elements are in no particular order.
 437      * If the queue fits in the specified array, it is returned therein.
 438      * Otherwise, a new array is allocated with the runtime type of the
 439      * specified array and the size of this queue.
 440      *
 441      * If this queue fits in the specified array with room to spare
 442      * (i.e., the array has more elements than this queue), the element in
 443      * the array immediately following the end of the queue is set to
 444      * {@code null}.
 445      *
 446      * Like the {@link #toArray()} method, this method acts as bridge between
 447      * array-based and collection-based APIs.  Further, this method allows
 448      * precise control over the runtime type of the output array, and may,
 449      * under certain circumstances, be used to save allocation costs.
 450      *
 451      * The following code can be used to dump a delay queue into a newly
 452      * allocated array of {@code Delayed}:
 453      *
 454      * 
 {@code Delayed[] a = q.toArray(new Delayed[0]);}

 455      *
 456      * Note that {@code toArray(new Object[0])} is identical in function to
 457      * {@code toArray()}.
 458      *
 459      * @param a the array into which the elements of the queue are to

 492     /**
 493      * Identity-based version for use in Itr.remove
 494      */
 495     void removeEQ(Object o) {
 496         final ReentrantLock lock = this.lock;
 497         lock.lock();
 498         try {
 499             for (Iterator<E> it = q.iterator(); it.hasNext(); ) {
 500                 if (o == it.next()) {
 501                     it.remove();
 502                     break;
 503                 }
 504             }
 505         } finally {
 506             lock.unlock();
 507         }
 508     }

 524     /**
 525      * Snapshot iterator that works off copy of underlying q array.
 526      */
 527     private class Itr implements Iterator<E> {
 528         final Object[] array; // Array of all elements
 529         int cursor;           // index of next element to return
 530         int lastRet;          // index of last element, or -1 if no such
 531 
 532         Itr(Object[] array) {
 533             lastRet = -1;
 534             this.array = array;
 535         }
 536 
 537         public boolean hasNext() {
 538             return cursor < array.length;
 539         }
 540 
 541         @SuppressWarnings("unchecked")
 542         public E next() {
 543             if (cursor >= array.length)
 544                 throw new NoSuchElementException();
 545             lastRet = cursor;
 546             return (E)array[cursor++];
 547         }
 548 
 549         public void remove() {
 550             if (lastRet < 0)
 551                 throw new IllegalStateException();
 552             removeEQ(array[lastRet]);
 553             lastRet = -1;
 554         }
 555     }
