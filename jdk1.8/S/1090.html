  41 /**
  42  * An object reference that may be updated atomically. See the {@link
  43  * java.util.concurrent.atomic} package specification for description
  44  * of the properties of atomic variables.
  45  * @since 1.5
  46  * @author Doug Lea
  47  * @param  The type of object referred to by this reference
  48  */
  49 public class AtomicReference<V> implements java.io.Serializable {
  50     private static final long serialVersionUID = -1848883965231344442L;
  51 
  52     private static final Unsafe unsafe = Unsafe.getUnsafe();
  53     private static final long valueOffset;
  54 
  55     static {
  56         try {
  57             valueOffset = unsafe.objectFieldOffset
  58                 (AtomicReference.class.getDeclaredField("value"));
  59         } catch (Exception ex) { throw new Error(ex); }
  60     }

  62     private volatile V value;

  65      * Creates a new AtomicReference with the given initial value.

  98      * Eventually sets to the given value.

 103     public final void lazySet(V newValue) {
 104         unsafe.putOrderedObject(this, valueOffset, newValue);
 105     }

 107     /**
 108      * Atomically sets the value to the given updated value
 109      * if the current value {@code ==} the expected value.
 110      * @param expect the expected value
 111      * @param update the new value
 112      * @return {@code true} if successful. False return indicates that
 113      * the actual value was not equal to the expected value.
 114      */
 115     public final boolean compareAndSet(V expect, V update) {
 116         return unsafe.compareAndSwapObject(this, valueOffset, expect, update);
 117     }

 119     /**

 121      * if the current value {@code ==} the expected value.

 131     public final boolean weakCompareAndSet(V expect, V update) {
 132         return unsafe.compareAndSwapObject(this, valueOffset, expect, update);
 133     }

 146     /**
 147      * Atomically updates the current value with the results of
 148      * applying the given function, returning the previous value. The
 149      * function should be side-effect-free, since it may be re-applied
 150      * when attempted updates fail due to contention among threads.
 151      *
 152      * @param updateFunction a side-effect-free function
 153      * @return the previous value
 154      * @since 1.8
 155      */
 156     public final V getAndUpdate(UnaryOperator<V> updateFunction) {
 157         V prev, next;
 158         do {
 159             prev = get();
 160             next = updateFunction.apply(prev);
 161         } while (!compareAndSet(prev, next));
 162         return prev;
 163     }

 165     /**

 170      *
 171      * @param updateFunction a side-effect-free function
 172      * @return the updated value

 175     public final V updateAndGet(UnaryOperator<V> updateFunction) {
 176         V prev, next;

 178             prev = get();
 179             next = updateFunction.apply(prev);
 180         } while (!compareAndSet(prev, next));
 181         return next;
 182     }

 184     /**
 185      * Atomically updates the current value with the results of
 186      * applying the given function to the current and given values,
 187      * returning the previous value. The function should be
 188      * side-effect-free, since it may be re-applied when attempted
 189      * updates fail due to contention among threads.  The function
 190      * is applied with the current value as its first argument,
 191      * and the given update as the second argument.
 192      *
 193      * @param x the update value
 194      * @param accumulatorFunction a side-effect-free function of two arguments
 195      * @return the previous value
 196      * @since 1.8
 197      */
 198     public final V getAndAccumulate(V x,
 199                                     BinaryOperator<V> accumulatorFunction) {
 200         V prev, next;
 201         do {
 202             prev = get();
 203             next = accumulatorFunction.apply(prev, x);
 204         } while (!compareAndSet(prev, next));
 205         return prev;
 206     }

 209      * Atomically updates the current value with the results of

 213      * updates fail due to contention among threads.  The function
 214      * is applied with the current value as its first argument,
 215      * and the given update as the second argument.
 216      *
 217      * @param x the update value
 218      * @param accumulatorFunction a side-effect-free function of two arguments

 222     public final V accumulateAndGet(V x,

 226             prev = get();

 230     }
