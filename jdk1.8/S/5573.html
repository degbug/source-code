  31 /**
  32  * A utility class for issuing error messages.
  33  *
  34  * A user of this class normally would create a singleton
  35  * instance of this class, passing the name
  36  * of the message class on the constructor. For example:
  37  * 
  38  * static Messages x = new Messages("org.package.MyMessages");
  39  * 
  40  * Later the message is typically generated this way if there are no
  41  * substitution arguments:
  42  * 
  43  * String msg = x.createMessage(org.package.MyMessages.KEY_ONE, null);
  44  * 
  45  * If there are arguments substitutions then something like this:
  46  * 
  47  * String filename = ...;
  48  * String directory = ...;
  49  * String msg = x.createMessage(org.package.MyMessages.KEY_TWO,
  50  *   new Object[] {filename, directory) );
  51  * 
  52  *
  53  * The constructor of an instance of this class must be given
  54  * the class name of a class that extends java.util.ListResourceBundle
  55  * ("org.package.MyMessages" in the example above).
  56  * The name should not have any language suffix
  57  * which will be added automatically by this utility class.
  58  *
  59  * The message class ("org.package.MyMessages")
  60  * must define the abstract method getContents() that is
  61  * declared in its base class, for example:
  62  * 
  63  * public Object[][] getContents() {return contents;}
  64  * 
  65  *
  66  * It is suggested that the message class expose its
  67  * message keys like this:
  68  * 
  69  *   public static final String KEY_ONE = "KEY1";
  70  *   public static final String KEY_TWO = "KEY2";
  71  *   . . .
  72  * 
  73  * and used through their names (KEY_ONE ...) rather than
  74  * their values ("KEY1" ...).
  75  *
  76  * The field contents (returned by getContents()
  77  * should be initialized something like this:
  78  * 
  79  * public static final Object[][] contents = {
  80  * { KEY_ONE, "Something has gone wrong!" },
  81  * { KEY_TWO, "The file ''{0}'' does not exist in directory ''{1}''." },
  82  * . . .
  83  * { KEY_N, "Message N" }  }
  84  * 
  85  *
  86  * Where that section of code with the KEY to Message mappings
  87  * (where the message classes 'contents' field is initialized)
  88  * can have the Message strings translated in an alternate language
  89  * in a errorResourceClass with a language suffix.
  90  *
  91  *
  92  * This class is not a public API, it is only public because it is
  93  * used in com.sun.org.apache.xml.internal.serializer.
  94  *
  95  *  @xsl.usage internal
  96  */

 139     /**
 140      * Creates a message from the specified key and replacement
 141      * arguments, localized to the given locale.
 142      *
 143      * @param msgKey  The key for the message text.
 144      * @param args    The arguments to be used as replacement text
 145      * in the message created.
 146      *
 147      * @return The formatted message string.
 148      * @xsl.usage internal
 149      */
 150     public final String createMessage(String msgKey, Object args[])
 151     {
 152         if (m_resourceBundle == null)
 153             m_resourceBundle = SecuritySupport.getResourceBundle(m_resourceBundleName);
 154 
 155         if (m_resourceBundle != null)
 156         {
 157             return createMsg(m_resourceBundle, msgKey, args);
 158         }
 159         else
 160             return "Could not load the resource bundles: "+ m_resourceBundleName;
 161     }
 162 
 163     /**
 164      * Creates a message from the specified key and replacement
 165      * arguments, localized to the given locale.
 166      *
 167      * @param errorCode The key for the message text.
 168      *
 169      * @param fResourceBundle The resource bundle to use.
 170      * @param msgKey  The message key to use.
 171      * @param args      The arguments to be used as replacement text
 172      *                  in the message created.
 173      *
 174      * @return The formatted message string.
 175      * @xsl.usage internal
 176      */
 177     private final String createMsg(
 178         ListResourceBundle fResourceBundle,
 179         String msgKey,
 180         Object args[]) //throws Exception
 181     {
 182 
 183         String fmsg = null;
 184         boolean throwex = false;
 185         String msg = null;
 186 
 187         if (msgKey != null)
 188             msg = fResourceBundle.getString(msgKey);
 189         else
 190             msgKey = "";
 191 
 192         if (msg == null)
 193         {
 194             throwex = true;
 195             /* The message is not in the bundle . . . this is bad,
 196              * so try to get the message that the message is not in the bundle
 197              */
 198             try
 199             {
 200 
 201                 msg =
 202                     java.text.MessageFormat.format(
 203                         MsgKey.BAD_MSGKEY,
 204                         new Object[] { msgKey, m_resourceBundleName });
 205             }
 206             catch (Exception e)
 207             {
 208                 /* even the message that the message is not in the bundle is
 209                  * not there ... this is really bad
 210                  */
 211                 msg =
 212                     "The message key '"
 213                         + msgKey
 214                         + "' is not in the message class '"
 215                         + m_resourceBundleName+"'";
 216             }
 217         }
 218         else if (args != null)
 219         {
 220             try
 221             {
 222                 // Do this to keep format from crying.
 223                 // This is better than making a bunch of conditional
 224                 // code all over the place.
 225                 int n = args.length;
 226 
 227                 for (int i = 0; i < n; i++)
 228                 {
 229                     if (null == args[i])
 230                         args[i] = "";
 231                 }
 232 
 233                 fmsg = java.text.MessageFormat.format(msg, args);
 234                 // if we get past the line above we have create the message ... hurray!
 235             }
 236             catch (Exception e)
 237             {
 238                 throwex = true;
 239                 try
 240                 {
 241                     // Get the message that the format failed.
 242                     fmsg =
 243                         java.text.MessageFormat.format(
 244                             MsgKey.BAD_MSGFORMAT,
 245                             new Object[] { msgKey, m_resourceBundleName });
 246                     fmsg += " " + msg;
 247                 }
 248                 catch (Exception formatfailed)
 249                 {
 250                     // We couldn't even get the message that the format of
 251                     // the message failed ... so fall back to English.
 252                     fmsg =
 253                         "The format of message '"
 254                             + msgKey
 255                             + "' in message class '"
 256                             + m_resourceBundleName
 257                             + "' failed.";
 258                 }
 259             }
 260         }
 261         else
 262             fmsg = msg;
 263 
 264         if (throwex)
 265         {
 266             throw new RuntimeException(fmsg);
 267         }
 268 
 269         return fmsg;
 270     }
