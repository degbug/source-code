  58     /**
  59      * Converts HSV components of a color to a set of RGB components.
  60      *
  61      * @param hsv  a float array with length equal to
  62      *             the number of HSV components
  63      * @param rgb  a float array with length of at least 3
  64      *             that contains RGB components of a color
  65      * @return a float array that contains RGB components
  66      */
  67     private static float[] HSVtoRGB(float[] hsv, float[] rgb) {
  68         if (rgb == null) {
  69             rgb = new float[3];
  70         }
  71         float hue = hsv[0];
  72         float saturation = hsv[1];
  73         float value = hsv[2];
  74 
  75         rgb[0] = value;
  76         rgb[1] = value;
  77         rgb[2] = value;
  78 
  79         if (saturation > 0.0f) {
  80             hue = (hue < 1.0f) ? hue * 6.0f : 0.0f;
  81             int integer = (int) hue;
  82             float f = hue - (float) integer;
  83             switch (integer) {
  84                 case 0:
  85                     rgb[1] *= 1.0f - saturation * (1.0f - f);
  86                     rgb[2] *= 1.0f - saturation;
  87                     break;
  88                 case 1:
  89                     rgb[0] *= 1.0f - saturation * f;
  90                     rgb[2] *= 1.0f - saturation;
  91                     break;
  92                 case 2:
  93                     rgb[0] *= 1.0f - saturation;
  94                     rgb[2] *= 1.0f - saturation * (1.0f - f);
  95                     break;
  96                 case 3:
  97                     rgb[0] *= 1.0f - saturation;
  98                     rgb[1] *= 1.0f - saturation * f;
  99                     break;
 100                 case 4:
 101                     rgb[0] *= 1.0f - saturation * (1.0f - f);
 102                     rgb[1] *= 1.0f - saturation;
 103                     break;
 104                 case 5:
 105                     rgb[1] *= 1.0f - saturation;
 106                     rgb[2] *= 1.0f - saturation * f;
 107                     break;
 108             }
 109         }
 110         return rgb;
 111     }

 113     /**
 114      * Converts RGB components of a color to a set of HSV components.
 115      *
 116      * @param rgb  a float array with length of at least 3
 117      *             that contains RGB components of a color
 118      * @param hsv  a float array with length equal to
 119      *             the number of HSV components
 120      * @return a float array that contains HSV components
 121      */
 122     private static float[] RGBtoHSV(float[] rgb, float[] hsv) {
 123         if (hsv == null) {
 124             hsv = new float[3];
 125         }
 126         float max = ColorModelHSL.max(rgb[0], rgb[1], rgb[2]);
 127         float min = ColorModelHSL.min(rgb[0], rgb[1], rgb[2]);
 128 
 129         float saturation = max - min;
 130         if (saturation > 0.0f) {
 131             saturation /= max;
 132         }
 133         hsv[0] = ColorModelHSL.getHue(rgb[0], rgb[1], rgb[2], max, min);
 134         hsv[1] = saturation;
 135         hsv[2] = max;
 136         return hsv;
 137     }
