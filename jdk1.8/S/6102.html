  58     /**
  59      * Factory method to instantiate object types. Returns a pre-defined
  60      * instance for "java.lang.Object" and "java.lang.String".
  61      */
  62     public static Type newObjectType(String javaClassName) {
  63         if (javaClassName == "java.lang.Object") {
  64             return Type.Object;
  65         }
  66         else if (javaClassName == "java.lang.String") {
  67             return Type.ObjectString;
  68         }
  69         else {
  70             //
  71             java.security.AccessControlContext acc = java.security.AccessController.getContext();
  72             acc.checkPermission(new RuntimePermission("getContextClassLoader"));
  73             return new ObjectType(javaClassName);
  74         }
  75     }

 154     /**
 155      * Translates object of this type to an object of type type.
 156      * Expects an object of the former type and pushes an object of the latter
 157      * if not boolean. If type type is boolean then a branchhandle
 158      * list (to be appended to the false list) is returned.
 159      */
 160     public FlowList translateToDesynthesized(ClassGenerator classGen,
 161                                              MethodGenerator methodGen,
 162                                              Type type) {
 163         FlowList fl = null;
 164         if (type == Type.Boolean) {
 165             fl = translateToDesynthesized(classGen, methodGen,
 166                                           (BooleanType)type);
 167         }
 168         else {
 169             translateTo(classGen, methodGen, type);
 170         }
 171         return fl;
 172     }
