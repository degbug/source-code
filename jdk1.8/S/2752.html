 134     /**
 135      * Returns the value for key.  If the value is a
 136      * <code>UIDefaults.LazyValue</code> then the real
 137      * value is computed with <code>LazyValue.createValue()</code>,
 138      * the table entry is replaced, and the real value is returned.
 139      * If the value is an <code>UIDefaults.ActiveValue</code>
 140      * the table entry is not replaced - the value is computed
 141      * with <code>ActiveValue.createValue()</code> for each
 142      * <code>get()</code> call.
 143      *
 144      * If the key is not found in the table then it is searched for in the list
 145      * of resource bundles maintained by this object.  The resource bundles are
 146      * searched most recently added first using the locale returned by
 147      * <code>getDefaultLocale</code>.  <code>LazyValues</code> and
 148      * <code>ActiveValues</code> are not supported in the resource bundles.
 149 
 150      *
 151      * @param key the desired key
 152      * @return the value for <code>key</code>
 153      * @see LazyValue
 154      * @see ActiveValue
 155      * @see java.util.Hashtable#get
 156      * @see #getDefaultLocale
 157      * @see #addResourceBundle
 158      * @since 1.4
 159      */
 160     public Object get(Object key) {
 161         Object value = getFromHashtable( key );
 162         return (value != null) ? value : getFromResourceBundle(key, null);
 163     }

 165     /**
 166      * Looks up up the given key in our Hashtable and resolves LazyValues
 167      * or ActiveValues.
 168      */
 169     private Object getFromHashtable(final Object key) {
 170         /* Quickly handle the common case, without grabbing
 171          * a lock.
 172          */
 173         Object value = super.get(key);
 174         if ((value != PENDING) &&
 175             !(value instanceof ActiveValue) &&
 176             !(value instanceof LazyValue)) {
 177             return value;
 178         }
 179 
 180         /* If the LazyValue for key is being constructed by another
 181          * thread then wait and then return the new value, otherwise drop
 182          * the lock and construct the ActiveValue or the LazyValue.
 183          * We use the special value PENDING to mark LazyValues that
 184          * are being constructed.
 185          */
 186         synchronized(this) {
 187             value = super.get(key);
 188             if (value == PENDING) {
 189                 do {
 190                     try {
 191                         this.wait();
 192                     }
 193                     catch (InterruptedException e) {
 194                     }
 195                     value = super.get(key);
 196                 }
 197                 while(value == PENDING);
 198                 return value;
 199             }
 200             else if (value instanceof LazyValue) {
 201                 super.put(key, PENDING);
 202             }
 203             else if (!(value instanceof ActiveValue)) {
 204                 return value;
 205             }
 206         }
 207 
 208         /* At this point we know that the value of key was
 209          * a LazyValue or an ActiveValue.
 210          */
 211         if (value instanceof LazyValue) {
 212             try {
 213                 /* If an exception is thrown we'll just put the LazyValue
 214                  * back in the table.
 215                  */
 216                 value = ((LazyValue)value).createValue(this);
 217             }
 218             finally {
 219                 synchronized(this) {
 220                     if (value == null) {
 221                         super.remove(key);
 222                     }
 223                     else {
 224                         super.put(key, value);
 225                     }
 226                     this.notifyAll();
 227                 }
 228             }
 229         }
 230         else {
 231             value = ((ActiveValue)value).createValue(this);
 232         }
 233 
 234         return value;
 235     }

 268     /**
 269      * Looks up given key in our resource bundles.
 270      */
 271     private Object getFromResourceBundle(Object key, Locale l) {
 272 
 273         if( resourceBundles == null ||
 274             resourceBundles.isEmpty() ||
 275             !(key instanceof String) ) {
 276             return null;
 277         }
 278 
 279         // A null locale means use the default locale.
 280         if( l == null ) {
 281             if( defaultLocale == null )
 282                 return null;
 283             else
 284                 l = defaultLocale;
 285         }
 286 
 287         synchronized(this) {
 288             return getResourceCache(l).get(key);
 289         }
 290     }

 292     /**
 293      * Returns a Map of the known resources for the given locale.
 294      */
 295     private Map<String, Object> getResourceCache(Locale l) {
 296         Map<String, Object> values = resourceCache.get(l);
 297 
 298         if (values == null) {
 299             values = new TextAndMnemonicHashMap();
 300             for (int i=resourceBundles.size()-1; i >= 0; i--) {
 301                 String bundleName = resourceBundles.get(i);
 302                 try {
 303                     Control c = CoreResourceBundleControl.getRBControlInstance(bundleName);
 304                     ResourceBundle b;
 305                     if (c != null) {
 306                         b = ResourceBundle.getBundle(bundleName, l, c);
 307                     } else {
 308                         b = ResourceBundle.getBundle(bundleName, l);
 309                     }
 310                     Enumeration keys = b.getKeys();
 311 
 312                     while (keys.hasMoreElements()) {
 313                         String key = (String)keys.nextElement();
 314 
 315                         if (values.get(key) == null) {
 316                             Object value = b.getObject(key);
 317 
 318                             values.put(key, value);
 319                         }
 320                     }
 321                 } catch( MissingResourceException mre ) {
 322                     // Keep looking
 323                 }
 324             }
 325             resourceCache.put(l, values);
 326         }
 327         return values;
 328     }

 331      * Sets the value of <code>key</code> to <code>value</code> for all locales.
 332      * If <code>key</code> is a string and the new value isn't
 333      * equal to the old one, fire a <code>PropertyChangeEvent</code>.
 334      * If value is <code>null</code>, the key is removed from the table.
 335      *
 336      * @param key    the unique <code>Object</code> who's value will be used
 337      *          to retrieve the data value associated with it
 338      * @param value  the new <code>Object</code> to store as data under
 339      *          that key
 340      * @return the previous <code>Object</code> value, or <code>null</code>
 341      * @see #putDefaults
 342      * @see java.util.Hashtable#put
 343      */
 344     public Object put(Object key, Object value) {
 345         Object oldValue = (value == null) ? super.remove(key) : super.put(key, value);
 346         if (key instanceof String) {
 347             firePropertyChange((String)key, oldValue, value);
 348         }
 349         return oldValue;
 350     }

 353     /**
 354      * Puts all of the key/value pairs in the database and
 355      * unconditionally generates one <code>PropertyChangeEvent</code>.
 356      * The events oldValue and newValue will be <code>null</code> and its
 357      * <code>propertyName</code> will be "UIDefaults".  The key/value pairs are
 358      * added for all locales.
 359      *
 360      * @param keyValueList  an array of key/value pairs
 361      * @see #put
 362      * @see java.util.Hashtable#put
 363      */
 364     public void putDefaults(Object[] keyValueList) {
 365         for(int i = 0, max = keyValueList.length; i < max; i += 2) {
 366             Object value = keyValueList[i + 1];
 367             if (value == null) {
 368                 super.remove(keyValueList[i]);
 369             }
 370             else {
 371                 super.put(keyValueList[i], value);
 372             }
 373         }
 374         firePropertyChange("UIDefaults", null, null);
 375     }

 656      * class that implements the corresponding <code>ComponentUI</code>

 658      * up the class with <code>uiClassLoader.loadClass()</code> if a non
 659      * <code>null</code>
 660      * class loader is provided, <code>classForName()</code> otherwise.
 661      * <p>
 662      * If a mapping for <code>uiClassID</code> exists or if the specified
 663      * class can't be found, return <code>null</code>.
 664      * <p>
 665      * This method is used by <code>getUI</code>, it's usually
 666      * not necessary to call it directly.
 667      *
 668      * @param uiClassID  a string containing the class ID
 669      * @param uiClassLoader the object which will load the class
 670      * @return the value of <code>Class.forName(get(uidClassID))</code>
 671      * @see #getUI
 672      */
 673     public Class<? extends ComponentUI>
 674         getUIClass(String uiClassID, ClassLoader uiClassLoader)
 675     {
 676         try {
 677             String className = (String)get(uiClassID);
 678             if (className != null) {
 679                 ReflectUtil.checkPackageAccess(className);
 680 
 681                 Class cls = (Class)get(className);
 682                 if (cls == null) {
 683                     if (uiClassLoader == null) {
 684                         cls = SwingUtilities.loadSystemClass(className);
 685                     }
 686                     else {
 687                         cls = uiClassLoader.loadClass(className);
 688                     }
 689                     if (cls != null) {
 690                         // Save lookup for future use, as forName is slow.
 691                         put(className, cls);
 692                     }
 693                 }
 694                 return cls;
 695             }
 696         }
 697         catch (ClassNotFoundException e) {
 698             return null;
 699         }
 700         catch (ClassCastException e) {
 701             return null;
 702         }
 703         return null;

 739      * specified component.  In other words create the look
 740      * and feel specific delegate object for <code>target</code>.
 741      * This is done in two steps:
 742      * <ul>
 743      * <li> Look up the name of the <code>ComponentUI</code> implementation
 744      * class under the value returned by <code>target.getUIClassID()</code>.
 745      * <li> Use the implementation classes static <code>createUI()</code>

 747      * </ul>
 748      * @param target  the <code>JComponent</code> which needs a UI
 749      * @return the <code>ComponentUI</code> object
 750      */
 751     public ComponentUI getUI(JComponent target) {
 752 
 753         Object cl = get("ClassLoader");
 754         ClassLoader uiClassLoader =
 755             (cl != null) ? (ClassLoader)cl : target.getClass().getClassLoader();
 756         Class<? extends ComponentUI> uiClass = getUIClass(target.getUIClassID(), uiClassLoader);
 757         Object uiObject = null;
 758 
 759         if (uiClass == null) {
 760             getUIError("no ComponentUI class for: " + target);
 761         }
 762         else {
 763             try {
 764                 Method m = (Method)get(uiClass);
 765                 if (m == null) {
 766                     m = uiClass.getMethod("createUI", new Class[]{JComponent.class});
 767                     put(uiClass, m);
 768                 }
 769                 uiObject = MethodUtil.invoke(m, null, new Object[]{target});
 770             }
 771             catch (NoSuchMethodException e) {
 772                 getUIError("static createUI() method not found in " + uiClass);
 773             }
 774             catch (Exception e) {
 775                 getUIError("createUI() failed for " + target + " " + e);
 776             }
 777         }
 778 
 779         return (ComponentUI)uiObject;

1010         private String className;

1061          *              paramaters to the static method in class c

1072         /**
1073          * Creates the value retrieved from the <code>UIDefaults</code> table.
1074          * The object is created each time it is accessed.
1075          *
1076          * @param table  a <code>UIDefaults</code> table
1077          * @return the created <code>Object</code>
1078          */
1079         public Object createValue(final UIDefaults table) {
1080             // In order to pick up the security policy in effect at the
1081             // time of creation we use a doPrivileged with the
1082             // AccessControlContext that was in place when this was created.
1083             if (acc == null && System.getSecurityManager() != null) {
1084                 throw new SecurityException("null AccessControlContext");
1085             }
1086             return AccessController.doPrivileged(new PrivilegedAction<Object>() {
1087                 public Object run() {
1088                     try {
1089                         Class<?> c;
1090                         Object cl;
1091                         // See if we should use a separate ClassLoader
1092                         if (table == null || !((cl = table.get("ClassLoader"))
1093                                                instanceof ClassLoader)) {
1094                             cl = Thread.currentThread().
1095                                         getContextClassLoader();
1096                             if (cl == null) {
1097                                 // Fallback to the system class loader.
1098                                 cl = ClassLoader.getSystemClassLoader();
1099                             }
1100                         }
1101                         ReflectUtil.checkPackageAccess(className);
1102                         c = Class.forName(className, true, (ClassLoader)cl);
1103                         SwingUtilities2.checkAccess(c.getModifiers());
1104                         if (methodName != null) {
1105                             Class[] types = getClassArray(args);
1106                             Method m = c.getMethod(methodName, types);
1107                             return MethodUtil.invoke(m, c, args);
1108                         } else {
1109                             Class[] types = getClassArray(args);
1110                             Constructor constructor = c.getConstructor(types);
1111                             SwingUtilities2.checkAccess(constructor.getModifiers());
1112                             return constructor.newInstance(args);
1113                         }
1114                     } catch(Exception e) {
1115                         // Ideally we would throw an exception, unfortunately
1116                         // often times there are errors as an initial look and
1117                         // feel is loaded before one can be switched. Perhaps a
1118                         // flag should be added for debugging, so that if true
1119                         // the exception would be thrown.
1120                     }
1121                     return null;
1122                 }
1123             }, acc);
1124         }

1129          * by substituting primitive types for their Object counterparts,

1131          * <code>UIResource</code> tag.
1132          */
1133         private Class[] getClassArray(Object[] args) {
1134             Class[] types = null;
1135             if (args!=null) {
1136                 types = new Class[args.length];
1137                 for (int i = 0; i< args.length; i++) {
1138                     /* PENDING(ges): At present only the primitive types

1140                        handle all primitive types */
1141                     if (args[i] instanceof java.lang.Integer) {
1142                         types[i]=Integer.TYPE;
1143                     } else if (args[i] instanceof java.lang.Boolean) {
1144                         types[i]=Boolean.TYPE;
1145                     } else if (args[i] instanceof javax.swing.plaf.ColorUIResource) {
1146                         /* PENDING(ges) Currently the Reflection APIs do not
1147                            search superclasses of parameters supplied for

1150                            directly instead of adding a massive amount
1151                            of mechanism for this.  Eventually this will
1152                            probably need to handle the general case as well.
1153                            */
1154                         types[i]=java.awt.Color.class;
1155                     } else {
1156                         types[i]=args[i].getClass();
1157                     }
1158                 }
1159             }
1160             return types;
1161         }

1213     /**

1215      * can have a mnemonic. For example:
1216      *   FileChooser.saveButton.textAndMnemonic=&Save
1217      * For this case method get returns "Save" for the key "FileChooser.saveButtonText" and
1218      * mnemonic "S" for the key "FileChooser.saveButtonMnemonic"
1219      *
1220      * There are several patterns for the text and mnemonic suffixes which are checked by the
1221      * <code>TextAndMnemonicHashMap</code> class.
1222      * Patterns which are converted to the xxx.textAndMnemonic key:
1223      * (xxxNameText, xxxNameMnemonic)
1224      * (xxxNameText, xxxMnemonic)
1225      * (xxx.nameText, xxx.mnemonic)
1226      * (xxxText, xxxMnemonic)
1227      *
1228      * These patterns can have a mnemonic index in format
1229      * (xxxDisplayedMnemonicIndex)
1230      *
1231      * Pattern which is converted to the xxx.titleAndMnemonic key:
1232      * (xxxTitle, xxxMnemonic)
1233      *
1234      */
1235     private static class TextAndMnemonicHashMap extends HashMap<String, Object> {
1236 
1237         static final String AND_MNEMONIC = "AndMnemonic";
1238         static final String TITLE_SUFFIX = ".titleAndMnemonic";
1239         static final String TEXT_SUFFIX = ".textAndMnemonic";
1240 
1241         @Override
1242         public Object get(Object key) {
1243 
1244             Object value = super.get(key);
1245 
1246             if (value == null) {
1247 
1248                 boolean checkTitle = false;
1249 
1250                 String stringKey = key.toString();
1251                 String compositeKey = null;
1252 
1253                 if (stringKey.endsWith(AND_MNEMONIC)) {
1254                     return null;
1255                 }
1256 
1257                 if (stringKey.endsWith(".mnemonic")) {
1258                     compositeKey = composeKey(stringKey, 9, TEXT_SUFFIX);
1259                 } else if (stringKey.endsWith("NameMnemonic")) {
1260                     compositeKey = composeKey(stringKey, 12, TEXT_SUFFIX);
1261                 } else if (stringKey.endsWith("Mnemonic")) {
1262                     compositeKey = composeKey(stringKey, 8, TEXT_SUFFIX);
1263                     checkTitle = true;
1264                 }
1265 
1266                 if (compositeKey != null) {
1267                     value = super.get(compositeKey);
1268                     if (value == null && checkTitle) {
1269                         compositeKey = composeKey(stringKey, 8, TITLE_SUFFIX);
1270                         value = super.get(compositeKey);
1271                     }
1272 
1273                     return value == null ? null : getMnemonicFromProperty(value.toString());
1274                 }
1275 
1276                 if (stringKey.endsWith("NameText")) {
1277                     compositeKey = composeKey(stringKey, 8, TEXT_SUFFIX);
1278                 } else if (stringKey.endsWith(".nameText")) {
1279                     compositeKey = composeKey(stringKey, 9, TEXT_SUFFIX);
1280                 } else if (stringKey.endsWith("Text")) {
