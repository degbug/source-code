 382     int digit(int ch, int radix) {

 384         if (radix >= Character.MIN_RADIX && radix <= Character.MAX_RADIX) {
 385             int val = getProperties(ch);

 391                 // Java supradecimal digit
 392                 value = (ch + ((val & 0x3E0) >> 5) & 0x1F) + 10;

 394         }

 614      int findInCharMap(int ch) {
 615         if (charMap == null || charMap.length == 0) {
 616             return -1;
 617         }
 618         int top, bottom, current;
 619         bottom = 0;
 620         top = charMap.length;
 621         current = top/2;
 622         // invariant: top > current >= bottom && ch >= CharacterData.charMap[bottom][0]
 623         while (top - bottom > 1) {
 624             if (ch >= charMap[current][0][0]) {
 625                 bottom = current;
 626             } else {
 627                 top = current;
 628             }
 629             current = (top + bottom) / 2;
 630         }
 631         if (ch == charMap[current][0][0]) return current;
 632         else return -1;
