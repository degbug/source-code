  60 import java.util.Stack;
  61 import com.sun.org.apache.bcel.internal.classfile.JavaClass;
  62 
  63 /**
  64  * Utility class implementing a (typesafe) stack of JavaClass objects.
  65  *
  66  * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
  67  * @see Stack
  68 */
  69 public class ClassStack implements java.io.Serializable {
  70   private Stack stack = new Stack();
  71 
  72   public void      push(JavaClass clazz) { stack.push(clazz); }
  73   public JavaClass pop()                 { return (JavaClass)stack.pop(); }
  74   public JavaClass top()                 { return (JavaClass)stack.peek(); }
  75   public boolean   empty()               { return stack.empty(); }
  76 }
