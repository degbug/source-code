  54   /**
  55    * Set an argument expression for a function.  This method is called by the
  56    * XPath compiler.
  57    *
  58    * @param arg non-null expression that represents the argument.
  59    * @param argNum The argument number index.
  60    *
  61    * @throws WrongNumberArgsException If a derived class determines that the
  62    * number of arguments is incorrect.
  63    */
  64   public void setArg(Expression arg, int argNum)
  65           throws WrongNumberArgsException
  66   {
  67 
  68     if (argNum < 3)
  69       super.setArg(arg, argNum);
  70     else
  71     {
  72       if (null == m_args)
  73       {
  74         m_args = new Expression[1];
  75         m_args[0] = arg;
  76       }
  77       else
  78       {
  79 
  80         // Slow but space conservative.
  81         Expression[] args = new Expression[m_args.length + 1];
  82 
  83         System.arraycopy(m_args, 0, args, 0, m_args.length);
  84 
  85         args[m_args.length] = arg;
  86         m_args = args;
  87       }
  88       arg.exprSetParent(this);
  89     }
  90   }

  92   /**
  93    * This function is used to fixup variables from QNames to stack frame
  94    * indexes at stylesheet build time.
  95    * @param vars List of QNames that correspond to variables.  This list
  96    * should be searched backwards for the first qualified name that
  97    * corresponds to the variable reference qname.  The position of the
  98    * QName in the vector from the start of the vector will be its position
  99    * in the stack frame (but variables above the globalsTop value will need
 100    * to be offset to the current stack frame).
 101    */
 102   public void fixupVariables(java.util.Vector vars, int globalsSize)
 103   {
 104     super.fixupVariables(vars, globalsSize);
 105     if(null != m_args)
 106     {
 107       for (int i = 0; i < m_args.length; i++)
 108       {
 109         m_args[i].fixupVariables(vars, globalsSize);
 110       }
 111     }

 139   /**
 140    * Tell if this expression or it's subexpressions can traverse outside
 141    * the current subtree.
 142    *
 143    * @return true if traversal outside the context node's subtree can occur.
 144    */
 145   public boolean canTraverseOutsideSubtree()
 146   {
 147 
 148     if (super.canTraverseOutsideSubtree())
 149       return true;
 150     else
 151     {
 152       int n = m_args.length;
 153 
 154       for (int i = 0; i < n; i++)
 155       {
 156         if (m_args[i].canTraverseOutsideSubtree())
 157           return true;
 158       }
 159 
 160       return false;
 161     }
 162   }

 209     /**
 210      * @see Expression#deepEquals(Expression)
 211      */
 212     public boolean deepEquals(Expression expr)
 213     {
 214       if (!super.deepEquals(expr))
 215             return false;
 216 
 217       FunctionMultiArgs fma = (FunctionMultiArgs) expr;
 218       if (null != m_args)
 219       {
 220         int n = m_args.length;
 221         if ((null == fma) || (fma.m_args.length != n))
 222               return false;
 223 
 224         for (int i = 0; i < n; i++)
 225         {
 226           if (!m_args[i].deepEquals(fma.m_args[i]))
 227                 return false;
 228         }
 229 
 230       }
 231       else if (null != fma.m_args)
 232       {
 233           return false;
 234       }
 235 
 236       return true;
 237     }
