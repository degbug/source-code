  44   /**
  45    * Fill in a list with nodes that match a space delimited list if ID
  46    * ID references.
  47    *
  48    * @param xctxt The runtime XPath context.
  49    * @param docContext The document where the nodes are being looked for.
  50    * @param refval A space delimited list of ID references.
  51    * @param usedrefs List of references for which nodes were found.
  52    * @param nodeSet Node set where the nodes will be added to.
  53    * @param mayBeMore true if there is another set of nodes to be looked for.
  54    *
  55    * @return The usedrefs value.
  56    */
  57   private StringVector getNodesByID(XPathContext xctxt, int docContext,
  58                                     String refval, StringVector usedrefs,
  59                                     NodeSetDTM nodeSet, boolean mayBeMore)
  60   {
  61 
  62     if (null != refval)
  63     {
  64       String ref = null;
  65 //      DOMHelper dh = xctxt.getDOMHelper();
  66       StringTokenizer tokenizer = new StringTokenizer(refval);
  67       boolean hasMore = tokenizer.hasMoreTokens();
  68       DTM dtm = xctxt.getDTM(docContext);
  69 
  70       while (hasMore)
  71       {
  72         ref = tokenizer.nextToken();
  73         hasMore = tokenizer.hasMoreTokens();
  74 
  75         if ((null != usedrefs) && usedrefs.contains(ref))
  76         {
  77           ref = null;
  78 
  79           continue;
  80         }
  81 
  82         int node = dtm.getElementById(ref);
  83 
  84         if (DTM.NULL != node)
  85           nodeSet.addNodeInDocOrder(node, xctxt);
  86 
  87         if ((null != ref) && (hasMore || mayBeMore))
  88         {
  89           if (null == usedrefs)
  90             usedrefs = new StringVector();
  91 
  92           usedrefs.addElement(ref);
  93         }
  94       }
  95     }
  96 
  97     return usedrefs;
  98   }

 100   /**
 101    * Execute the function.  The function must return
 102    * a valid object.
 103    * @param xctxt The current execution context.
 104    * @return A valid XObject.
 105    *
 106    * @throws javax.xml.transform.TransformerException
 107    */
 108   public XObject execute(XPathContext xctxt) throws javax.xml.transform.TransformerException
 109   {
 110 
 111     int context = xctxt.getCurrentNode();
 112     DTM dtm = xctxt.getDTM(context);
 113     int docContext = dtm.getDocument();
 114 
 115     if (DTM.NULL == docContext)
 116       error(xctxt, XPATHErrorResources.ER_CONTEXT_HAS_NO_OWNERDOC, null);
 117 
 118     XObject arg = m_arg0.execute(xctxt);
 119     int argType = arg.getType();
 120     XNodeSet nodes = new XNodeSet(xctxt.getDTMManager());
 121     NodeSetDTM nodeSet = nodes.mutableNodeset();
 122 
 123     if (XObject.CLASS_NODESET == argType)
 124     {
 125       DTMIterator ni = arg.iter();
 126       StringVector usedrefs = null;
 127       int pos = ni.nextNode();
 128 
 129       while (DTM.NULL != pos)
 130       {
 131         DTM ndtm = ni.getDTM(pos);
 132         String refval = ndtm.getStringValue(pos).toString();
 133 
 134         pos = ni.nextNode();
 135         usedrefs = getNodesByID(xctxt, docContext, refval, usedrefs, nodeSet,
 136                                 DTM.NULL != pos);
 137       }
 138       // ni.detach();
 139     }
 140     else if (XObject.CLASS_NULL == argType)
 141     {
 142       return nodes;
 143     }
 144     else
 145     {
 146       String refval = arg.str();
 147 
 148       getNodesByID(xctxt, docContext, refval, null, nodeSet, false);
 149     }
 150 
 151     return nodes;
 152   }
