  26 package org.omg.CORBA;
  27 
  28 /**
  29  * An object that represents an ORB service: its service_detail_type
  30  * field contains the type of the ORB service, and its service_detail
  31  * field contains a description of the ORB service.
  32 
  33  *
  34  * @author RIP Team
  35  */
  36 public final class ServiceDetail implements org.omg.CORBA.portable.IDLEntity
  37 {
  38     /**
  39      * The type of the ORB service that this ServiceDetail
  40      * object represents.
  41      */
  42     public int service_detail_type;
  43 
  44     /**
  45      * The data describing the ORB service that this ServiceDetail
  46      * object represents.
  47      */
  48     public byte[] service_detail;
  49 
  50     /**
  51      * Constructs a ServiceDetail object with 0 for the type of
  52      * ORB service and an empty description.
  53      */
  54     public ServiceDetail() { }
  55 
  56     /**
  57      * Constructs a ServiceDetail object with the given
  58      * ORB service type and the given description.
  59      *
  60      * @param service_detail_type an int specifying the type of
  61      *                            ORB service
  62      * @param service_detail a byte array describing the ORB service
  63      */
  64     public ServiceDetail(int service_detail_type, byte[] service_detail) {
  65         this.service_detail_type = service_detail_type;
  66         this.service_detail = service_detail;
  67     }
  68 }
