 105   private final void setFlag(int flag, boolean set) {
 106     if((access_flags & flag) != 0) { // Flag is set already
 107       if(!set) // Delete flag ?
 108         access_flags ^= flag;
 109     } else {   // Flag not set
 110       if(set)  // Set flag ?
 111         access_flags |= flag;
 112     }
 113   }
