  28 /**

  31 public class NotEquals extends Operation

  34 
  35   /**
  36    * Apply the operation to two operands, and return the result.
  37    *
  38    *
  39    * @param left non-null reference to the evaluated left operand.
  40    * @param right non-null reference to the evaluated right operand.
  41    *
  42    * @return non-null reference to the XObject that represents the result of the operation.
  43    *
  44    * @throws javax.xml.transform.TransformerException
  45    */
  46   public XObject operate(XObject left, XObject right)
  47           throws javax.xml.transform.TransformerException
  48   {
  49     return (left.notEquals(right)) ? XBoolean.S_TRUE : XBoolean.S_FALSE;
  50   }
  51 }
