  94          * <p>Is specified object model supported by this
  95          * <code>XPathFactory</code>?</p>
  96          *
  97          * @param objectModel Specifies the object model which the returned
  98          * <code>XPathFactory</code> will understand.
  99          *
 100          * @return <code>true</code> if <code>XPathFactory</code> supports
 101          * <code>objectModel</code>, else <code>false</code>.
 102          *
 103          * @throws NullPointerException If <code>objectModel</code> is <code>null</code>.
 104          * @throws IllegalArgumentException If <code>objectModel.length() == 0</code>.
 105          */
 106         public boolean isObjectModelSupported(String objectModel) {
 107 
 108             if (objectModel == null) {
 109                 String fmsg = XSLMessages.createXPATHMessage(
 110                         XPATHErrorResources.ER_OBJECT_MODEL_NULL,
 111                         new Object[] { this.getClass().getName() } );
 112 
 113                 throw new NullPointerException( fmsg );
 114             }
 115 
 116             if (objectModel.length() == 0) {
 117                 String fmsg = XSLMessages.createXPATHMessage(
 118                         XPATHErrorResources.ER_OBJECT_MODEL_EMPTY,
 119                         new Object[] { this.getClass().getName() } );
 120                 throw new IllegalArgumentException( fmsg );
 121             }
 122 
 123             // know how to support default object model, W3C DOM
 124             if (objectModel.equals(XPathFactory.DEFAULT_OBJECT_MODEL_URI)) {
 125                 return true;
 126             }
 127 
 128             // don't know how to support anything else
 129             return false;
 130         }

 145         /**
 146          * <p>Set a feature for this <code>XPathFactory</code> and
 147          * <code>XPath</code>s created by this factory.</p>
 148          *
 149          * <p>
 150          * Feature names are fully qualified {@link java.net.URI}s.
 151          * Implementations may define their own features.
 152          * An {@link XPathFactoryConfigurationException} is thrown if this
 153          * <code>XPathFactory</code> or the <code>XPath</code>s
 154          *  it creates cannot support the feature.
 155          * It is possible for an <code>XPathFactory</code> to expose a feature
 156          * value but be unable to change its state.
 157          * </p>
 158          *
 159          * <p>See {@link javax.xml.xpath.XPathFactory} for full documentation
 160          * of specific features.</p>
 161          *
 162          * @param name Feature name.
 163          * @param value Is feature state <code>true</code> or <code>false</code>.
 164          *
 165          * @throws XPathFactoryConfigurationException if this
 166          * <code>XPathFactory</code> or the <code>XPath</code>s
 167          *   it creates cannot support this feature.
 168          * @throws NullPointerException if <code>name</code> is
 169          * <code>null</code>.
 170          */
 171         public void setFeature(String name, boolean value)
 172                 throws XPathFactoryConfigurationException {
 173 
 174             // feature name cannot be null
 175             if (name == null) {
 176                 String fmsg = XSLMessages.createXPATHMessage(
 177                         XPATHErrorResources.ER_FEATURE_NAME_NULL,
 178                         new Object[] { CLASS_NAME, new Boolean( value) } );
 179                 throw new NullPointerException( fmsg );
 180              }
 181 
 182             // secure processing?
 183             if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
 184                 if ((_isSecureMode) && (!value)) {
 185                     String fmsg = XSLMessages.createXPATHMessage(
 186                             XPATHErrorResources.ER_SECUREPROCESSING_FEATURE,
 187                             new Object[] { name, CLASS_NAME, new Boolean(value) } );
 188                     throw new XPathFactoryConfigurationException( fmsg );
 189                 }
 190 
 191                 _isNotSecureProcessing = !value;
 192                 if (value && _featureManager != null) {
 193                     _featureManager.setValue(FeatureManager.Feature.ORACLE_ENABLE_EXTENSION_FUNCTION,
 194                             FeaturePropertyBase.State.FSP, XalanConstants.FEATURE_FALSE);
 195                 }
 196 
 197                 // all done processing feature
 198                 return;
 199             }
 200             if (name.equals(XalanConstants.ORACLE_FEATURE_SERVICE_MECHANISM)) {
 201                 //in secure mode, let _useServicesMechanism be determined by the constructor
 202                 if (!_isSecureMode)
 203                     _useServicesMechanism = value;
 204                 return;
 205             }
 206 
 207             if (_featureManager != null &&
 208                     _featureManager.setValue(name, FeaturePropertyBase.State.APIPROPERTY, value)) {
 209                 return;
 210             }
 211 
 212             // unknown feature
 213             String fmsg = XSLMessages.createXPATHMessage(
 214                     XPATHErrorResources.ER_FEATURE_UNKNOWN,
 215                     new Object[] { name, CLASS_NAME, new Boolean(value) } );
 216             throw new XPathFactoryConfigurationException( fmsg );
 217         }

 219         /**
 220          * <p>Get the state of the named feature.</p>
 221          *
 222          * <p>
 223          * Feature names are fully qualified {@link java.net.URI}s.
 224          * Implementations may define their own features.
 225          * An {@link XPathFactoryConfigurationException} is thrown if this
 226          * <code>XPathFactory</code> or the <code>XPath</code>s
 227          * it creates cannot support the feature.
 228          * It is possible for an <code>XPathFactory</code> to expose a feature
 229          * value but be unable to change its state.
 230          * </p>
 231          *
 232          * @param name Feature name.
 233          *
 234          * @return State of the named feature.
 235          *
 236          * @throws XPathFactoryConfigurationException if this
 237          * <code>XPathFactory</code> or the <code>XPath</code>s
 238          *   it creates cannot support this feature.
 239          * @throws NullPointerException if <code>name</code> is
 240          * <code>null</code>.
 241          */
 242         public boolean getFeature(String name)
 243                 throws XPathFactoryConfigurationException {
 244 
 245             // feature name cannot be null
 246             if (name == null) {
 247                 String fmsg = XSLMessages.createXPATHMessage(
 248                         XPATHErrorResources.ER_GETTING_NULL_FEATURE,
 249                         new Object[] { CLASS_NAME } );
 250                 throw new NullPointerException( fmsg );
 251             }
 252 
 253             // secure processing?
 254             if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
 255                 return !_isNotSecureProcessing;
 256             }
 257             if (name.equals(XalanConstants.ORACLE_FEATURE_SERVICE_MECHANISM)) {
 258                 return _useServicesMechanism;
 259             }
 260 
 261             /** Check to see if the property is managed by the security manager **/
 262             String propertyValue = (_featureManager != null) ?
 263                     _featureManager.getValueAsString(name) : null;
 264             if (propertyValue != null) {
 265                 return _featureManager.isFeatureEnabled(name);
 266             }
 267 
 268             // unknown feature
 269             String fmsg = XSLMessages.createXPATHMessage(
 270                     XPATHErrorResources.ER_GETTING_UNKNOWN_FEATURE,
 271                     new Object[] { name, CLASS_NAME } );
 272 
 273             throw new XPathFactoryConfigurationException( fmsg );
 274         }
