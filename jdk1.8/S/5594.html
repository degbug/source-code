  23 package com.sun.org.apache.xml.internal.serializer;

  25 import java.io.IOException;
  26 import java.io.OutputStream;
  27 import java.io.Writer;

  31 /**
  32  * This class writes ASCII to a byte stream as quickly as possible.  For the
  33  * moment it does not do buffering, though I reserve the right to do some
  34  * buffering down the line if I can prove that it will be faster even if the
  35  * output stream is buffered.
  36  *
  37  * This class is only used internally within Xalan.
  38  *
  39  * @xsl.usage internal
  40  */
  41 class WriterToASCI extends Writer implements WriterChain
  42 {
  43 
  44   /** The byte stream to write to.  */
  45   private final OutputStream m_os;
  46 
  47   /**
  48    * Create an unbuffered ASCII writer.
  49    *
  50    *
  51    * @param os The byte stream to write to.
  52    */
  53   public WriterToASCI(OutputStream os)
  54   {
  55     m_os = os;
  56   }
  57 
  58   /**
  59    * Write a portion of an array of characters.
  60    *
  61    * @param  chars  Array of characters
  62    * @param  start   Offset from which to start writing characters
  63    * @param  length   Number of characters to write
  64    *
  65    * @exception  IOException  If an I/O error occurs
  66    *
  67    * @throws java.io.IOException
  68    */
  69   public void write(char chars[], int start, int length)
  70           throws java.io.IOException
  71   {
  72 
  73     int n = length+start;
  74 
  75     for (int i = start; i < n; i++)
  76     {
  77       m_os.write(chars[i]);
  78     }
  79   }
  80 
  81   /**
  82    * Write a single character.  The character to be written is contained in
  83    * the 16 low-order bits of the given integer value; the 16 high-order bits
  84    * are ignored.
  85    *
  86    * <p> Subclasses that intend to support efficient single-character output
  87    * should override this method.
  88    *
  89    * @param c  int specifying a character to be written.
  90    * @exception  IOException  If an I/O error occurs
  91    */
  92   public void write(int c) throws IOException
  93   {
  94     m_os.write(c);
  95   }
  96 
  97   /**
  98    * Write a string.
  99    *
 100    * @param  s String to be written
 101    *
 102    * @exception  IOException  If an I/O error occurs
 103    */
 104   public void write(String s) throws IOException
 105   {
 106     int n = s.length();
 107     for (int i = 0; i < n; i++)
 108     {
 109       m_os.write(s.charAt(i));
 110     }
 111   }
 112 
 113   /**
 114    * Flush the stream.  If the stream has saved any characters from the
 115    * various write() methods in a buffer, write them immediately to their
 116    * intended destination.  Then, if that destination is another character or
 117    * byte stream, flush it.  Thus one flush() invocation will flush all the
 118    * buffers in a chain of Writers and OutputStreams.
 119    *
 120    * @exception  IOException  If an I/O error occurs
 121    */
 122   public void flush() throws java.io.IOException
 123   {
 124     m_os.flush();
 125   }
 126 
 127   /**
 128    * Close the stream, flushing it first.  Once a stream has been closed,
 129    * further write() or flush() invocations will cause an IOException to be
 130    * thrown.  Closing a previously-closed stream, however, has no effect.
 131    *
 132    * @exception  IOException  If an I/O error occurs
 133    */
 134   public void close() throws java.io.IOException
 135   {
 136     m_os.close();
 137   }
 138 
 139   /**
 140    * Get the output stream where the events will be serialized to.
 141    *
 142    * @return reference to the result stream, or null of only a writer was
 143    * set.
 144    */
 145   public OutputStream getOutputStream()
 146   {
 147     return m_os;
 148   }
 149 
 150   /**

 153   public Writer getWriter()

 155       return null;
 156   }
 157 }
