      Appends numPages pages to the end of this
      Book.  Each of the pages is associated with
      page.
      @param painter   the Printable instance that renders
                       the page
      @param page      the size and orientation of the page
      @param numPages  the number of pages to be added to the
                       this Book.
      @throws NullPointerException
               If the painter or page
               argument is null

 153     public void append(Printable painter, PageFormat page, int numPages) {
 154         BookPage bookPage = new BookPage(painter, page);
 155         int pageIndex = mPages.size();
 156         int newSize = pageIndex + numPages;
 157 
 158         mPages.setSize(newSize);
 159         for(int i = pageIndex; i < newSize; i++){
 160             mPages.setElementAt(bookPage, i);
 161         }
 162     }
