  29 /**
  30  * Represents "user defined" exceptions thrown by MBean methods
  31  * in the agent. It "wraps" the actual "user defined" exception thrown.
  32  * This exception will be built by the MBeanServer when a call to an
  33  * MBean method results in an unknown exception.

  35  * @since 1.5
  36  */

  42

  44      * @serial Encapsulated {@link Exception}

  46     private java.lang.Exception exception ;
  47 
  48 
  49     /**
  50      * Creates an MBeanException that wraps the actual java.lang.Exception.
  51      *
  52      * @param e the wrapped exception.
  53      */
  54     public MBeanException(java.lang.Exception e) {
  55         super() ;
  56         exception = e ;
  57     }

  59     /**
  60      * Creates an MBeanException that wraps the actual java.lang.Exception with

  64      * @param message the detail message.
  65      */
  66     public MBeanException(java.lang.Exception e, String message) {
  67         super(message) ;

  69     }
