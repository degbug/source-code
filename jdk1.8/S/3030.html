 105     @Override
 106     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
 107         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 108         componentColors = extendedCacheKeys;
 109         //generate this entire method. Each state/bg/fg/border combo that has
 110         //been painted gets its own KEY and paint method.
 111         switch(state) {
 112             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
 113             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
 114             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
 115             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
 116             case BACKGROUND_ENABLED_WINDOWNOTFOCUSED: paintBackgroundEnabledAndWindowNotFocused(g); break;
 117             case BACKGROUND_MOUSEOVER_WINDOWNOTFOCUSED: paintBackgroundMouseOverAndWindowNotFocused(g); break;
 118             case BACKGROUND_PRESSED_WINDOWNOTFOCUSED: paintBackgroundPressedAndWindowNotFocused(g); break;
 119 
 120         }
 121     }
