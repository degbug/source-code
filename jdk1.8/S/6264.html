  57   /**
  58    * The str:align function aligns a string within another string.
  59    * <p>
  60    * The first argument gives the target string to be aligned. The second argument gives
  61    * the padding string within which it is to be aligned.
  62    * <p>
  63    * If the target string is shorter than the padding string then a range of characters
  64    * in the padding string are repaced with those in the target string. Which characters
  65    * are replaced depends on the value of the third argument, which gives the type of
  66    * alignment. It can be one of 'left', 'right' or 'center'. If no third argument is
  67    * given or if it is not one of these values, then it defaults to left alignment.
  68    * <p>
  69    * With left alignment, the range of characters replaced by the target string begins
  70    * with the first character in the padding string. With right alignment, the range of
  71    * characters replaced by the target string ends with the last character in the padding
  72    * string. With center alignment, the range of characters replaced by the target string
  73    * is in the middle of the padding string, such that either the number of unreplaced
  74    * characters on either side of the range is the same or there is one less on the left
  75    * than there is on the right.
  76    * <p>
  77    * If the target string is longer than the padding string, then it is truncated to be
  78    * the same length as the padding string and returned.
  79    *
  80    * @param targetStr The target string
  81    * @param paddingStr The padding string
  82    * @param type The type of alignment
  83    *
  84    * @return The string after alignment
  85    */
  86   public static String align(String targetStr, String paddingStr, String type)
  87   {
  88     if (targetStr.length() >= paddingStr.length())
  89       return targetStr.substring(0, paddingStr.length());
  90 
  91     if (type.equals("right"))
  92     {
  93       return paddingStr.substring(0, paddingStr.length() - targetStr.length()) + targetStr;
  94     }
  95     else if (type.equals("center"))
  96     {
  97       int startIndex = (paddingStr.length() - targetStr.length()) / 2;
  98       return paddingStr.substring(0, startIndex) + targetStr + paddingStr.substring(startIndex + targetStr.length());
  99     }
 100     // Default is left
 101     else
 102     {
 103       return targetStr + paddingStr.substring(targetStr.length());
 104     }
 105   }

 115   /**
 116    * The str:concat function takes a node set and returns the concatenation of the
 117    * string values of the nodes in that node set. If the node set is empty, it returns
 118    * an empty string.
 119    *
 120    * @param nl A node set
 121    * @return The concatenation of the string values of the nodes in that node set
 122    */
 123   public static String concat(NodeList nl)
 124   {
 125     StringBuffer sb = new StringBuffer();
 126     for (int i = 0; i < nl.getLength(); i++)
 127     {
 128       Node node = nl.item(i);
 129       String value = toString(node);
 130 
 131       if (value != null && value.length() > 0)
 132         sb.append(value);
 133     }
 134 
 135     return sb.toString();
 136   }

 138   /**
 139    * The str:padding function creates a padding string of a certain length.
 140    * The first argument gives the length of the padding string to be created.
 141    * The second argument gives a string to be used to create the padding. This
 142    * string is repeated as many times as is necessary to create a string of the
 143    * length specified by the first argument; if the string is more than a character
 144    * long, it may have to be truncated to produce the required length. If no second
 145    * argument is specified, it defaults to a space (' '). If the second argument is
 146    * an empty string, str:padding returns an empty string.
 147    *
 148    * @param length The length of the padding string to be created
 149    * @param pattern The string to be used as pattern
 150    *
 151    * @return A padding string of the given length
 152    */
 153   public static String padding(double length, String pattern)
 154   {
 155     if (pattern == null || pattern.length() == 0)
 156       return "";
 157 
 158     StringBuffer sb = new StringBuffer();
 159     int len = (int)length;
 160     int numAdded = 0;
 161     int index = 0;
 162     while (numAdded < len)
 163     {
 164       if (index == pattern.length())
 165         index = 0;
 166 
 167       sb.append(pattern.charAt(index));
 168       index++;
 169       numAdded++;
 170     }
 171 
 172     return sb.toString();
 173   }

 183   /**
 184    * The str:split function splits up a string and returns a node set of token
 185    * elements, each containing one token from the string.
 186    * <p>
 187    * The first argument is the string to be split. The second argument is a pattern
 188    * string. The string given by the first argument is split at any occurrence of
 189    * this pattern. For example:
 190    * <pre>
 191    * str:split('a, simple, list', ', ') gives the node set consisting of:
 192    *
 193    * <token>a</token>
 194    * <token>simple</token>
 195    * <token>list</token>
 196    * </pre>
 197    * If the second argument is omitted, the default is the string '&#x20;' (i.e. a space).
 198    *
 199    * @param str The string to be split
 200    * @param pattern The pattern
 201    *
 202    * @return A node set of split tokens
 203    */
 204   public static NodeList split(String str, String pattern)
 205   {
 206 
 207 
 208     NodeSet resultSet = new NodeSet();
 209     resultSet.setShouldCacheNodes(true);
 210 
 211     boolean done = false;
 212     int fromIndex = 0;
 213     int matchIndex = 0;
 214     String token = null;
 215 
 216     while (!done && fromIndex < str.length())
 217     {
 218       matchIndex = str.indexOf(pattern, fromIndex);
 219       if (matchIndex >= 0)
 220       {
 221         token = str.substring(fromIndex, matchIndex);
 222         fromIndex = matchIndex + pattern.length();
 223       }
 224       else
 225       {
 226         done = true;
 227         token = str.substring(fromIndex);
 228       }
 229 
 230       Document doc = getDocument();
 231       synchronized (doc)
 232       {
 233         Element element = doc.createElement("token");
 234         Text text = doc.createTextNode(token);
 235         element.appendChild(text);
 236         resultSet.addNode(element);
 237       }
 238     }
 239 
 240     return resultSet;
 241   }

 251   /**
 252    * The str:tokenize function splits up a string and returns a node set of token
 253    * elements, each containing one token from the string.
 254    * <p>
 255    * The first argument is the string to be tokenized. The second argument is a
 256    * string consisting of a number of characters. Each character in this string is
 257    * taken as a delimiting character. The string given by the first argument is split
 258    * at any occurrence of any of these characters. For example:
 259    * <pre>
 260    * str:tokenize('2001-06-03T11:40:23', '-T:') gives the node set consisting of:
 261    *
 262    * <token>2001</token>
 263    * <token>06</token>
 264    * <token>03</token>
 265    * <token>11</token>
 266    * <token>40</token>
 267    * <token>23</token>
 268    * </pre>
 269    * If the second argument is omitted, the default is the string '&#x9;&#xA;&#xD;&#x20;'
 270    * (i.e. whitespace characters).
 271    * <p>
 272    * If the second argument is an empty string, the function returns a set of token
 273    * elements, each of which holds a single character.
 274    * <p>
 275    * Note: This one is different from the tokenize extension function in the Xalan
 276    * namespace. The one in Xalan returns a set of Text nodes, while this one wraps
 277    * the Text nodes inside the token Element nodes.
 278    *
 279    * @param toTokenize The string to be tokenized
 280    * @param delims The delimiter string
 281    *
 282    * @return A node set of split token elements
 283    */
 284   public static NodeList tokenize(String toTokenize, String delims)
 285   {
 286 
 287 
 288     NodeSet resultSet = new NodeSet();
 289 
 290     if (delims != null && delims.length() > 0)
 291     {
 292       StringTokenizer lTokenizer = new StringTokenizer(toTokenize, delims);
 293 
 294       Document doc = getDocument();
 295       synchronized (doc)
 296       {
 297         while (lTokenizer.hasMoreTokens())
 298         {
 299           Element element = doc.createElement("token");
 300           element.appendChild(doc.createTextNode(lTokenizer.nextToken()));
 301           resultSet.addNode(element);
 302         }
 303       }
 304     }
 305     // If the delimiter is an empty string, create one token Element for
 306     // every single character.
 307     else
 308     {
 309 
 310       Document doc = getDocument();
 311       synchronized (doc)
 312       {
 313         for (int i = 0; i < toTokenize.length(); i++)
 314         {
 315           Element element = doc.createElement("token");
 316           element.appendChild(doc.createTextNode(toTokenize.substring(i, i+1)));
 317           resultSet.addNode(element);
 318         }
 319       }
 320     }
 321 
 322     return resultSet;
 323   }
