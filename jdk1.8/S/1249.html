  41 /**
  42  *
  43  *
  44  * This ProtectionDomain class encapsulates the characteristics of a domain,
  45  * which encloses a set of classes whose instances are granted a set
  46  * of permissions when being executed on behalf of a given set of Principals.
  47  * 
  48  * A static set of permissions can be bound to a ProtectionDomain when it is
  49  * constructed; such permissions are granted to the domain regardless of the
  50  * Policy in force. However, to support dynamic security policies, a
  51  * ProtectionDomain can also be constructed such that it is dynamically
  52  * mapped to a set of permissions by the current Policy whenever a permission
  53  * is checked.
  54  * 
  55  *
  56  * @author Li Gong
  57  * @author Roland Schemers
  58  * @author Gary Ellison
  59  */

 128     /**
 129      * Creates a new ProtectionDomain with the given CodeSource and
 130      * Permissions. If the permissions object is not null, then
 131      *  {@code setReadOnly())} will be called on the passed in
 132      * Permissions object. The only permissions granted to this domain
 133      * are the ones specified; the current Policy will not be consulted.
 134      *
 135      * @param codesource the codesource associated with this domain
 136      * @param permissions the permissions granted to this domain
 137      */
 138     public ProtectionDomain(CodeSource codesource,
 139                             PermissionCollection permissions) {
 140         this.codesource = codesource;
 141         if (permissions != null) {
 142             this.permissions = permissions;
 143             this.permissions.setReadOnly();
 144             if (permissions instanceof Permissions &&
 145                 ((Permissions)permissions).allPermission != null) {
 146                 hasAllPerm = true;
 147             }
 148         }
 149         this.classloader = null;
 150         this.principals = new Principal[0];
 151         staticPermissions = true;
 152     }

 154     /**
 155      * Creates a new ProtectionDomain qualified by the given CodeSource,
 156      * Permissions, ClassLoader and array of Principals. If the
 157      * permissions object is not null, then {@code setReadOnly()}
 158      * will be called on the passed in Permissions object.
 159      * The permissions granted to this domain are dynamic; they include
 160      * both the static permissions passed to this constructor, and any
 161      * permissions granted to this domain by the current Policy at the
 162      * time a permission is checked.
 163      * 
 164      * This constructor is typically used by
 165      * {@link SecureClassLoader ClassLoaders}
 166      * and {@link DomainCombiner DomainCombiners} which delegate to
 167      * {@code Policy} to actively associate the permissions granted to
 168      * this domain. This constructor affords the
 169      * Policy provider the opportunity to augment the supplied
 170      * PermissionCollection to reflect policy changes.
 171      * 
 172      *
 173      * @param codesource the CodeSource associated with this domain
 174      * @param permissions the permissions granted to this domain
 175      * @param classloader the ClassLoader associated with this domain
 176      * @param principals the array of Principals associated with this
 177      * domain. The contents of the array are copied to protect against
 178      * subsequent modification.
 179      * @see Policy#refresh
 180      * @see Policy#getPermissions(ProtectionDomain)
 181      * @since 1.4
 182      */
 183     public ProtectionDomain(CodeSource codesource,
 184                             PermissionCollection permissions,
 185                             ClassLoader classloader,
 186                             Principal[] principals) {
 187         this.codesource = codesource;
 188         if (permissions != null) {
 189             this.permissions = permissions;
 190             this.permissions.setReadOnly();
 191             if (permissions instanceof Permissions &&
 192                 ((Permissions)permissions).allPermission != null) {
 193                 hasAllPerm = true;
 194             }
 195         }
 196         this.classloader = classloader;
 197         this.principals = (principals != null ? principals.clone():
 198                            new Principal[0]);
 199         staticPermissions = false;
 200     }

 245     /**
 246      * Check and see if this ProtectionDomain implies the permissions
 247      * expressed in the Permission object.
 248      * 
 249      * The set of permissions evaluated is a function of whether the
 250      * ProtectionDomain was constructed with a static set of permissions
 251      * or it was bound to a dynamically mapped set of permissions.
 252      * 
 253      * If the ProtectionDomain was constructed to a
 254      * {@link #ProtectionDomain(CodeSource, PermissionCollection)
 255      * statically bound} PermissionCollection then the permission will
 256      * only be checked against the PermissionCollection supplied at
 257      * construction.
 258      * 
 259      * However, if the ProtectionDomain was constructed with
 260      * the constructor variant which supports
 261      * {@link #ProtectionDomain(CodeSource, PermissionCollection,
 262      * ClassLoader, java.security.Principal[]) dynamically binding}
 263      * permissions, then the permission will be checked against the
 264      * combination of the PermissionCollection supplied at construction and
 265      * the current Policy binding.
 266      * 
 267      *
 268      * @param permission the Permission object to check.
 269      *
 270      * @return true if "permission" is implicit to this ProtectionDomain.
 271      */
 272     public boolean implies(Permission permission) {
 273 
 274         if (hasAllPerm) {
 275             // internal permission collection already has AllPermission -
 276             // no need to go to policy
 277             return true;
 278         }
 279 
 280         if (!staticPermissions &&
 281             Policy.getPolicyNoCheck().implies(this, permission))
 282             return true;
 283         if (permissions != null)
 284             return permissions.implies(permission);
 285 
 286         return false;
 287     }

 295      * Convert a ProtectionDomain to a String.

 297     @Override public String toString() {
 298         String pals = "<no principals>";
 299         if (principals != null && principals.length > 0) {
 300             StringBuilder palBuf = new StringBuilder("(principals ");
 301 
 302             for (int i = 0; i < principals.length; i++) {
 303                 palBuf.append(principals[i].getClass().getName() +
 304                             " \"" + principals[i].getName() +
 305                             "\"");
 306                 if (i < principals.length-1)
 307                     palBuf.append(",\n");
 308                 else
 309                     palBuf.append(")\n");
 310             }
 311             pals = palBuf.toString();
 312         }
 313 
 314         // Check if policy is set; we don't want to load
 315         // the policy prematurely here
 316         PermissionCollection pc = Policy.isSet() && seeAllp() ?
 317                                       mergePermissions():
 318                                       getPermissions();
 319 
 320         return "ProtectionDomain "+
 321             " "+codesource+"\n"+
 322             " "+classloader+"\n"+
 323             " "+pals+"\n"+
 324             " "+pc+"\n";
 325     }

 327     /**
 328      * Return true (merge policy permissions) in the following cases:
 329      *
 330      * . SecurityManager is null
 331      *
 332      * . SecurityManager is not null,
 333      *          debug is not null,
 334      *          SecurityManager impelmentation is in bootclasspath,
 335      *          Policy implementation is in bootclasspath
 336      *          (the bootclasspath restrictions avoid recursion)
 337      *
 338      * . SecurityManager is not null,
 339      *          debug is null,
 340      *          caller has Policy.getPolicy permission
 341      */
 342     private static boolean seeAllp() {
 343         SecurityManager sm = System.getSecurityManager();
 344 
 345         if (sm == null) {
 346             return true;
 347         } else {
 348             if (debug != null) {
 349                 if (sm.getClass().getClassLoader() == null &&
 350                     Policy.getPolicyNoCheck().getClass().getClassLoader()
 351                                                                 == null) {
 352                     return true;
 353                 }
 354             } else {
 355                 try {
 356                     sm.checkPermission(SecurityConstants.GET_POLICY_PERMISSION);
 357                     return true;
 358                 } catch (SecurityException se) {
 359                     // fall thru and return false
 360                 }
 361             }
 362         }
 363 
 364         return false;
 365     }

 367     private PermissionCollection mergePermissions() {
 368         if (staticPermissions)
 369             return permissions;
 370 
 371         PermissionCollection perms =
 372             java.security.AccessController.doPrivileged
 373             (new java.security.PrivilegedAction<PermissionCollection>() {
 374                     public PermissionCollection run() {
 375                         Policy p = Policy.getPolicyNoCheck();
 376                         return p.getPermissions(ProtectionDomain.this);
 377                     }
 378                 });
 379 
 380         Permissions mergedPerms = new Permissions();
 381         int swag = 32;
 382         int vcap = 8;
 383         Enumeration<Permission> e;
 384         List<Permission> pdVector = new ArrayList<>(vcap);
 385         List<Permission> plVector = new ArrayList<>(swag);
 386 
 387         //
 388         // Build a vector of domain permissions for subsequent merge
 389         if (permissions != null) {
 390             synchronized (permissions) {
 391                 e = permissions.elements();
 392                 while (e.hasMoreElements()) {
 393                     pdVector.add(e.nextElement());
 394                 }
 395             }
 396         }
 397 
 398         //
 399         // Build a vector of Policy permissions for subsequent merge
 400         if (perms != null) {
 401             synchronized (perms) {
 402                 e = perms.elements();
 403                 while (e.hasMoreElements()) {
 404                     plVector.add(e.nextElement());
 405                     vcap++;
 406                 }
 407             }
 408         }
 409 
 410         if (perms != null && permissions != null) {
 411             //
 412             // Weed out the duplicates from the policy. Unless a refresh
 413             // has occurred since the pd was consed this should result in
 414             // an empty vector.
 415             synchronized (permissions) {
 416                 e = permissions.elements();   // domain vs policy
 417                 while (e.hasMoreElements()) {
 418                     Permission pdp = e.nextElement();
 419                     Class<?> pdpClass = pdp.getClass();
 420                     String pdpActions = pdp.getActions();
 421                     String pdpName = pdp.getName();
 422                     for (int i = 0; i < plVector.size(); i++) {
 423                         Permission pp = plVector.get(i);
 424                         if (pdpClass.isInstance(pp)) {
 425                             // The equals() method on some permissions
 426                             // have some side effects so this manual
 427                             // comparison is sufficient.
 428                             if (pdpName.equals(pp.getName()) &&
 429                                 pdpActions.equals(pp.getActions())) {
 430                                 plVector.remove(i);
 431                                 break;
 432                             }
 433                         }
 434                     }
 435                 }
 436             }
 437         }
 438 
 439         if (perms !=null) {
 440             // the order of adding to merged perms and permissions
 441             // needs to preserve the bugfix 4301064
 442 
 443             for (int i = plVector.size()-1; i >= 0; i--) {
 444                 mergedPerms.add(plVector.get(i));
 445             }
 446         }
 447         if (permissions != null) {
 448             for (int i = pdVector.size()-1; i >= 0; i--) {
 449                 mergedPerms.add(pdVector.get(i));
 450             }
 451         }
 452 
 453         return mergedPerms;
 454     }
