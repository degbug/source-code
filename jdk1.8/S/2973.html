  35 /**
  36  * Provides the Synth L&F UI delegate for
  37  * {@link javax.swing.JPopupMenu}.
  38  *
  39  * @author Georges Saab
  40  * @author David Karlton
  41  * @author Arnaud Weber
  42  * @since 1.7
  43  */
  44 public class SynthPopupMenuUI extends BasicPopupMenuUI
  45                               implements PropertyChangeListener, SynthUI {
  46     private SynthStyle style;
  47 
  48     /**
  49      * Creates a new UI object for the given component.
  50      *
  51      * @param x component to create UI object for
  52      * @return the UI object
  53      */
  54     public static ComponentUI createUI(JComponent x) {
  55         return new SynthPopupMenuUI();
  56     }
  57 
  58     /**
  59      * {@inheritDoc}
  60      */
  61     @Override
  62     public void installDefaults() {
  63         if (popupMenu.getLayout() == null ||
  64             popupMenu.getLayout() instanceof UIResource) {
  65             popupMenu.setLayout(new SynthMenuLayout(popupMenu, BoxLayout.Y_AXIS));
  66         }
  67         updateStyle(popupMenu);
  68     }
  69 
  70     private void updateStyle(JComponent c) {
  71         SynthContext context = getContext(c, ENABLED);
  72         SynthStyle oldStyle = style;
  73         style = SynthLookAndFeel.updateStyle(context, this);
  74         if (style != oldStyle) {
  75             if (oldStyle != null) {
  76                 uninstallKeyboardActions();
  77                 installKeyboardActions();
  78             }
  79         }
  80         context.dispose();
  81     }
  82 
  83     /**
  84      * {@inheritDoc}
  85      */
  86     @Override
  87     protected void installListeners() {
  88         super.installListeners();
  89         popupMenu.addPropertyChangeListener(this);
  90     }
  91 
  92     /**
  93      * {@inheritDoc}
  94      */
  95     @Override
  96     protected void uninstallDefaults() {
  97         SynthContext context = getContext(popupMenu, ENABLED);
  98 
  99         style.uninstallDefaults(context);
 100         context.dispose();
 101         style = null;
 102 
 103         if (popupMenu.getLayout() instanceof UIResource) {
 104             popupMenu.setLayout(null);
 105         }
 106     }
 107 
 108     /**
 109      * {@inheritDoc}
 110      */
 111     @Override

 113         super.uninstallListeners();

 115     }

 133     /**
 134      * Notifies this UI delegate to repaint the specified component.
 135      * This method paints the component background, then calls
 136      * the {@link #paint(SynthContext,Graphics)} method.
 137      *
 138      * In general, this method does not need to be overridden by subclasses.
 139      * All Look and Feel rendering code should reside in the {@code paint} method.
 140      *
 141      * @param g the {@code Graphics} object used for painting
 142      * @param c the component being painted
 143      * @see #paint(SynthContext,Graphics)
 144      */
 145     @Override
 146     public void update(Graphics g, JComponent c) {
 147         SynthContext context = getContext(c);
 148 
 149         SynthLookAndFeel.update(context, g);
 150         context.getPainter().paintPopupMenuBackground(context,
 151                           g, 0, 0, c.getWidth(), c.getHeight());
 152         paint(context, g);
 153         context.dispose();
 154     }
 155 
 156     /**
 157      * Paints the specified component according to the Look and Feel.

 160      *
 161      * @param g the {@code Graphics} object used for painting
 162      * @param c the component being painted
 163      * @see #paint(SynthContext,Graphics)
 164      */
 165     @Override
 166     public void paint(Graphics g, JComponent c) {
 167         SynthContext context = getContext(c);
 168 
 169         paint(context, g);
 170         context.dispose();
 171     }
