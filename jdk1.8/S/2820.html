 113     static boolean isWindows() {
 114         if (!checkedWindows) {
 115             OSInfo.OSType osType = AccessController.doPrivileged(OSInfo.getOSTypeAction());
 116             if (osType == OSInfo.OSType.WINDOWS) {
 117                 isWindows = true;
 118                 String systemFonts = AccessController.doPrivileged(
 119                     new GetPropertyAction("swing.useSystemFontSettings"));
 120                 useSystemFonts = (systemFonts != null &&
 121                                (Boolean.valueOf(systemFonts).booleanValue()));
 122             }
 123             checkedWindows = true;
 124         }
 125         return isWindows;
 126     }

 128     /**
 129      * Returns true if system fonts should be used, this is only useful
 130      * for windows.
 131      */
 132     static boolean useSystemFonts() {
 133         if (isWindows() && useSystemFonts) {
 134             if (METAL_LOOK_AND_FEEL_INITED) {
 135                 Object value = UIManager.get(
 136                                  "Application.useSystemFontSettings");
 137 
 138                 return (value == null || Boolean.TRUE.equals(value));
 139             }
 140             // If an instanceof MetalLookAndFeel hasn't been inited yet, we
 141             // don't want to trigger loading of a UI by asking the UIManager
 142             // for a property, assume the user wants system fonts. This will
 143             // be properly adjusted when install is invoked on the
 144             // MetalTheme
 145             return true;
 146         }
 147         return false;
 148     }

1596      * After the theme is set, {@code MetalLookAndFeel} needs to be
1597      * re-installed and the uis need to be recreated. The following
1598      * shows how to do this:
1599      * <pre>
1600      *   MetalLookAndFeel.setCurrentTheme(theme);
1601      *
1602      *   // re-install the Metal Look and Feel
1603      *   UIManager.setLookAndFeel(new MetalLookAndFeel());
1604      *
1605      *   // Update the ComponentUIs for all Components. This
1606      *   // needs to be invoked for all windows.
1607      *   SwingUtilities.updateComponentTreeUI(rootComponent);
1608      * </pre>
1609      * If this is not done the results are undefined.
1610      *

1613      * @see #getCurrentTheme
1614      */

1617         // this step, we don't bother blowing away any potential windows
1618         // values.
1619         if (theme == null) {
1620             throw new NullPointerException("Can't have null theme");
1621         }
1622         AppContext.getAppContext().put( "currentMetalTheme", theme );

1626      * Return the theme currently being used by <code>MetalLookAndFeel</code>.
1627      * If the current theme is {@code null}, the default theme is created.
1628      *
1629      * @return the current theme
1630      * @see #setCurrentTheme
1631      * @since 1.5
1632      */
1633     public static MetalTheme getCurrentTheme() {
1634         MetalTheme currentTheme;
1635         AppContext context = AppContext.getAppContext();
1636         currentTheme = (MetalTheme) context.get( "currentMetalTheme" );
1637         if (currentTheme == null) {
1638             // This will happen in two cases:
1639             // . When MetalLookAndFeel is first being initialized.
1640             // . When a new AppContext has been created that hasn't
1641             //   triggered UIManager to load a LAF. Rather than invoke
1642             //   a method on the UIManager, which would trigger the loading
1643             //   of a potentially different LAF, we directly set the
1644             //   Theme here.
1645             if (useHighContrastTheme()) {
1646                 currentTheme = new MetalHighContrastTheme();
1647             }
1648             else {
1649                 // Create the default theme. We prefer Ocean, but will
1650                 // use DefaultMetalTheme if told to.
1651                 String theme = AccessController.doPrivileged(
1652                                new GetPropertyAction("swing.metalTheme"));
1653                 if ("steel".equals(theme)) {
1654                     currentTheme = new DefaultMetalTheme();
1655                 }
1656                 else {
1657                     currentTheme = new OceanTheme();
1658                 }
1659             }
1660             setCurrentTheme(currentTheme);
1661         }
1662         return currentTheme;
1663     }
