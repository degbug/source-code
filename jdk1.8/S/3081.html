  79     @Override
  80     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
  81         //populate componentColors array with colors calculated in getExtendedCacheKeys call
  82         componentColors = extendedCacheKeys;
  83         //generate this entire method. Each state/bg/fg/border combo that has
  84         //been painted gets its own KEY and paint method.
  85         switch(state) {
  86             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
  87             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
  88             case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
  89 
  90         }
  91     }
