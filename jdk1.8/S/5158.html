  The DTD validator. The validator implements a document
  filter: receiving document events from the scanner; validating
  the content and structure; augmenting the InfoSet, if applicable;
  and notifying the parser of the information resulting from the
  validation process.
  
 Formerly, this component also handled DTD events and grammar construction.
  To facilitate the development of a meaningful DTD grammar caching/preparsing
  framework, this functionality has been moved into the XMLDTDLoader
  class.  Therefore, this class no longer implements the DTDFilter
  or DTDContentModelFilter interfaces.
  

  This component requires the following features and properties from the
  component manager that uses it:
  
  • http://xml.org/sax/features/namespaces
  • http://xml.org/sax/features/validation
  • http://apache.org/xml/features/validation/dynamic
  • http://apache.org/xml/properties/internal/symbol-table
  • http://apache.org/xml/properties/internal/error-reporter
  • http://apache.org/xml/properties/internal/grammar-pool
  • http://apache.org/xml/properties/internal/datatype-validator-factory

     /** Bind namespaces */

 112     protected final void startNamespaceScope (QName element, XMLAttributes attributes,
 113                                       Augmentations augs) throws XNIException {
 114 
 115         // add new namespace context
 116         fNamespaceContext.pushContext();
 117 
 118         if (element.prefix == XMLSymbols.PREFIX_XMLNS) {
 119             fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 120                                        "ElementXMLNSPrefix",
 121                                        new Object[]{element.rawname},
 122                                        XMLErrorReporter.SEVERITY_FATAL_ERROR);
 123         }
 124 
 125         // search for new namespace bindings
 126         int length = attributes.getLength();
 127         for (int i = 0; i < length; i++) {
 128             String localpart = attributes.getLocalName(i);
 129             String prefix = attributes.getPrefix(i);
 130             // when it's of form xmlns="..." or xmlns:prefix="...",
 131             // it's a namespace declaration. but prefix:xmlns="..." isn't.
 132             if (prefix == XMLSymbols.PREFIX_XMLNS ||
 133                 prefix == XMLSymbols.EMPTY_STRING && localpart == XMLSymbols.PREFIX_XMLNS) {
 134 
 135                 // get the internalized value of this attribute
 136                 String uri = fSymbolTable.addSymbol(attributes.getValue(i));
 137 
 138                 // 1. "xmlns" can't be bound to any namespace
 139                 if (prefix == XMLSymbols.PREFIX_XMLNS && localpart == XMLSymbols.PREFIX_XMLNS) {
 140                     fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 141                                                "CantBindXMLNS",
 142                                                new Object[]{attributes.getQName(i)},
 143                                                XMLErrorReporter.SEVERITY_FATAL_ERROR);
 144                 }
 145 
 146                 // 2. the namespace for "xmlns" can't be bound to any prefix
 147                 if (uri == NamespaceContext.XMLNS_URI) {
 148                     fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 149                                                "CantBindXMLNS",
 150                                                new Object[]{attributes.getQName(i)},
 151                                                XMLErrorReporter.SEVERITY_FATAL_ERROR);
 152                 }
 153 
 154                 // 3. "xml" can't be bound to any other namespace than it's own
 155                 if (localpart == XMLSymbols.PREFIX_XML) {
 156                     if (uri != NamespaceContext.XML_URI) {
 157                         fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 158                                                    "CantBindXML",
 159                                                    new Object[]{attributes.getQName(i)},
 160                                                    XMLErrorReporter.SEVERITY_FATAL_ERROR);
 161                     }
 162                 }
 163                 // 4. the namespace for "xml" can't be bound to any other prefix
 164                 else {
 165                     if (uri ==NamespaceContext.XML_URI) {
 166                         fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 167                                                    "CantBindXML",
 168                                                    new Object[]{attributes.getQName(i)},
 169                                                    XMLErrorReporter.SEVERITY_FATAL_ERROR);
 170                     }
 171                 }
 172 
 173                 prefix = localpart != XMLSymbols.PREFIX_XMLNS ? localpart : XMLSymbols.EMPTY_STRING;
 174 
 175                 // http://www.w3.org/TR/1999/REC-xml-names-19990114/#dt-prefix
 176                 // We should only report an error if there is a prefix,
 177                 // that is, the local part is not "xmlns". -SG
 178                 if (uri == XMLSymbols.EMPTY_STRING && localpart != XMLSymbols.PREFIX_XMLNS) {
 179                     fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 180                                                "EmptyPrefixedAttName",
 181                                                new Object[]{attributes.getQName(i)},
 182                                                XMLErrorReporter.SEVERITY_FATAL_ERROR);
 183                     continue;
 184                 }
 185 
 186                 // declare prefix in context
 187                 fNamespaceContext.declarePrefix(prefix, uri.length() != 0 ? uri : null);
 188             }
 189         }
 190 
 191         // bind the element
 192         String prefix = element.prefix != null
 193                       ? element.prefix : XMLSymbols.EMPTY_STRING;
 194         element.uri = fNamespaceContext.getURI(prefix);
 195         if (element.prefix == null && element.uri != null) {
 196             element.prefix = XMLSymbols.EMPTY_STRING;
 197         }
 198         if (element.prefix != null && element.uri == null) {
 199             fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 200                                        "ElementPrefixUnbound",
 201                                        new Object[]{element.prefix, element.rawname},
 202                                        XMLErrorReporter.SEVERITY_FATAL_ERROR);
 203         }
 204 
 205         // bind the attributes
 206         for (int i = 0; i < length; i++) {
 207             attributes.getName(i, fAttributeQName);
 208             String aprefix = fAttributeQName.prefix != null
 209                            ? fAttributeQName.prefix : XMLSymbols.EMPTY_STRING;
 210             String arawname = fAttributeQName.rawname;
 211             if (arawname == XMLSymbols.PREFIX_XMLNS) {
 212                 fAttributeQName.uri = fNamespaceContext.getURI(XMLSymbols.PREFIX_XMLNS);
 213                 attributes.setName(i, fAttributeQName);
 214             }
 215             else if (aprefix != XMLSymbols.EMPTY_STRING) {
 216                 fAttributeQName.uri = fNamespaceContext.getURI(aprefix);
 217                 if (fAttributeQName.uri == null) {
 218                     fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 219                                                "AttributePrefixUnbound",
 220                                                new Object[]{element.rawname,arawname,aprefix},
 221                                                XMLErrorReporter.SEVERITY_FATAL_ERROR);
 222                 }
 223                 attributes.setName(i, fAttributeQName);
 224             }
 225         }
 226 
 227         // verify that duplicate attributes don't exist
 228         // Example: <foo xmlns:a='NS' xmlns:b='NS' a:attr='v1' b:attr='v2'/>
 229         int attrCount = attributes.getLength();
 230         for (int i = 0; i < attrCount - 1; i++) {
 231             String auri = attributes.getURI(i);
 232             if (auri == null || auri == NamespaceContext.XMLNS_URI) {
 233                 continue;
 234             }
 235             String alocalpart = attributes.getLocalName(i);
 236             for (int j = i + 1; j < attrCount; j++) {
 237                 String blocalpart = attributes.getLocalName(j);
 238                 String buri = attributes.getURI(j);
 239                 if (alocalpart == blocalpart && auri == buri) {
 240                     fErrorReporter.reportError(XMLMessageFormatter.XMLNS_DOMAIN,
 241                                                "AttributeNSNotUnique",
 242                                                new Object[]{element.rawname,alocalpart, auri},
 243                                                XMLErrorReporter.SEVERITY_FATAL_ERROR);
 244                 }
 245             }
 246         }
 247 
 248 
 249     } // startNamespaceScope(QName,XMLAttributes)

     /** Handles end element. */

 253     protected void endNamespaceScope(QName element, Augmentations augs, boolean isEmpty)
 254         throws XNIException {
 255 
 256         // bind element
 257         String eprefix = element.prefix != null ? element.prefix : XMLSymbols.EMPTY_STRING;
 258         element.uri = fNamespaceContext.getURI(eprefix);
 259         if (element.uri != null) {
 260             element.prefix = eprefix;
 261         }
 262 
 263         // call handlers
 264         if (fDocumentHandler != null) {
 265             if (!isEmpty) {
 266                 fDocumentHandler.endElement(element, augs);
 267             }
 268         }
 269 
 270         // pop context
 271         fNamespaceContext.popContext();
 272 
 273     } // endNamespaceScope(QName,boolean)
