  34 /**
  35  * A representation of the XML <code>X509Data</code> element as defined in
  36  * the <a href="http://www.w3.org/TR/xmldsig-core/">
  37  * W3C Recommendation for XML-Signature Syntax and Processing</a>. An
  38  * <code>X509Data</code> object contains one or more identifers of keys
  39  * or X.509 certificates (or certificates' identifiers or a revocation list).
  40  * The XML Schema Definition is defined as:
  41  *
  42  * <pre>
  43  *    &lt;element name="X509Data" type="ds:X509DataType"/&gt;
  44  *    &lt;complexType name="X509DataType"&gt;
  45  *        &lt;sequence maxOccurs="unbounded"&gt;
  46  *          &lt;choice&gt;
  47  *            &lt;element name="X509IssuerSerial" type="ds:X509IssuerSerialType"/&gt;
  48  *            &lt;element name="X509SKI" type="base64Binary"/&gt;
  49  *            &lt;element name="X509SubjectName" type="string"/&gt;
  50  *            &lt;element name="X509Certificate" type="base64Binary"/&gt;
  51  *            &lt;element name="X509CRL" type="base64Binary"/&gt;
  52  *            &lt;any namespace="##other" processContents="lax"/&gt;
  53  *          &lt;/choice&gt;
  54  *        &lt;/sequence&gt;
  55  *    &lt;/complexType&gt;
  56  *
  57  *    &lt;complexType name="X509IssuerSerialType"&gt;
  58  *      &lt;sequence&gt;
  59  *        &lt;element name="X509IssuerName" type="string"/&gt;
  60  *        &lt;element name="X509SerialNumber" type="integer"/&gt;
  61  *      &lt;/sequence&gt;
  62  *    &lt;/complexType&gt;
  63  * </pre>
  64  *
  65  * An <code>X509Data</code> instance may be created by invoking the
  66  * {@link KeyInfoFactory#newX509Data newX509Data} methods of the
  67  * {@link KeyInfoFactory} class and passing it a list of one or more
  68  * {@link XMLStructure}s representing X.509 content; for example:
  69  * <pre>
  70  *   KeyInfoFactory factory = KeyInfoFactory.getInstance("DOM");
  71  *   X509Data x509Data = factory.newX509Data
  72  *       (Collections.singletonList("cn=Alice"));
  73  * </pre>
  74  *
  75  * @author Sean Mullan
  76  * @author JSR 105 Expert Group
  77  * @since 1.6
  78  * @see KeyInfoFactory#newX509Data(List)
  79  */

  90     final static String TYPE = "http://www.w3.org/2000/09/xmldsig#X509Data";

  96      * {@link RetrievalMethod} class to describe a remote X509 Certificate.

  98     final static String RAW_X509_CERTIFICATE_TYPE =
