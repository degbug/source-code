  42 /**
  43  * <code>PopupFactory</code>, as the name implies, is used to obtain
  44  * instances of <code>Popup</code>s. <code>Popup</code>s are used to
  45  * display a <code>Component</code> above all other <code>Component</code>s
  46  * in a particular containment hierarchy. The general contract is that
  47  * once you have obtained a <code>Popup</code> from a
  48  * <code>PopupFactory</code>, you must invoke <code>hide</code> on the
  49  * <code>Popup</code>. The typical usage is:
  50  * <pre>
  51  *   PopupFactory factory = PopupFactory.getSharedInstance();
  52  *   Popup popup = factory.getPopup(owner, contents, x, y);
  53  *   popup.show();
  54  *   ...
  55  *   popup.hide();
  56  * </pre>
  57  *
  58  * @see Popup
  59  *
  60  * @since 1.4
  61  */

 147     /**
 148      * Creates a <code>Popup</code> for the Component <code>owner</code>
 149      * containing the Component <code>contents</code>. <code>owner</code>
 150      * is used to determine which <code>Window</code> the new
 151      * <code>Popup</code> will parent the <code>Component</code> the
 152      * <code>Popup</code> creates to. A null <code>owner</code> implies there
 153      * is no valid parent. <code>x</code> and
 154      * <code>y</code> specify the preferred initial location to place
 155      * the <code>Popup</code> at. Based on screen size, or other paramaters,
 156      * the <code>Popup</code> may not display at <code>x</code> and
 157      * <code>y</code>.
 158      *
 159      * @param owner    Component mouse coordinates are relative to, may be null
 160      * @param contents Contents of the Popup
 161      * @param x        Initial x screen coordinate
 162      * @param y        Initial y screen coordinate
 163      * @exception IllegalArgumentException if contents is null
 164      * @return Popup containing Contents
 165      */
 166     public Popup getPopup(Component owner, Component contents,
 167                           int x, int y) throws IllegalArgumentException {
 168         if (contents == null) {
 169             throw new IllegalArgumentException(
 170                           "Popup.getPopup must be passed non-null contents");
 171         }
 172 
 173         int popupType = getPopupType(owner, contents, x, y);
 174         Popup popup = getPopup(owner, contents, x, y, popupType);
 175 
 176         if (popup == null) {
 177             // Didn't fit, force to heavy.
 178             popup = getPopup(owner, contents, x, y, HEAVY_WEIGHT_POPUP);
 179         }
 180         return popup;
 181     }

 183     /**
 184      * Returns the popup type to use for the specified parameters.
 185      */
 186     private int getPopupType(Component owner, Component contents,
 187                              int ownerX, int ownerY) {
 188         int popupType = getPopupType();
 189 
 190         if (owner == null || invokerInHeavyWeightPopup(owner)) {
 191             popupType = HEAVY_WEIGHT_POPUP;
 192         }
 193         else if (popupType == LIGHT_WEIGHT_POPUP &&
 194                  !(contents instanceof JToolTip) &&
 195                  !(contents instanceof JPopupMenu)) {
 196             popupType = MEDIUM_WEIGHT_POPUP;
 197         }
 198 
 199         // Check if the parent component is an option pane.  If so we need to
 200         // force a heavy weight popup in order to have event dispatching work
 201         // correctly.
 202         Component c = owner;
 203         while (c != null) {
 204             if (c instanceof JComponent) {
 205                 if (((JComponent)c).getClientProperty(
 206                             PopupFactory_FORCE_HEAVYWEIGHT_POPUP) == Boolean.TRUE) {
 207                     popupType = HEAVY_WEIGHT_POPUP;
 208                     break;
 209                 }
 210             }
 211             c = c.getParent();
 212         }
 213 
 214         return popupType;
 215     }

 217     /**
 218      * Obtains the appropriate <code>Popup</code> based on
 219      * <code>popupType</code>.
 220      */
 221     private Popup getPopup(Component owner, Component contents,
 222                            int ownerX, int ownerY, int popupType) {
 223         if (GraphicsEnvironment.isHeadless()) {
 224             return getHeadlessPopup(owner, contents, ownerX, ownerY);
 225         }
 226 
 227         switch(popupType) {
 228         case LIGHT_WEIGHT_POPUP:
 229             return getLightWeightPopup(owner, contents, ownerX, ownerY);
 230         case MEDIUM_WEIGHT_POPUP:
 231             return getMediumWeightPopup(owner, contents, ownerX, ownerY);
 232         case HEAVY_WEIGHT_POPUP:
 233             Popup popup = getHeavyWeightPopup(owner, contents, ownerX, ownerY);
 234             if ((AccessController.doPrivileged(OSInfo.getOSTypeAction()) ==
 235                 OSInfo.OSType.MACOSX) && (owner != null) &&
 236                 (EmbeddedFrame.getAppletIfAncestorOf(owner) != null)) {
 237                 ((HeavyWeightPopup)popup).setCacheEnabled(false);
 238             }
 239             return popup;
 240         }
 241         return null;
 242     }

 303     private static class HeavyWeightPopup extends Popup {
 304         private static final Object heavyWeightPopupCacheKey =
 305                  new StringBuffer("PopupFactory.heavyWeightPopupCache");

 307         private volatile boolean isCacheEnabled = true;
 308 
 309         /**
 310          * Returns either a new or recycled <code>Popup</code> containing
 311          * the specified children.
 312          */
 313         static Popup getHeavyWeightPopup(Component owner, Component contents,
 314                                          int ownerX, int ownerY) {
 315             Window window = (owner != null) ? SwingUtilities.
 316                               getWindowAncestor(owner) : null;
 317             HeavyWeightPopup popup = null;
 318 
 319             if (window != null) {
 320                 popup = getRecycledHeavyWeightPopup(window);
 321             }
 322 
 323             boolean focusPopup = false;
 324             if(contents != null && contents.isFocusable()) {
 325                 if(contents instanceof JPopupMenu) {
 326                     JPopupMenu jpm = (JPopupMenu) contents;
 327                     Component popComps[] = jpm.getComponents();
 328                     for (Component popComp : popComps) {
 329                         if (!(popComp instanceof MenuElement) &&
 330                                 !(popComp instanceof JSeparator)) {
 331                             focusPopup = true;
 332                             break;
 333                         }
 334                     }
 335                 }
 336             }
 337 
 338             if (popup == null ||
 339                 ((JWindow) popup.getComponent())
 340                  .getFocusableWindowState() != focusPopup) {
 341 
 342                 if(popup != null) {
 343                     // The recycled popup can't serve us well
 344                     // dispose it and create new one
 345                     popup._dispose();
 346                 }
 347 
 348                 popup = new HeavyWeightPopup();
 349             }
 350 
 351             popup.reset(owner, contents, ownerX, ownerY);
 352 
 353             if(focusPopup) {
 354                 JWindow wnd = (JWindow) popup.getComponent();
 355                 wnd.setFocusableWindowState(true);
 356                 // Set window name. We need this in BasicPopupMenuUI
 357                 // to identify focusable popup window.
 358                 wnd.setName("###focusableSwingPopup###");
 359             }
 360 
 361             return popup;
 362         }

 364         /**
 365          * Returns a previously disposed heavy weight <code>Popup</code>
 366          * associated with <code>window</code>. This will return null if
 367          * there is no <code>HeavyWeightPopup</code> associated with
 368          * <code>window</code>.
 369          */
 370         private static HeavyWeightPopup getRecycledHeavyWeightPopup(Window w) {
 371             synchronized (HeavyWeightPopup.class) {
 372                 List<HeavyWeightPopup> cache;
 373                 Map<Window, List<HeavyWeightPopup>> heavyPopupCache = getHeavyWeightPopupCache();
 374 
 375                 if (heavyPopupCache.containsKey(w)) {
 376                     cache = heavyPopupCache.get(w);
 377                 } else {
 378                     return null;
 379                 }
 380                 if (cache.size() > 0) {
 381                     HeavyWeightPopup r = cache.get(0);
 382                     cache.remove(0);
 383                     return r;
 384                 }
 385                 return null;
 386             }
 387         }

 389         /**
 390          * Returns the cache to use for heavy weight popups. Maps from
 391          * <code>Window</code> to a <code>List</code> of
 392          * <code>HeavyWeightPopup</code>s.
 393          */
 394         private static Map<Window, List<HeavyWeightPopup>> getHeavyWeightPopupCache() {
 395             synchronized (HeavyWeightPopup.class) {
 396                 Map<Window, List<HeavyWeightPopup>> cache = (Map<Window, List<HeavyWeightPopup>>)SwingUtilities.appContextGet(
 397                                   heavyWeightPopupCacheKey);
 398 
 399                 if (cache == null) {
 400                     cache = new HashMap<Window, List<HeavyWeightPopup>>(2);
 401                     SwingUtilities.appContextPut(heavyWeightPopupCacheKey,
 402                                                  cache);
 403                 }
 404                 return cache;
 405             }
 406         }

 408         /**
 409          * Recycles the passed in <code>HeavyWeightPopup</code>.
 410          */
 411         private static void recycleHeavyWeightPopup(HeavyWeightPopup popup) {
 412             synchronized (HeavyWeightPopup.class) {
 413                 List<HeavyWeightPopup> cache;
 414                 Window window = SwingUtilities.getWindowAncestor(
 415                                      popup.getComponent());
 416                 Map<Window, List<HeavyWeightPopup>> heavyPopupCache = getHeavyWeightPopupCache();
 417 
 418                 if (window instanceof Popup.DefaultFrame ||
 419                                       !window.isVisible()) {
 420                     // If the Window isn't visible, we don't cache it as we
 421                     // likely won't ever get a windowClosed event to clean up.
 422                     // We also don't cache DefaultFrames as this indicates
 423                     // there wasn't a valid Window parent, and thus we don't
 424                     // know when to clean up.
 425                     popup._dispose();
 426                     return;
 427                 } else if (heavyPopupCache.containsKey(window)) {
 428                     cache = heavyPopupCache.get(window);
 429                 } else {
 430                     cache = new ArrayList<HeavyWeightPopup>();
 431                     heavyPopupCache.put(window, cache);
 432                     // Clean up if the Window is closed
 433                     final Window w = window;
 434 
 435                     w.addWindowListener(new WindowAdapter() {
 436                         public void windowClosed(WindowEvent e) {
 437                             List<HeavyWeightPopup> popups;
 438 
 439                             synchronized(HeavyWeightPopup.class) {
 440                                 Map<Window, List<HeavyWeightPopup>> heavyPopupCache2 =
 441                                               getHeavyWeightPopupCache();
 442 
 443                                 popups = heavyPopupCache2.remove(w);
 444                             }
 445                             if (popups != null) {
 446                                 for (int counter = popups.size() - 1;
 447                                                    counter >= 0; counter--) {
 448                                     popups.get(counter)._dispose();
 449                                 }
 450                             }
 451                         }
 452                     });
 453                 }
 454 
 455                 if(cache.size() < MAX_CACHE_SIZE) {
 456                     cache.add(popup);
 457                 } else {
 458                     popup._dispose();
 459                 }
 460             }
 461         }

 509         public void hide() {
 510             Component component = getComponent();
 511 
 512             if (component != null) {
 513                 Container parent = component.getParent();
 514 
 515                 if (parent != null) {
 516                     Rectangle bounds = component.getBounds();
 517

 519                     parent.repaint(bounds.x, bounds.y, bounds.width,
 520                                    bounds.height);
 521                 }
 522             }
 523             owner = null;
 524         }
 525         public void pack() {
 526             Component component = getComponent();
 527 
 528             if (component != null) {
 529                 component.setSize(component.getPreferredSize());
 530             }
 531         }

 533         void reset(Component owner, Component contents, int ownerX,
 534                    int ownerY) {
 535             if ((owner instanceof JFrame) || (owner instanceof JDialog) ||
 536                                                  (owner instanceof JWindow)) {
 537                 // Force the content to be added to the layered pane, otherwise
 538                 // we'll get an exception when adding to the RootPaneContainer.
 539                 owner = ((RootPaneContainer)owner).getLayeredPane();
 540             }
 541             super.reset(owner, contents, ownerX, ownerY);
 542 
 543             x = ownerX;
 544             y = ownerY;
 545             this.owner = owner;
 546         }

 548         boolean overlappedByOwnedWindow() {
 549             Component component = getComponent();
 550             if(owner != null && component != null) {
 551                 Window w = SwingUtilities.getWindowAncestor(owner);
 552                 if (w == null) {
 553                     return false;
 554                 }
 555                 Window[] ownedWindows = w.getOwnedWindows();
 556                 if(ownedWindows != null) {
 557                     Rectangle bnd = component.getBounds();
 558                     for (Window window : ownedWindows) {
 559                         if (window.isVisible() &&
 560                                 bnd.intersects(window.getBounds())) {
 561 
 562                             return true;
 563                         }
 564                     }
 565                 }
 566             }
 567             return false;
 568         }

 570         /**
 571          * Returns true if popup can fit the screen and the owner's top parent.
 572          * It determines can popup be lightweight or mediumweight.
 573          */
 574         boolean fitsOnScreen() {
 575             boolean result = false;
 576             Component component = getComponent();
 577             if (owner != null && component != null) {
 578                 int popupWidth = component.getWidth();
 579                 int popupHeight = component.getHeight();
 580 
 581                 Container parent = (Container) SwingUtilities.getRoot(owner);
 582                 if (parent instanceof JFrame ||
 583                     parent instanceof JDialog ||
 584                     parent instanceof JWindow) {
 585 
 586                     Rectangle parentBounds = parent.getBounds();
 587                     Insets i = parent.getInsets();
 588                     parentBounds.x += i.left;
 589                     parentBounds.y += i.top;
 590                     parentBounds.width -= i.left + i.right;
 591                     parentBounds.height -= i.top + i.bottom;
 592 
 593                     if (JPopupMenu.canPopupOverlapTaskBar()) {
 594                         GraphicsConfiguration gc =
 595                                 parent.getGraphicsConfiguration();
 596                         Rectangle popupArea = getContainerPopupArea(gc);
 597                         result = parentBounds.intersection(popupArea)
 598                                 .contains(x, y, popupWidth, popupHeight);
 599                     } else {
 600                         result = parentBounds
 601                                 .contains(x, y, popupWidth, popupHeight);
 602                     }
 603                 } else if (parent instanceof JApplet) {
 604                     Rectangle parentBounds = parent.getBounds();
 605                     Point p = parent.getLocationOnScreen();
 606                     parentBounds.x = p.x;
 607                     parentBounds.y = p.y;
 608                     result = parentBounds.contains(x, y, popupWidth, popupHeight);
 609                 }
 610             }
 611             return result;
 612         }
 613 
 614         Rectangle getContainerPopupArea(GraphicsConfiguration gc) {
 615             Rectangle screenBounds;
 616             Toolkit toolkit = Toolkit.getDefaultToolkit();
 617             Insets insets;
 618             if(gc != null) {
 619                 // If we have GraphicsConfiguration use it
 620                 // to get screen bounds
 621                 screenBounds = gc.getBounds();
 622                 insets = toolkit.getScreenInsets(gc);
 623             } else {
 624                 // If we don't have GraphicsConfiguration use primary screen
 625                 screenBounds = new Rectangle(toolkit.getScreenSize());
 626                 insets = new Insets(0, 0, 0, 0);
 627             }
 628             // Take insets into account
 629             screenBounds.x += insets.left;
 630             screenBounds.y += insets.top;
 631             screenBounds.width -= (insets.left + insets.right);
 632             screenBounds.height -= (insets.top + insets.bottom);
 633             return screenBounds;
 634         }

 742         public void show() {
 743             Container parent = null;
 744 
 745             if (owner != null) {
 746                 parent = (owner instanceof Container? (Container)owner : owner.getParent());
 747             }
 748 
 749             // Try to find a JLayeredPane and Window to add
 750             for (Container p = parent; p != null; p = p.getParent()) {
 751                 if (p instanceof JRootPane) {
 752                     if (p.getParent() instanceof JInternalFrame) {
 753                         continue;
 754                     }
 755                     parent = ((JRootPane)p).getLayeredPane();
 756                     // Continue, so that if there is a higher JRootPane, we'll
 757                     // pick it up.
 758                 } else if(p instanceof Window) {
 759                     if (parent == null) {
 760                         parent = p;
 761                     }
 762                     break;
 763                 } else if (p instanceof JApplet) {
 764                     // Painting code stops at Applets, we don't want
 765                     // to add to a Component above an Applet otherwise
 766                     // you'll never see it painted.
 767                     break;
 768                 }
 769             }
 770 
 771             Point p = SwingUtilities.convertScreenLocationToParent(parent, x,
 772                                                                    y);
 773             Component component = getComponent();
 774 
 775             component.setLocation(p.x, p.y);
 776             if (parent instanceof JLayeredPane) {
 777                 parent.add(component, JLayeredPane.POPUP_LAYER, 0);
 778             } else {
 779                 parent.add(component);
 780             }
 781         }

 823         /**
 824          * Returns a medium weight <code>Popup</code> implementation. If
 825          * the <code>Popup</code> needs more space that in available in
 826          * <code>owner</code>, this will return null.
 827          */
 828         static Popup getMediumWeightPopup(Component owner, Component contents,
 829                                           int ownerX, int ownerY) {
 830             MediumWeightPopup popup = getRecycledMediumWeightPopup();
 831 
 832             if (popup == null) {
 833                 popup = new MediumWeightPopup();
 834             }
 835             popup.reset(owner, contents, ownerX, ownerY);
 836             if (!popup.fitsOnScreen() ||
 837                  popup.overlappedByOwnedWindow()) {
 838                 popup.hide();
 839                 return null;
 840             }
 841             return popup;
 842         }

 896         public void show() {
 897             Component component = getComponent();
 898             Container parent = null;
 899 
 900             if (owner != null) {
 901                 parent = owner.getParent();
 902             }
 903             /*
 904               Find the top level window,
 905               if it has a layered pane,
 906               add to that, otherwise
 907               add to the window. */
 908             while (!(parent instanceof Window || parent instanceof Applet) &&
 909                    (parent!=null)) {
 910                 parent = parent.getParent();
 911             }
 912             // Set the visibility to false before adding to workaround a
 913             // bug in Solaris in which the Popup gets added at the wrong
 914             // location, which will result in a mouseExit, which will then
 915             // result in the ToolTip being removed.
 916             if (parent instanceof RootPaneContainer) {
 917                 parent = ((RootPaneContainer)parent).getLayeredPane();
 918                 Point p = SwingUtilities.convertScreenLocationToParent(parent,
 919                                                                        x, y);
 920                 component.setVisible(false);
 921                 component.setLocation(p.x, p.y);
 922                 parent.add(component, JLayeredPane.POPUP_LAYER,
 923                                            0);
 924             } else {
 925                 Point p = SwingUtilities.convertScreenLocationToParent(parent,
 926                                                                        x, y);
 927 
 928                 component.setLocation(p.x, p.y);
 929                 component.setVisible(false);
 930                 parent.add(component);
 931             }
 932             component.setVisible(true);
 933         }
