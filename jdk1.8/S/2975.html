      Paints the specified component.
      @param context context for the component being painted
      @param g the Graphics object used for painting
      @see #update(Graphics,JComponent)

 178     protected void paint(SynthContext context, Graphics g) {
 179         JToolTip tip = (JToolTip)context.getComponent();
 180 
 181         Insets insets = tip.getInsets();
 182         View v = (View)tip.getClientProperty(BasicHTML.propertyKey);
 183         if (v != null) {
 184             Rectangle paintTextR = new Rectangle(insets.left, insets.top,
 185                   tip.getWidth() - (insets.left + insets.right),
 186                   tip.getHeight() - (insets.top + insets.bottom));
 187             v.paint(g, paintTextR);
 188         } else {
 189             g.setColor(context.getStyle().getColor(context,
 190                                                    ColorType.TEXT_FOREGROUND));
 191             g.setFont(style.getFont(context));
 192             context.getStyle().getGraphicsUtils(context).paintText(
 193                 context, g, tip.getTipText(), insets.left, insets.top, -1);
 194         }
 195     }

 200     @Override
 201     public Dimension getPreferredSize(JComponent c) {
 202         SynthContext context = getContext(c);
 203         Insets insets = c.getInsets();
 204         Dimension prefSize = new Dimension(insets.left+insets.right,
 205                                            insets.top+insets.bottom);
 206         String text = ((JToolTip)c).getTipText();
 207 
 208         if (text != null) {
 209             View v = (c != null) ? (View) c.getClientProperty("html") : null;
 210             if (v != null) {
 211                 prefSize.width += (int) v.getPreferredSpan(View.X_AXIS);
 212                 prefSize.height += (int) v.getPreferredSpan(View.Y_AXIS);
 213             } else {
 214                 Font font = context.getStyle().getFont(context);
 215                 FontMetrics fm = c.getFontMetrics(font);
 216                 prefSize.width += context.getStyle().getGraphicsUtils(context).
 217                                   computeStringWidth(context, font, fm, text);
 218                 prefSize.height += fm.getHeight();
 219             }
 220         }
 221         context.dispose();
 222         return prefSize;
 223     }

      {@inheritDoc}

 228     @Override
 229     public void propertyChange(PropertyChangeEvent e) {
 230         if (SynthLookAndFeel.shouldUpdateStyle(e)) {
 231             updateStyle((JToolTip)e.getSource());
 232         }
 233         String name = e.getPropertyName();
 234         if (name.equals("tiptext") || "font".equals(name) ||
 235                 "foreground".equals(name)) {
 236             // remove the old html view client property if one
 237             // existed, and install a new one if the text installed
 238             // into the JLabel is html source.
 239             JToolTip tip = ((JToolTip) e.getSource());
 240             String text = tip.getTipText();
 241             BasicHTML.updateRenderer(tip, text);
 242         }
 243     }
