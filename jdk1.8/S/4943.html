  38 /**
  39  * <p>The NodeSetDTM class can act as either a NodeVector,
  40  * NodeList, or NodeIterator.  However, in order for it to
  41  * act as a NodeVector or NodeList, it's required that
  42  * setShouldCacheNodes(true) be called before the first
  43  * nextNode() is called, in order that nodes can be added
  44  * as they are fetched.  Derived classes that implement iterators
  45  * must override runTo(int index), in order that they may
  46  * run the iteration to the given index. </p>
  47  *
  48  * <p>Note that we directly implement the DOM's NodeIterator
  49  * interface. We do not emulate all the behavior of the
  50  * standard NodeIterator. In particular, we do not guarantee
  51  * to present a "live view" of the document ... but in XSLT,
  52  * the source document should never be mutated, so this should
  53  * never be an issue.</p>
  54  *
  55  * <p>Thought: Should NodeSetDTM really implement NodeList and NodeIterator,
  56  * or should there be specific subclasses of it which do so? The
  57  * advantage of doing it all here is that all NodeSetDTMs will respond
  58  * to the same calls; the disadvantage is that some of them may return
  59  * less-than-enlightening results when you do so.</p>
  60  * @xsl.usage advanced
  61  */
  62 public class NodeSetDTM extends NodeVector
  63         implements /* NodeList, NodeIterator, */ DTMIterator,

 377   /**
 378    *  Returns the next node in the set and advances the position of the
 379    * iterator in the set. After a DTMIterator is created, the first call
 380    * to nextNode() returns the first node in the set.
 381    * @return  The next <code>Node</code> in the set being iterated over, or
 382    *   <code>DTM.NULL</code> if there are no more members in that set.
 383    * @throws DOMException
 384    *    INVALID_STATE_ERR: Raised if this method is called after the
 385    *   <code>detach</code> method was invoked.
 386    */
 387   public int nextNode()
 388   {
 389 
 390     if ((m_next) < this.size())
 391     {
 392       int next = this.elementAt(m_next);
 393 
 394       m_next++;
 395 
 396       return next;
 397     }
 398     else
 399       return DTM.NULL;
 400   }

 481   {
 482

 487     if ((index >= 0) && (m_next < m_firstFree))
 488       m_next = index;
 489     else
 490       m_next = m_firstFree - 1;

 784   /**
 785    * Add the node into a vector of nodes where it should occur in
 786    * document order.
 787    * @param node The node to be added.
 788    * @param test true if we should test for doc order
 789    * @param support The XPath runtime context.
 790    * @return insertIndex.
 791    * @throws RuntimeException thrown if this NodeSetDTM is not of
 792    * a mutable type.
 793    */
 794   public int addNodeInDocOrder(int node, boolean test, XPathContext support)
 795   {
 796 
 797     if (!m_mutable)
 798       throw new RuntimeException(XSLMessages.createXPATHMessage(XPATHErrorResources.ER_NODESETDTM_NOT_MUTABLE, null)); //"This NodeSetDTM is not mutable!");
 799 
 800     int insertIndex = -1;
 801 
 802     if (test)
 803     {
 804 
 805       // This needs to do a binary search, but a binary search
 806       // is somewhat tough because the sequence test involves
 807       // two nodes.
 808       int size = size(), i;
 809 
 810       for (i = size - 1; i >= 0; i--)
 811       {
 812         int child = elementAt(i);
 813 
 814         if (child == node)
 815         {
 816           i = -2;  // Duplicate, suppress insert
 817 
 818           break;
 819         }
 820 
 821         DTM dtm = support.getDTM(node);
 822         if (!dtm.isNodeAfter(node, child))
 823         {
 824           break;
 825         }
 826       }
 827 
 828       if (i != -2)
 829       {
 830         insertIndex = i + 1;
 831 
 832         insertElementAt(node, insertIndex);
 833       }
 834     }
 835     else
 836     {
 837       insertIndex = this.size();
 838 
 839       boolean foundit = false;
 840 
 841       for (int i = 0; i < insertIndex; i++)
 842       {
 843         if (i == node)
 844         {
 845           foundit = true;
 846 
 847           break;
 848         }
 849       }
 850 
 851       if (!foundit)
 852         addElement(node);
 853     }
 854 
 855     // checkDups();
 856     return insertIndex;
 857   }  // end addNodeInDocOrder(Vector v, Object obj)
