  This class implements xni.grammars.XMLGrammarLoader.
  It also serves as implementation of xs.XSLoader interface and DOMConfiguration interface.
  This class is designed to interact either with a proxy for a user application
  which wants to preparse schemas, or with our own Schema validator.
  It is hoped that none of these "external" classes will therefore need to communicate directly
  with XSDHandler in future.
  
This class only knows how to make XSDHandler do its thing.
  The caller must ensure that all its properties (schemaLocation, JAXPSchemaSource
  etc.) have been properly set.

 315     XMLSchemaLoader(SymbolTable symbolTable,
 316             XMLErrorReporter errorReporter,
 317             XMLEntityManager entityResolver,
 318             XSGrammarBucket grammarBucket,
 319             SubstitutionGroupHandler sHandler,
 320             CMBuilder builder) {
 321 
 322         // store properties and features in configuration
 323         fLoaderConfig.addRecognizedFeatures(RECOGNIZED_FEATURES);
 324         fLoaderConfig.addRecognizedProperties(RECOGNIZED_PROPERTIES);
 325         if (symbolTable != null){
 326             fLoaderConfig.setProperty(SYMBOL_TABLE, symbolTable);
 327         }
 328 
 329         if(errorReporter == null) {
 330             errorReporter = new XMLErrorReporter ();
 331             errorReporter.setLocale(fLocale);
 332             errorReporter.setProperty(ERROR_HANDLER, new DefaultErrorHandler());
 333 
 334         }
 335         fErrorReporter = errorReporter;
 336         // make sure error reporter knows about schemas...
 337         if(fErrorReporter.getMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN) == null) {
 338             fErrorReporter.putMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN, new XSMessageFormatter());
 339         }
 340         fLoaderConfig.setProperty(ERROR_REPORTER, fErrorReporter);
 341         fEntityManager = entityResolver;
 342         // entity manager is null if XMLSchemaValidator creates the loader
 343         if (fEntityManager != null){
 344             fLoaderConfig.setProperty(ENTITY_MANAGER, fEntityManager);
 345         }
 346 
 347         // by default augment PSVI (i.e. don't use declaration pool)
 348         fLoaderConfig.setFeature(AUGMENT_PSVI, true);
 349 
 350         if(grammarBucket == null ) {
 351             grammarBucket = new XSGrammarBucket();
 352         }
 353         fGrammarBucket = grammarBucket;
 354         if(sHandler == null) {
 355             sHandler = new SubstitutionGroupHandler(fGrammarBucket);
 356         }
 357         fSubGroupHandler = sHandler;
 358 
 359         if(builder == null) {
 360             builder = new CMBuilder(fNodeFactory);
 361         }
 362         fCMBuilder = builder;
 363         fSchemaHandler = new XSDHandler(fGrammarBucket);
 364         if (fDeclPool != null) {
 365             fDeclPool.reset();
 366         }
 367         fJAXPCache = new HashMap();
 368 
 369         fSettingsChanged = true;
 370     }

      Returns a Grammar object by parsing the contents of the
      entity pointed to by source.
      @param source        the location of the entity which forms
                               the starting point of the grammar to be constructed.
      @throws IOException      When a problem is encountered reading the entity
               XNIException    When a condition arises (such as a FatalError) that requires parsing
                                   of the entity be terminated.

 555     public Grammar loadGrammar(XMLInputSource source)
 556     throws IOException, XNIException {
 557 
 558         // REVISIT: this method should have a namespace parameter specified by
 559         // user. In this case we can easily detect if a schema asked to be loaded
 560         // is already in the local cache.
 561 
 562         reset(fLoaderConfig);
 563         fSettingsChanged = false;
 564         XSDDescription desc = new XSDDescription();
 565         desc.fContextType = XSDDescription.CONTEXT_PREPARSE;
 566         desc.setBaseSystemId(source.getBaseSystemId());
 567         desc.setLiteralSystemId( source.getSystemId());
 568         // none of the other fields make sense for preparsing
 569         Map locationPairs = new HashMap();
 570         // Process external schema location properties.
 571         // We don't call tokenizeSchemaLocationStr here, because we also want
 572         // to check whether the values are valid URI.
 573         processExternalHints(fExternalSchemas, fExternalNoNSSchema,
 574                 locationPairs, fErrorReporter);
 575         SchemaGrammar grammar = loadSchema(desc, source, locationPairs);
 576 
 577         if(grammar != null && fGrammarPool != null) {
 578             fGrammarPool.cacheGrammars(XMLGrammarDescription.XML_SCHEMA, fGrammarBucket.getGrammars());
 579             // NOTE: we only need to verify full checking in case the schema was not provided via JAXP
 580             // since full checking already verified for all JAXP schemas
 581             if(fIsCheckedFully && fJAXPCache.get(grammar) != grammar) {
 582                 XSConstraints.fullSchemaChecking(fGrammarBucket, fSubGroupHandler, fCMBuilder, fErrorReporter);
 583             }
 584         }
 585         return grammar;
 586     } // loadGrammar(XMLInputSource):  Grammar
