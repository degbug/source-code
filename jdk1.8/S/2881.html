      Special thin border for rollover toolbar buttons.
      @since 1.4

 149     public static class RolloverButtonBorder extends ButtonBorder {
 150 
 151         public RolloverButtonBorder(Color shadow, Color darkShadow,
 152                                   Color highlight, Color lightHighlight) {
 153             super(shadow, darkShadow, highlight, lightHighlight);
 154         }
 155 
 156         public void paintBorder( Component c, Graphics g, int x, int y, int w, int h ) {
 157             AbstractButton b = (AbstractButton) c;
 158             ButtonModel model = b.getModel();
 159 
 160             Color shade = shadow;
 161             Component p = b.getParent();
 162             if (p != null && p.getBackground().equals(shadow)) {
 163                 shade = darkShadow;
 164             }
 165 
 166             if ((model.isRollover() && !(model.isPressed() && !model.isArmed())) ||
 167                 model.isSelected()) {
 168 
 169                 Color oldColor = g.getColor();
 170                 g.translate(x, y);
 171 
 172                 if (model.isPressed() && model.isArmed() || model.isSelected()) {
 173                     // Draw the pressd button
 174                     g.setColor(shade);
 175                     g.drawRect(0, 0, w-1, h-1);
 176                     g.setColor(lightHighlight);
 177                     g.drawLine(w-1, 0, w-1, h-1);
 178                     g.drawLine(0, h-1, w-1, h-1);
 179                 } else {
 180                     // Draw a rollover button
 181                     g.setColor(lightHighlight);
 182                     g.drawRect(0, 0, w-1, h-1);
 183                     g.setColor(shade);
 184                     g.drawLine(w-1, 0, w-1, h-1);
 185                     g.drawLine(0, h-1, w-1, h-1);
 186                 }
 187                 g.translate(-x, -y);
 188                 g.setColor(oldColor);
 189             }
 190         }
 191     }

      A border which is like a Margin border but it will only honor the margin
      if the margin has been explicitly set by the developer.
      Note: This is identical to the package private class
      MetalBorders.RolloverMarginBorder and should probably be consolidated.

 201     static class RolloverMarginBorder extends EmptyBorder {
 202 
 203         public RolloverMarginBorder() {
 204             super(3,3,3,3); // hardcoded margin for JLF requirements.
 205         }
 206 
 207         public Insets getBorderInsets(Component c, Insets insets) {
 208             Insets margin = null;
 209 
 210             if (c instanceof AbstractButton) {
 211                 margin = ((AbstractButton)c).getMargin();
 212             }
 213             if (margin == null || margin instanceof UIResource) {
 214                 // default margin so replace
 215                 insets.left = left;
 216                 insets.top = top;
 217                 insets.right = right;
 218                 insets.bottom = bottom;
 219             } else {
 220                 // Margin which has been explicitly set by the user.
 221                 insets.left = margin.left;
 222                 insets.top = margin.top;
 223                 insets.right = margin.right;
 224                 insets.bottom = margin.bottom;
 225             }
 226             return insets;
 227         }
 228     }

 355     public static class MarginBorder extends AbstractBorder implements UIResource {
 356         public Insets getBorderInsets(Component c, Insets insets)       {
 357             Insets margin = null;
 358             //
 359             // Ideally we'd have an interface defined for classes which
 360             // support margins (to avoid this hackery), but we've
 361             // decided against it for simplicity
 362             //
 363            if (c instanceof AbstractButton) {
 364                AbstractButton b = (AbstractButton)c;
 365                margin = b.getMargin();
 366            } else if (c instanceof JToolBar) {
 367                JToolBar t = (JToolBar)c;
 368                margin = t.getMargin();
 369            } else if (c instanceof JTextComponent) {
 370                JTextComponent t = (JTextComponent)c;
 371                margin = t.getMargin();
 372            }
 373            insets.top = margin != null? margin.top : 0;
 374            insets.left = margin != null? margin.left : 0;
 375            insets.bottom = margin != null? margin.bottom : 0;
 376            insets.right = margin != null? margin.right : 0;
 377 
 378            return insets;
 379         }

      Draws the border around the divider in a splitpane
      (when BasicSplitPaneUI is used). To get the appropriate effect, this
      needs to be used with a SplitPaneBorder.

 423     static class SplitPaneDividerBorder implements Border, UIResource {
 424         Color highlight;
 425         Color shadow;
 426 
 427         SplitPaneDividerBorder(Color highlight, Color shadow) {
 428             this.highlight = highlight;
 429             this.shadow = shadow;
 430         }
 431 
 432         public void paintBorder(Component c, Graphics g, int x, int y,
 433                                 int width, int height) {
 434             if (!(c instanceof BasicSplitPaneDivider)) {
 435                 return;
 436             }
 437             Component          child;
 438             Rectangle          cBounds;
 439             JSplitPane         splitPane = ((BasicSplitPaneDivider)c).
 440                                          getBasicSplitPaneUI().getSplitPane();
 441             Dimension          size = c.getSize();
 442 
 443             child = splitPane.getLeftComponent();
 444             // This is needed for the space between the divider and end of
 445             // splitpane.
 446             g.setColor(c.getBackground());
 447             g.drawRect(x, y, width - 1, height - 1);
 448             if(splitPane.getOrientation() == JSplitPane.HORIZONTAL_SPLIT) {
 449                 if(child != null) {
 450                     g.setColor(highlight);
 451                     g.drawLine(0, 0, 0, size.height);
 452                 }
 453                 child = splitPane.getRightComponent();
 454                 if(child != null) {
 455                     g.setColor(shadow);
 456                     g.drawLine(size.width - 1, 0, size.width - 1, size.height);
 457                 }
 458             } else {
 459                 if(child != null) {
 460                     g.setColor(highlight);
 461                     g.drawLine(0, 0, size.width, 0);
 462                 }
 463                 child = splitPane.getRightComponent();
 464                 if(child != null) {
 465                     g.setColor(shadow);
 466                     g.drawLine(0, size.height - 1, size.width,
 467                                size.height - 1);
 468                 }
 469             }
 470         }
 471         public Insets getBorderInsets(Component c) {
 472             Insets insets = new Insets(0,0,0,0);
 473             if (c instanceof BasicSplitPaneDivider) {
 474                 BasicSplitPaneUI bspui = ((BasicSplitPaneDivider)c).
 475                                          getBasicSplitPaneUI();
 476 
 477                 if (bspui != null) {
 478                     JSplitPane splitPane = bspui.getSplitPane();
 479 
 480                     if (splitPane != null) {
 481                         if (splitPane.getOrientation() ==
 482                             JSplitPane.HORIZONTAL_SPLIT) {
 483                             insets.top = insets.bottom = 0;
 484                             insets.left = insets.right = 1;
 485                             return insets;
 486                         }
 487                         // VERTICAL_SPLIT
 488                         insets.top = insets.bottom = 1;
 489                         insets.left = insets.right = 0;
 490                         return insets;
 491                     }
 492                 }
 493             }
 494             insets.top = insets.bottom = insets.left = insets.right = 1;
 495             return insets;
 496         }
 497         public boolean isBorderOpaque() { return true; }
 498     }

      Draws the border around the splitpane. To work correctly you should
      also install a border on the divider (property SplitPaneDivider.border).

 505     public static class SplitPaneBorder implements Border, UIResource {
 506         protected Color highlight;
 507         protected Color shadow;
 508 
 509         public SplitPaneBorder(Color highlight, Color shadow) {
 510             this.highlight = highlight;
 511             this.shadow = shadow;
 512         }
 513 
 514         public void paintBorder(Component c, Graphics g, int x, int y,
 515                                 int width, int height) {
 516             if (!(c instanceof JSplitPane)) {
 517                 return;
 518             }
 519             // The only tricky part with this border is that the divider is
 520             // not positioned at the top (for horizontal) or left (for vert),
 521             // so this border draws to where the divider is:
 522             // -----------------
 523             // |xxxxxxx xxxxxxx|
 524             // |x     ---     x|
 525             // |x     | |     x|
 526             // |x     |D|     x|
 527             // |x     | |     x|
 528             // |x     ---     x|
 529             // |xxxxxxx xxxxxxx|
 530             // -----------------
 531             // The above shows (rather excessively) what this looks like for
 532             // a horizontal orientation. This border then draws the x's, with
 533             // the SplitPaneDividerBorder drawing its own border.
 534 
 535             Component          child;
 536             Rectangle          cBounds;
 537 
 538             JSplitPane splitPane = (JSplitPane)c;
 539 
 540             child = splitPane.getLeftComponent();
 541             // This is needed for the space between the divider and end of
 542             // splitpane.
 543             g.setColor(c.getBackground());
 544             g.drawRect(x, y, width - 1, height - 1);
 545             if(splitPane.getOrientation() == JSplitPane.HORIZONTAL_SPLIT) {
 546                 if(child != null) {
 547                     cBounds = child.getBounds();
 548                     g.setColor(shadow);
 549                     g.drawLine(0, 0, cBounds.width + 1, 0);
 550                     g.drawLine(0, 1, 0, cBounds.height + 1);
 551 
 552                     g.setColor(highlight);
 553                     g.drawLine(0, cBounds.height + 1, cBounds.width + 1,
 554                                cBounds.height + 1);
 555                 }
 556                 child = splitPane.getRightComponent();
 557                 if(child != null) {
 558                     cBounds = child.getBounds();
 559 
 560                     int             maxX = cBounds.x + cBounds.width;
 561                     int             maxY = cBounds.y + cBounds.height;
 562 
 563                     g.setColor(shadow);
 564                     g.drawLine(cBounds.x - 1, 0, maxX, 0);
 565                     g.setColor(highlight);
 566                     g.drawLine(cBounds.x - 1, maxY, maxX, maxY);
 567                     g.drawLine(maxX, 0, maxX, maxY + 1);
 568                 }
 569             } else {
 570                 if(child != null) {
 571                     cBounds = child.getBounds();
 572                     g.setColor(shadow);
 573                     g.drawLine(0, 0, cBounds.width + 1, 0);
 574                     g.drawLine(0, 1, 0, cBounds.height);
 575                     g.setColor(highlight);
 576                     g.drawLine(1 + cBounds.width, 0, 1 + cBounds.width,
 577                                cBounds.height + 1);
 578                     g.drawLine(0, cBounds.height + 1, 0, cBounds.height + 1);
 579                 }
 580                 child = splitPane.getRightComponent();
 581                 if(child != null) {
 582                     cBounds = child.getBounds();
 583 
 584                     int             maxX = cBounds.x + cBounds.width;
 585                     int             maxY = cBounds.y + cBounds.height;
 586 
 587                     g.setColor(shadow);
 588                     g.drawLine(0, cBounds.y - 1, 0, maxY);
 589                     g.drawLine(maxX, cBounds.y - 1, maxX, cBounds.y - 1);
 590                     g.setColor(highlight);
 591                     g.drawLine(0, maxY, cBounds.width + 1, maxY);
 592                     g.drawLine(maxX, cBounds.y, maxX, maxY);
 593                 }
 594             }
 595         }
 596         public Insets getBorderInsets(Component c) {
 597             return new Insets(1, 1, 1, 1);
 598         }
 599         public boolean isBorderOpaque() { return true; }
 600     }
