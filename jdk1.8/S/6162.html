  37 final class BooleanCall extends FunctionCall {
  38 
  39     private Expression _arg = null;
  40 
  41     public BooleanCall(QName fname, Vector arguments) {
  42         super(fname, arguments);
  43         _arg = argument(0);
  44     }
  45 
  46     public Type typeCheck(SymbolTable stable) throws TypeCheckError {
  47         _arg.typeCheck(stable);
  48         return _type = Type.Boolean;
  49     }
  50 
  51     public void translate(ClassGenerator classGen, MethodGenerator methodGen) {
  52         _arg.translate(classGen, methodGen);
  53         final Type targ = _arg.getType();
  54         if (!targ.identicalTo(Type.Boolean)) {
  55             _arg.startIterator(classGen, methodGen);
  56             targ.translateTo(classGen, methodGen, Type.Boolean);
  57         }
  58     }
