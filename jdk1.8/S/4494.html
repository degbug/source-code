  55 /**
  56  * Provides a JAAS login module that prompts for a key store alias and
  57  * populates the subject with the alias's principal and credentials. Stores
  58  * an X500Principal for the subject distinguished name of the
  59  * first certificate in the alias's credentials in the subject's principals,
  60  * the alias's certificate path in the subject's public credentials, and a
  61  * X500PrivateCredential whose certificate is the first
  62  * certificate in the alias's certificate path and whose private key is the
  63  * alias's private key in the subject's private credentials. 
  64  *
  65  * Recognizes the following options in the configuration file:
  66  * 

  67  *
  68  * 
 keyStoreURL 

  69  * 
 A URL that specifies the location of the key store.  Defaults to

  70  *      a URL pointing to the .keystore file in the directory specified by the
  71  *      user.home system property.  The input stream from this
  72  *      URL is passed to the KeyStore.load method.
  73  *      "NONE" may be specified if a null stream must be
  74  *      passed to the KeyStore.load method.
  75  *      "NONE" should be specified if the KeyStore resides
  76  *      on a hardware token device, for example.
  77  *
  78  * 
 keyStoreType 

  79  * 
 The key store type.  If not specified, defaults to the result of

  80  *      calling KeyStore.getDefaultType().
  81  *      If the type is "PKCS11", then keyStoreURL must be "NONE"
  82  *      and privateKeyPasswordURL must not be specified.
  83  *
  84  * 
 keyStoreProvider 

  85  * 
 The key store provider.  If not specified, uses the standard search

  86  *      order to find the provider. 
  87  *
  88  * 
 keyStoreAlias 

  89  * 
 The alias in the key store to login as.  Required when no callback

  90  *      handler is provided.  No default value. 
  91  *
  92  * 
 keyStorePasswordURL 

  93  * 
 A URL that specifies the location of the key store password.  Required

  94  *      when no callback handler is provided and
  95  *      protected is false.
  96  *      No default value. 
  97  *
  98  * 
 privateKeyPasswordURL 

  99  * 
 A URL that specifies the location of the specific private key password

 100  *      needed to access the private key for this alias.
 101  *      The keystore password
 102  *      is used if this value is needed and not specified. 
 103  *
 104  * 
 protected 

 105  * 
 This value should be set to "true" if the KeyStore

 106  *      has a separate, protected authentication path
 107  *      (for example, a dedicated PIN-pad attached to a smart card).
 108  *      Defaults to "false". If "true" keyStorePasswordURL and
 109  *      privateKeyPasswordURL must not be specified.
 110  *
 111  * 
 112  */

 269     public boolean login() throws LoginException {
 270         switch (status) {
 271         case UNINITIALIZED:
 272         default:
 273             throw new LoginException("The login module is not initialized");
 274         case INITIALIZED:
 275         case AUTHENTICATED:
 276 
 277             if (token && !nullStream) {
 278                 throw new LoginException
 279                         ("if keyStoreType is " + P11KEYSTORE +
 280                         " then keyStoreURL must be " + NONE);
 281             }
 282 
 283             if (token && privateKeyPasswordURL != null) {
 284                 throw new LoginException
 285                         ("if keyStoreType is " + P11KEYSTORE +
 286                         " then privateKeyPasswordURL must not be specified");
 287             }
 288 
 289             if (protectedPath &&
 290                 (keyStorePasswordURL != null ||
 291                         privateKeyPasswordURL != null)) {
 292                 throw new LoginException
 293                         ("if protected is true then keyStorePasswordURL and " +
 294                         "privateKeyPasswordURL must not be specified");
 295             }
 296 
 297             // get relevant alias and password info
 298 
 299             if (protectedPath) {
 300                 getAliasAndPasswords(PROTECTED_PATH);
 301             } else if (token) {
 302                 getAliasAndPasswords(TOKEN);
 303             } else {
 304                 getAliasAndPasswords(NORMAL);
 305             }
 306 
 307             // log into KeyStore to retrieve data,
 308             // then clear passwords
 309 
 310             try {
 311                 getKeyStoreInfo();
 312             } finally {
 313                 if (privateKeyPassword != null &&
 314                     privateKeyPassword != keyStorePassword) {
 315                     Arrays.fill(privateKeyPassword, '\0');
 316                     privateKeyPassword = null;
 317                 }
 318                 if (keyStorePassword != null) {
 319                     Arrays.fill(keyStorePassword, '\0');
 320                     keyStorePassword = null;
 321                 }
 322             }
 323             status = AUTHENTICATED;
 324             return true;
 325         case LOGGED_IN:
 326             return true;
 327         }
 328     }

 331     @SuppressWarnings("fallthrough")
 332     private void getAliasAndPasswords(int env) throws LoginException {
 333         if (callbackHandler == null) {
 334 
 335             // No callback handler - check for alias and password options
 336 
 337             switch (env) {
 338             case PROTECTED_PATH:
 339                 checkAlias();
 340                 break;
 341             case TOKEN:
 342                 checkAlias();
 343                 checkStorePass();
 344                 break;
 345             case NORMAL:
 346                 checkAlias();
 347                 checkStorePass();
 348                 checkKeyPass();
 349                 break;
 350             }
 351 
 352         } else {
 353 
 354             // Callback handler available - prompt for alias and passwords
 355 
 356             NameCallback aliasCallback;
 357             if (keyStoreAlias == null || keyStoreAlias.length() == 0) {
 358                 aliasCallback = new NameCallback(
 359                                         rb.getString("Keystore.alias."));
 360             } else {
 361                 aliasCallback =
 362                     new NameCallback(rb.getString("Keystore.alias."),
 363                                      keyStoreAlias);
 364             }
 365 
 366             PasswordCallback storePassCallback = null;
 367             PasswordCallback keyPassCallback = null;
 368 
 369             switch (env) {
 370             case PROTECTED_PATH:
 371                 break;
 372             case NORMAL:
 373                 keyPassCallback = new PasswordCallback
 374                     (rb.getString("Private.key.password.optional."), false);
 375                 // fall thru
 376             case TOKEN:
 377                 storePassCallback = new PasswordCallback
 378                     (rb.getString("Keystore.password."), false);
 379                 break;
 380             }
 381             prompt(aliasCallback, storePassCallback, keyPassCallback);
 382         }
 383 
 384         if (debug) {
 385             debugPrint("alias=" + keyStoreAlias);
 386         }
 387     }

 397     private void checkStorePass() throws LoginException {
 398         if (keyStorePasswordURL == null) {
 399             throw new LoginException
 400                 ("Need to specify keyStorePasswordURL option to use " +
 401                 "KeyStoreLoginModule non-interactively.");
 402         }
 403         InputStream in = null;
 404         try {
 405             in = new URL(keyStorePasswordURL).openStream();
 406             keyStorePassword = Password.readPassword(in);
 407         } catch (IOException e) {
 408             LoginException le = new LoginException
 409                 ("Problem accessing keystore password \"" +
 410                 keyStorePasswordURL + "\"");
 411             le.initCause(e);
 412             throw le;
 413         } finally {
 414             if (in != null) {
 415                 try {
 416                     in.close();
 417                 } catch (IOException ioe) {
 418                     LoginException le = new LoginException(
 419                         "Problem closing the keystore password stream");
 420                     le.initCause(ioe);
 421                     throw le;
 422                 }
 423             }
 424         }
 425     }

 485                 throw new LoginException("Login cancelled");

 491

 493

 495                 callbackHandler.handle(
 496                     new Callback[] {
 497                         bannerCallback, aliasCallback,
 498                         storePassCallback, confirmationCallback
 499                     });
 500             } catch (IOException e) {
 501                 LoginException le = new LoginException
 502                         ("Problem retrieving keystore alias and password");
 503                 le.initCause(e);
 504                 throw le;
 505             } catch (UnsupportedCallbackException e) {
 506                 throw new LoginException(
 507                     "Error: " + e.getCallback().toString() +
 508                     " is not available to retrieve authentication " +
 509                     " information from the user");
 510             }
 511 
 512             int confirmationResult = confirmationCallback.getSelectedIndex();
 513 
 514             if (confirmationResult == ConfirmationCallback.CANCEL) {
 515                 throw new LoginException("Login cancelled");
 516             }
 517 
 518             saveAlias(aliasCallback);
 519             saveStorePass(storePassCallback);
 520 
 521         } else {
 522 
 523             // prompt for alias, key store password, and key password
 524 
 525             try {
 526                 callbackHandler.handle(
 527                     new Callback[] {
 528                         bannerCallback, aliasCallback,

 532             } catch (IOException e) {

 536                 throw le;
 537             } catch (UnsupportedCallbackException e) {
 538                 throw new LoginException(
 539                     "Error: " + e.getCallback().toString() +

 541                     " information from the user");
 542             }

 545 
 546             if (confirmationResult == ConfirmationCallback.CANCEL) {

 548             }

 581     /** Get the credentials from the KeyStore. */
 582     private void getKeyStoreInfo() throws LoginException {
 583 
 584         /* Get KeyStore instance */
 585         try {
 586             if (keyStoreProvider == null) {
 587                 keyStore = KeyStore.getInstance(keyStoreType);
 588             } else {
 589                 keyStore =
 590                     KeyStore.getInstance(keyStoreType, keyStoreProvider);
 591             }
 592         } catch (KeyStoreException e) {
 593             LoginException le = new LoginException
 594                 ("The specified keystore type was not available");
 595             le.initCause(e);
 596             throw le;
 597         } catch (NoSuchProviderException e) {
 598             LoginException le = new LoginException
 599                 ("The specified keystore provider was not available");
 600             le.initCause(e);
 601             throw le;
 602         }
 603 
 604         /* Load KeyStore contents from file */
 605         InputStream in = null;
 606         try {
 607             if (nullStream) {
 608                 // if using protected auth path, keyStorePassword will be null
 609                 keyStore.load(null, keyStorePassword);
 610             } else {
 611                 in = new URL(keyStoreURL).openStream();
 612                 keyStore.load(in, keyStorePassword);
 613             }
 614         } catch (MalformedURLException e) {
 615             LoginException le = new LoginException
 616                                 ("Incorrect keyStoreURL option");
 617             le.initCause(e);
 618             throw le;
 619         } catch (GeneralSecurityException e) {
 620             LoginException le = new LoginException
 621                                 ("Error initializing keystore");
 622             le.initCause(e);
 623             throw le;
 624         } catch (IOException e) {
 625             LoginException le = new LoginException
 626                                 ("Error initializing keystore");
 627             le.initCause(e);
 628             throw le;
 629         } finally {
 630             if (in != null) {
 631                 try {
 632                     in.close();
 633                 } catch (IOException ioe) {
 634                     LoginException le = new LoginException
 635                                 ("Error initializing keystore");
 636                     le.initCause(ioe);
 637                     throw le;
 638                 }
 639             }
 640         }
 641 
 642         /* Get certificate chain and create a certificate path */
 643         try {
 644             fromKeyStore =
 645                 keyStore.getCertificateChain(keyStoreAlias);
 646             if (fromKeyStore == null
 647                 || fromKeyStore.length == 0
 648                 || !(fromKeyStore[0] instanceof X509Certificate))
 649             {
 650                 throw new FailedLoginException(
 651                     "Unable to find X.509 certificate chain in keystore");
 652             } else {
 653                 LinkedList<Certificate> certList = new LinkedList<>();
 654                 for (int i=0; i < fromKeyStore.length; i++) {
 655                     certList.add(fromKeyStore[i]);
 656                 }
 657                 CertificateFactory certF=
 658                     CertificateFactory.getInstance("X.509");
 659                 certP =
 660                     certF.generateCertPath(certList);
 661             }
 662         } catch (KeyStoreException e) {
 663             LoginException le = new LoginException("Error using keystore");
 664             le.initCause(e);
 665             throw le;
 666         } catch (CertificateException ce) {
 667             LoginException le = new LoginException
 668                 ("Error: X.509 Certificate type unavailable");
 669             le.initCause(ce);
 670             throw le;
 671         }
 672 
 673         /* Get principal and keys */
 674         try {
 675             X509Certificate certificate = (X509Certificate)fromKeyStore[0];
 676             principal = new javax.security.auth.x500.X500Principal
 677                 (certificate.getSubjectDN().getName());
 678 
 679             // if token, privateKeyPassword will be null
 680             Key privateKey = keyStore.getKey(keyStoreAlias, privateKeyPassword);
 681             if (privateKey == null
 682                 || !(privateKey instanceof PrivateKey))
 683             {
 684                 throw new FailedLoginException(
 685                     "Unable to recover key from keystore");
 686             }
 687 
 688             privateCredential = new X500PrivateCredential(
 689                 certificate, (PrivateKey) privateKey, keyStoreAlias);
 690         } catch (KeyStoreException e) {
 691             LoginException le = new LoginException("Error using keystore");
 692             le.initCause(e);
 693             throw le;
 694         } catch (NoSuchAlgorithmException e) {
 695             LoginException le = new LoginException("Error using keystore");
 696             le.initCause(e);
 697             throw le;
 698         } catch (UnrecoverableKeyException e) {
 699             FailedLoginException fle = new FailedLoginException
 700                                 ("Unable to recover key from keystore");
 701             fle.initCause(e);
 702             throw fle;
 703         }
 704         if (debug) {
 705             debugPrint("principal=" + principal +
 706                        "\n certificate="
 707                        + privateCredential.getCertificate() +
 708                        "\n alias =" + privateCredential.getAlias());
 709         }
 710     }

 712     /**
 713      * Abstract method to commit the authentication process (phase 2).
 714      *
 715      *  This method is called if the LoginContext's
 716      * overall authentication succeeded
 717      * (the relevant REQUIRED, REQUISITE, SUFFICIENT and OPTIONAL LoginModules
 718      * succeeded).
 719      *
 720      *  If this LoginModule's own authentication attempt
 721      * succeeded (checked by retrieving the private state saved by the
 722      * login method), then this method associates a
 723      * X500Principal for the subject distinguished name of the
 724      * first certificate in the alias's credentials in the subject's
 725      * principals,the alias's certificate path in the subject's public
 726      * credentials, and aX500PrivateCredential whose certificate
 727      * is the first  certificate in the alias's certificate path and whose
 728      * private key is the alias's private key in the subject's private
 729      * credentials.  If this LoginModule's own
 730      * authentication attempted failed, then this method removes
 731      * any state that was originally saved.
 732      *
 733      * 
 734      *
 735      * @exception LoginException if the commit fails
 736      *
 737      * @return true if this LoginModule's own login and commit
 738      *          attempts succeeded, or false otherwise.
 739      */
 740 
 741     public boolean commit() throws LoginException {
 742         switch (status) {
 743         case UNINITIALIZED:
 744         default:
 745             throw new LoginException("The login module is not initialized");
 746         case INITIALIZED:
 747             logoutInternal();
 748             throw new LoginException("Authentication failed");
 749         case AUTHENTICATED:
 750             if (commitInternal()) {
 751                 return true;
 752             } else {
 753                 logoutInternal();
 754                 throw new LoginException("Unable to retrieve certificates");
 755             }
 756         case LOGGED_IN:
 757             return true;
 758         }
 759     }
 760 
 761     private boolean commitInternal() throws LoginException {
 762         /* If the subject is not readonly add to the principal and credentials
 763          * set; otherwise just return true
 764          */
 765         if (subject.isReadOnly()) {
 766             throw new LoginException ("Subject is set readonly");
 767         } else {
 768             subject.getPrincipals().add(principal);
 769             subject.getPublicCredentials().add(certP);
 770             subject.getPrivateCredentials().add(privateCredential);
 771             status = LOGGED_IN;
 772             return true;
 773         }
 774     }

 776     /**
 777      *  This method is called if the LoginContext's
 778      * overall authentication failed.
 779      * (the relevant REQUIRED, REQUISITE, SUFFICIENT and OPTIONAL LoginModules
 780      * did not succeed).
 781      *
 782      *  If this LoginModule's own authentication attempt
 783      * succeeded (checked by retrieving the private state saved by the
 784      * login and commit methods),
 785      * then this method cleans up any state that was originally saved.
 786      *
 787      *  If the loaded KeyStore's provider extends
 788      * java.security.AuthProvider,
 789      * then the provider's logout method is invoked.
 790      *
 791      * 
 792      *
 793      * @exception LoginException if the abort fails.
 794      *
 795      * @return false if this LoginModule's own login and/or commit attempts
 796      *          failed, and true otherwise.
 797      */
 798 
 799     public boolean abort() throws LoginException {
 800         switch (status) {
 801         case UNINITIALIZED:
 802         default:
 803             return false;
 804         case INITIALIZED:
 805             return false;
 806         case AUTHENTICATED:
 807             logoutInternal();
 808             return true;
 809         case LOGGED_IN:
 810             logoutInternal();
 811             return true;
 812         }
 813     }
 814     /**
 815      * Logout a user.
 816      *
 817      *  This method removes the Principals, public credentials and the
 818      * private credentials that were added by the commit method.
 819      *
 820      *  If the loaded KeyStore's provider extends
 821      * java.security.AuthProvider,
 822      * then the provider's logout method is invoked.
 823      *
 824      * 
 825      *
 826      * @exception LoginException if the logout fails.
 827      *
 828      * @return true in all cases since this LoginModule
 829      *          should not be ignored.
 830      */
 831 
 832     public boolean logout() throws LoginException {
 833         if (debug)
 834             debugPrint("Entering logout " + status);
 835         switch (status) {
 836         case UNINITIALIZED:
 837             throw new LoginException
 838                 ("The login module is not initialized");
 839         case INITIALIZED:
 840         case AUTHENTICATED:
 841         default:
 842            // impossible for LoginModule to be in AUTHENTICATED
 843            // state
 844            // assert status != AUTHENTICATED;
 845             return false;
 846         case LOGGED_IN:
 847             logoutInternal();
 848             return true;
 849         }
 850     }
 851 
 852     private void logoutInternal() throws LoginException {
 853         if (debug) {
 854             debugPrint("Entering logoutInternal");
 855         }
 856 
 857         // assumption is that KeyStore.load did a login -
 858         // perform explicit logout if possible
 859         LoginException logoutException = null;
 860         Provider provider = keyStore.getProvider();
 861         if (provider instanceof AuthProvider) {
 862             AuthProvider ap = (AuthProvider)provider;
 863             try {
 864                 ap.logout();
 865                 if (debug) {
 866                     debugPrint("logged out of KeyStore AuthProvider");
 867                 }
 868             } catch (LoginException le) {
 869                 // save but continue below
 870                 logoutException = le;
 871             }
 872         }
 873 
 874         if (subject.isReadOnly()) {
 875             // attempt to destroy the private credential
 876             // even if the Subject is read-only
 877             principal = null;
 878             certP = null;
 879             status = INITIALIZED;
 880             // destroy the private credential
 881             Iterator<Object> it = subject.getPrivateCredentials().iterator();
 882             while (it.hasNext()) {
 883                 Object obj = it.next();
 884                 if (privateCredential.equals(obj)) {
 885                     privateCredential = null;
 886                     try {
 887                         ((Destroyable)obj).destroy();
 888                         if (debug)
 889                             debugPrint("Destroyed private credential, " +
 890                                        obj.getClass().getName());
 891                         break;
 892                     } catch (DestroyFailedException dfe) {
 893                         LoginException le = new LoginException
 894                             ("Unable to destroy private credential, "
 895                              + obj.getClass().getName());
 896                         le.initCause(dfe);
 897                         throw le;
 898                     }
 899                 }
 900             }
 901 
 902             // throw an exception because we can not remove
 903             // the principal and public credential from this
 904             // read-only Subject
 905             throw new LoginException
 906                 ("Unable to remove Principal ("
 907                  + "X500Principal "
 908                  + ") and public credential (certificatepath) "
 909                  + "from read-only Subject");
 910         }
 911         if (principal != null) {
 912             subject.getPrincipals().remove(principal);
 913             principal = null;
 914         }
 915         if (certP != null) {
 916             subject.getPublicCredentials().remove(certP);
 917             certP = null;
 918         }
 919         if (privateCredential != null) {
 920             subject.getPrivateCredentials().remove(privateCredential);
 921             privateCredential = null;
 922         }
 923 
 924         // throw pending logout exception if there is one
 925         if (logoutException != null) {
 926             throw logoutException;
 927         }
 928         status = INITIALIZED;
 929     }
