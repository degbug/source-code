  28 abstract class POAPolicyMediatorFactory {
  29     // create an appropriate policy mediator based on the policies.
  30     // Note that the policies object has already been validated before
  31     // this call, so it can only contain valid combinations of POA policies.
  32     static POAPolicyMediator create( Policies policies, POAImpl poa )
  33     {
  34         if (policies.retainServants()) {
  35             if (policies.useActiveMapOnly())
  36                 return new POAPolicyMediatorImpl_R_AOM( policies, poa ) ;
  37             else if (policies.useDefaultServant())
  38                 return new POAPolicyMediatorImpl_R_UDS( policies, poa ) ;
  39             else if (policies.useServantManager())
  40                 return new POAPolicyMediatorImpl_R_USM( policies, poa ) ;
  41             else
  42                 throw poa.invocationWrapper().pmfCreateRetain() ;
  43         } else {
  44             if (policies.useDefaultServant())
  45                 return new POAPolicyMediatorImpl_NR_UDS( policies, poa ) ;
  46             else if (policies.useServantManager())
  47                 return new POAPolicyMediatorImpl_NR_USM( policies, poa ) ;
  48             else
  49                 throw poa.invocationWrapper().pmfCreateNonRetain() ;
  50         }
  51     }
  52 }
