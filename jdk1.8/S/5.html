  55 /*
  56  * The following environment variable is used to influence the behavior
  57  * of the jre exec'd through the SelectVersion routine.  The command line
  58  * options which specify the version are not passed to the exec'd version,
  59  * because that jre may be an older version which wouldn't recognize them.
  60  * This environment variable is known to this (and later) version and serves
  61  * to suppress the version selection code.  This is not only for efficiency,
  62  * but also for correctness, since any command line options have been
  63  * removed which would cause any value found in the manifest to be used.
  64  * This would be incorrect because the command line options are defined
  65  * to take precedence.
  66  *
  67  * The value associated with this environment variable is the MainClass
  68  * name from within the executable jar file (if any). This is strictly a
  69  * performance enhancement to avoid re-reading the jar file manifest.
  70  *
  71  */
  72 #define ENV_ENTRY "_JAVA_VERSION_SET"

 245 #define NULL_CHECK_RETURN_VALUE(NCRV_check_pointer, NCRV_return_value) \
 246     do { \
 247         if ((NCRV_check_pointer) == NULL) { \
 248             JLI_ReportErrorMessage(JNI_ERROR); \
 249             return NCRV_return_value; \
 250         } \
 251     } while (JNI_FALSE)

 253 #define NULL_CHECK0(NC0_check_pointer) \
 254     NULL_CHECK_RETURN_VALUE(NC0_check_pointer, 0)

 257     NULL_CHECK_RETURN_VALUE(NC_check_pointer, )

 259 /*
 260  * For JNI calls :
 261  *  - check for thrown exceptions
 262  *  - check for null return
 263  *
 264  *  JNI calls can return null and/or throw an exception.  Check for these.
 265  *
 266  *  : CHECK_JNI_RETURN_EXCEPTION()
 267  *    return the specified RETURNVALUE if exception was generated
 268  *  : CHECK_JNI_RETURN_0(JNISTATEMENT)        : check if JNISTATEMENT was successful, return 0 if not
 269  *  : CHECK_JNI_RETURN_VOID(JNISTATEMENT)     : check if JNISTATEMENT was successful, return void if not
 270  *  : CHECK_JNI_RETURN_VALUE(JNISTATEMENT,n)  : check if JNISTATEMENT was successful, return n if not
 271  *
 272  *  These macros need at least one parameter, the JNI statement [ JNISTATEMENT ].
 273  *
 274  *  E.G.: check the JNI statement, and specify a value to return if a failure was detected.
 275  *
 276  *      CHECK_JNI_RETURN_VALUE(str = (*env)->CallStaticObjectMethod(env, cls,
 277  *                                               makePlatformStringMID, USE_STDERR, ary), -1);
 278  */
 279 
 280 #define RETURNVOID return
 281 #define RETURN0 return 0
 282 #define RETURN(N) return (N)
 283 
 284 #define CHECK_JNI_RETURN_EXCEPTION(RETURNVALUE) \
 285         if ((((*env)->ExceptionOccurred(env))!=NULL)) { \
 286             RETURNVALUE; \
 287         }
 288 
 289 #define CHECK_JNI_RETURN_0(JNISTATEMENT) \
 290     CHECK_JNI_RETURN_EXCEPTION(RETURN0); \
 291     NULL_CHECK0(JNISTATEMENT);
 292 
 293 #define CHECK_JNI_RETURN_VOID(JNISTATEMENT) \
 294     CHECK_JNI_RETURN_EXCEPTION(RETURNVOID); \
 295     NULL_CHECK(JNISTATEMENT);
 296 
 297 #define CHECK_JNI_RETURN_VALUE(JNISTATEMENT, NCRV_return_value) \
 298     CHECK_JNI_RETURN_EXCEPTION(RETURN(NCRV_return_value)); \
 299     NULL_CHECK_RETURN_VALUE(JNISTATEMENT, NCRV_return_value);
