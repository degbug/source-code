  26 package org.omg.CORBA;

  28 /**
  29  * An object containing a modifiable list of String objects
  30  * that represent property names.
  31  * This class is used in Request operations to
  32  * describe the contexts that need to be resolved and sent with the
  33  * invocation.  (A context is resolved by giving a property name
  34  * and getting back the value associated with it.)  This is done
  35  * by calling the Context method
  36  * get_values and supplying a string from a
  37  * ContextList object as the third parameter.
  38  * The method get_values returns an NVList
  39  * object containing the NamedValue objects that hold
  40  * the value(s) identified by the given string.
  41  * 

  42  * A ContextList object is created by the ORB, as
  43  * illustrated here:
  44  * 

  45  *   ORB orb = ORB.init(args, null);
  46  *   org.omg.CORBA.ContextList ctxList = orb.create_context_list();
  47  * 
  48  * The variable ctxList represents an empty
  49  * ContextList object.  Strings are added to
  50  * the list with the method add, accessed
  51  * with the method item, and removed with the
  52  * method remove.
  53  *
  54  * @see Context
  55  * @since   JDK1.2
  56  */
  57 
  58 public abstract class ContextList {
  59 
  60     /**
  61      * Returns the number of String objects in this
  62      * ContextList object.
  63      *
  64      * @return                  an int representing the number of
  65      * Strings in this ContextList object
  66      */
  67 
  68     public abstract int count();
  69 
  70     /**
  71      * Adds a String object to this ContextList
  72      * object.
  73      *
  74      * @param ctx               the String object to be added
  75      */
  76 
  77     public abstract void add(String ctx);
  78 
  79     /**
  80      * Returns the String object at the given index.
  81      *
  82      * @param index             the index of the string desired, with 0 being the
  83      index of the first string
  84      * @return                  the string at the given index
  85      * @exception org.omg.CORBA.Bounds  if the index is greater than
  86      *                          or equal to the number of strings in this
  87      *                ContextList object
  88      */
  89 
  90     public abstract String item(int index) throws org.omg.CORBA.Bounds;
  91 
  92     /**
  93      * Removes the String object at the given index. Note that
  94      * the indices of all strings following the one removed are
  95      * shifted down by one.
  96      *
  97      * @param index     the index of the String object to be removed,
  98      *                with 0 designating the first string
  99      * @exception org.omg.CORBA.Bounds  if the index is greater than
 100      *                          or equal to the number of String objects in
 101      *                this ContextList object
 102      */
 103 
 104     public abstract void remove(int index) throws org.omg.CORBA.Bounds;
 105 
 106 }
