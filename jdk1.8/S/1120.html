  48 /**
  49  * An optionally-bounded {@linkplain BlockingDeque blocking deque} based on
  50  * linked nodes.
  51  *
  52  * <p>The optional capacity bound constructor argument serves as a
  53  * way to prevent excessive expansion. The capacity, if unspecified,
  54  * is equal to {@link Integer#MAX_VALUE}.  Linked nodes are
  55  * dynamically created upon each insertion unless this would bring the
  56  * deque above capacity.
  57  *
  58  * <p>Most operations run in constant time (ignoring time spent
  59  * blocking).  Exceptions include {@link #remove(Object) remove},
  60  * {@link #removeFirstOccurrence removeFirstOccurrence}, {@link
  61  * #removeLastOccurrence removeLastOccurrence}, {@link #contains
  62  * contains}, {@link #iterator iterator.remove()}, and the bulk
  63  * operations, all of which run in linear time.
  64  *
  65  * <p>This class and its iterator implement all of the
  66  * <em>optional</em> methods of the {@link Collection} and {@link
  67  * Iterator} interfaces.
  68  *
  69  * <p>This class is a member of the
  70  * <a href="{@docRoot}/../technotes/guides/collections/index.html">
  71  * Java Collections Framework</a>.
  72  *
  73  * @since 1.6
  74  * @author  Doug Lea
  75  * @param <E> the type of elements held in this collection
  76  */

  78     extends AbstractQueue<E>

  81     /*
  82      * Implemented as a simple doubly-linked list protected by a
  83      * single lock and using conditions to manage blocking.
  84      *
  85      * To implement weakly consistent iterators, it appears we need to
  86      * keep all Nodes GC-reachable from a predecessor dequeued Node.
  87      * That would cause two problems:
  88      * - allow a rogue Iterator to cause unbounded memory retention
  89      * - cause cross-generational linking of old Nodes to new Nodes if
  90      *   a Node was tenured while live, which generational GCs have a
  91      *   hard time dealing with, causing repeated major collections.
  92      * However, only non-deleted Nodes need to be reachable from
  93      * dequeued Nodes, and reachability does not necessarily have to
  94      * be of the kind understood by the GC.  We use the trick of
  95      * linking a Node that has just been dequeued to itself.  Such a
  96      * self-link implicitly means to jump to "first" (for next links)
  97      * or "last" (for prev links).
  98      */
  99

 185     /**
 186      * Creates a {@code LinkedBlockingDeque} with a capacity of
 187      * {@link Integer#MAX_VALUE}, initially containing the elements of
 188      * the given collection, added in traversal order of the
 189      * collection's iterator.
 190      *
 191      * @param c the collection of elements to initially contain
 192      * @throws NullPointerException if the specified collection or any
 193      *         of its elements are null
 194      */
 195     public LinkedBlockingDeque(Collection<? extends E> c) {
 196         this(Integer.MAX_VALUE);
 197         final ReentrantLock lock = this.lock;
 198         lock.lock(); // Never contended, but necessary for visibility
 199         try {
 200             for (E e : c) {
 201                 if (e == null)
 202                     throw new NullPointerException();
 203                 if (!linkLast(new Node<E>(e)))
 204                     throw new IllegalStateException("Deque full");
 205             }
 206         } finally {
 207             lock.unlock();
 208         }
 209     }

 214     /**
 215      * Links node as first element, or returns false if full.
 216      */
 217     private boolean linkFirst(Node<E> node) {
 218         // assert lock.isHeldByCurrentThread();
 219         if (count >= capacity)
 220             return false;
 221         Node<E> f = first;
 222         node.next = f;
 223         first = node;
 224         if (last == null)
 225             last = node;
 226         else
 227             f.prev = node;
 228         ++count;
 229         notEmpty.signal();
 230         return true;
 231     }
 232 
 233     /**
 234      * Links node as last element, or returns false if full.
 235      */
 236     private boolean linkLast(Node<E> node) {
 237         // assert lock.isHeldByCurrentThread();
 238         if (count >= capacity)
 239             return false;
 240         Node<E> l = last;
 241         node.prev = l;
 242         last = node;
 243         if (first == null)
 244             first = node;
 245         else
 246             l.next = node;
 247         ++count;
 248         notEmpty.signal();
 249         return true;
 250     }
 251 
 252     /**
 253      * Removes and returns first element, or null if empty.
 254      */
 255     private E unlinkFirst() {
 256         // assert lock.isHeldByCurrentThread();
 257         Node<E> f = first;
 258         if (f == null)
 259             return null;
 260         Node<E> n = f.next;
 261         E item = f.item;
 262         f.item = null;
 263         f.next = f; // help GC
 264         first = n;
 265         if (n == null)
 266             last = null;
 267         else
 268             n.prev = null;
 269         --count;
 270         notFull.signal();
 271         return item;
 272     }
 273 
 274     /**
 275      * Removes and returns last element, or null if empty.
 276      */
 277     private E unlinkLast() {
 278         // assert lock.isHeldByCurrentThread();
 279         Node<E> l = last;
 280         if (l == null)
 281             return null;
 282         Node<E> p = l.prev;
 283         E item = l.item;
 284         l.item = null;
 285         l.prev = l; // help GC
 286         last = p;
 287         if (p == null)
 288             first = null;
 289         else
 290             p.next = null;
 291         --count;
 292         notFull.signal();
 293         return item;
 294     }
 295 
 296     /**
 297      * Unlinks x.
 298      */
 299     void unlink(Node<E> x) {
 300         // assert lock.isHeldByCurrentThread();
 301         Node<E> p = x.prev;
 302         Node<E> n = x.next;
 303         if (p == null) {
 304             unlinkFirst();
 305         } else if (n == null) {
 306             unlinkLast();
 307         } else {
 308             p.next = n;
 309             n.prev = p;
 310             x.item = null;
 311             // Don't mess with x's links.  They may still be in use by
 312             // an iterator.
 313             --count;
 314             notFull.signal();
 315         }
 316     }

 586     public boolean removeFirstOccurrence(Object o) {
 587         if (o == null) return false;
 588         final ReentrantLock lock = this.lock;
 589         lock.lock();
 590         try {
 591             for (Node<E> p = first; p != null; p = p.next) {
 592                 if (o.equals(p.item)) {

 594                     return true;
 595                 }

 736     /**
 737      * @throws UnsupportedOperationException {@inheritDoc}
 738      * @throws ClassCastException            {@inheritDoc}
 739      * @throws NullPointerException          {@inheritDoc}
 740      * @throws IllegalArgumentException      {@inheritDoc}
 741      */
 742     public int drainTo(Collection<? super E> c, int maxElements) {
 743         if (c == null)
 744             throw new NullPointerException();
 745         if (c == this)
 746             throw new IllegalArgumentException();
 747         if (maxElements <= 0)
 748             return 0;
 749         final ReentrantLock lock = this.lock;
 750         lock.lock();
 751         try {
 752             int n = Math.min(maxElements, count);
 753             for (int i = 0; i < n; i++) {
 754                 c.add(first.item);   // In this order, in case add() throws.
 755                 unlinkFirst();
 756             }
 757             return n;
 758         } finally {
 759             lock.unlock();
 760         }

1034     /**
1035      * Base class for Iterators for LinkedBlockingDeque
1036      */
1037     private abstract class AbstractItr implements Iterator<E> {
1038         /**
1039          * The next node to return in next()
1040          */
1041         Node<E> next;
1042 
1043         /**
1044          * nextItem holds on to item fields because once we claim that
1045          * an element exists in hasNext(), we must return item read
1046          * under lock (in advance()) even if it was in the process of
1047          * being removed when hasNext() was called.
1048          */
1049         E nextItem;
1050 
1051         /**
1052          * Node returned by most recent call to next. Needed by remove.
1053          * Reset to null if this element is deleted by a call to remove.
1054          */
1055         private Node<E> lastRet;
1056 
1057         abstract Node<E> firstNode();
1058         abstract Node<E> nextNode(Node<E> n);
1059 
1060         AbstractItr() {
1061             // set to initial position
1062             final ReentrantLock lock = LinkedBlockingDeque.this.lock;
1063             lock.lock();
1064             try {
1065                 next = firstNode();
1066                 nextItem = (next == null) ? null : next.item;
1067             } finally {
1068                 lock.unlock();
1069             }
1070         }
1071 
1072         /**
1073          * Returns the successor node of the given non-null, but
1074          * possibly previously deleted, node.
1075          */
1076         private Node<E> succ(Node<E> n) {
1077             // Chains of deleted nodes ending in null or self-links
1078             // are possible if multiple interior nodes are removed.
1079             for (;;) {
1080                 Node<E> s = nextNode(n);
1081                 if (s == null)
1082                     return null;
1083                 else if (s.item != null)
1084                     return s;
1085                 else if (s == n)
1086                     return firstNode();
1087                 else
1088                     n = s;
1089             }
1090         }
1091 
1092         /**
1093          * Advances next.
1094          */
1095         void advance() {
1096             final ReentrantLock lock = LinkedBlockingDeque.this.lock;
1097             lock.lock();
1098             try {
1099                 // assert next != null;
1100                 next = succ(next);
1101                 nextItem = (next == null) ? null : next.item;
1102             } finally {
1103                 lock.unlock();
1104             }
1105         }
1106 
1107         public boolean hasNext() {
1108             return next != null;
1109         }
1110 
1111         public E next() {
1112             if (next == null)
1113                 throw new NoSuchElementException();
1114             lastRet = next;
1115             E x = nextItem;
1116             advance();
1117             return x;
1118         }
1119 
1120         public void remove() {
1121             Node<E> n = lastRet;
1122             if (n == null)
1123                 throw new IllegalStateException();
1124             lastRet = null;
1125             final ReentrantLock lock = LinkedBlockingDeque.this.lock;
1126             lock.lock();
1127             try {
1128                 if (n.item != null)
1129                     unlink(n);
1130             } finally {
1131                 lock.unlock();
1132             }
1133         }
1134     }

1148     /** A customized variant of Spliterators.IteratorSpliterator */
1149     static final class LBDSpliterator<E> implements Spliterator<E> {
1150         static final int MAX_BATCH = 1 << 25;  // max batch array size;
1151         final LinkedBlockingDeque<E> queue;
1152         Node<E> current;    // current node; null until initialized
1153         int batch;          // batch size for splits
1154         boolean exhausted;  // true when no more nodes
1155         long est;           // size estimate
1156         LBDSpliterator(LinkedBlockingDeque<E> queue) {
1157             this.queue = queue;
1158             this.est = queue.size();
1159         }
1160 
1161         public long estimateSize() { return est; }

1163         public Spliterator<E> trySplit() {
1164             Node<E> h;
1165             final LinkedBlockingDeque<E> q = this.queue;
1166             int b = batch;
1167             int n = (b <= 0) ? 1 : (b >= MAX_BATCH) ? MAX_BATCH : b + 1;
1168             if (!exhausted &&
1169                 ((h = current) != null || (h = q.first) != null) &&
1170                 h.next != null) {
1171                 Object[] a = new Object[n];
1172                 final ReentrantLock lock = q.lock;
1173                 int i = 0;
1174                 Node<E> p = current;
1175                 lock.lock();
1176                 try {
1177                     if (p != null || (p = q.first) != null) {
1178                         do {
1179                             if ((a[i] = p.item) != null)
1180                                 ++i;
1181                         } while ((p = p.next) != null && i < n);
1182                     }
1183                 } finally {
1184                     lock.unlock();
1185                 }
1186                 if ((current = p) == null) {
1187                     est = 0L;
1188                     exhausted = true;
1189                 }
1190                 else if ((est -= i) < 0L)
1191                     est = 0L;
1192                 if (i > 0) {
1193                     batch = i;
1194                     return Spliterators.spliterator
1195                         (a, 0, i, Spliterator.ORDERED | Spliterator.NONNULL |
1196                          Spliterator.CONCURRENT);
1197                 }
1198             }
1199             return null;
1200         }

1202         public void forEachRemaining(Consumer<? super E> action) {
1203             if (action == null) throw new NullPointerException();
1204             final LinkedBlockingDeque<E> q = this.queue;
1205             final ReentrantLock lock = q.lock;
1206             if (!exhausted) {
1207                 exhausted = true;
1208                 Node<E> p = current;
1209                 do {
1210                     E e = null;
1211                     lock.lock();
1212                     try {
1213                         if (p == null)
1214                             p = q.first;
1215                         while (p != null) {
1216                             e = p.item;
1217                             p = p.next;
1218                             if (e != null)
1219                                 break;
1220                         }
1221                     } finally {
1222                         lock.unlock();
1223                     }
1224                     if (e != null)
1225                         action.accept(e);
1226                 } while (p != null);
1227             }
1228         }

1230         public boolean tryAdvance(Consumer<? super E> action) {
1231             if (action == null) throw new NullPointerException();
1232             final LinkedBlockingDeque<E> q = this.queue;
1233             final ReentrantLock lock = q.lock;
1234             if (!exhausted) {
1235                 E e = null;
1236                 lock.lock();
1237                 try {
1238                     if (current == null)
1239                         current = q.first;
1240                     while (current != null) {
1241                         e = current.item;
1242                         current = current.next;
1243                         if (e != null)
1244                             break;
1245                     }
1246                 } finally {
1247                     lock.unlock();
1248                 }
1249                 if (current == null)
1250                     exhausted = true;
1251                 if (e != null) {
1252                     action.accept(e);
1253                     return true;
1254                 }
1255             }
1256             return false;
1257         }
