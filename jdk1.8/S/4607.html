  64 /**
  65  * This class represents the field info structure, i.e., the representation
  66  * for a variable in the class. See JVM specification for details.
  67  *
  68  * @author  M. Dahm
  69  */
  70 public final class Field extends FieldOrMethod {
  71   /**
  72    * Initialize from another object. Note that both objects use the same
  73    * references (shallow copy). Use clone() for a physical copy.
  74    */
  75   public Field(Field c) {
  76     super(c);
  77   }

  87   }

  89   /**
  90    * @param access_flags Access rights of field
  91    * @param name_index Points to field name in constant pool
  92    * @param signature_index Points to encoded signature
  93    * @param attributes Collection of attributes
  94    * @param constant_pool Array of constants
  95    */
  96   public Field(int access_flags, int name_index, int signature_index,
  97                Attribute[] attributes, ConstantPool constant_pool)
  98   {
  99     super(access_flags, name_index, signature_index, attributes, constant_pool);
 100   }

 102   /**
 103    * Called by objects that are traversing the nodes of the tree implicitely
 104    * defined by the contents of a Java class. I.e., the hierarchy of methods,
 105    * fields, attributes, etc. spawns a tree of objects.
 106    *
 107    * @param v Visitor object
 108    */
 109   public void accept(Visitor v) {
 110     v.visitField(this);
 111   }
 112 
 113   /**
 114    * @return constant value associated with this field (may be null)
 115    */
 116   public final ConstantValue getConstantValue() {
 117     for(int i=0; i < attributes_count; i++)
 118       if(attributes[i].getTag() == Constants.ATTR_CONSTANT_VALUE)
 119         return (ConstantValue)attributes[i];
 120 
 121     return null;
 122   }
 123 
 124   /**
 125    * Return string representation close to declaration format,
 126    * `public static final short MAX = 100', e.g..
 127    *
 128    * @return String representation of field, including the signature.
 129    */
 130   public final String toString() {
 131     String name, signature, access; // Short cuts to constant pool
 132 
 133     // Get names from constant pool
 134     access    = Utility.accessToString(access_flags);
 135     access    = access.equals("")? "" : (access + " ");
 136     signature = Utility.signatureToString(getSignature());
 137     name      = getName();
 138 
 139     StringBuffer  buf = new StringBuffer(access + signature + " " + name);
 140     ConstantValue cv  = getConstantValue();
 141 
 142     if(cv != null)
 143       buf.append(" = " + cv);
 144 
 145     for(int i=0; i < attributes_count; i++) {
 146       Attribute a = attributes[i];
 147 
 148       if(!(a instanceof ConstantValue))
 149         buf.append(" [" + a.toString() + "]");
 150     }
 151 
 152     return buf.toString();
 153   }
 154 
 155   /**
 156    * @return deep copy of this field
 157    */
 158   public final Field copy(ConstantPool constant_pool) {
 159     return (Field)copy_(constant_pool);
 160   }

 162   /**
 163    * @return type of field
 164    */
 165   public Type getType() {
 166     return Type.getReturnType(getSignature());
 167   }
