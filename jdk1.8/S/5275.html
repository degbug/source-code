  53 /**
  54  * This is the DTD-only parser configuration.  It extends the basic
  55  * configuration with a standard set of parser components appropriate
  56  * to DTD-centric validation. Since
  57  * the Xerces2 reference implementation document and DTD scanner
  58  * implementations are capable of acting as pull parsers, this
  59  * configuration implements the
  60  * <code>XMLPullParserConfiguration</code> interface.
  61  * <p>
  62  * In addition to the features and properties recognized by the base
  63  * parser configuration, this class recognizes these additional
  64  * features and properties:
  65  * <ul>
  66  * <li>Features
  67  *  <ul>
  68  *   <li>http://apache.org/xml/features/validation/warn-on-duplicate-attdef</li>
  69  *   <li>http://apache.org/xml/features/validation/warn-on-undeclared-elemdef</li>
  70  *   <li>http://apache.org/xml/features/allow-java-encodings</li>
  71  *   <li>http://apache.org/xml/features/continue-after-fatal-error</li>
  72  *   <li>http://apache.org/xml/features/load-external-dtd</li>
  73  *  </ul>
  74  * <li>Properties
  75  *  <ul>
  76  *   <li>http://apache.org/xml/properties/internal/error-reporter</li>
  77  *   <li>http://apache.org/xml/properties/internal/entity-manager</li>
  78  *   <li>http://apache.org/xml/properties/internal/document-scanner</li>
  79  *   <li>http://apache.org/xml/properties/internal/dtd-scanner</li>
  80  *   <li>http://apache.org/xml/properties/internal/grammar-pool</li>
  81  *   <li>http://apache.org/xml/properties/internal/validator/dtd</li>
  82  *   <li>http://apache.org/xml/properties/internal/datatype-validator-factory</li>
  83  *  </ul>
  84  * </ul>
  85  *
  86  * @author Arnaud  Le Hors, IBM
  87  * @author Andy Clark, IBM
  88  * @author Neil Graham, IBM
  89  *
  90  * @version $Id: DTDConfiguration.java,v 1.7 2010-11-01 04:40:09 joehw Exp $
  91  */

 485     /**
 486      * Parses the document in a pull parsing fashion.
 487      *
 488      * @param complete True if the pull parser should parse the
 489      *                 remaining document completely.
 490      *
 491      * @return True if there is more document to parse.
 492      *
 493      * @exception XNIException Any XNI exception, possibly wrapping
 494      *                         another exception.
 495      * @exception IOException  An IO exception from the parser, possibly
 496      *                         from a byte stream or character stream
 497      *                         supplied by the parser.
 498      *
 499      * @see #setInputSource
 500      */
 501     public boolean parse(boolean complete) throws XNIException, IOException {
 502         //
 503         // reset and configure pipeline and set InputSource.
 504         if (fInputSource !=null) {
 505             try {
 506                 // resets and sets the pipeline.
 507                 reset();
 508                 fScanner.setInputSource(fInputSource);
 509                 fInputSource = null;
 510             }
 511             catch (XNIException ex) {
 512                 if (PRINT_EXCEPTION_STACK_TRACE)
 513                     ex.printStackTrace();
 514                 throw ex;
 515             }
 516             catch (IOException ex) {
 517                 if (PRINT_EXCEPTION_STACK_TRACE)
 518                     ex.printStackTrace();
 519                 throw ex;
 520             }
 521             catch (RuntimeException ex) {
 522                 if (PRINT_EXCEPTION_STACK_TRACE)
 523                     ex.printStackTrace();
 524                 throw ex;
 525             }
 526             catch (Exception ex) {
 527                 if (PRINT_EXCEPTION_STACK_TRACE)
 528                     ex.printStackTrace();
 529                 throw new XNIException(ex);
 530             }
 531         }
 532 
 533         try {
 534             return fScanner.scanDocument(complete);
 535         }
 536         catch (XNIException ex) {
 537             if (PRINT_EXCEPTION_STACK_TRACE)
 538                 ex.printStackTrace();
 539             throw ex;
 540         }
 541         catch (IOException ex) {
 542             if (PRINT_EXCEPTION_STACK_TRACE)
 543                 ex.printStackTrace();
 544             throw ex;
 545         }
 546         catch (RuntimeException ex) {
 547             if (PRINT_EXCEPTION_STACK_TRACE)
 548                 ex.printStackTrace();
 549             throw ex;
 550         }
 551         catch (Exception ex) {
 552             if (PRINT_EXCEPTION_STACK_TRACE)
 553                 ex.printStackTrace();
 554             throw new XNIException(ex);
 555         }
 556 
 557     } // parse(boolean):boolean

 572     /**
 573      * Parses the specified input source.
 574      *
 575      * @param source The input source.
 576      *
 577      * @exception XNIException Throws exception on XNI error.
 578      * @exception java.io.IOException Throws exception on i/o error.
 579      */
 580     public void parse(XMLInputSource source) throws XNIException, IOException {
 581 
 582         if (fParseInProgress) {
 583             // REVISIT - need to add new error message
 584             throw new XNIException("FWK005 parse may not be called while parsing.");
 585         }
 586         fParseInProgress = true;
 587 
 588         try {
 589             setInputSource(source);
 590             parse(true);
 591         }
 592         catch (XNIException ex) {
 593             if (PRINT_EXCEPTION_STACK_TRACE)
 594                 ex.printStackTrace();
 595             throw ex;
 596         }
 597         catch (IOException ex) {
 598             if (PRINT_EXCEPTION_STACK_TRACE)
 599                 ex.printStackTrace();
 600             throw ex;
 601         }
 602         catch (RuntimeException ex) {
 603             if (PRINT_EXCEPTION_STACK_TRACE)
 604                 ex.printStackTrace();
 605             throw ex;
 606         }
 607         catch (Exception ex) {
 608             if (PRINT_EXCEPTION_STACK_TRACE)
 609                 ex.printStackTrace();
 610             throw new XNIException(ex);
 611         }
 612         finally {
 613             fParseInProgress = false;
 614             // close all streams opened by xerces
 615             this.cleanup();
 616         }
 617

 721      * Check a feature. If feature is know and supported, this method simply

 723      *
 724      * @param featureId The unique identifier (URI) of the feature.
 725      *
 726      * @throws XMLConfigurationException Thrown for configuration error.

 728      *                                   only throw this exception if
 729      *                                   it is <strong>really</strong>
 730      *                                   a critical error.
 731      */
 732     protected FeatureState checkFeature(String featureId)
 733         throws XMLConfigurationException {
 734 
 735         //
 736         // Xerces Features
 737         //
 738 
 739         if (featureId.startsWith(Constants.XERCES_FEATURE_PREFIX)) {
 740             final int suffixLength = featureId.length() - Constants.XERCES_FEATURE_PREFIX.length();
 741 
 742             //
 743             // http://apache.org/xml/features/validation/dynamic
 744             //   Allows the parser to validate a document only when it
 745             //   contains a grammar. Validation is turned on/off based
 746             //   on each document instance, automatically.
 747             //
 748             if (suffixLength == Constants.DYNAMIC_VALIDATION_FEATURE.length() &&
 749                 featureId.endsWith(Constants.DYNAMIC_VALIDATION_FEATURE)) {
 750                 return FeatureState.RECOGNIZED;
 751             }
 752 
 753             //
 754             // http://apache.org/xml/features/validation/default-attribute-values
 755             //
 756             if (suffixLength == Constants.DEFAULT_ATTRIBUTE_VALUES_FEATURE.length() &&
 757                 featureId.endsWith(Constants.DEFAULT_ATTRIBUTE_VALUES_FEATURE)) {
 758                 // REVISIT
 759                 return FeatureState.NOT_SUPPORTED;
 760             }
 761             //
 762             // http://apache.org/xml/features/validation/default-attribute-values
 763             //
 764             if (suffixLength == Constants.VALIDATE_CONTENT_MODELS_FEATURE.length() &&
 765                 featureId.endsWith(Constants.VALIDATE_CONTENT_MODELS_FEATURE)) {
 766                 // REVISIT
 767                 return FeatureState.NOT_SUPPORTED;
 768             }
 769             //
 770             // http://apache.org/xml/features/validation/nonvalidating/load-dtd-grammar
 771             //
 772             if (suffixLength == Constants.LOAD_DTD_GRAMMAR_FEATURE.length() &&
 773                 featureId.endsWith(Constants.LOAD_DTD_GRAMMAR_FEATURE)) {
 774                 return FeatureState.RECOGNIZED;
 775             }
 776             //
 777             // http://apache.org/xml/features/validation/nonvalidating/load-external-dtd
 778             //
 779             if (suffixLength == Constants.LOAD_EXTERNAL_DTD_FEATURE.length() &&
 780                 featureId.endsWith(Constants.LOAD_EXTERNAL_DTD_FEATURE)) {
 781                 return FeatureState.RECOGNIZED;
 782             }
 783 
 784             //
 785             // http://apache.org/xml/features/validation/default-attribute-values
 786             //
 787             if (suffixLength == Constants.VALIDATE_DATATYPES_FEATURE.length() &&
 788                 featureId.endsWith(Constants.VALIDATE_DATATYPES_FEATURE)) {
 789                 return FeatureState.NOT_SUPPORTED;
 790             }
 791         }
 792 
 793         //
 794         // Not recognized
 795         //
 796 
 797         return super.checkFeature(featureId);
