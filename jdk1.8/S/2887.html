  32 /**
  33   * Center-positioning layout manager.
  34   * @author Tom Santos
  35   * @author Steve Wilson
  36   */
  37 class CenterLayout implements LayoutManager, Serializable {
  38     public void addLayoutComponent(String name, Component comp) { }
  39     public void removeLayoutComponent(Component comp) { }
  40 
  41     public Dimension preferredLayoutSize( Container container ) {
  42         Component c = container.getComponent( 0 );
  43         if ( c != null ) {
  44             Dimension size = c.getPreferredSize();
  45             Insets insets = container.getInsets();
  46 
  47             return new Dimension(size.width + insets.left + insets.right,
  48                                  size.height + insets.top + insets.bottom);
  49         }
  50         else {
  51             return new Dimension( 0, 0 );
  52         }
  53     }
  54 
  55     public Dimension minimumLayoutSize(Container cont) {
  56         return preferredLayoutSize(cont);
  57     }
  58 
  59     public void layoutContainer(Container container) {
  60         if (container.getComponentCount() > 0) {
  61             Component c = container.getComponent(0);
  62             Dimension pref = c.getPreferredSize();
  63             int containerWidth = container.getWidth();
  64             int containerHeight = container.getHeight();
  65             Insets containerInsets = container.getInsets();
  66 
  67             containerWidth -= containerInsets.left +
  68                               containerInsets.right;
  69             containerHeight -= containerInsets.top +
  70                                containerInsets.bottom;
  71 
  72             int left = (containerWidth - pref.width) / 2 +
  73                             containerInsets.left;
  74             int right = (containerHeight - pref.height) / 2 +
  75                             containerInsets.top;
  76 
  77             c.setBounds(left, right, pref.width, pref.height);
  78         }
  79     }
  80 }
