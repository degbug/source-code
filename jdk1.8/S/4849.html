  25 import javax.xml.transform.TransformerException;

  27 import org.w3c.dom.Element;
  28 
  29 /**
  30  * A class that implements this interface can tell if a given element should
  31  * strip whitespace nodes from it's children.
  32  */
  33 public interface WhitespaceStrippingElementMatcher
  34 {
  35   /**
  36    * Get information about whether or not an element should strip whitespace.
  37    * @see <a href="http://www.w3.org/TR/xslt#strip">strip in XSLT Specification</a>
  38    *
  39    * @param support The XPath runtime state.
  40    * @param targetElement Element to check
  41    *
  42    * @return true if the whitespace should be stripped.
  43    *
  44    * @throws TransformerException
  45    */
  46   public boolean shouldStripWhiteSpace(
  47           XPathContext support, Element targetElement) throws TransformerException;
  48 
  49   /**
  50    * Get information about whether or not whitespace can be stripped.
  51    * @see <a href="http://www.w3.org/TR/xslt#strip">strip in XSLT Specification</a>
  52    *
  53    * @return true if the whitespace can be stripped.
  54    */
  55   public boolean canStripWhiteSpace();
  56 }
