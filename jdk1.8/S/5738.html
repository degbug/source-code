  75     /**
  76      * Method enginePerformTransform
  77      * @inheritDoc
  78      * @param input
  79      *
  80      * @throws TransformationException
  81      */
  82     protected XMLSignatureInput enginePerformTransform(
  83         XMLSignatureInput input, OutputStream os, Transform transformObject
  84     ) throws TransformationException {
  85         try {
  86             List<NodeList> unionNodes = new ArrayList<NodeList>();
  87             List<NodeList> subtractNodes = new ArrayList<NodeList>();
  88             List<NodeList> intersectNodes = new ArrayList<NodeList>();
  89 
  90             Element[] xpathElements =
  91                 XMLUtils.selectNodes(
  92                     transformObject.getElement().getFirstChild(),
  93                     XPath2FilterContainer.XPathFilter2NS,
  94                     XPath2FilterContainer._TAG_XPATH2
  95                 );
  96             if (xpathElements.length == 0) {
  97                 Object exArgs[] = { Transforms.TRANSFORM_XPATH2FILTER, "XPath" };
  98 
  99                 throw new TransformationException("xml.WrongContent", exArgs);
 100             }
 101 
 102             Document inputDoc = null;
 103             if (input.getSubNode() != null) {
 104                 inputDoc = XMLUtils.getOwnerDocument(input.getSubNode());
 105             } else {
 106                 inputDoc = XMLUtils.getOwnerDocument(input.getNodeSet());
 107             }
 108 
 109             for (int i = 0; i < xpathElements.length; i++) {
 110                 Element xpathElement = xpathElements[i];
 111 
 112                 XPath2FilterContainer xpathContainer =
 113                     XPath2FilterContainer.newInstance(xpathElement, input.getSourceURI());
 114 
 115                 String str =
 116                     XMLUtils.getStrFromNode(xpathContainer.getXPathFilterTextNode());
 117 
 118                 XPathFactory xpathFactory = XPathFactory.newInstance();
 119                 XPathAPI xpathAPIInstance = xpathFactory.newXPathAPI();
 120 
 121                 NodeList subtreeRoots =
 122                     xpathAPIInstance.selectNodeList(
 123                         inputDoc,
 124                         xpathContainer.getXPathFilterTextNode(),
 125                         str,
 126                         xpathContainer.getElement());
 127                 if (xpathContainer.isIntersect()) {
 128                     intersectNodes.add(subtreeRoots);
 129                 } else if (xpathContainer.isSubtract()) {
 130                     subtractNodes.add(subtreeRoots);
 131                 } else if (xpathContainer.isUnion()) {
 132                     unionNodes.add(subtreeRoots);
 133                 }
 134             }
 135 
 136             input.addNodeFilter(
 137                 new XPath2NodeFilter(unionNodes, subtractNodes, intersectNodes)
 138             );
 139             input.setNodeSet(true);
 140             return input;
 141         } catch (TransformerException ex) {
 142             throw new TransformationException("empty", ex);
 143         } catch (DOMException ex) {
 144             throw new TransformationException("empty", ex);
 145         } catch (CanonicalizationException ex) {
 146             throw new TransformationException("empty", ex);
 147         } catch (InvalidCanonicalizerException ex) {
 148             throw new TransformationException("empty", ex);
 149         } catch (XMLSecurityException ex) {
 150             throw new TransformationException("empty", ex);
 151         } catch (SAXException ex) {
 152             throw new TransformationException("empty", ex);
 153         } catch (IOException ex) {
 154             throw new TransformationException("empty", ex);
 155         } catch (ParserConfigurationException ex) {
 156             throw new TransformationException("empty", ex);
 157         }
 158     }

 161 class XPath2NodeFilter implements NodeFilter {
 162 
 163     boolean hasUnionFilter;
 164     boolean hasSubtractFilter;
 165     boolean hasIntersectFilter;
 166     Set<Node> unionNodes;
 167     Set<Node> subtractNodes;
 168     Set<Node> intersectNodes;
 169     int inSubtract = -1;
 170     int inIntersect = -1;
 171     int inUnion = -1;
 172 
 173     XPath2NodeFilter(List<NodeList> unionNodes, List<NodeList> subtractNodes,
 174                      List<NodeList> intersectNodes) {
 175         hasUnionFilter = !unionNodes.isEmpty();
 176         this.unionNodes = convertNodeListToSet(unionNodes);
 177         hasSubtractFilter = !subtractNodes.isEmpty();
 178         this.subtractNodes = convertNodeListToSet(subtractNodes);
 179         hasIntersectFilter = !intersectNodes.isEmpty();
 180         this.intersectNodes = convertNodeListToSet(intersectNodes);
 181     }
 182 
 183     /**
 184      * @see com.sun.org.apache.xml.internal.security.signature.NodeFilter#isNodeInclude(org.w3c.dom.Node)
 185      */
 186     public int isNodeInclude(Node currentNode) {
 187         int result = 1;
 188 
 189         if (hasSubtractFilter && rooted(currentNode, subtractNodes)) {
 190             result = -1;
 191         } else if (hasIntersectFilter && !rooted(currentNode, intersectNodes)) {
 192             result = 0;
 193         }
 194 
 195         //TODO OPTIMIZE
 196         if (result == 1) {
 197             return 1;
 198         }
 199         if (hasUnionFilter) {
 200             if (rooted(currentNode, unionNodes)) {
 201                 return 1;
 202             }
 203             result = 0;
 204         }
 205         return result;
 206     }
 207 
 208     public int isNodeIncludeDO(Node n, int level) {
 209         int result = 1;
 210         if (hasSubtractFilter) {
 211             if ((inSubtract == -1) || (level <= inSubtract)) {
 212                 if (inList(n, subtractNodes)) {
 213                     inSubtract = level;
 214                 } else {
 215                     inSubtract = -1;
 216                 }
 217             }
 218             if (inSubtract != -1){
 219                 result = -1;
 220             }
 221         }
 222         if (result != -1 && hasIntersectFilter
 223             && ((inIntersect == -1) || (level <= inIntersect))) {
 224             if (!inList(n, intersectNodes)) {
 225                 inIntersect = -1;
 226                 result = 0;
 227             } else {
 228                 inIntersect = level;
 229             }
 230         }
 231 
 232         if (level <= inUnion) {
 233             inUnion = -1;
 234         }
 235         if (result == 1) {
 236             return 1;
 237         }
 238         if (hasUnionFilter) {
 239             if ((inUnion == -1) && inList(n, unionNodes)) {
 240                 inUnion = level;
 241             }
 242             if (inUnion != -1) {
 243                 return 1;
 244             }
 245             result=0;
 246         }
 247 
 248         return result;
 249     }

 251     /**
 252      * Method rooted
 253      * @param currentNode
 254      * @param nodeList
 255      *
 256      * @return if rooted bye the rootnodes
 257      */
 258     static boolean rooted(Node currentNode, Set<Node> nodeList) {
 259         if (nodeList.isEmpty()) {
 260             return false;
 261         }
 262         if (nodeList.contains(currentNode)) {
 263             return true;
 264         }
 265         for (Node rootNode : nodeList) {
 266             if (XMLUtils.isDescendantOrSelf(rootNode, currentNode)) {
 267                 return true;
 268             }
 269         }
 270         return false;
 271     }
 272 
 273     /**
 274      * Method rooted
 275      * @param currentNode
 276      * @param nodeList
 277      *
 278      * @return if rooted bye the rootnodes
 279      */
 280     static boolean inList(Node currentNode, Set<Node> nodeList) {
 281         return nodeList.contains(currentNode);
 282     }
 283 
 284     private static Set<Node> convertNodeListToSet(List<NodeList> l) {
 285         Set<Node> result = new HashSet<Node>();
 286         for (NodeList rootNodes : l) {
 287             int length = rootNodes.getLength();
 288 
 289             for (int i = 0; i < length; i++) {
 290                 Node rootNode = rootNodes.item(i);
 291                 result.add(rootNode);
 292             }
 293         }
 294         return result;
 295     }
