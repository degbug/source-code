      Paint the component using the Nimbus Table Header Background Painter

  49     @Override protected void paintComponent(Graphics g) {
  50         Painter painter = (Painter) UIManager.get(
  51             "TableHeader:\"TableHeader.renderer\"[Enabled].backgroundPainter");
  52         if (painter != null){
  53             if (g instanceof Graphics2D){
  54                 painter.paint((Graphics2D)g,this,getWidth()+1,getHeight());
  55             } else {
  56                 // paint using image to not Graphics2D to support
  57                 // Java 1.1 printing API
  58                 BufferedImage img =  new BufferedImage(getWidth(),getHeight(),
  59                         BufferedImage.TYPE_INT_ARGB);
  60                 Graphics2D g2 = (Graphics2D)img.getGraphics();
  61                 painter.paint(g2,this,getWidth()+1,getHeight());
  62                 g2.dispose();
  63                 g.drawImage(img,0,0,null);
  64                 img = null;
  65             }
  66         }
  67     }
  68 }
