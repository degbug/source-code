  26 package java.util;
  27 
  28 /**
  29  * FomattableFlags are passed to the {@link Formattable#formatTo
  30  * Formattable.formatTo()} method and modify the output format for {@linkplain
  31  * Formattable Formattables}.  Implementations of {@link Formattable} are
  32  * responsible for interpreting and validating any flags.
  33  *
  34  * @since  1.5
  35  */
  36 public class FormattableFlags {
  37 
  38     // Explicit instantiation of this class is prohibited.
  39     private FormattableFlags() {}
  40 
  41     /**
  42      * Left-justifies the output.  Spaces ('\u0020') will be added
  43      * at the end of the converted value as required to fill the minimum width
  44      * of the field.  If this flag is not set then the output will be
  45      * right-justified.
  46      *
  47      *  This flag corresponds to '-' ('\u002d') in
  48      * the format specifier.
  49      */
  50     public static final int LEFT_JUSTIFY = 1<<0; // '-'
  51 
  52     /**
  53      * Converts the output to upper case according to the rules of the
  54      * {@linkplain java.util.Locale locale} given during creation of the
  55      * formatter argument of the {@link Formattable#formatTo
  56      * formatTo()} method.  The output should be equivalent the following
  57      * invocation of {@link String#toUpperCase(java.util.Locale)}
  58      *
  59      * 

  60      *     out.toUpperCase() 
  61      *
  62      *  This flag corresponds to 'S' ('\u0053') in
  63      * the format specifier.
  64      */
  65     public static final int UPPERCASE = 1<<1;    // 'S'
  66 
  67     /**
  68      * Requires the output to use an alternate form.  The definition of the
  69      * form is specified by the Formattable.
  70      *
  71      *  This flag corresponds to '#' ('\u0023') in
  72      * the format specifier.
  73      */
  74     public static final int ALTERNATE = 1<<2;    // '#'
  75 }
