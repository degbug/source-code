 133         //populate componentColors array with colors calculated in getExtendedCacheKeys call
 134         componentColors = extendedCacheKeys;
 135         //generate this entire method. Each state/bg/fg/border combo that has

 138             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
 139             case BACKGROUND_DISABLED_PRESSED: paintBackgroundDisabledAndPressed(g); break;
 140             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
