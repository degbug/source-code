  Template class for building up a method. This is done by defining exception
  handlers, adding thrown exceptions, local variables and attributes, whereas
  the `LocalVariableTable' and `LineNumberTable' attributes will be set
  automatically for the code. Use stripAttributes() if you don't like this.
  While generating code it may be necessary to insert NOP operations. You can
  use the `removeNOPs' method to get rid off them.
  The resulting method object can be obtained via the `getMethod()' method.

 366   private static final void sort(LocalVariableGen[] vars, int l, int r) {
 367     int i = l, j = r;
 368     int m = vars[(l + r) / 2].getIndex();
 369     LocalVariableGen h;
 370 
 371     do {
 372       while(vars[i].getIndex() < m) i++;
 373       while(m < vars[j].getIndex()) j--;
 374 
 375       if(i <= j) {
 376         h=vars[i]; vars[i]=vars[j]; vars[j]=h; // Swap elements
 377         i++; j--;
 378       }
 379     } while(i <= j);
 380 
 381     if(l < j) sort(vars, l, j);
 382     if(i < r) sort(vars, i, r);
 383   }

    Computes stack usage of an instruction list by performing control flow analysis.
    @return maximum stack depth used by method

 955   public static int getMaxStack(ConstantPoolGen cp, InstructionList il, CodeExceptionGen[] et) {
 956     BranchStack branchTargets = new BranchStack();
 957 


 Initially, populate the branch stack with the exception
      handlers, because these aren't (necessarily) branched to
      explicitly. in each case, the stack will have depth 1,
      containing the exception object.

 963     for (int i = 0; i < et.length; i++) {
 964       InstructionHandle handler_pc = et[i].getHandlerPC();
 965       if (handler_pc != null)
 966         branchTargets.push(handler_pc, 1);
 967     }
 968 
 969     int               stackDepth = 0, maxStackDepth = 0;
 970     InstructionHandle ih         = il.getStart();
 971 
 972     while(ih != null) {
 973       Instruction instruction = ih.getInstruction();
 974       short opcode = instruction.getOpcode();
 975       int delta = instruction.produceStack(cp) - instruction.consumeStack(cp);
 976 
 977       stackDepth += delta;
 978       if(stackDepth > maxStackDepth)
 979         maxStackDepth = stackDepth;
 980 
 981       // choose the next instruction based on whether current is a branch.
 982       if(instruction instanceof BranchInstruction) {
 983         BranchInstruction branch = (BranchInstruction) instruction;
 984         if(instruction instanceof Select) {
 985           // explore all of the select's targets. the default target is handled below.
 986           Select select = (Select) branch;
 987           InstructionHandle[] targets = select.getTargets();
 988           for (int i = 0; i < targets.length; i++)
 989             branchTargets.push(targets[i], stackDepth);
 990           // nothing to fall through to.
 991           ih = null;
 992         } else if(!(branch instanceof IfInstruction)) {
 993           // if an instruction that comes back to following PC,
 994           // push next instruction, with stack depth reduced by 1.
 995           if(opcode == Constants.JSR || opcode == Constants.JSR_W)
 996             branchTargets.push(ih.getNext(), stackDepth - 1);
 997           ih = null;
 998         }
 999         // for all branches, the target of the branch is pushed on the branch stack.
1000         // conditional branches have a fall through case, selects don't, and
1001         // jsr/jsr_w return to the next instruction.
1002         branchTargets.push(branch.getTarget(), stackDepth);
1003       } else {
1004         // check for instructions that terminate the method.
1005         if(opcode == Constants.ATHROW || opcode == Constants.RET ||
1006            (opcode >= Constants.IRETURN && opcode <= Constants.RETURN))
1007           ih = null;
1008       }
1009       // normal case, go to the next instruction.
1010       if(ih != null)
1011         ih = ih.getNext();
1012       // if we have no more instructions, see if there are any deferred branches to explore.
1013       if(ih == null) {
1014         BranchTarget bt = branchTargets.pop();
1015         if (bt != null) {
1016           ih = bt.target;
1017           stackDepth = bt.stackDepth;
1018         }
1019       }
1020     }
1021 
1022     return maxStackDepth;
1023   }
