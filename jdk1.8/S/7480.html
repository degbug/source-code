  35 /** An ObjectKeyTemplate represents the part of an Object Key

  37  * object reference.  The template is shared between many
  38  * object references.
  39  */

  41 {
  42     public ORBVersion getORBVersion() ;
  43 
  44     /** An ID used to determine how to perform operations on this
  45      * ObjectKeyTemplate.  This id determines how to process requests
  46      * on this object reference, and what object adapter type to use.
  47      */
  48     public int getSubcontractId();
  49 
  50     /** Return the server ID for this template.
  51     * For CORBA 3.0, this should be a String, but it is currently
  52     * an int in the object key template.
  53     */
  54     public int getServerId() ;
  55 
  56     /** Return the ORB ID for this template.
  57     */
  58     public String getORBId() ;
  59 
  60     /** Return the object adapter ID for this template.
  61     */
  62     public ObjectAdapterId getObjectAdapterId() ;
  63 
  64     /** Compute an adapter ID for this template than includes
  65     * all of the template information.
  66     * This value is cached to avoid the expense of recomputing
  67     * it.
  68     */
  69     public byte[] getAdapterId() ;

  71     public void write(ObjectId objectId, OutputStream os);
