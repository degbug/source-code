  27 package java.util.logging;

  29 import java.io.*;
  30 import java.net.*;

  32 /**
  33  * Simple network logging Handler.
  34  * 
  35  * LogRecords are published to a network stream connection.  By default
  36  * the XMLFormatter class is used for formatting.
  37  * 
  38  * Configuration:
  39  * By default each SocketHandler is initialized using the following
  40  * LogManager configuration properties where <handler-name>
  41  * refers to the fully-qualified class name of the handler.
  42  * If properties are not defined
  43  * (or have invalid values) then the specified default values are used.
  44  * 
    
  45  * 
  •    <handler-name>.level
    • 
  46  *        specifies the default level for the Handler
  47  *        (defaults to Level.ALL). 
  48  * 
  •    <handler-name>.filter
    • 
  49  *        specifies the name of a Filter class to use
  50  *        (defaults to no Filter). 
  51  * 
  •    <handler-name>.formatter
    • 
  52  *        specifies the name of a Formatter class to use
  53  *        (defaults to java.util.logging.XMLFormatter). 
  54  * 
  •    <handler-name>.encoding
    • 
  55  *        the name of the character set encoding to use (defaults to
  56  *        the default platform encoding). 
  57  * 
  •    <handler-name>.host
    • 
  58  *        specifies the target host name to connect to (no default). 
  59  * 
  •    <handler-name>.port
    • 
  60  *        specifies the target TCP port to use (no default). 
  61  * 
  62  * 
  63  * For example, the properties for {@code SocketHandler} would be:
  64  * 
      
  65  * 
    •    java.util.logging.SocketHandler.level=INFO 
      • 
  66  * 
    •    java.util.logging.SocketHandler.formatter=java.util.logging.SimpleFormatter 
      • 
  67  * 
  68  * 
  69  * For a custom handler, e.g. com.foo.MyHandler, the properties would be:
  70  * 
        
  71  * 
      •    com.foo.MyHandler.level=INFO 
        • 
  72  * 
      •    com.foo.MyHandler.formatter=java.util.logging.SimpleFormatter 
        • 
  73  * 
  74  * 
  75  * The output IO stream is buffered, but is flushed after each
  76  * LogRecord is written.
  77  *
  78  * @since 1.4
  79  */
  80 
  81 public class SocketHandler extends StreamHandler {
  82     private Socket sock;
  83     private String host;
  84     private int port;
  85 
  86     // Private method to configure a SocketHandler from LogManager
  87     // properties and/or default values as specified in the class
  88     // javadoc.
  89     private void configure() {
  90         LogManager manager = LogManager.getLogManager();
  91         String cname = getClass().getName();
  92 
  93         setLevel(manager.getLevelProperty(cname +".level", Level.ALL));
  94         setFilter(manager.getFilterProperty(cname +".filter", null));
  95         setFormatter(manager.getFormatterProperty(cname +".formatter", new XMLFormatter()));
  96         try {
  97             setEncoding(manager.getStringProperty(cname +".encoding", null));
  98         } catch (Exception ex) {
  99             try {
 100                 setEncoding(null);
 101             } catch (Exception ex2) {
 102                 // doing a setEncoding with null should always work.
 103                 // assert false;
 104             }
 105         }
 106         port = manager.getIntProperty(cname + ".port", 0);
 107         host = manager.getStringProperty(cname + ".host", null);
 108     }

 111     /**
 112      * Create a SocketHandler, using only LogManager properties
 113      * (or their defaults).
 114      * @throws IllegalArgumentException if the host or port are invalid or
 115      *          are not specified as LogManager properties.
 116      * @throws IOException if we are unable to connect to the target
 117      *         host and port.
 118      */
 119     public SocketHandler() throws IOException {
 120         // We are going to use the logging defaults.
 121         sealed = false;
 122         configure();
 123 
 124         try {
 125             connect();
 126         } catch (IOException ix) {
 127             System.err.println("SocketHandler: connect failed to " + host + ":" + port);
 128             throw ix;
 129         }
 130         sealed = true;
 131     }
 132 
 133     /**
 134      * Construct a SocketHandler using a specified host and port.
 135      *
 136      * The SocketHandler is configured based on LogManager
 137      * properties (or their default values) except that the given target host
 138      * and port arguments are used. If the host argument is empty, but not
 139      * null String then the localhost is used.
 140      *
 141      * @param host target host.
 142      * @param port target port.
 143      *
 144      * @throws IllegalArgumentException if the host or port are invalid.
 145      * @throws IOException if we are unable to connect to the target
 146      *         host and port.
 147      */
 148     public SocketHandler(String host, int port) throws IOException {
 149         sealed = false;
 150         configure();
 151         sealed = true;
 152         this.port = port;
 153         this.host = host;
 154         connect();
 155     }
 156 
 157     private void connect() throws IOException {
 158         // Check the arguments are valid.
 159         if (port == 0) {
 160             throw new IllegalArgumentException("Bad port: " + port);
 161         }
 162         if (host == null) {
 163             throw new IllegalArgumentException("Null host name: " + host);
 164         }
 165 
 166         // Try to open a new socket.
 167         sock = new Socket(host, port);
 168         OutputStream out = sock.getOutputStream();
 169         BufferedOutputStream bout = new BufferedOutputStream(out);
 170         setOutputStream(bout);
 171     }
 172 
 173     /**
 174      * Close this output stream.
 175      *
 176      * @exception  SecurityException  if a security manager exists and if
 177      *             the caller does not have LoggingPermission("control").
 178      */
 179     @Override
 180     public synchronized void close() throws SecurityException {
 181         super.close();
 182         if (sock != null) {
 183             try {
 184                 sock.close();
 185             } catch (IOException ix) {
 186                 // drop through.
 187             }
 188         }
 189         sock = null;
 190     }
 191 
 192     /**
 193      * Format and publish a LogRecord.
 194      *
 195      * @param  record  description of the log event. A null record is
 196      *                 silently ignored and is not published
 197      */
 198     @Override
 199     public synchronized void publish(LogRecord record) {
 200         if (!isLoggable(record)) {
 201             return;
 202         }
 203         super.publish(record);
 204         flush();
 205     }
