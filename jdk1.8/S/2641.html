 111     /**
 112      * Update the relevant ProtectionDomains with the Principals
 113      * from the {@code Subject} associated with this
 114      * {@code SubjectDomainCombiner}.
 115      *
 116      *  A new {@code ProtectionDomain} instance is created
 117      * for each {@code ProtectionDomain} in the
 118      * currentDomains array.  Each new {@code ProtectionDomain}
 119      * instance is created using the {@code CodeSource},
 120      * {@code Permission}s and {@code ClassLoader}
 121      * from the corresponding {@code ProtectionDomain} in
 122      * currentDomains, as well as with the Principals from
 123      * the {@code Subject} associated with this
 124      * {@code SubjectDomainCombiner}.
 125      *
 126      *  All of the newly instantiated ProtectionDomains are
 127      * combined into a new array.  The ProtectionDomains from the
 128      * assignedDomains array are appended to this new array,
 129      * and the result is returned.
 130      *
 131      *  Note that optimizations such as the removal of duplicate
 132      * ProtectionDomains may have occurred.
 133      * In addition, caching of ProtectionDomains may be permitted.
 134      *
 135      * 
 136      *
 137      * @param currentDomains the ProtectionDomains associated with the
 138      *          current execution Thread, up to the most recent
 139      *          privileged {@code ProtectionDomain}.
 140      *          The ProtectionDomains are are listed in order of execution,
 141      *          with the most recently executing {@code ProtectionDomain}
 142      *          residing at the beginning of the array. This parameter may
 143      *          be {@code null} if the current execution Thread
 144      *          has no associated ProtectionDomains.
 145      *
 146      * @param assignedDomains the ProtectionDomains inherited from the
 147      *          parent Thread, or the ProtectionDomains from the
 148      *          privileged context, if a call to
 149      *          AccessController.doPrivileged(..., context)
 150      *          had occurred  This parameter may be {@code null}
 151      *          if there were no ProtectionDomains inherited from the
 152      *          parent Thread, or from the privileged context.
 153      *
 154      * @return a new array consisting of the updated ProtectionDomains,
 155      *          or {@code null}.
 156      */
 157     public ProtectionDomain[] combine(ProtectionDomain[] currentDomains,
 158                                 ProtectionDomain[] assignedDomains) {
 159         if (debug != null) {
 160             if (subject == null) {
 161                 debug.println("null subject");
 162             } else {
 163                 final Subject s = subject;
 164                 AccessController.doPrivileged
 165                     (new java.security.PrivilegedAction<Void>() {
 166                     public Void run() {
 167                         debug.println(s.toString());
 168                         return null;
 169                     }
 170                 });
 171             }
 172             printInputDomains(currentDomains, assignedDomains);
 173         }
 174 
 175         if (currentDomains == null || currentDomains.length == 0) {
 176             // No need to optimize assignedDomains because it should
 177             // have been previously optimized (when it was set).
 178 
 179             // Note that we are returning a direct reference
 180             // to the input array - since ACC does not clone
 181             // the arrays when it calls combiner.combine,
 182             // multiple ACC instances may share the same
 183             // array instance in this case
 184 
 185             return assignedDomains;
 186         }
 187 
 188         // optimize currentDomains
 189         //
 190         // No need to optimize assignedDomains because it should
 191         // have been previously optimized (when it was set).
 192 
 193         currentDomains = optimize(currentDomains);
 194         if (debug != null) {
 195             debug.println("after optimize");
 196             printInputDomains(currentDomains, assignedDomains);
 197         }
 198 
 199         if (currentDomains == null && assignedDomains == null) {
 200             return null;
 201         }
 202 
 203         // maintain backwards compatibility for developers who provide
 204         // their own custom javax.security.auth.Policy implementations
 205         if (useJavaxPolicy) {
 206             return combineJavaxPolicy(currentDomains, assignedDomains);
 207         }
 208 
 209         int cLen = (currentDomains == null ? 0 : currentDomains.length);
 210         int aLen = (assignedDomains == null ? 0 : assignedDomains.length);
 211 
 212         // the ProtectionDomains for the new AccessControlContext
 213         // that we will return
 214         ProtectionDomain[] newDomains = new ProtectionDomain[cLen + aLen];
 215 
 216         boolean allNew = true;
 217         synchronized(cachedPDs) {
 218             if (!subject.isReadOnly() &&
 219                 !subject.getPrincipals().equals(principalSet)) {
 220 
 221                 // if the Subject was mutated, clear the PD cache
 222                 Set<Principal> newSet = subject.getPrincipals();
 223                 synchronized(newSet) {
 224                     principalSet = new java.util.HashSet<Principal>(newSet);
 225                 }
 226                 principals = principalSet.toArray
 227                         (new Principal[principalSet.size()]);
 228                 cachedPDs.clear();
 229 
 230                 if (debug != null) {
 231                     debug.println("Subject mutated - clearing cache");
 232                 }
 233             }
 234 
 235             ProtectionDomain subjectPd;
 236             for (int i = 0; i < cLen; i++) {
 237                 ProtectionDomain pd = currentDomains[i];
 238 
 239                 subjectPd = cachedPDs.getValue(pd);
 240 
 241                 if (subjectPd == null) {
 242                     subjectPd = new ProtectionDomain(pd.getCodeSource(),
 243                                                 pd.getPermissions(),
 244                                                 pd.getClassLoader(),
 245                                                 principals);
 246                     cachedPDs.putValue(pd, subjectPd);
 247                 } else {
 248                     allNew = false;
 249                 }
 250                 newDomains[i] = subjectPd;
 251             }
 252         }
 253 
 254         if (debug != null) {
 255             debug.println("updated current: ");
 256             for (int i = 0; i < cLen; i++) {
 257                 debug.println("\tupdated[" + i + "] = " +
 258                                 printDomain(newDomains[i]));
 259             }
 260         }
 261 
 262         // now add on the assigned domains
 263         if (aLen > 0) {
 264             System.arraycopy(assignedDomains, 0, newDomains, cLen, aLen);
 265 
 266             // optimize the result (cached PDs might exist in assignedDomains)
 267             if (!allNew) {
 268                 newDomains = optimize(newDomains);
 269             }
 270         }
 271 
 272         // if aLen == 0 || allNew, no need to further optimize newDomains
 273 
 274         if (debug != null) {
 275             if (newDomains == null || newDomains.length == 0) {
 276                 debug.println("returning null");
 277             } else {
 278                 debug.println("combinedDomains: ");
 279                 for (int i = 0; i < newDomains.length; i++) {
 280                     debug.println("newDomain " + i + ": " +
 281                                   printDomain(newDomains[i]));
 282                 }
 283             }
 284         }
 285 
 286         // return the new ProtectionDomains
 287         if (newDomains == null || newDomains.length == 0) {
 288             return null;
 289         } else {
 290             return newDomains;
 291         }
 292     }

 294     /**
 295      * Use the javax.security.auth.Policy implementation
 296      */
 297     private ProtectionDomain[] combineJavaxPolicy(
 298         ProtectionDomain[] currentDomains,
 299         ProtectionDomain[] assignedDomains) {
 300 
 301         if (!allowCaching) {
 302             java.security.AccessController.doPrivileged
 303                 (new PrivilegedAction<Void>() {
 304                     @SuppressWarnings("deprecation")
 305                     public Void run() {
 306                         // Call refresh only caching is disallowed
 307                         javax.security.auth.Policy.getPolicy().refresh();
 308                         return null;
 309                     }
 310                 });
 311         }
 312 
 313 
 314         int cLen = (currentDomains == null ? 0 : currentDomains.length);
 315         int aLen = (assignedDomains == null ? 0 : assignedDomains.length);
 316 
 317         // the ProtectionDomains for the new AccessControlContext
 318         // that we will return
 319         ProtectionDomain[] newDomains = new ProtectionDomain[cLen + aLen];
 320 
 321         synchronized(cachedPDs) {
 322             if (!subject.isReadOnly() &&
 323                 !subject.getPrincipals().equals(principalSet)) {
 324 
 325                 // if the Subject was mutated, clear the PD cache
 326                 Set<Principal> newSet = subject.getPrincipals();
 327                 synchronized(newSet) {
 328                     principalSet = new java.util.HashSet<Principal>(newSet);
 329                 }
 330                 principals = principalSet.toArray
 331                         (new Principal[principalSet.size()]);
 332                 cachedPDs.clear();
 333 
 334                 if (debug != null) {
 335                     debug.println("Subject mutated - clearing cache");
 336                 }
 337             }
 338 
 339             for (int i = 0; i < cLen; i++) {
 340                 ProtectionDomain pd = currentDomains[i];
 341                 ProtectionDomain subjectPd = cachedPDs.getValue(pd);
 342 
 343                 if (subjectPd == null) {
 344 
 345                     // XXX
 346                     // we must first add the original permissions.
 347                     // that way when we later add the new JAAS permissions,
 348                     // any unresolved JAAS-related permissions will
 349                     // automatically get resolved.
 350 
 351                     // get the original perms
 352                     Permissions perms = new Permissions();
 353                     PermissionCollection coll = pd.getPermissions();
 354                     java.util.Enumeration<Permission> e;
 355                     if (coll != null) {
 356                         synchronized (coll) {
 357                             e = coll.elements();
 358                             while (e.hasMoreElements()) {
 359                                 Permission newPerm =
 360                                         e.nextElement();
 361                                  perms.add(newPerm);
 362                             }
 363                         }
 364                     }
 365 
 366                     // get perms from the policy
 367 
 368                     final java.security.CodeSource finalCs = pd.getCodeSource();
 369                     final Subject finalS = subject;
 370                     PermissionCollection newPerms =
 371                         java.security.AccessController.doPrivileged
 372                         (new PrivilegedAction<PermissionCollection>() {
 373                         @SuppressWarnings("deprecation")
 374                         public PermissionCollection run() {
 375                           return
 376                           javax.security.auth.Policy.getPolicy().getPermissions
 377                                 (finalS, finalCs);
 378                         }
 379                     });
 380 
 381                     // add the newly granted perms,
 382                     // avoiding duplicates
 383                     synchronized (newPerms) {
 384                         e = newPerms.elements();
 385                         while (e.hasMoreElements()) {
 386                             Permission newPerm = e.nextElement();
 387                             if (!perms.implies(newPerm)) {
 388                                 perms.add(newPerm);
 389                                 if (debug != null)
 390                                     debug.println (
 391                                         "Adding perm " + newPerm + "\n");
 392                             }
 393                         }
 394                     }
 395                     subjectPd = new ProtectionDomain
 396                         (finalCs, perms, pd.getClassLoader(), principals);
 397 
 398                     if (allowCaching)
 399                         cachedPDs.putValue(pd, subjectPd);
 400                 }
 401                 newDomains[i] = subjectPd;
 402             }
 403         }
 404 
 405         if (debug != null) {
 406             debug.println("updated current: ");
 407             for (int i = 0; i < cLen; i++) {
 408                 debug.println("\tupdated[" + i + "] = " + newDomains[i]);
 409             }
 410         }
 411 
 412         // now add on the assigned domains
 413         if (aLen > 0) {
 414             System.arraycopy(assignedDomains, 0, newDomains, cLen, aLen);
 415         }
 416 
 417         if (debug != null) {
 418             if (newDomains == null || newDomains.length == 0) {
 419                 debug.println("returning null");
 420             } else {
 421                 debug.println("combinedDomains: ");
 422                 for (int i = 0; i < newDomains.length; i++) {
 423                     debug.println("newDomain " + i + ": " +
 424                         newDomains[i].toString());
 425                 }
 426             }
 427         }
 428 
 429         // return the new ProtectionDomains
 430         if (newDomains == null || newDomains.length == 0) {
 431             return null;
 432         } else {
 433             return newDomains;
 434         }
 435     }

 437     private static ProtectionDomain[] optimize(ProtectionDomain[] domains) {
 438         if (domains == null || domains.length == 0)
 439             return null;
 440 
 441         ProtectionDomain[] optimized = new ProtectionDomain[domains.length];
 442         ProtectionDomain pd;
 443         int num = 0;
 444         for (int i = 0; i < domains.length; i++) {
 445 
 446             // skip domains with AllPermission
 447             // XXX
 448             //
 449             //  if (domains[i].implies(ALL_PERMISSION))
 450             //  continue;
 451 
 452             // skip System Domains
 453             if ((pd = domains[i]) != null) {
 454 
 455                 // remove duplicates
 456                 boolean found = false;
 457                 for (int j = 0; j < num && !found; j++) {
 458                     found = (optimized[j] == pd);
 459                 }
 460                 if (!found) {
 461                     optimized[num++] = pd;
 462                 }
 463             }
 464         }
 465 
 466         // resize the array if necessary
 467         if (num > 0 && num < domains.length) {
 468             ProtectionDomain[] downSize = new ProtectionDomain[num];
 469             System.arraycopy(optimized, 0, downSize, 0, downSize.length);
 470             optimized = downSize;
 471         }
 472 
 473         return ((num == 0 || optimized.length == 0) ? null : optimized);
 474     }

 478             (new PrivilegedAction<String>() {

 484             return Boolean.parseBoolean(s);

 488         return true;

 534     /**
 535      * A HashMap that has weak keys and values.
 536      *
 537      * Key objects in this map are the "current" ProtectionDomain instances
 538      * received via the combine method.  Each "current" PD is mapped to a
 539      * new PD instance that holds both the contents of the "current" PD,
 540      * as well as the principals from the Subject associated with this combiner.
 541      *
 542      * The newly created "principal-based" PD values must be stored as
 543      * WeakReferences since they contain strong references to the
 544      * corresponding key object (the "current" non-principal-based PD),
 545      * which will prevent the key from being GC'd.  Specifically,
 546      * a "principal-based" PD contains strong references to the CodeSource,
 547      * signer certs, PermissionCollection and ClassLoader objects
 548      * in the "current PD".
 549      */
 550     private static class WeakKeyValueMap<K,V> extends
 551                                         WeakHashMap<K,WeakReference<V>> {
 552 
 553         public V getValue(K key) {
 554             WeakReference<V> wr = super.get(key);
 555             if (wr != null) {
 556                 return wr.get();
 557             }
 558             return null;
 559         }
 560 
 561         public V putValue(K key, V value) {
 562             WeakReference<V> wr = super.put(key, new WeakReference<V>(value));
 563             if (wr != null) {
 564                 return wr.get();
 565             }
 566             return null;
 567         }
 568     }
