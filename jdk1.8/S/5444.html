  30 /**
  31  * This class represents a Document Type <em>declaraction</em> in
  32  * the document itself, <em>not</em> a Document Type Definition (DTD).

 232         /**
 233           * DOM Level 3 WD- Experimental.
 234           * Override inherited behavior from ParentNodeImpl to support deep equal.
 235           */
 236     public boolean isEqualNode(Node arg) {
 237 
 238         if (!super.isEqualNode(arg)) {
 239             return false;
 240         }
 241 
 242         if (needsSyncData()) {
 243             synchronizeData();
 244         }
 245         DocumentTypeImpl argDocType = (DocumentTypeImpl) arg;
 246 
 247         //test if the following string attributes are equal: publicId,
 248         //systemId, internalSubset.
 249         if ((getPublicId() == null && argDocType.getPublicId() != null)
 250             || (getPublicId() != null && argDocType.getPublicId() == null)
 251             || (getSystemId() == null && argDocType.getSystemId() != null)
 252             || (getSystemId() != null && argDocType.getSystemId() == null)
 253             || (getInternalSubset() == null
 254                 && argDocType.getInternalSubset() != null)
 255             || (getInternalSubset() != null
 256                 && argDocType.getInternalSubset() == null)) {
 257             return false;
 258         }
 259 
 260         if (getPublicId() != null) {
 261             if (!getPublicId().equals(argDocType.getPublicId())) {
 262                 return false;
 263             }
 264         }
 265 
 266         if (getSystemId() != null) {
 267             if (!getSystemId().equals(argDocType.getSystemId())) {
 268                 return false;
 269             }
 270         }
 271 
 272         if (getInternalSubset() != null) {
 273             if (!getInternalSubset().equals(argDocType.getInternalSubset())) {
 274                 return false;
 275             }
 276         }
 277 
 278         //test if NamedNodeMaps entities and notations are equal
 279         NamedNodeMapImpl argEntities = argDocType.entities;
 280 
 281         if ((entities == null && argEntities != null)
 282             || (entities != null && argEntities == null))
 283             return false;
 284 
 285         if (entities != null && argEntities != null) {
 286             if (entities.getLength() != argEntities.getLength())
 287                 return false;
 288 
 289             for (int index = 0; entities.item(index) != null; index++) {
 290                 Node entNode1 = entities.item(index);
 291                 Node entNode2 =
 292                     argEntities.getNamedItem(entNode1.getNodeName());
 293 
 294                 if (!((NodeImpl) entNode1).isEqualNode((NodeImpl) entNode2))
 295                     return false;
 296             }
 297         }
 298 
 299         NamedNodeMapImpl argNotations = argDocType.notations;
 300 
 301         if ((notations == null && argNotations != null)
 302             || (notations != null && argNotations == null))
 303             return false;
 304 
 305         if (notations != null && argNotations != null) {
 306             if (notations.getLength() != argNotations.getLength())
 307                 return false;
 308 
 309             for (int index = 0; notations.item(index) != null; index++) {
 310                 Node noteNode1 = notations.item(index);
 311                 Node noteNode2 =
 312                     argNotations.getNamedItem(noteNode1.getNodeName());
 313 
 314                 if (!((NodeImpl) noteNode1).isEqualNode((NodeImpl) noteNode2))
 315                     return false;
 316             }
 317         }
 318 
 319         return true;
 320     } //end isEqualNode

 334     /** NON-DOM
 335         Get the number associated with this doctype.
 336     */
 337     protected int getNodeNumber() {
 338          // If the doctype has a document owner, get the node number
 339          // relative to the owner doc
 340          if (getOwnerDocument()!=null)
 341             return super.getNodeNumber();
 342 
 343          // The doctype is disconnected and not associated with any document.
 344          // Assign the doctype a number relative to the implementation.
 345          if (doctypeNumber==0) {
 346 
 347             CoreDOMImplementationImpl cd = (CoreDOMImplementationImpl)CoreDOMImplementationImpl.getDOMImplementation();
 348             doctypeNumber = cd.assignDocTypeNumber();
 349          }
 350          return doctypeNumber;
 351     }

 370     /**
 371      * Access the collection of general Entities, both external and
 372      * internal, defined in the DTD. For example, in:
 373      * <p>
 374      * <pre>
 375      *   &lt;!doctype example SYSTEM "ex.dtd" [
 376      *     &lt;!ENTITY foo "foo"&gt;
 377      *     &lt;!ENTITY bar "bar"&gt;
 378      *     &lt;!ENTITY % baz "baz"&gt;
 379      *     ]&gt;
 380      * </pre>
 381      * <p>
 382      * The Entities map includes foo and bar, but not baz. It is promised that
 383      * only Nodes which are Entities will exist in this NamedNodeMap.
 384      * <p>
 385      * For HTML, this will always be null.
 386      * <p>
 387      * Note that "built in" entities such as &amp; and &lt; should be
 388      * converted to their actual characters before being placed in the DOM's
 389      * contained text, and should be converted back when the DOM is rendered
 390      * as XML or HTML, and hence DO NOT appear here.
 391      */
 392     public NamedNodeMap getEntities() {
 393         if (needsSyncChildren()) {
 394             synchronizeChildren();
 395             }
 396         return entities;
 397     }

 445     public Object setUserData(String key,
 446     Object data, UserDataHandler handler) {
 447         if(userData == null)
 448             userData = new Hashtable();
 449         if (data == null) {
 450             if (userData != null) {
 451                 Object o = userData.remove(key);
 452                 if (o != null) {
 453                     UserDataRecord r = (UserDataRecord) o;
 454                     return r.fData;
 455                 }
 456             }
 457             return null;
 458         }
 459         else {
 460             Object o = userData.put(key, new UserDataRecord(data, handler));
 461             if (o != null) {
 462                 UserDataRecord r = (UserDataRecord) o;
 463                 return r.fData;
 464             }
 465         }
 466         return null;
 467     }
 468 
 469     public Object getUserData(String key) {
 470         if (userData == null) {
 471             return null;
 472         }
 473         Object o = userData.get(key);
 474         if (o != null) {
 475             UserDataRecord r = (UserDataRecord) o;
 476             return r.fData;
 477         }
 478         return null;
 479     }
