  25 /**
  26  * Administrative class to keep track of the version number of
  27  * the Serializer release.
  28  * 
This class implements the upcoming standard of having

  29  * org.apache.project-name.Version.getVersion() be a standard way
  30  * to get version information.

  31  * @xsl.usage general
  32  */
  33 public final class Version

  36   /**
  37    * Get the basic version string for the current Serializer.
  38    * Version String formatted like
  39    * "Serializer Java v.r[.dd| Dnn]".
  40    *
  41    * Futurework: have this read version info from jar manifest.
  42    *
  43    * @return String denoting our current version
  44    */
  45   public static String getVersion()
  46   {
  47      return getProduct()+" "+getImplementationLanguage()+" "
  48            +getMajorVersionNum()+"."+getReleaseVersionNum()+"."
  49            +( (getDevelopmentVersionNum() > 0) ?
  50                ("D"+getDevelopmentVersionNum()) : (""+getMaintenanceVersionNum()));
  51   }
  52 
  53   /**

  55    *
  56    * @param argv command line arguments, unused.
  57    */
  58   public static void _main(String argv[])
  59   {
  60     System.out.println(getVersion());
  61   }
