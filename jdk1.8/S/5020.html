      @throws XMLStreamException
      @return

 549     public int next() throws XMLStreamException {
 550         if( !hasNext() ) {
 551             if (fEventType != -1) {
 552                 throw new java.util.NoSuchElementException( "END_DOCUMENT reached: no more elements on the stream." );
 553             } else {
 554                 throw new XMLStreamException( "Error processing input source. The input stream is not complete." );
 555             }
 556         }
 557         try {
 558             fEventType = fScanner.next();
 559 
 560             if (versionStr == null) {
 561                 versionStr = getVersion();
 562             }
 563 
 564             if (fEventType == XMLStreamConstants.START_DOCUMENT
 565                     && versionStr != null
 566                     && versionStr.equals("1.1")) {
 567                 switchToXML11Scanner();
 568             }
 569 
 570             return fEventType;
 571         } catch (IOException ex) {
 572             // if this error occured trying to resolve the external DTD subset
 573             // and IS_VALIDATING == false, then this is not an XML error
 574             if (fScanner.fScannerState == fScanner.SCANNER_STATE_DTD_EXTERNAL) {
 575                 Boolean isValidating = (Boolean) fPropertyManager.getProperty(
 576                         XMLInputFactory.IS_VALIDATING);
 577                 if (isValidating != null
 578                         && !isValidating.booleanValue()) {
 579                     // ignore the error, set scanner to known state
 580                     fEventType = XMLEvent.DTD;
 581                     fScanner.setScannerState(fScanner.SCANNER_STATE_PROLOG);
 582                     fScanner.setDriver(fScanner.fPrologDriver);
 583                     if (fDTDDecl == null
 584                             || fDTDDecl.length() == 0) {
 585                         fDTDDecl = "<!-- "
 586                                 + "Exception scanning External DTD Subset.  "
 587                                 + "True contents of DTD cannot be determined.  "
 588                                 + "Processing will continue as XMLInputFactory.IS_VALIDATING == false."
 589                                 + " -->";
 590                     }
 591                     return XMLEvent.DTD;
 592                 }
 593             }
 594 
 595             // else real error
 596             throw new XMLStreamException(ex.getMessage(), getLocation(), ex);
 597         } catch (XNIException ex) {
 598             throw new XMLStreamException(
 599                     ex.getMessage(),
 600                     getLocation(),
 601                     ex.getException());
 602         }
 603     } //next()

 605     private void switchToXML11Scanner() throws IOException{
 606 
 607         int oldEntityDepth = fScanner.fEntityDepth;
 608         com.sun.org.apache.xerces.internal.xni.NamespaceContext oldNamespaceContext = fScanner.fNamespaceContext;
 609 
 610         fScanner = new XML11NSDocumentScannerImpl();
 611 
 612         //get the new scanner state to old scanner's previous state
 613         fScanner.reset(fPropertyManager);
 614         fScanner.setPropertyManager(fPropertyManager);
 615         fEntityScanner = (XMLEntityScanner)fEntityManager.getEntityScanner()  ;
 616         fEntityManager.fCurrentEntity.mayReadChunks = true;
 617         fScanner.setScannerState(XMLEvent.START_DOCUMENT);
 618 
 619         fScanner.fEntityDepth = oldEntityDepth;
 620         fScanner.fNamespaceContext = oldNamespaceContext;
 621         fEventType = fScanner.next();
 622     }

 Gets the the text associated with a CHARACTERS, SPACE or CDATA event.
      Text starting a "sourceStart" is copied into "destination" starting at "targetStart".
      Up to "length" characters are copied.  The number of characters actually copied is returned.
      The "sourceStart" argument must be greater or equal to 0 and less than or equal to
      the number of characters associated with the event.  Usually, one requests text starting at a "sourceStart" of 0.
      If the number of characters actually copied is less than the "length", then there is no more text.
      Otherwise, subsequent calls need to be made until all text has been retrieved. For example:
      
      int length = 1024;
      char[] myBuffer = new char[ length ];
      for ( int sourceStart = 0 ; ; sourceStart += length )
      {
         int nCopied = stream.getTextCharacters( sourceStart, myBuffer, 0, length );
        if (nCopied < length)
            break;
      }

      The "targetStart" argument must be greater than or equal to 0 and less than the length of "target",
      Length must be greater than 0 and "targetStart + length" must be less than or equal to length of "target".

      @throws IndexOutOfBoundsException if targetStart < 0 or > than the length of target

      @throws NullPointerException is if target is null

1146             (targetStart + length ) > target.length) {
