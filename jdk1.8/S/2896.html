  31 import java.awt.Container;

  34 /**
  35  * The default layout manager for Popup menus and menubars.  This
  36  * class is an extension of BoxLayout which adds the UIResource tag
  37  * so that pluggable L&Fs can distinguish it from user-installed
  38  * layout managers on menus.
  39  *
  40  * @author Georges Saab
  41  */
  42 
  43 public class DefaultMenuLayout extends BoxLayout implements UIResource {
  44     public DefaultMenuLayout(Container target, int axis) {
  45         super(target, axis);
  46     }
  47 
  48     public Dimension preferredLayoutSize(Container target) {
  49         if (target instanceof JPopupMenu) {
  50             JPopupMenu popupMenu = (JPopupMenu) target;
  51             sun.swing.MenuItemLayoutHelper.clearUsedClientProperties(popupMenu);
  52             if (popupMenu.getComponentCount() == 0) {
  53                 return new Dimension(0, 0);
  54             }
  55         }
  56 
  57         // Make BoxLayout recalculate cached preferred sizes
  58         super.invalidateLayout(target);
  59 
  60         return super.preferredLayoutSize(target);
  61     }
  62 }
