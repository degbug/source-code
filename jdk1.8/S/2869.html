      @throws NullPointerException {@inheritDoc}
      @throws IllegalArgumentException {@inheritDoc}
      @see javax.swing.JComponent#getBaseline(int, int)
      @since 1.6

 364     public int getBaseline(JComponent c, int width, int height) {
 365         if (c == null) {
 366             throw new NullPointerException("Component must be non-null");
 367         }
 368 
 369         if (width < 0 || height < 0) {
 370             throw new IllegalArgumentException("Width and height must be >= 0");
 371         }
 372 
 373         JViewport viewport = scrollpane.getViewport();
 374         Insets spInsets = scrollpane.getInsets();
 375         int y = spInsets.top;
 376         height = height - spInsets.top - spInsets.bottom;
 377         width = width - spInsets.left - spInsets.right;
 378         JViewport columnHeader = scrollpane.getColumnHeader();
 379         if (columnHeader != null && columnHeader.isVisible()) {
 380             Component header = columnHeader.getView();
 381             if (header != null && header.isVisible()) {
 382                 // Header is always given it's preferred size.
 383                 Dimension headerPref = header.getPreferredSize();
 384                 int baseline = header.getBaseline(headerPref.width,
 385                                                   headerPref.height);
 386                 if (baseline >= 0) {
 387                     return y + baseline;
 388                 }
 389             }
 390             Dimension columnPref = columnHeader.getPreferredSize();
 391             height -= columnPref.height;
 392             y += columnPref.height;
 393         }
 394         Component view = (viewport == null) ? null : viewport.getView();
 395         if (view != null && view.isVisible() &&
 396                 view.getBaselineResizeBehavior() ==
 397                 Component.BaselineResizeBehavior.CONSTANT_ASCENT) {
 398             Border viewportBorder = scrollpane.getViewportBorder();
 399             if (viewportBorder != null) {
 400                 Insets vpbInsets = viewportBorder.getBorderInsets(scrollpane);
 401                 y += vpbInsets.top;
 402                 height = height - vpbInsets.top - vpbInsets.bottom;
 403                 width = width - vpbInsets.left - vpbInsets.right;
 404             }
 405             if (view.getWidth() > 0 && view.getHeight() > 0) {
 406                 Dimension min = view.getMinimumSize();
 407                 width = Math.max(min.width, view.getWidth());
 408                 height = Math.max(min.height, view.getHeight());
 409             }
 410             if (width > 0 && height > 0) {
 411                 int baseline = view.getBaseline(width, height);
 412                 if (baseline > 0) {
 413                     return y + baseline;
 414                 }
 415             }
 416         }
 417         return -1;
 418     }

 574     protected void updateViewport(PropertyChangeEvent e)
 575     {
 576         JViewport oldViewport = (JViewport)(e.getOldValue());
 577         JViewport newViewport = (JViewport)(e.getNewValue());
 578 
 579         if (oldViewport != null) {
 580             oldViewport.removeChangeListener(viewportChangeListener);
 581         }
 582 
 583         if (newViewport != null) {
 584             Point p = newViewport.getViewPosition();
 585             if (scrollpane.getComponentOrientation().isLeftToRight()) {
 586                 p.x = Math.max(p.x, 0);
 587             } else {
 588                 int max = newViewport.getViewSize().width;
 589                 int extent = newViewport.getExtentSize().width;
 590                 if (extent > max) {
 591                     p.x = max - extent;
 592                 } else {
 593                     p.x = Math.max(0, Math.min(max - extent, p.x));
 594                 }
 595             }
 596             p.y = Math.max(p.y, 0);
 597             newViewport.setViewPosition(p);
 598             newViewport.addChangeListener(viewportChangeListener);
 599         }
 600     }

 797         private void scroll(JScrollPane scrollpane, int orientation,
 798                             int direction, boolean block) {
 799             JViewport vp = scrollpane.getViewport();
 800             Component view;
 801             if (vp != null && (view = vp.getView()) != null) {
 802                 Rectangle visRect = vp.getViewRect();
 803                 Dimension vSize = view.getSize();
 804                 int amount;
 805 
 806                 if (view instanceof Scrollable) {
 807                     if (block) {
 808                         amount = ((Scrollable)view).getScrollableBlockIncrement
 809                                  (visRect, orientation, direction);
 810                     }
 811                     else {
 812                         amount = ((Scrollable)view).getScrollableUnitIncrement
 813                                  (visRect, orientation, direction);
 814                     }
 815                 }
 816                 else {
 817                     if (block) {
 818                         if (orientation == SwingConstants.VERTICAL) {
 819                             amount = visRect.height;
 820                         }
 821                         else {
 822                             amount = visRect.width;
 823                         }
 824                     }
 825                     else {
 826                         amount = 10;
 827                     }
 828                 }
 829                 if (orientation == SwingConstants.VERTICAL) {
 830                     visRect.y += (amount * direction);
 831                     if ((visRect.y + visRect.height) > vSize.height) {
 832                         visRect.y = Math.max(0, vSize.height - visRect.height);
 833                     }
 834                     else if (visRect.y < 0) {
 835                         visRect.y = 0;
 836                     }
 837                 }
 838                 else {
 839                     if (scrollpane.getComponentOrientation().isLeftToRight()) {
 840                         visRect.x += (amount * direction);
 841                         if ((visRect.x + visRect.width) > vSize.width) {
 842                             visRect.x = Math.max(0, vSize.width - visRect.width);
 843                         } else if (visRect.x < 0) {
 844                             visRect.x = 0;
 845                         }
 846                     } else {
 847                         visRect.x -= (amount * direction);
 848                         if (visRect.width > vSize.width) {
 849                             visRect.x = vSize.width - visRect.width;
 850                         } else {
 851                             visRect.x = Math.max(0, Math.min(vSize.width - visRect.width, visRect.x));
 852                         }
 853                     }
 854                 }
 855                 vp.setViewPosition(visRect.getLocation());
 856             }
 857         }

 865         public void mouseWheelMoved(MouseWheelEvent e) {
 866             if (scrollpane.isWheelScrollingEnabled() &&
 867                 e.getWheelRotation() != 0) {
 868                 JScrollBar toScroll = scrollpane.getVerticalScrollBar();
 869                 int direction = e.getWheelRotation() < 0 ? -1 : 1;
 870                 int orientation = SwingConstants.VERTICAL;
 871 
 872                 // find which scrollbar to scroll, or return if none
 873                 if (toScroll == null || !toScroll.isVisible()) {
 874                     toScroll = scrollpane.getHorizontalScrollBar();
 875                     if (toScroll == null || !toScroll.isVisible()) {
 876                         return;
 877                     }
 878                     orientation = SwingConstants.HORIZONTAL;
 879                 } else if(e.isShiftDown()){
 880                     JScrollBar hScroll = scrollpane.getHorizontalScrollBar();
 881                     if (hScroll != null && hScroll.isVisible()) {
 882                         toScroll = hScroll;
 883                         orientation = SwingConstants.HORIZONTAL;
 884                     }
 885                 }
 886 
 887                 e.consume();
 888 
 889                 if (e.getScrollType() == MouseWheelEvent.WHEEL_UNIT_SCROLL) {
 890                     JViewport vp = scrollpane.getViewport();
 891                     if (vp == null) { return; }
 892                     Component comp = vp.getView();
 893                     int units = Math.abs(e.getUnitsToScroll());
 894 
 895                     // When the scrolling speed is set to maximum, it's possible
 896                     // for a single wheel click to scroll by more units than
 897                     // will fit in the visible area.  This makes it
 898                     // hard/impossible to get to certain parts of the scrolling
 899                     // Component with the wheel.  To make for more accurate
 900                     // low-speed scrolling, we limit scrolling to the block
 901                     // increment if the wheel was only rotated one click.
 902                     boolean limitScroll = Math.abs(e.getWheelRotation()) == 1;
 903 
 904                     // Check if we should use the visibleRect trick
 905                     Object fastWheelScroll = toScroll.getClientProperty(
 906                                                "JScrollBar.fastWheelScrolling");
 907                     if (Boolean.TRUE == fastWheelScroll &&
 908                         comp instanceof Scrollable) {
 909                         // 5078454: Under maximum acceleration, we may scroll
 910                         // by many 100s of units in ~1 second.
 911                         //
 912                         // BasicScrollBarUI.scrollByUnits() can bog down the EDT
 913                         // with repaints in this situation.  However, the
 914                         // Scrollable interface allows us to pass in an
 915                         // arbitrary visibleRect.  This allows us to accurately
 916                         // calculate the total scroll amount, and then update
 917                         // the GUI once.  This technique provides much faster
 918                         // accelerated wheel scrolling.
 919                         Scrollable scrollComp = (Scrollable) comp;
 920                         Rectangle viewRect = vp.getViewRect();
 921                         int startingX = viewRect.x;
 922                         boolean leftToRight =
 923                                  comp.getComponentOrientation().isLeftToRight();
 924                         int scrollMin = toScroll.getMinimum();
 925                         int scrollMax = toScroll.getMaximum() -
 926                                         toScroll.getModel().getExtent();
 927 
 928                         if (limitScroll) {
 929                             int blockIncr =
 930                                 scrollComp.getScrollableBlockIncrement(viewRect,
 931                                                                     orientation,
 932                                                                     direction);
 933                             if (direction < 0) {
 934                                 scrollMin = Math.max(scrollMin,
 935                                                toScroll.getValue() - blockIncr);
 936                             }
 937                             else {
 938                                 scrollMax = Math.min(scrollMax,
 939                                                toScroll.getValue() + blockIncr);
 940                             }
 941                         }
 942 
 943                         for (int i = 0; i < units; i++) {
 944                             int unitIncr =
 945                                 scrollComp.getScrollableUnitIncrement(viewRect,
 946                                                         orientation, direction);
 947                             // Modify the visible rect for the next unit, and
 948                             // check to see if we're at the end already.
 949                             if (orientation == SwingConstants.VERTICAL) {
 950                                 if (direction < 0) {
 951                                     viewRect.y -= unitIncr;
 952                                     if (viewRect.y <= scrollMin) {
 953                                         viewRect.y = scrollMin;
 954                                         break;
 955                                     }
 956                                 }
 957                                 else { // (direction > 0
 958                                     viewRect.y += unitIncr;
 959                                     if (viewRect.y >= scrollMax) {
 960                                         viewRect.y = scrollMax;
 961                                         break;
 962                                     }
 963                                 }
 964                             }
 965                             else {
 966                                 // Scroll left
 967                                 if ((leftToRight && direction < 0) ||
 968                                     (!leftToRight && direction > 0)) {
 969                                     viewRect.x -= unitIncr;
 970                                     if (leftToRight) {
 971                                         if (viewRect.x < scrollMin) {
 972                                             viewRect.x = scrollMin;
 973                                             break;
 974                                         }
 975                                     }
 976                                 }
 977                                 // Scroll right
 978                                 else if ((leftToRight && direction > 0) ||
 979                                     (!leftToRight && direction < 0)) {
 980                                     viewRect.x += unitIncr;
 981                                     if (leftToRight) {
 982                                         if (viewRect.x > scrollMax) {
 983                                             viewRect.x = scrollMax;
 984                                             break;
 985                                         }
 986                                     }
 987                                 }
 988                                 else {
 989                                     assert false : "Non-sensical ComponentOrientation / scroll direction";
 990                                 }
 991                             }
 992                         }
 993                         // Set the final view position on the ScrollBar
 994                         if (orientation == SwingConstants.VERTICAL) {
 995                             toScroll.setValue(viewRect.y);
 996                         }
 997                         else {
 998                             if (leftToRight) {
 999                                 toScroll.setValue(viewRect.x);
1000                             }
1001                             else {
1002                                 // rightToLeft scrollbars are oriented with
1003                                 // minValue on the right and maxValue on the
1004                                 // left.
1005                                 int newPos = toScroll.getValue() -
1006                                                        (viewRect.x - startingX);
1007                                 if (newPos < scrollMin) {
1008                                     newPos = scrollMin;
1009                                 }
1010                                 else if (newPos > scrollMax) {
1011                                     newPos = scrollMax;
1012                                 }
1013                                 toScroll.setValue(newPos);
1014                             }
1015                         }
1016                     }
1017                     else {
1018                         // Viewport's view is not a Scrollable, or fast wheel
1019                         // scrolling is not enabled.
1020                         BasicScrollBarUI.scrollByUnits(toScroll, direction,
1021                                                        units, limitScroll);
1022                     }
1023                 }
1024                 else if (e.getScrollType() ==
1025                          MouseWheelEvent.WHEEL_BLOCK_SCROLL) {
1026                     BasicScrollBarUI.scrollByBlock(toScroll, direction);
1027                 }
1028             }
1029         }
