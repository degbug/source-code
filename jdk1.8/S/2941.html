 211     public void propertyChange(PropertyChangeEvent evt) {
 212         if (evt.getSource() instanceof JInternalFrame.JDesktopIcon) {
 213             if (SynthLookAndFeel.shouldUpdateStyle(evt)) {
 214                 updateStyle((JInternalFrame.JDesktopIcon)evt.getSource());
 215             }
 216         } else if (evt.getSource() instanceof JInternalFrame) {
 217             JInternalFrame frame = (JInternalFrame)evt.getSource();
 218             if (iconPane instanceof JToggleButton) {
 219                 JToggleButton button = (JToggleButton)iconPane;
 220                 String prop = evt.getPropertyName();
 221                 if (prop == "title") {
 222                     button.setText((String)evt.getNewValue());
 223                 } else if (prop == "frameIcon") {
 224                     button.setIcon((Icon)evt.getNewValue());
 225                 } else if (prop == JInternalFrame.IS_ICON_PROPERTY ||
 226                            prop == JInternalFrame.IS_SELECTED_PROPERTY) {
 227                     button.setSelected(!frame.isIcon() && frame.isSelected());
 228                 }
 229             }
 230         }
 231     }

 233     private final class Handler implements ActionListener {
 234         public void actionPerformed(ActionEvent evt) {
 235             if (evt.getSource() instanceof JToggleButton) {
 236                 // Either iconify the frame or deiconify and activate it.
 237                 JToggleButton button = (JToggleButton)evt.getSource();
 238                 try {
 239                     boolean selected = button.isSelected();
 240                     if (!selected && !frame.isIconifiable()) {
 241                         button.setSelected(true);
 242                     } else {
 243                         frame.setIcon(!selected);
 244                         if (selected) {
 245                             frame.setSelected(true);
 246                         }
 247                     }
 248                 } catch (PropertyVetoException e2) {
 249                 }
 250             }
 251         }
 252     }
