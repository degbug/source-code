  47 
  48     /**

  50      * detail message and nested error.
  51      *
  52      * @param s the detail message.
  53      * @param err the nested error.  An instance of this class can be
  54      * constructed where this parameter is null, but the standard
  55      * connectors will never do so.
  56      */
  57     public JMXServerErrorException(String s, Error err) {
  58         super(s);
  59         cause = err;
  60     }
  61
