 469      * Decodes as many bytes as possible from the given input buffer,
 470      * writing the results to the given output buffer.
 471      *
 472      * <p> The buffers are read from, and written to, starting at their current
 473      * positions.  At most {@link Buffer#remaining in.remaining()} bytes
 474      * will be read and at most {@link Buffer#remaining out.remaining()}

 476      * reflect the bytes read and the characters written, but their marks and
 477      * limits will not be modified.
 478      *
 479      * <p> In addition to reading bytes from the input buffer and writing
 480      * characters to the output buffer, this method returns a {@link CoderResult}
 481      * object to describe its reason for termination:
 482      *
 483      * <ul>
 484      *
 485      *   <li><p> {@link CoderResult#UNDERFLOW} indicates that as much of the
 486      *   input buffer as possible has been decoded.  If there is no further
 487      *   input then the invoker can proceed to the next step of the
 488      *   <a href="#steps">decoding operation</a>.  Otherwise this method
 489      *   should be invoked again with further input.  </p></li>
 490      *
 491      *   <li><p> {@link CoderResult#OVERFLOW} indicates that there is
 492      *   insufficient space in the output buffer to decode any more bytes.
 493      *   This method should be invoked again with an output buffer that has
 494      *   more {@linkplain Buffer#remaining remaining} characters. This is
 495      *   typically done by draining any decoded characters from the output
 496      *   buffer.  </p></li>
 497      *
 498      *   <li><p> A {@linkplain CoderResult#malformedForLength
 499      *   malformed-input} result indicates that a malformed-input
 500      *   error has been detected.  The malformed bytes begin at the input
 501      *   buffer's (possibly incremented) position; the number of malformed
 502      *   bytes may be determined by invoking the result object's {@link
 503      *   CoderResult#length() length} method.  This case applies only if the
 504      *   {@linkplain #onMalformedInput malformed action} of this decoder
 505      *   is {@link CodingErrorAction#REPORT}; otherwise the malformed input
 506      *   will be ignored or replaced, as requested.  </p></li>
 507      *
 508      *   <li><p> An {@linkplain CoderResult#unmappableForLength
 509      *   unmappable-character} result indicates that an
 510      *   unmappable-character error has been detected.  The bytes that
 511      *   decode the unmappable character begin at the input buffer's (possibly
 512      *   incremented) position; the number of such bytes may be determined
 513      *   by invoking the result object's {@link CoderResult#length() length}
 514      *   method.  This case applies only if the {@linkplain #onUnmappableCharacter
 515      *   unmappable action} of this decoder is {@link
 516      *   CodingErrorAction#REPORT}; otherwise the unmappable character will be
 517      *   ignored or replaced, as requested.  </p></li>

 519      * </ul>

 567                                     boolean endOfInput)

 569         int newState = endOfInput ? ST_END : ST_CODING;
 570         if ((state != ST_RESET) && (state != ST_CODING)
 571             && !(endOfInput && (state == ST_END)))
 572             throwIllegalStateException(state, newState);
 573         state = newState;
 574 
 575         for (;;) {
 576 
 577             CoderResult cr;
 578             try {
 579                 cr = decodeLoop(in, out);
 580             } catch (BufferUnderflowException x) {
 581                 throw new CoderMalfunctionError(x);
 582             } catch (BufferOverflowException x) {
 583                 throw new CoderMalfunctionError(x);
 584             }
 585 
 586             if (cr.isOverflow())
 587                 return cr;
 588 
 589             if (cr.isUnderflow()) {
 590                 if (endOfInput && in.hasRemaining()) {
 591                     cr = CoderResult.malformedForLength(in.remaining());
 592                     // Fall through to malformed-input case
 593                 } else {
 594                     return cr;
 595                 }
 596             }
 597 
 598             CodingErrorAction action = null;
 599             if (cr.isMalformed())
 600                 action = malformedInputAction;
 601             else if (cr.isUnmappable())
 602                 action = unmappableCharacterAction;
 603             else
 604                 assert false : cr.toString();
 605 
 606             if (action == CodingErrorAction.REPORT)
 607                 return cr;
 608 
 609             if (action == CodingErrorAction.REPLACE) {
 610                 if (out.remaining() < replacement.length())
 611                     return CoderResult.OVERFLOW;
 612                 out.put(replacement);
 613             }
 614 
 615             if ((action == CodingErrorAction.IGNORE)
 616                 || (action == CodingErrorAction.REPLACE)) {
 617                 // Skip erroneous input either way
 618                 in.position(in.position() + cr.length());
 619                 continue;
 620             }
 621 
 622             assert false;
 623         }

 630      * <p> Some decoders maintain internal state and may need to write some

 658      * @throws  IllegalStateException

 660      *          invocation neither of the {@link #flush flush} method nor of

 672         }

 675             throwIllegalStateException(state, ST_FLUSHED);

 677         return CoderResult.UNDERFLOW; // Already flushed

 722     /**
 723      * Decodes one or more bytes into one or more characters.
 724      *
 725      * <p> This method encapsulates the basic decoding loop, decoding as many
 726      * bytes as possible until it either runs out of input, runs out of room
 727      * in the output buffer, or encounters a decoding error.  This method is
 728      * invoked by the {@link #decode decode} method, which handles result
 729      * interpretation and error recovery.
 730      *
 731      * <p> The buffers are read from, and written to, starting at their current
 732      * positions.  At most {@link Buffer#remaining in.remaining()} bytes
 733      * will be read, and at most {@link Buffer#remaining out.remaining()}
 734      * characters will be written.  The buffers' positions will be advanced to
 735      * reflect the bytes read and the characters written, but their marks and
 736      * limits will not be modified.
 737      *
 738      * <p> This method returns a {@link CoderResult} object to describe its
 739      * reason for termination, in the same manner as the {@link #decode decode}
 740      * method.  Most implementations of this method will handle decoding errors
 741      * by returning an appropriate result object for interpretation by the
 742      * {@link #decode decode} method.  An optimized implementation may instead
 743      * examine the relevant error action and implement that action itself.
 744      *
 745      * <p> An implementation of this method may perform arbitrary lookahead by
 746      * returning {@link CoderResult#UNDERFLOW} until it receives sufficient
 747      * input.  </p>
 748      *
 749      * @param  in
 750      *         The input byte buffer
 751      *
 752      * @param  out
 753      *         The output character buffer
 754      *
 755      * @return  A coder-result object describing the reason for termination
 756      */
 757     protected abstract CoderResult decodeLoop(ByteBuffer in,
 758                                               CharBuffer out);

 760     /**
 761      * Convenience method that decodes the remaining content of a single input
 762      * byte buffer into a newly-allocated character buffer.
 763      *
 764      * <p> This method implements an entire <a href="#steps">decoding
 765      * operation</a>; that is, it resets this decoder, then it decodes the
 766      * bytes in the given byte buffer, and finally it flushes this
 767      * decoder.  This method should therefore not be invoked if a decoding
 768      * operation is already in progress.  </p>
 769      *
 770      * @param  in
 771      *         The input byte buffer
 772      *
 773      * @return A newly-allocated character buffer containing the result of the
 774      *         decoding operation.  The buffer's position will be zero and its
 775      *         limit will follow the last character written.
 776      *
 777      * @throws  IllegalStateException
 778      *          If a decoding operation is already in progress
 779      *
 780      * @throws  MalformedInputException
 781      *          If the byte sequence starting at the input buffer's current
 782      *          position is not legal for this charset and the current malformed-input action
 783      *          is {@link CodingErrorAction#REPORT}
 784      *
 785      * @throws  UnmappableCharacterException
 786      *          If the byte sequence starting at the input buffer's current
 787      *          position cannot be mapped to an equivalent character sequence and
 788      *          the current unmappable-character action is {@link
 789      *          CodingErrorAction#REPORT}
 790      */
 791     public final CharBuffer decode(ByteBuffer in)
 792         throws CharacterCodingException
 793     {
 794         int n = (int)(in.remaining() * averageCharsPerByte());
 795         CharBuffer out = CharBuffer.allocate(n);
 796 
 797         if ((n == 0) && (in.remaining() == 0))
 798             return out;
 799         reset();
 800         for (;;) {
 801             CoderResult cr = in.hasRemaining() ?
 802                 decode(in, out, true) : CoderResult.UNDERFLOW;
 803             if (cr.isUnderflow())
 804                 cr = flush(out);
 805 
 806             if (cr.isUnderflow())
 807                 break;
 808             if (cr.isOverflow()) {
 809                 n = 2*n + 1;    // Ensure progress; n might be 0!
 810                 CharBuffer o = CharBuffer.allocate(n);
 811                 out.flip();
 812                 o.put(out);
 813                 out = o;
 814                 continue;
 815             }
 816             cr.throwException();
 817         }
 818         out.flip();
 819         return out;
 820     }
