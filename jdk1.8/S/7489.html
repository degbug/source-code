  73     {
  74         return new EncapsulationFactoryBase(ORBConstants.TAG_REQUEST_PARTITIONING_ID) {
  75             public Identifiable readContents(InputStream in)
  76             {
  77                 int threadPoolToUse = in.read_ulong();
