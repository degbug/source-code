  25 /**
  26  * The class ExtendedType represents an extended type object used by
  27  * ExpandedNameTable.
  28  */
  29 public final class ExtendedType
  30 {
  31     private int nodetype;
  32     private String namespace;
  33     private String localName;
  34     private int hash;
  35 
  36     /**
  37      * Create an ExtendedType object from node type, namespace and local name.
  38      * The hash code is calculated from the node type, namespace and local name.
  39      *
  40      * @param nodetype Type of the node
  41      * @param namespace Namespace of the node
  42      * @param localName Local name of the node
  43      */
  44     public ExtendedType (int nodetype, String namespace, String localName)
  45     {
  46       this.nodetype = nodetype;
  47       this.namespace = namespace;
  48       this.localName = localName;
  49       this.hash = nodetype + namespace.hashCode() + localName.hashCode();
  50     }
  51 
  52     /**
  53      * Create an ExtendedType object from node type, namespace, local name
  54      * and a given hash code.
  55      *
  56      * @param nodetype Type of the node
  57      * @param namespace Namespace of the node
  58      * @param localName Local name of the node
  59      * @param hash The given hash code
  60      */
  61     public ExtendedType (int nodetype, String namespace, String localName, int hash)
  62     {
  63       this.nodetype = nodetype;
  64       this.namespace = namespace;
  65       this.localName = localName;
  66       this.hash = hash;
  67     }
  68 
  69     /**
  70      * Redefine this ExtendedType object to represent a different extended type.
  71      * This is intended to be used ONLY on the hashET object. Using it elsewhere
  72      * will mess up existing hashtable entries!
  73      */
  74     protected void redefine(int nodetype, String namespace, String localName)
  75     {
  76       this.nodetype = nodetype;
  77       this.namespace = namespace;
  78       this.localName = localName;
  79       this.hash = nodetype + namespace.hashCode() + localName.hashCode();
  80     }
  81 
  82     /**
  83      * Redefine this ExtendedType object to represent a different extended type.
  84      * This is intended to be used ONLY on the hashET object. Using it elsewhere
  85      * will mess up existing hashtable entries!
  86      */
  87     protected void redefine(int nodetype, String namespace, String localName, int hash)
  88     {
  89       this.nodetype = nodetype;
  90       this.namespace = namespace;

 103     /**
 104      * Test if this ExtendedType object is equal to the given ExtendedType.

 106      * @param other The other ExtendedType object to test for equality

 108      */
 109     public boolean equals(ExtendedType other)
 110     {
