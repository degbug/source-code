  26 /**
  27  *
  28  * Like IntVector, but used only for the OpMap array.  Length of array
  29  * is kept in the m_lengthPos position of the array.  Only the required methods
  30  * are in included here.
  31  * @xsl.usage internal
  32  */
  33 public class OpMapVector {
  34 
  35  /** Size of blocks to allocate          */
  36   protected int m_blocksize;
  37 
  38   /** Array of ints          */
  39   protected int m_map[]; // IntStack is trying to see this directly
  40 
  41   /** Position where size of array is kept          */
  42   protected int m_lengthPos = 0;
  43 
  44   /** Size of array          */
  45   protected int m_mapSize;
  46 
  47     /**
  48    * Construct a OpMapVector, using the given block size.
  49    *
  50    * @param blocksize Size of block to allocate
  51    */
  52   public OpMapVector(int blocksize, int increaseSize, int lengthPos)
  53   {
  54 
  55     m_blocksize = increaseSize;
  56     m_mapSize = blocksize;
  57     m_lengthPos = lengthPos;
  58     m_map = new int[blocksize];
  59   }
  60 
  61   /**
  62    * Get the nth element.
  63    *
  64    * @param i index of object to get
  65    *
  66    * @return object at given index
  67    */
  68   public final int elementAt(int i)
  69   {
  70     return m_map[i];
  71   }
  72 
  73     /**
  74    * Sets the component at the specified index of this vector to be the
  75    * specified object. The previous component at that position is discarded.
  76    *
  77    * The index must be a value greater than or equal to 0 and less
  78    * than the current size of the vector.
  79    *
  80    * @param value object to set
  81    * @param index Index of where to set the object
  82    */
  83   public final void setElementAt(int value, int index)
  84   {
  85     if (index >= m_mapSize)
  86     {
  87       int oldSize = m_mapSize;
  88 
  89       m_mapSize += m_blocksize;
  90 
  91       int newMap[] = new int[m_mapSize];
  92 
  93       System.arraycopy(m_map, 0, newMap, 0, oldSize);
  94 
  95       m_map = newMap;
  96     }
  97 
  98     m_map[index] = value;
  99   }
 100 
 101 
 102   /*
 103    * Reset the array to the supplied size.  No checking is done.
 104    *
 105    * @param size The size to trim to.
 106    */
 107   public final void setToSize(int size) {
 108 
 109     int newMap[] = new int[size];
 110 
 111     System.arraycopy(m_map, 0, newMap, 0, m_map[m_lengthPos]);
 112 
 113     m_mapSize = size;
 114     m_map = newMap;
 115 
 116   }
 117 
 118 }
