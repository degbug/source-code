  21 package com.sun.org.apache.regexp.internal;
  22 
  23 /**
  24  * Exception thrown to indicate a syntax error in a regular expression.
  25  * This is a non-checked exception because you should only have problems compiling
  26  * a regular expression during development.
  27  * If you are making regular expresion programs dynamically then you can catch it
  28  * if you wish. But should not be forced to.
  29  *
  30  * @author <a href="mailto:jonl@muppetlabs.com">Jonathan Locke</a>
  31  * @author <a href="mailto:gholam@xtra.co.nz>Michael McCallum</a>
  32  */
  33 public class RESyntaxException extends RuntimeException
  34 {
  35     /**
  36      * Constructor.
  37      * @param s Further description of the syntax error
  38      */
  39     public RESyntaxException(String s)
  40     {
  41         super("Syntax error: " + s);
  42     }
  43 }
