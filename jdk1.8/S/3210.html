  58     /**
  59      * Converts HSL components of a color to a set of RGB components.
  60      *
  61      * @param hsl  a float array with length equal to
  62      *             the number of HSL components
  63      * @param rgb  a float array with length of at least 3
  64      *             that contains RGB components of a color
  65      * @return a float array that contains RGB components
  66      */
  67     private static float[] HSLtoRGB(float[] hsl, float[] rgb) {
  68         if (rgb == null) {
  69             rgb = new float[3];
  70         }
  71         float hue = hsl[0];
  72         float saturation = hsl[1];
  73         float lightness = hsl[2];
  74 
  75         if (saturation > 0.0f) {
  76             hue = (hue < 1.0f) ? hue * 6.0f : 0.0f;
  77             float q = lightness + saturation * ((lightness > 0.5f) ? 1.0f - lightness : lightness);
  78             float p = 2.0f * lightness - q;
  79             rgb[0]= normalize(q, p, (hue < 4.0f) ? (hue + 2.0f) : (hue - 4.0f));
  80             rgb[1]= normalize(q, p, hue);
  81             rgb[2]= normalize(q, p, (hue < 2.0f) ? (hue + 4.0f) : (hue - 2.0f));
  82         }
  83         else {
  84             rgb[0] = lightness;
  85             rgb[1] = lightness;
  86             rgb[2] = lightness;
  87         }
  88         return rgb;
  89     }

  91     /**
  92      * Converts RGB components of a color to a set of HSL components.
  93      *
  94      * @param rgb  a float array with length of at least 3
  95      *             that contains RGB components of a color
  96      * @param hsl  a float array with length equal to
  97      *             the number of HSL components
  98      * @return a float array that contains HSL components
  99      */
 100     private static float[] RGBtoHSL(float[] rgb, float[] hsl) {
 101         if (hsl == null) {
 102             hsl = new float[3];
 103         }
 104         float max = max(rgb[0], rgb[1], rgb[2]);
 105         float min = min(rgb[0], rgb[1], rgb[2]);
 106 
 107         float summa = max + min;
 108         float saturation = max - min;
 109         if (saturation > 0.0f) {
 110             saturation /= (summa > 1.0f)
 111                     ? 2.0f - summa
 112                     : summa;
 113         }
 114         hsl[0] = getHue(rgb[0], rgb[1], rgb[2], max, min);
 115         hsl[1] = saturation;
 116         hsl[2] = summa / 2.0f;
 117         return hsl;
 118     }

 146     /**
 147      * Calculates the hue component for HSL and HSV color spaces.
 148      *
 149      * @param red    the red component of the color
 150      * @param green  the green component of the color
 151      * @param blue   the blue component of the color
 152      * @param max    the larger of {@code red}, {@code green} and {@code blue}
 153      * @param min    the smaller of {@code red}, {@code green} and {@code blue}
 154      * @return the hue component
 155      */
 156     static float getHue(float red, float green, float blue, float max, float min) {
 157         float hue = max - min;
 158         if (hue > 0.0f) {
 159             if (max == red) {
 160                 hue = (green - blue) / hue;
 161                 if (hue < 0.0f) {
 162                     hue += 6.0f;
 163                 }
 164             }
 165             else if (max == green) {
 166                 hue = 2.0f + (blue - red) / hue;
 167             }
 168             else /*max == blue*/ {
 169                 hue = 4.0f + (red - green) / hue;
 170             }
 171             hue /= 6.0f;
 172         }
 173         return hue;
 174     }
 175 
 176     private static float normalize(float q, float p, float color) {
 177         if (color < 1.0f) {
 178             return p + (q - p) * color;
 179         }
 180         if (color < 3.0f) {
 181             return q;
 182         }
 183         if (color < 4.0f) {
 184             return p + (q - p) * (4.0f - color);
 185         }
 186         return p;
 187     }
