    @see Expression#deepEquals(Expression)

 220   public boolean deepEquals(Expression expr)
 221   {
 222         if(!isSameClass(expr))
 223                 return false;
 224 
 225         NodeTest nt = (NodeTest)expr;
 226 
 227         if(null != nt.m_name)
 228         {
 229                 if(null == m_name)
 230                         return false;
 231                 else if(!nt.m_name.equals(m_name))
 232                         return false;
 233         }
 234         else if(null != m_name)
 235                 return false;
 236 
 237         if(null != nt.m_namespace)
 238         {
 239                 if(null == m_namespace)
 240                         return false;
 241                 else if(!nt.m_namespace.equals(m_namespace))
 242                         return false;
 243         }
 244         else if(null != m_namespace)
 245                 return false;
 246 
 247         if(m_whatToShow != nt.m_whatToShow)
 248                 return false;
 249 
 250         if(m_isTotallyWild != nt.m_isTotallyWild)
 251                 return false;
 252 
 253         return true;
 254   }

    Static calc of match score.

 323   protected void calcScore()
 324   {
 325 
 326     if ((m_namespace == null) && (m_name == null))
 327       m_score = SCORE_NODETEST;
 328     else if (((m_namespace == WILD) || (m_namespace == null))
 329              && (m_name == WILD))
 330       m_score = SCORE_NODETEST;
 331     else if ((m_namespace != WILD) && (m_name == WILD))
 332       m_score = SCORE_NSWILD;
 333     else
 334       m_score = SCORE_QNAME;
 335 
 336     m_isTotallyWild = (m_namespace == null && m_name == WILD);
 337   }

    Tell what the test score is for the given node.
    @param xctxt XPath runtime context.
    @param context The node being tested.
    @return {@link com.sun.org.apache.xpath.internal.patterns.NodeTest#SCORE_NODETEST},
            {@link com.sun.org.apache.xpath.internal.patterns.NodeTest#SCORE_NONE},
            {@link com.sun.org.apache.xpath.internal.patterns.NodeTest#SCORE_NSWILD},
            {@link com.sun.org.apache.xpath.internal.patterns.NodeTest#SCORE_QNAME}, or
            {@link com.sun.org.apache.xpath.internal.patterns.NodeTest#SCORE_OTHER}.
    @throws javax.xml.transform.TransformerException

 527   public XObject execute(XPathContext xctxt, int context)
 528           throws javax.xml.transform.TransformerException
 529   {
 530 
 531     DTM dtm = xctxt.getDTM(context);
 532     short nodeType = dtm.getNodeType(context);
 533 
 534     if (m_whatToShow == DTMFilter.SHOW_ALL)
 535       return m_score;
 536 
 537     int nodeBit = (m_whatToShow & (0x00000001 << (nodeType - 1)));
 538 
 539     switch (nodeBit)
 540     {
 541     case DTMFilter.SHOW_DOCUMENT_FRAGMENT :
 542     case DTMFilter.SHOW_DOCUMENT :
 543       return SCORE_OTHER;
 544     case DTMFilter.SHOW_COMMENT :
 545       return m_score;
 546     case DTMFilter.SHOW_CDATA_SECTION :
 547     case DTMFilter.SHOW_TEXT :
 548 
 549       // was:
 550       // return (!xctxt.getDOMHelper().shouldStripSourceNode(context))
 551       //       ? m_score : SCORE_NONE;
 552       return m_score;
 553     case DTMFilter.SHOW_PROCESSING_INSTRUCTION :
 554       return subPartMatch(dtm.getNodeName(context), m_name)
 555              ? m_score : SCORE_NONE;
 556 
 557     // From the draft: "Two expanded names are equal if they
 558     // have the same local part, and either both have no URI or
 559     // both have the same URI."
 560     // "A node test * is true for any node of the principal node type.
 561     // For example, child::* will select all element children of the
 562     // context node, and attribute::* will select all attributes of
 563     // the context node."
 564     // "A node test can have the form NCName:*. In this case, the prefix
 565     // is expanded in the same way as with a QName using the context
 566     // namespace declarations. The node test will be true for any node
 567     // of the principal type whose expanded name has the URI to which
 568     // the prefix expands, regardless of the local part of the name."
 569     case DTMFilter.SHOW_NAMESPACE :
 570     {
 571       String ns = dtm.getLocalName(context);
 572 
 573       return (subPartMatch(ns, m_name)) ? m_score : SCORE_NONE;
 574     }
 575     case DTMFilter.SHOW_ATTRIBUTE :
 576     case DTMFilter.SHOW_ELEMENT :
 577     {
 578       return (m_isTotallyWild || (subPartMatchNS(dtm.getNamespaceURI(context), m_namespace) && subPartMatch(dtm.getLocalName(context), m_name)))
 579              ? m_score : SCORE_NONE;
 580     }
 581     default :
 582       return SCORE_NONE;
 583     }  // end switch(testType)
 584   }
