  96     protected void paintTabBorder( Graphics g, int tabPlacement,
  97                                    int tabIndex, int x, int y, int w, int h,
  98                                    boolean isSelected) {
  99         int bottom = y + (h-1);
 100         int right = x + (w-1);
 101 
 102         switch ( tabPlacement ) {
 103         case LEFT:
 104             paintLeftTabBorder(tabIndex, g, x, y, w, h, bottom, right, isSelected);
 105             break;
 106         case BOTTOM:
 107             paintBottomTabBorder(tabIndex, g, x, y, w, h, bottom, right, isSelected);
 108             break;
 109         case RIGHT:
 110             paintRightTabBorder(tabIndex, g, x, y, w, h, bottom, right, isSelected);
 111             break;
 112         case TOP:
 113         default:
 114             paintTopTabBorder(tabIndex, g, x, y, w, h, bottom, right, isSelected);
 115         }
 116     }

 258                 //  first tab in run but not first tab in last run
 259                 if (tabPane.getSelectedIndex()==tabRuns[currentRun+1]) {
 260                     // tab in front of selected tab
 261                     g.setColor( selectHighlight );
 262                 }
 263                 else {

 271             // Paint slant

 273 
 274             // Paint top

 276 
 277             // Paint left

 279                 // last tab in run
 280                 g.drawLine( 1, 1, 1, bottom );
 281             } else {
 282                 g.drawLine( 0, 1, 0, bottom );
 283             }
 284         }
 285 
 286         g.translate( -x, -y );
 287     }

 289     protected boolean shouldFillGap( int currentRun, int tabIndex, int x, int y ) {
 290         boolean result = false;
 291 
 292         if (!tabsOpaque) {
 293             return false;
 294         }
 295 
 296         if ( currentRun == runCount - 2 ) {  // If it's the second to last row.
 297             Rectangle lastTabBounds = getTabBounds( tabPane, tabPane.getTabCount() - 1 );
 298             Rectangle tabBounds = getTabBounds( tabPane, tabIndex );
 299             if (MetalUtils.isLeftToRight(tabPane)) {
 300                 int lastTabRight = lastTabBounds.x + lastTabBounds.width - 1;
 301 
 302                 // is the right edge of the last tab to the right
 303                 // of the left edge of the current tab?
 304                 if ( lastTabRight > tabBounds.x + 2 ) {
 305                     return true;
 306                 }
 307             } else {
 308                 int lastTabLeft = lastTabBounds.x;
 309                 int currentTabRight = tabBounds.x + tabBounds.width - 1;
 310 
 311                 // is the left edge of the last tab to the left
 312                 // of the right edge of the current tab?
 313                 if ( lastTabLeft < currentTabRight - 2 ) {
 314                     return true;
 315                 }
 316             }
 317         } else {
 318             // fill in gap for all other rows except last row
 319             result = currentRun != runCount - 1;
 320         }
 321 
 322         return result;
 323     }

 325     protected Color getColorForGap( int currentRun, int x, int y ) {
 326         final int shadowWidth = 4;
 327         int selectedIndex = tabPane.getSelectedIndex();
 328         int startIndex = tabRuns[ currentRun + 1 ];
 329         int endIndex = lastTabInRun( tabPane.getTabCount(), currentRun + 1 );
 330         int tabOverGap = -1;
 331         // Check each tab in the row that is 'on top' of this row
 332         for ( int i = startIndex; i <= endIndex; ++i ) {
 333             Rectangle tabBounds = getTabBounds( tabPane, i );
 334             int tabLeft = tabBounds.x;
 335             int tabRight = (tabBounds.x + tabBounds.width) - 1;
 336             // Check to see if this tab is over the gap
 337             if ( MetalUtils.isLeftToRight(tabPane) ) {
 338                 if ( tabLeft <= x && tabRight - shadowWidth > x ) {
 339                     return selectedIndex == i ? selectColor : getUnselectedBackgroundAt( i );
 340                 }
 341             }
 342             else {
 343                 if ( tabLeft + shadowWidth < x && tabRight >= x ) {
 344                     return selectedIndex == i ? selectColor : getUnselectedBackgroundAt( i );
 345                 }
 346             }
 347         }
 348 
 349         return tabPane.getBackground();
 350     }

 352     protected void paintLeftTabBorder( int tabIndex, Graphics g,
 353                                        int x, int y, int w, int h,
 354                                        int btm, int rght,
 355                                        boolean isSelected ) {
 356         int tabCount = tabPane.getTabCount();

 358         int lastIndex = lastTabInRun( tabCount, currentRun );
 359         int firstIndex = tabRuns[ currentRun ];
 360 
 361         g.translate( x, y );
 362 
 363         int bottom = h - 1;
 364         int right = w - 1;
 365 
 366         //
 367         // Paint part of the tab above
 368         //
 369 
 370         if ( tabIndex != firstIndex && tabsOpaque ) {
 371             g.setColor( tabPane.getSelectedIndex() == tabIndex - 1 ?
 372                         selectColor :
 373                         getUnselectedBackgroundAt( tabIndex - 1 ) );
 374             g.fillRect( 2, 0, 4, 3 );
 375             g.drawLine( 2, 3, 2, 3 );
 376         }
 377 
 378 
 379         //
 380         // Paint Highlight
 381         //
 382 
 383         if (ocean) {
 384             g.setColor(isSelected ? selectHighlight :
 385                        MetalLookAndFeel.getWhite());
 386         }
 387         else {
 388             g.setColor( isSelected ? selectHighlight : highlight );
 389         }
 390 
 391         // Paint slant
 392         g.drawLine( 1, 6, 6, 1 );
 393 
 394         // Paint left
 395         g.drawLine( 1, 6, 1, bottom );
 396 
 397         // Paint top
 398         g.drawLine( 6, 1, right, 1 );
 399 
 400         if ( tabIndex != firstIndex ) {
 401             if (tabPane.getSelectedIndex() == tabIndex - 1) {
 402                 g.setColor(selectHighlight);
 403             } else {
 404                 g.setColor(ocean ? MetalLookAndFeel.getWhite() : highlight);
 405             }
 406 
 407             g.drawLine( 1, 0, 1, 4 );
 408         }
 409 
 410         //
 411         // Paint Border
 412         //
 413 
 414         if (ocean) {
 415             if (isSelected) {
 416                 g.setColor(oceanSelectedBorderColor);
 417             }
 418             else {
 419                 g.setColor( darkShadow );
 420             }
 421         }
 422         else {
 423             g.setColor( darkShadow );
 424         }
 425 
 426         // Paint slant
 427         g.drawLine( 1, 5, 6, 0 );
 428 
 429         // Paint top
 430         g.drawLine( 6, 0, right, 0 );
 431 
 432         // Paint bottom
 433         if ( tabIndex == lastIndex ) {
 434             g.drawLine( 0, bottom, right, bottom );
 435         }
 436 
 437         // Paint left
 438         if (ocean) {
 439             if (tabPane.getSelectedIndex() == tabIndex - 1) {
 440                 g.drawLine(0, 5, 0, bottom);
 441                 g.setColor(oceanSelectedBorderColor);
 442                 g.drawLine(0, 0, 0, 5);
 443             }
 444             else if (isSelected) {
 445                 g.drawLine( 0, 6, 0, bottom );
 446                 if (tabIndex != 0) {
 447                     g.setColor(darkShadow);
 448                     g.drawLine(0, 0, 0, 5);
 449                 }
 450             }
 451             else if ( tabIndex != firstIndex ) {
 452                 g.drawLine( 0, 0, 0, bottom );
 453             } else {
 454                 g.drawLine( 0, 6, 0, bottom );
 455             }
 456         }
 457         else { // metal
 458             if ( tabIndex != firstIndex ) {
 459                 g.drawLine( 0, 0, 0, bottom );
 460             } else {
 461                 g.drawLine( 0, 6, 0, bottom );
 462             }
 463         }
 464 
 465         g.translate( -x, -y );
 466     }

1132     protected int calculateMaxTabHeight( int tabPlacement ) {
1133         FontMetrics metrics = getFontMetrics();
1134         int height = metrics.getHeight();
1135         boolean tallerIcons = false;
1136 
1137         for ( int i = 0; i < tabPane.getTabCount(); ++i ) {
1138             Icon icon = tabPane.getIconAt( i );
1139             if ( icon != null ) {
1140                 if ( icon.getIconHeight() > height ) {
1141                     tallerIcons = true;
1142                     break;
1143                 }
1144             }
1145         }
1146         return super.calculateMaxTabHeight( tabPlacement ) -
1147                   (tallerIcons ? (tabInsets.top + tabInsets.bottom) : 0);
1148     }
