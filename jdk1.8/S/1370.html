  34 /**
  35  * A PrintStream adds functionality to another output stream,
  36  * namely the ability to print representations of various data values
  37  * conveniently.  Two other features are provided as well.  Unlike other output
  38  * streams, a PrintStream never throws an
  39  * IOException; instead, exceptional situations merely set an
  40  * internal flag that can be tested via the checkError method.
  41  * Optionally, a PrintStream can be created so as to flush
  42  * automatically; this means that the flush method is
  43  * automatically invoked after a byte array is written, one of the
  44  * println methods is invoked, or a newline character or byte
  45  * ('\n') is written.
  46  *
  47  *  All characters printed by a PrintStream are converted into
  48  * bytes using the platform's default character encoding.  The {@link
  49  * PrintWriter} class should be used in situations that require writing
  50  * characters rather than bytes.
  51  *
  52  * @author     Frank Yellin
  53  * @author     Mark Reinhold
  54  * @since      JDK1.0
  55  */

 372     /**
 373      * Flushes the stream and checks its error state. The internal error state
 374      * is set to true when the underlying output stream throws an
 375      * IOException other than InterruptedIOException,
 376      * and when the setError method is invoked.  If an operation
 377      * on the underlying output stream throws an
 378      * InterruptedIOException, then the PrintStream
 379      * converts the exception back into an interrupt by doing:
 380      * 

 381      *     Thread.currentThread().interrupt();
 382      * 
 383      * or the equivalent.
 384      *
 385      * @return true if and only if this stream has encountered an
 386      *         IOException other than
 387      *         InterruptedIOException, or the
 388      *         setError method has been invoked
 389      */
 390     public boolean checkError() {
 391         if (out != null)
 392             flush();
 393         if (out instanceof java.io.PrintStream) {
 394             PrintStream ps = (PrintStream) out;
 395             return ps.checkError();
 396         }
 397         return trouble;
 398     }

 431     /**
 432      * Writes the specified byte to this stream.  If the byte is a newline and
 433      * automatic flushing is enabled then the flush method will be
 434      * invoked.
 435      *
 436      *  Note that the byte is written as given; to write a character that
 437      * will be translated according to the platform's default character
 438      * encoding, use the print(char) or println(char)
 439      * methods.
 440      *
 441      * @param  b  The byte to be written
 442      * @see #print(char)
 443      * @see #println(char)
 444      */
 445     public void write(int b) {
 446         try {
 447             synchronized (this) {
 448                 ensureOpen();
 449                 out.write(b);
 450                 if ((b == '\n') && autoFlush)
 451                     out.flush();
 452             }
 453         }
 454         catch (InterruptedIOException x) {
 455             Thread.currentThread().interrupt();
 456         }
 457         catch (IOException x) {
 458             trouble = true;
 459         }
 460     }

 499     private void write(char buf[]) {
 500         try {
 501             synchronized (this) {
 502                 ensureOpen();
 503                 textOut.write(buf);
 504                 textOut.flushBuffer();
 505                 charOut.flushBuffer();
 506                 if (autoFlush) {
 507                     for (int i = 0; i < buf.length; i++)
 508                         if (buf[i] == '\n')
 509                             out.flush();
 510                 }
 511             }
 512         }
 513         catch (InterruptedIOException x) {
 514             Thread.currentThread().interrupt();
 515         }
 516         catch (IOException x) {
 517             trouble = true;
 518         }
 519     }
