  25 package org.omg.CORBA.portable;
  26 
  27 /**
  28 This class is used for reporting locate forward exceptions and object forward
  29 GIOP messages back to the ORB. In this case the ORB must remarshal the request
  30 before trying again.
  31 Stubs which use the stream-based model shall catch the RemarshalException
  32 which is potentially thrown from the _invoke() method of ObjectImpl.
  33 Upon catching the exception, the stub shall immediately remarshal the request by calling
  34 _request(), marshalling the arguments (if any), and then calling
  35 _invoke(). The stub shall repeat this process until _invoke()
  36 returns normally or raises some exception other than RemarshalException.
  37 */
  38 
  39 public final class RemarshalException extends Exception {
  40     /**
  41      * Constructs a RemarshalException.
  42      */
  43     public RemarshalException() {
  44         super();
  45     }
  46 }
