  30 /** The Transient Object Adapter is used for standard RMI-IIOP and Java-IDL
  31  * (legacy JDK 1.2) object implementations.  Its protocol for managing objects is very
  32  * simple: just connect and disconnect.  There is only a single TOA instance per ORB,
  33  * and its lifetime is the same as the ORB.  The TOA instance is always ready to receive
  34  * messages except when the ORB is shutting down.
  35  */
  36 public interface TOA extends ObjectAdapter {
  37     /** Connect the given servant to the ORB by allocating a transient object key
  38      *  and creating an IOR and object reference using the current factory.
  39      */
  40     void connect( org.omg.CORBA.Object servant ) ;
  41 
  42     /** Disconnect the object from this ORB.
  43     */
  44     void disconnect( org.omg.CORBA.Object obj ) ;
  45 }
