  28 // Introduce more information about WHY we are re-trying a request
  29 // so we can properly handle the two cases:
  30 // - BEFORE_RESPONSE means that the retry is caused by
  31 //   something that happened BEFORE the message was sent: either
  32 //   an exception from the SocketFactory, or one from the
  33 //   Client side send_request interceptor point.
  34 // - AFTER_RESPONSE means that the retry is a result either of the
  35 //   request sent to the server (from the response), or from the
  36 //   Client side receive_xxx interceptor point.
  37 public enum RetryType {
  38     NONE( false ),
  39     BEFORE_RESPONSE( true ),
  40     AFTER_RESPONSE( true ) ;
  41 
  42     private final boolean isRetry ;
  43 
  44     RetryType( boolean isRetry ) {
  45         this.isRetry = isRetry ;
  46     }
  47 
  48     public boolean isRetry() {
  49         return this.isRetry ;
  50     }
  51 } ;
